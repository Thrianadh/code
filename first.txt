import os

def extract_paragraphs(doc):
    """Extract all paragraphs from the document."""
    paragraphs = []
    for para in doc.paragraphs:
        paragraphs.append({
            "text": para.text,
            "style": para.style.name,
            "alignment": para.alignment,
            "bold": any(run.bold for run in para.runs),
            "italic": any(run.italic for run in para.runs),
        })
    return paragraphs

def extract_tables(doc):
    """Extract all tables from the document."""
    tables = []
    for table in doc.tables:
        table_data = []
        for row in table.rows:
            row_data = [cell.text for cell in row.cells]
            table_data.append(row_data)
        tables.append(table_data)
    return tables

def extract_headings(doc):
    """Extract headings (paragraphs with Heading styles)."""
    headings = []
    for para in doc.paragraphs:
        if para.style.name.startswith("Heading"):
            headings.append({
                "text": para.text,
                "level": int(para.style.name.split(" ")[-1]),  # e.g., "Heading 1" -> 1
            })
    return headings

def save_to_json(data, output_file="extracted_data.json"):
    """Save extracted data to a JSON file."""
    with open(output_file, "w", encoding="utf-8") as f:
        json.dump(data, f, indent=4, ensure_ascii=False)
    print(f"Data saved to {output_file}")

def extract_docx(file_path):
    """Main function to extract data from a .docx file."""
    doc = Document(file_path)
    data = {
        "metadata": {
            "file_name": os.path.basename(file_path),
            "num_paragraphs": len(doc.paragraphs),
            "num_tables": len(doc.tables),
        },
        "headings": extract_headings(doc),
        "paragraphs": extract_paragraphs(doc),
        "tables": extract_tables(doc),
    }
    return data

if __name__ == "__main__":
    # Example usage
    input_file = "onenote_data.docx"  # Replace with your file path

    if os.path.exists(input_file):
        extracted_data = extract_docx(input_file)
        save_to_json(extracted_data, output_file)

#!/usr/bin/env python3
"""
Simplified DOCX Diagnostic Extractor
======================================
Uses mammoth to convert .docx → markdown, then analyzes structure.
Reuses helpers from common.py style.

Outputs:
  - report.txt          Human-readable diagnostic report
  - sections.json       Detected sections with content
  - tables.md           Tables found in markdown format
  - raw.md              Full markdown text

Usage:
  python docx_diagnostic_simple.py --input file.docx --output ./report
"""

import argparse
import json
import os
import re
from pathlib import Path
from typing import List, Dict, Tuple, Optional

import mammoth


def load_docx_as_markdown(path: Path) -> str:
    """Read .docx and return markdown (mammoth). Skips broken images."""
    def _ignore_image(_image):
        return []
    with path.open("rb") as fh:
        result = mammoth.convert_to_markdown(
            fh, convert_image=mammoth.images.img_element(_ignore_image)
        )
    return result.value


def looks_like_heading(line: str) -> Optional[str]:
    """Detect __bold__ lines as pseudo-headings (OneNote copy-paste style)."""
    match = re.match(r'^\s*__([^_].{2,120}?)__\s*$', line)
    if not match:
        return None
    text = match.group(1).strip()
    if len(text) > 120 or text.endswith('.'):
        return None
    return text


def normalize_whitespace(text: str) -> str:
    """Collapse blank lines and strip trailing whitespace."""
    lines = [line.rstrip() for line in text.splitlines()]
    return re.sub(r'\n{3,}', '\n\n', '\n'.join(lines)).strip()


def strip_markdown_bold(text: str) -> str:
    """Remove __bold__ markers."""
    return re.sub(r'__([^_]+)__', r'\1', text)

# DIAGNOSTIC ANALYZER
# =============================================================================
class DocxDiagnosticAnalyzer:
    """Analyzes markdown from mammoth to find structure and issues."""

    def __init__(self, markdown_text: str, filename: str):
        self.text = normalize_whitespace(markdown_text)
        self.filename = filename
        self.lines = self.text.splitlines()
        self.sections: List[Dict] = []
        self.tables: List[str] = []
        self.issues: List[str] = []

    def analyze(self):
        """Run all analysis steps."""
        self._find_tables()
        self._find_sections()
        self._find_issues()

    def _find_tables(self):
        """Extract markdown tables (lines starting with |)."""
        in_table = False
        current_table = []

        for line in self.lines:
            stripped = line.strip()
            if stripped.startswith('|'):
                in_table = True
                current_table.append(stripped)
            elif in_table:
                # Table ended
                if len(current_table) >= 2:  # Need at least header + separator
                    self.tables.append('\n'.join(current_table))
                in_table = False
                current_table = []

        # Catch last table
        if in_table and len(current_table) >= 2:
            self.tables.append('\n'.join(current_table))

    def _find_sections(self):
        """Split markdown into sections by heading lines."""
        current = {
            'heading': 'Introduction',
            'heading_level': 0,
            'lines': [],
            'tables': [],
            'word_count': 0
        }
        table_idx = 0
        tables_in_section = []

        for line in self.lines:
            heading_text = looks_like_heading(line)

            if heading_text:
                # Save previous section
                current['word_count'] = len(' '.join(current['lines']).split())
                current['tables'] = tables_in_section
                self.sections.append(current)

                # Start new section
                current = {
                    'heading': heading_text,
                    'heading_level': 1,  # All __bold__ treated as H1
                    'lines': [],
                    'tables': [],
                    'word_count': 0
                }
                tables_in_section = []

            elif line.strip().startswith('|'):
                # Table line - associate with current section
                if table_idx < len(self.tables):
                    tables_in_section.append(table_idx)
                    table_idx += 1
            else:
                current['lines'].append(line)

        # Save last section
        current['word_count'] = len(' '.join(current['lines']).split())
        current['tables'] = tables_in_section
        self.sections.append(current)

    def _find_issues(self):
        """Detect common problems in the markdown."""
        # 1. Check for images (mammoth usually drops them, but check for placeholders)
        image_markers = ['[image]', '![]', 'Image:', 'Figure:']
        image_count = sum(1 for line in self.lines if any(m in line for m in image_markers))
        if image_count > 0:
            self.issues.append(
                f"🔴 Found {image_count} image reference(s). "
                f"Images were skipped by mammoth — content may be missing. "
                f"Consider manual transcription or OCR."
            )

        # 2. Check for tables
        if self.tables:
            self.issues.append(
                f"🟡 Found {len(self.tables)} table(s). "
                f"Verify row alignment in markdown output."
            )

            # Check for very wide tables (many columns)
            for i, table in enumerate(self.tables):
                first_line = table.split('\n')[0]
                col_count = first_line.count('|') - 1
                if col_count > 5:
                    self.issues.append(
                        f"🟡 Table {i+1} has {col_count} columns — "
                        f"may be hard for LLM to read. Consider simplifying."
                    )

        # 3. Check for very long sections
        long_sections = [s for s in self.sections if s['word_count'] > 1000]
        if long_sections:
            self.issues.append(
                f"🟡 {len(long_sections)} section(s) have >1000 words. "
                f"Consider splitting for better RAG retrieval."
            )

        # 4. Check for very short sections
        short_sections = [s for s in self.sections if 0 < s['word_count'] < 20]
        if short_sections:
            self.issues.append(
                f"🟡 {len(short_sections)} section(s) have <20 words. "
                f"These may be formatting artifacts or empty headings."
            )

        # 5. Check for no headings at all
        heading_sections = [s for s in self.sections if s['heading_level'] > 0]
        if len(heading_sections) <= 1:
            self.issues.append(
                f"🔴 Only {len(heading_sections)} heading(s) detected. "
                f"Document may lack structure. Check if bold text was preserved in OneNote transfer."
            )

        # 6. Check for potential multi-column (alternating short lines)
        short_lines = [l for l in self.lines if 10 < len(l.strip()) < 60]
        if len(short_lines) > 20:
            # Check for pairs of lines with similar length (side-by-side)
            pairs = 0
            for i in range(0, len(short_lines)-1, 2):
                if abs(len(short_lines[i]) - len(short_lines[i+1])) < 10:
                    pairs += 1
            if pairs > 5:
                self.issues.append(
                    "🟡 Potential multi-column layout detected (pairs of similar-length short lines). "
                    "Verify addresses and paired data are not interleaved."
                )

        # 7. Check for embedded document references
        doc_refs = ['.docx', '.pdf', '.xlsx', 'Template', 'Attachment']
        ref_count = sum(1 for line in self.lines if any(r in line for r in doc_refs))
        if ref_count > 0:
            self.issues.append(
                f"🟡 Found {ref_count} reference(s) to embedded documents/templates. "
                f"Content from these files is not included in the markdown."
            )

        # 8. Check for URLs
        urls = re.findall(r'https?://[^\s]+', self.text)
        if urls:
            self.issues.append(
                f"🟡 Found {len(urls)} URL(s). External content is not included: "
                f"{', '.join(urls[:3])}{'...' if len(urls) > 3 else ''}"
            )

        # 9. Check for numbered lists that might be procedures
        numbered = [l for l in self.lines if re.match(r'^\s*\d+[\.\)]\s+', l.strip())]
        if len(numbered) > 5:
            self.issues.append(
                f"🟡 Found {len(numbered)} numbered list items. "
                f"Ensure these are grouped under correct section headings."
            )

    def generate_report(self) -> str:
        """Generate human-readable diagnostic report."""
        lines = [
            "=" * 70,
            "DOCX DIAGNOSTIC REPORT (Markdown-based)",
            "=" * 70,
            "",
            f"File: {self.filename}",
            f"",
            f"STATISTICS",
            f"  Total Lines:        {len(self.lines)}",
            f"  Total Words:        {len(self.text.split())}",
            f"  Sections Found:     {len(self.sections)}",
            f"  Tables Found:       {len(self.tables)}",
            f"  Issues Found:       {len(self.issues)}",
            f"",
            f"SECTIONS BREAKDOWN",
        ]

        for i, sec in enumerate(self.sections):
            lines.append(
                f"  [{i+1}] {sec['heading'][:50]} | "
                f"{len(sec['lines'])} lines | {sec['word_count']} words | "
                f"{len(sec['tables'])} tables"
            )

        lines.extend([
            f"",
            f"TABLES BREAKDOWN",
        ])

        for i, table in enumerate(self.tables):
            first_line = table.split('\n')[0]
            col_count = first_line.count('|') - 1
            row_count = len([l for l in table.split('\n') if l.strip().startswith('|')])
            lines.append(f"  [T{i+1}] {row_count} rows x {col_count} cols")

        lines.extend([
            f"",
            f"ISSUES & RECOMMENDATIONS",
        ])

        if self.issues:
            for issue in self.issues:
                lines.append(f"  {issue}")
        else:
            lines.append("  ✅ No major issues detected.")

        lines.extend([
            f"",
            f"RECOMMENDATIONS FOR RAG INGESTION",
            f"  1. If images are missing: Manually transcribe critical screenshots or use OCR.",
            f"  2. If tables are complex: Verify markdown row alignment before ingestion.",
            f"  3. If sections are too long (>1000 words): Split into sub-sections.",
            f"  4. If no headings detected: Add __bold__ lines as section markers in the docx.",
            f"  5. If URLs are present: Decide whether to crawl them or keep as references.",
            f"",
            f"=" * 70,
            f"END OF REPORT",
            f"=" * 70,
        ])

        return '\n'.join(lines)

def main():
    parser = argparse.ArgumentParser(description="Simplified DOCX Diagnostic Extractor")
    parser.add_argument("--input", "-i", required=True, help="Path to .docx file")
    parser.add_argument("--output", "-o", default="./docx_diagnostic", help="Output directory")
    args = parser.parse_args()

    input_path = Path(args.input)
    output_dir = Path(args.output)
    output_dir.mkdir(parents=True, exist_ok=True)

    print(f"Loading: {input_path}")
    markdown = load_docx_as_markdown(input_path)
    print(f"  Markdown length: {len(markdown)} chars, {len(markdown.splitlines())} lines")

    print(f"Analyzing...")
    analyzer = DocxDiagnosticAnalyzer(markdown, input_path.name)
    analyzer.analyze()

    # Save outputs
    base = input_path.stem

    # 1. Report
    report_path = output_dir / f"{base}_report.txt"
    report_path.write_text(analyzer.generate_report(), encoding="utf-8")
    print(f"  ✓ Report: {report_path}")

    # 2. Sections JSON
    sections_path = output_dir / f"{base}_sections.json"
    sections_data = [
        {
            "id": i+1,
            "heading": s["heading"],
            "word_count": s["word_count"],
            "line_count": len(s["lines"]),
            "table_count": len(s["tables"]),
            "content": "\\n".join(s["lines"])[:500] + "..." if len("\\n".join(s["lines"])) > 500 else "\\n".join(s["lines"])
        }
        for i, s in enumerate(analyzer.sections)
    ]
    sections_path.write_text(json.dumps(sections_data, indent=2, ensure_ascii=False), encoding="utf-8")
    print(f"  ✓ Sections: {sections_path}")

    # 3. Tables markdown
    if analyzer.tables:
        tables_path = output_dir / f"{base}_tables.md"
        tables_md = "\n\n---\n\n".join(
            f"## Table {i+1}\n\n{t}" for i, t in enumerate(analyzer.tables)
        )
        tables_path.write_text(tables_md, encoding="utf-8")
        print(f"  ✓ Tables: {tables_path}")

    # 4. Raw markdown
    raw_path = output_dir / f"{base}_raw.md"
    raw_path.write_text(markdown, encoding="utf-8")
    print(f"  ✓ Raw markdown: {raw_path}")

    # Print summary
    print(f"\n{'='*70}")
    print(f"SUMMARY")
    print(f"{'='*70}")
    print(f"Sections: {len(analyzer.sections)}")
    print(f"Tables:   {len(analyzer.tables)}")
    print(f"Issues:   {len(analyzer.issues)}")
    print(f"\nTop issues:")
    for issue in analyzer.issues[:5]:
        print(f"  {issue}")
    print(f"\nSee {output_dir} for full details.")


if __name__ == "__main__":
    main()

from pathlib import Path
from docling.datamodel.base_models import InputFormat
from docling.datamodel.pipeline_options import PdfPipelineOptions
from docling.document_converter import DocumentConverter, PdfFormatOption

def extract_document_data(file_path: str):
    path = Path(file_path)
    if not path.exists():
        print(f"Error: File {file_path} not found.")
        return

    # 1. Configure OCR options for PDFs and Images
    pipeline_options = PdfPipelineOptions()
    pipeline_options.do_ocr = True  # Enables OCR for scanned text/images
    pipeline_options.do_table_structure = True  # Enables advanced table structure extraction
    
    # 2. Initialize Converter with options
    converter = DocumentConverter(
        format_options={
            InputFormat.PDF: PdfFormatOption(pipeline_options=pipeline_options),
            InputFormat.IMAGE: PdfFormatOption(pipeline_options=pipeline_options)
        }
    )

    # 3. Convert the document (Works for .docx, .pdf, .png, .jpg, etc.)
    print(f"Processing {path.name}...")
    result = converter.convert(path)
    doc = result.document

    # 4. Extract and print normal text
    print("\n--- EXTRACTED TEXT ---")
    print(doc.export_to_markdown()) # Markdown preserves headings and list formats

    # 5. Extract and print tables specifically
    print("\n--- EXTRACTED TABLES ---")
    if hasattr(doc, 'tables') and doc.tables:
        for i, table in enumerate(doc.tables):
            print(f"\nTable {i+1}:")
            # Convert the table data into a clean text grid layout
            for row in table.data:
                print(" | ".join([str(cell.text).strip() for cell in row]))
    else:
        # Fallback for structured document models
        table_count = 0
        for element, _ in doc.iterate_items():
            if element.label == "Table":
                table_count += 1
                print(f"\nTable {table_count}:")
                # Docling stores tables as structured grid elements
                if hasattr(element, 'data'):
                    for row in element.data:
                        print(" | ".join([str(cell.text).strip() for cell in row]))
