"""
ECMC unified scraper — v5.

Changes from v4:
  - HTML saved as markdown, narrowed to <main>/<article> before conversion.
    v4 saved the whole page via soup.get_text(), which meant every saved
    file was dominated by the ~150-link global nav menu present on every
    page, with actual page content buried at the bottom.
  - Language filtering now applies to HTML pages too, not just PDFs.
  - No Bedrock anywhere. Language detection via langdetect (offline, free).
    PDF text extraction removed entirely — textract_pipeline.py already
    does this, tested and far more capable (Layout+Tables+Forms) than
    anything worth rebuilding here. This scraper downloads PDFs and does
    a cheap pre-filter (PyPDF2 + langdetect) to skip Spanish PDFs before
    they hit S3/Textract cost; PyPDF2 failure (image PDFs) lets the file
    through rather than guessing — Textract handles those downstream.
  - _same_site no longer auto-includes subdomains by default. The site has
    an auth-walled portal at borroweraccess.ecmc.org; with allow_full_domain
    a crawler could wander into it. Opt in explicitly via allow_subdomains.

Handoff: PDFs land in S3 under the same run's s3_prefix/pdfs/. Run
textract_pipeline.process_batch(source_prefix=<that prefix>) separately —
same loose S3-based coupling tree_builder.py and index_builder.py already
use against process_document's output.
"""

from datetime import datetime
from io import BytesIO
from pathlib import Path
from typing import Dict, Iterable, List, Optional, Set, Tuple
from urllib.parse import urljoin, urlparse
from warnings import filterwarnings

import boto3
import requests
from bs4 import BeautifulSoup
from langdetect import DetectorFactory, LangDetectException, detect_langs
from markdownify import markdownify as md

filterwarnings("ignore", category=UserWarning, module="bs4")
DetectorFactory.seed = 0  # deterministic langdetect results

S3_BUCKET = "ecmc-dev-bucket"
S3_BASE_PREFIX = "Ecmc_scrape_understand_bankruptcy"

SKIP_EXTENSIONS = (
    ".jpg", ".jpeg", ".png", ".gif", ".svg", ".webp", ".ico",
    ".css", ".js", ".zip", ".doc", ".docx", ".xls", ".xlsx",
    ".mp3", ".mp4", ".avi", ".mov", ".xml", ".json", ".rss", ".atom",
)

_LANG_ALIASES: Dict[str, Set[str]] = {
    "spanish":    {"spanish", "español", "espanol", "castellano",
                   "es", "es-es", "es-mx", "es-la", "es-419", "sp", "esp"},
    "english":    {"english", "en", "en-us", "en-gb", "eng"},
    "french":     {"french", "français", "francais", "fr", "fr-fr", "fr-ca"},
    "portuguese": {"portuguese", "português", "portugues", "pt", "pt-br", "pt-pt"},
    "german":     {"german", "deutsch", "de", "de-de"},
    "italian":    {"italian", "italiano", "it", "it-it"},
    "chinese":    {"chinese", "mandarin", "zh", "zh-cn", "zh-tw"},
}

s3_client = boto3.client("s3", region_name="us-east-1")


# ----------------------------------------------------------------------------
# Helpers
# ----------------------------------------------------------------------------
def generate_folder_name(url: str) -> str:
    parsed = urlparse(url)
    path = parsed.path.strip("/")
    timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
    if path:
        last = path.split("/")[-1] if "/" in path else path
        last = re.sub(r"\.(html|htm|php|asp|aspx|jsp)$", "", last, flags=re.IGNORECASE)
        clean = re.sub(r"[^a-zA-Z0-9]+", "_", last).strip("_")[:80]
        if not clean:
            clean = re.sub(r"[^a-zA-Z0-9]+", "_", parsed.netloc).strip("_")
    else:
        clean = re.sub(r"[^a-zA-Z0-9]+", "_", parsed.netloc).strip("_")
    return f"{clean}_{timestamp}"


def derive_section_prefix(url: str) -> str:
    parsed = urlparse(url)
    path = parsed.path.rstrip("/")
    if "." in path.split("/")[-1]:
        path = "/".join(path.split("/")[:-1])
    return path or "/"


def normalize_url(url: str) -> str:
    return url.split("#")[0].rstrip("/")


def normalize_prefix(prefix: str) -> str:
    p = prefix.strip()
    if not p.startswith("/"):
        p = "/" + p
    return p.rstrip("/") or "/"


def canon_language(s: str) -> str:
    if not s:
        return ""
    cleaned = re.sub(r"[^\w\-]", "", s.strip().lower())
    for canon, aliases in _LANG_ALIASES.items():
        if cleaned in aliases:
            return canon
    return cleaned


def is_excluded_language(detected: str, exclude_list: Iterable[str]) -> bool:
    if not exclude_list:
        return False
    detected_canon = canon_language(detected)
    if not detected_canon:
        return False
    return detected_canon in {canon_language(x) for x in exclude_list}


def detect_language(text: str) -> Tuple[str, float]:
    """Returns (canonical_name, confidence). ("unknown", 0.0) on failure."""
    if not text or len(text.strip()) < 20:
        return "unknown", 0.0
    try:
        langs = detect_langs(text[:2000])
        if not langs:
            return "unknown", 0.0
        top = langs[0]
        return canon_language(top.lang), float(top.prob)
    except LangDetectException:
        return "unknown", 0.0


def clean_pdf_filename(name: str) -> str:
    clean = "".join(c for c in name if c.isalnum() or c in (".", "-", "_")).strip()
    if not clean.endswith(".pdf"):
        clean += ".pdf"
    return clean or "document.pdf"


def s3_put_object(key: str, body, content_type: str = "text/plain") -> str:
    if isinstance(body, str):
        body = body.encode("utf-8")
    s3_client.put_object(Bucket=S3_BUCKET, Key=key, Body=body, ContentType=content_type)
    return f"s3://{S3_BUCKET}/{key}"


# ----------------------------------------------------------------------------
# HTML -> Markdown, narrowed to the actual content region
# ----------------------------------------------------------------------------
def html_to_markdown(html: str, source_url: str) -> str:
    """Strip nav/header/footer/script/style, keep only <main>/<article> (or
    <body> as a last resort), convert to markdown. Narrowing BEFORE
    conversion is the fix — v4 converted/saved the whole page, which meant
    the ~150-link global nav dominated every saved file.
    """
    soup = BeautifulSoup(html, "html.parser")
    for tag in soup(["script", "style", "noscript", "meta", "link",
                     "header", "footer", "nav", "aside", "form", "iframe"]):
        tag.decompose()

    content = soup.find("main") or soup.find("article") or soup.body or soup
    if content is None:
        return ""

    md_text = md(str(content), heading_style="ATX", strip=["a"], bullets="-")
    md_text = re.sub(r"\n{3,}", "\n\n", md_text).strip()
    return f"<!-- source_url: {source_url} -->\n\n{md_text}"

class SectionCrawler:
    """Walks HTML pages within scope and harvests every PDF link along the way."""

    def __init__(
        self,
        s3_prefix: str,
        base_domain: str,
        section_prefixes: Optional[List[str]] = None,
        allow_full_domain: bool = False,
        allow_subdomains: bool = False,
        delay: float = 0.5,
        max_pages: Optional[int] = None,
        max_depth: Optional[int] = None,
        request_timeout: int = 20,
        exclude_languages: Optional[List[str]] = None,
        verbose: bool = False,
    ):
        self.s3_prefix = s3_prefix
        self.base_domain = base_domain.replace("www.", "")
        self.allow_full_domain = allow_full_domain
        self.allow_subdomains = allow_subdomains
        self.section_prefixes = [normalize_prefix(p) for p in (section_prefixes or [])]
        self.delay = delay
        self.max_pages = max_pages
        self.max_depth = max_depth
        self.request_timeout = request_timeout
        self.exclude_languages = [canon_language(x) for x in (exclude_languages or [])]
        self.verbose = verbose

        self.visited: Set[str] = set()
        self.queued: Set[str] = set()
        self.pdf_catalog: Dict[str, Dict] = {}
        self.reject_stats: Counter = Counter()
        self._to_visit: deque = deque()

        self.session = requests.Session()
        self.session.headers.update({
            "User-Agent": ("Mozilla/5.0 (Windows NT 10.0; Win64; x64) "
                          "AppleWebKit/537.36 (KHTML, like Gecko) Chrome/124.0 Safari/537.36")
        })

    def _same_site(self, url: str) -> bool:
        parsed = urlparse(url)
        if parsed.scheme not in ("http", "https"):
            return False
        host = parsed.netloc.replace("www.", "")
        if host == self.base_domain:
            return True
        if self.allow_subdomains and host.endswith("." + self.base_domain):
            return True
        return False

    def _in_scope(self, url: str) -> bool:
        if self.allow_full_domain:
            return True
        if not self.section_prefixes:
            return True
        path = urlparse(url).path.rstrip("/") or "/"
        for prefix in self.section_prefixes:
            if path == prefix or path.startswith(prefix + "/"):
                return True
        return False

    def _classify_link(self, url: str) -> Tuple[str, str]:
        parsed = urlparse(url)
        if parsed.scheme not in ("http", "https"):
            return "reject", "bad_scheme"
        if not self._same_site(url):
            return "reject", "off_site"
        path_lower = parsed.path.lower()
        if path_lower.endswith(".pdf"):
            return "pdf", "pdf"
        if path_lower.endswith(SKIP_EXTENSIONS):
            return "reject", "binary_extension"
        if not self._in_scope(url):
            return "reject", "out_of_scope"
        return "crawl", "in_scope"

    def _html_filename(self, url: str) -> str:
        parsed = urlparse(url)
        path = parsed.path.strip("/")
        base = re.sub(r"[^a-zA-Z0-9._-]+", "_", path.replace("/", "_"))[:100] if path else "index"
        domain = parsed.netloc.replace("www.", "")
        full = f"{domain}_{base}"
        if len(full) > 150:
            full = f"{domain}_{hashlib.md5(url.encode()).hexdigest()[:8]}"
        return f"{full}.md"

    def _record_pdf(self, anchor_tag, full_url: str, source_page: str) -> None:
        if full_url in self.pdf_catalog:
            self.pdf_catalog[full_url]["referrers"].add(source_page)
            return
        title = anchor_tag.get_text(strip=True)
        if not title:
            parent = anchor_tag.find_parent(["h1", "h2", "h3", "h4", "li"])
            if parent:
                title = parent.get_text(strip=True)
        if not title:
            title = Path(urlparse(full_url).path).stem.replace("-", " ").title()
        self.pdf_catalog[full_url] = {
            "title": title[:150],
            "url": full_url,
            "filename": Path(urlparse(full_url).path).name,
            "first_seen_on": source_page,
            "referrers": {source_page},
        }

    def _scrape_page(self, url: str, depth: int) -> Tuple[bool, str]:
        self.visited.add(url)
        link_counts: Counter = Counter()

        try:
            print(f"  [HTML d={depth}] {url}")
            resp = self.session.get(url, timeout=self.request_timeout, allow_redirects=True)
            resp.raise_for_status()

            ctype = resp.headers.get("content-type", "").lower()
            if "text/html" not in ctype and "application/xhtml" not in ctype:
                return False, f"Non-HTML: {ctype}"

            final_url = normalize_url(resp.url)
            if not self._same_site(final_url):
                return False, f"Redirected off-site: {final_url}"

            soup = BeautifulSoup(resp.text, "html.parser")

            for a in soup.find_all("a", href=True):
                href = a["href"].strip()
                if not href or href.startswith(("javascript:", "mailto:", "tel:", "#")):
                    link_counts["non_http"] += 1
                    continue
                full = normalize_url(urljoin(url, href))
                category, reason = self._classify_link(full)
                link_counts[reason] += 1

                if category == "pdf":
                    self._record_pdf(a, full, source_page=url)
                elif category == "crawl":
                    if full not in self.visited and full not in self.queued:
                        if self.max_depth is None or depth + 1 <= self.max_depth:
                            self.queued.add(full)
                            self._to_visit.append((full, depth + 1))
                        else:
                            link_counts["depth_capped"] += 1
                            self.reject_stats["depth_capped"] += 1
                else:
                    self.reject_stats[reason] += 1
                    if self.verbose:
                        print(f"        rejected ({reason}): {full}")

            page_markdown = html_to_markdown(resp.text, source_url=url)
            language, confidence = detect_language(page_markdown)

            if is_excluded_language(language, self.exclude_languages):
                print(f"        SKIPPED — language '{language}' (conf={confidence:.2f}) "
                     f"is in exclude list. Not saved.")
                return False, f"Excluded language: {language}"

            key = f"{self.s3_prefix}/web_md/{self._html_filename(url)}"
            s3_path = s3_put_object(key, page_markdown, "text/markdown")

            print(f"        language={language} (conf={confidence:.2f}) | "
                 f"links: {sum(link_counts.values())} | "
                 f"in_scope={link_counts.get('in_scope', 0)} | "
                 f"pdf={link_counts.get('pdf', 0)} | "
                 f"out_of_scope={link_counts.get('out_of_scope', 0)} | "
                 f"off_site={link_counts.get('off_site', 0)}")
            return True, s3_path

        except Exception as e:  # noqa: BLE001
            return False, f"Error: {e}"

    def crawl(self, seed_urls: List[str]) -> Dict:
        scope_desc = "FULL DOMAIN" if self.allow_full_domain else f"prefixes={self.section_prefixes}"
        print(f"\n{'=' * 60}\nCrawling {self.base_domain}\n  scope:      {scope_desc}\n"
             f"  subdomains: {self.allow_subdomains}\n"
             f"  max_depth:  {self.max_depth if self.max_depth is not None else 'unlimited'}\n"
             f"  max_pages:  {self.max_pages if self.max_pages else 'unlimited'}\n"
             f"  exclude:    {self.exclude_languages or 'none'}\n"
             f"  seeds:      {seed_urls}\n{'=' * 60}")

        for seed in seed_urls:
            seed = normalize_url(seed)
            if seed not in self.queued:
                self.queued.add(seed)
                self._to_visit.append((seed, 0))

        result = {"scraped": 0, "failed": 0, "skipped_language": 0,
                 "s3_files": [], "max_depth_seen": 0}

        while self._to_visit:
            if self.max_pages and result["scraped"] >= self.max_pages:
                print(f"  Reached max_pages={self.max_pages}; stopping.")
                break

            url, depth = self._to_visit.popleft()
            if url in self.visited:
                continue

            ok, msg = self._scrape_page(url, depth)
            result["max_depth_seen"] = max(result["max_depth_seen"], depth)

            if ok:
                result["scraped"] += 1
                result["s3_files"].append(msg)
            elif msg.startswith("Excluded language"):
                result["skipped_language"] += 1
            else:
                result["failed"] += 1
                print(f"      skipped: {msg}")

            if self._to_visit and self.delay > 0:
                time.sleep(self.delay)

        result["pdf_count"] = len(self.pdf_catalog)
        result["reject_stats"] = dict(self.reject_stats)

        print(f"\n  HTML pages scraped: {result['scraped']} | failed: {result['failed']} | "
             f"skipped (language): {result['skipped_language']} | "
             f"unique PDFs: {result['pdf_count']} | deepest: d={result['max_depth_seen']}")
        if self.reject_stats:
            print(f"  Rejected-link tally: {dict(self.reject_stats)}")
        return result


# ----------------------------------------------------------------------------
# PDF handling — download + cheap language pre-filter + upload. No text
# extraction here; textract_pipeline.process_batch() does that downstream.
# ----------------------------------------------------------------------------
class PDFProcessor:
    def __init__(self, s3_prefix: str, exclude_languages: Optional[List[str]] = None,
                request_timeout: int = 60):
        self.s3_prefix = s3_prefix
        self.exclude_languages = [canon_language(x) for x in (exclude_languages or [])]
        self.request_timeout = request_timeout
        self.session = requests.Session()
        self.session.headers.update({
            "User-Agent": ("Mozilla/5.0 (Windows NT 10.0; Win64; x64) "
                          "AppleWebKit/537.36 (KHTML, like Gecko) Chrome/124.0 Safari/537.36")
        })

    @staticmethod
    def _sample_text_pypdf2(pdf_bytes: bytes, max_chars: int = 3000) -> str:
        """Cheap sample for language detection only — not a real extraction
        path. Returns "" on any failure OR when no real text is found (e.g.
        image-only PDFs produce only whitespace) — callers treat "" as
        "can't pre-filter, let it through to Textract."
        """
        try:
            from PyPDF2 import PdfReader
            reader = PdfReader(BytesIO(pdf_bytes))
            text = ""
            for page in reader.pages[:3]:
                text += (page.extract_text() or "") + "\n"
                if len(text) > max_chars:
                    break
            text = text.strip()
            return text[:max_chars] if text else ""
        except Exception:  # noqa: BLE001
            return ""

    def process(self, pdf_catalog: Dict[str, Dict]) -> List[Dict]:
        if not pdf_catalog:
            print("  No PDFs to process.")
            return []
        if self.exclude_languages:
            print(f"  Language exclude filter active: {self.exclude_languages}")

        results: List[Dict] = []
        total = len(pdf_catalog)
        skipped_by_language = 0
        seen_filenames: Dict[str, str] = {}

        for i, (pdf_url, meta) in enumerate(pdf_catalog.items(), 1):
            print(f"\n  [PDF {i}/{total}] {meta['title']}\n    url: {pdf_url}")
            try:
                resp = self.session.get(pdf_url, timeout=self.request_timeout)
                resp.raise_for_status()
                pdf_bytes = resp.content
                size_mb = len(pdf_bytes) / (1024 * 1024)

                sample = self._sample_text_pypdf2(pdf_bytes)
                if sample:
                    language, confidence = detect_language(sample)
                else:
                    language, confidence = "unknown_could_not_sample", 0.0

                print(f"    language sample: {language} (conf={confidence:.2f})")

                if sample and is_excluded_language(language, self.exclude_languages):
                    print(f"    SKIPPED — '{language}' is in exclude list.")
                    results.append({
                        "title": meta["title"], "url": pdf_url,
                        "first_seen_on": meta["first_seen_on"],
                        "referrers": sorted(meta["referrers"]),
                        "language": language, "status": "skipped_excluded_language",
                        "size_mb": round(size_mb, 3),
                    })
                    skipped_by_language += 1
                    continue

                clean_name = clean_pdf_filename(meta["filename"])
                if clean_name in seen_filenames and seen_filenames[clean_name] != pdf_url:
                    suffix = hashlib.md5(pdf_url.encode()).hexdigest()[:6]
                    clean_name = clean_name.replace(".pdf", f"_{suffix}.pdf")
                seen_filenames[clean_name] = pdf_url

                pdf_key = f"{self.s3_prefix}/pdfs/{clean_name}"
                s3_client.put_object(Bucket=S3_BUCKET, Key=pdf_key, Body=pdf_bytes,
                                     ContentType="application/pdf")
                pdf_s3 = f"s3://{S3_BUCKET}/{pdf_key}"
                print(f"    saved: {pdf_s3} ({size_mb:.2f} MB)")

                results.append({
                    "title": meta["title"], "pdf_s3_path": pdf_s3, "url": pdf_url,
                    "first_seen_on": meta["first_seen_on"],
                    "referrers": sorted(meta["referrers"]),
                    "language": language, "size_mb": round(size_mb, 3),
                    "status": "downloaded",
                })

            except Exception as e:  # noqa: BLE001
                print(f"    failed: {e}")
                results.append({"title": meta["title"], "url": pdf_url,
                               "first_seen_on": meta["first_seen_on"],
                               "status": "error", "error": str(e)})

        print("\n  PDF download summary:")
        status_counter = Counter(r.get("status", "unknown") for r in results)
        for status, count in status_counter.most_common():
            print(f"    {status}: {count}")
        if self.exclude_languages:
            print(f"  PDFs filtered by language: {skipped_by_language}/{total}")
        return results

# Unified pipeline
# ----------------------------------------------------------------------------
class UnifiedScraperPipeline:
    def run(
        self,
        url: str,
        section_prefixes: Optional[List[str]] = None,
        allow_full_domain: bool = False,
        allow_subdomains: bool = False,
        max_pages: Optional[int] = None,
        max_depth: Optional[int] = None,
        web_delay: float = 0.5,
        scrape_pdfs: bool = True,
        verbose: bool = False,
        extra_seed_urls: Optional[List[str]] = None,
        exclude_languages: Optional[List[str]] = None,
    ) -> Dict:
        if section_prefixes is None and not allow_full_domain:
            section_prefixes = [derive_section_prefix(url)]

        base_domain = urlparse(url).netloc.replace("www.", "")
        folder_name = generate_folder_name(url)
        s3_prefix = f"{S3_BASE_PREFIX}/{folder_name}"
        excluded_canon = [canon_language(x) for x in (exclude_languages or [])]

        print("=" * 70 + "\nUNIFIED SCRAPER PIPELINE (v5 — markdown, no Bedrock)\n" + "=" * 70)
        print(f"Start URL:         {url}")
        print(f"Full domain?:      {allow_full_domain}")
        print(f"Allow subdomains?: {allow_subdomains}")
        print(f"Section prefixes:  {section_prefixes}")
        print(f"Exclude languages: {excluded_canon or 'none'}")
        print(f"S3 output:         s3://{S3_BUCKET}/{s3_prefix}")
        print("=" * 70)

        results = {
            "url": url, "section_prefixes": section_prefixes,
            "allow_full_domain": allow_full_domain, "allow_subdomains": allow_subdomains,
            "max_depth": max_depth, "exclude_languages": excluded_canon,
            "s3_bucket": S3_BUCKET, "s3_prefix": s3_prefix,
            "s3_full_path": f"s3://{S3_BUCKET}/{s3_prefix}", "folder_name": folder_name,
            "web_results": None, "pdf_results": None,
        }

        print("\n[STEP 1] Crawling HTML (-> markdown) and discovering PDFs...")
        crawler = SectionCrawler(
            s3_prefix=s3_prefix, base_domain=base_domain, section_prefixes=section_prefixes,
            allow_full_domain=allow_full_domain, allow_subdomains=allow_subdomains,
            delay=web_delay, max_pages=max_pages, max_depth=max_depth,
            exclude_languages=exclude_languages, verbose=verbose,
        )
        seeds = [url] + (extra_seed_urls or [])
        results["web_results"] = crawler.crawl(seeds)

        if scrape_pdfs and crawler.pdf_catalog:
            print("\n[STEP 2] Downloading PDFs (text extraction happens later, "
                 "via textract_pipeline.process_batch on this S3 prefix)...")
            processor = PDFProcessor(s3_prefix=s3_prefix, exclude_languages=excluded_canon)
            pdf_results = processor.process(crawler.pdf_catalog)
            results["pdf_results"] = pdf_results
            s3_put_object(f"{s3_prefix}/pdf_catalog.json",
                         __import__("json").dumps(pdf_results, indent=2), "application/json")
        elif not crawler.pdf_catalog:
            print("\n[STEP 2] No PDFs discovered.")

        wr = results["web_results"]
        print(f"\n{'=' * 70}\nFINAL REPORT\n{'=' * 70}")
        print(f"S3 output:      s3://{S3_BUCKET}/{s3_prefix}")
        print(f"HTML scraped:   {wr['scraped']} (skipped by language: {wr['skipped_language']})")
        print(f"PDFs found:     {wr.get('pdf_count', 0)}")
        if results["pdf_results"]:
            stats = Counter(r.get("status", "unknown") for r in results["pdf_results"])
            print(f"PDF outcomes:   {dict(stats)}")
        print("\nNext step: textract_pipeline.process_batch(")
        print(f"    source_bucket='{S3_BUCKET}', source_prefix='{s3_prefix}/pdfs/',")
        print(f"    output_bucket='{S3_BUCKET}', output_prefix='textract-output')")
        print("=" * 70)
        return results

"""Textract-only ingestion pipeline: Layout + Tables extraction, ordered by
physical page position, written as three timestamped output files per
document (.json/.md/.txt). No Bedrock, Claude, or Nova model of any kind —
text, tables, and figure locations come exclusively from Amazon Textract.

Flow per document:
    start_analysis          -> submit to Textract (LAYOUT + TABLES), get job_id
    wait_for_job             -> poll until SUCCEEDED/FAILED, paginate all Blocks
    tables_to_markdown       -> TABLE/CELL blocks -> list of markdown tables
    find_figures             -> LAYOUT_FIGURE blocks -> page + location only,
                                 no captioning (Textract does not read images)
    build_sequential_content -> orders text/tables/figures by (Page, Top) —
                                 physical position on the page, not by an
                                 assumed block relationship
    process_document         -> writes {source}_{timestamp}.json/.md/.txt,
                                 all three sharing one timestamp, and updates
                                 the job tracking map

process_document() runs the full flow for one S3 PDF and returns a summary.
process_batch() runs it across every PDF under an S3 prefix.
"""

class TextractTransientError(Exception):
    """Retryable Textract failure (throttling, transient service errors)."""


def _is_transient(exc: BaseException) -> bool:
    code = getattr(exc, "response", {}).get("Error", {}).get("Code", "")
    return code in (
        "ProvisionedThroughputExceededException",
        "ThrottlingException",
        "InternalServerError",
        "LimitExceededException",
    )

# Textract job lifecycle

def _request_token(bucket: str, key: str, feature_types: List[str]) -> str:
    # Deterministic per (bucket, key, feature_types) so re-running a batch
    # with the SAME request doesn't start duplicate (billable) jobs — but a
    # genuinely different request (e.g. FeatureTypes changed) gets a new
    # token instead of colliding with an old one and raising
    # IdempotentParameterMismatchException.
    features_part = ",".join(sorted(feature_types))
    digest = hashlib.sha256(f"{bucket}/{key}/{features_part}".encode()).hexdigest()
    return digest[:64]


DEFAULT_FEATURE_TYPES = ["LAYOUT", "TABLES", "FORMS"]


@retry(
    retry=retry_if_exception_type(TextractTransientError),
    wait=wait_exponential(multiplier=2, min=2, max=60),
    stop=stop_after_attempt(5),
    reraise=True,
)
def start_analysis(textract, bucket: str, key: str) -> str:
    try:
        resp = textract.start_document_analysis(
            DocumentLocation={"S3Object": {"Bucket": bucket, "Name": key}},
            FeatureTypes=DEFAULT_FEATURE_TYPES,
            ClientRequestToken=_request_token(bucket, key, DEFAULT_FEATURE_TYPES),
        )
    except Exception as exc:
        if _is_transient(exc):
            raise TextractTransientError(str(exc)) from exc
        raise
    return resp["JobId"]


def wait_for_job(textract, job_id: str) -> Dict[str, Any]:
    started = time.time()
    while True:
        resp = textract.get_document_analysis(JobId=job_id)
        status = resp["JobStatus"]
        if status in ("SUCCEEDED", "FAILED"):
            break
        if time.time() - started > MAX_POLL_SECONDS:
            return {"JobStatus": "TIMEOUT", "Blocks": [], "StatusMessage": "Polling timed out"}
        time.sleep(POLL_SECONDS)

    if status != "SUCCEEDED":
        return resp

    blocks = list(resp.get("Blocks", []))
    next_token = resp.get("NextToken")
    while next_token:
        page = textract.get_document_analysis(JobId=job_id, NextToken=next_token)
        blocks.extend(page.get("Blocks", []))
        next_token = page.get("NextToken")

    resp["Blocks"] = blocks
    return resp

# Table reconstruction — TABLE/CELL blocks -> markdown grid
# ---------------------------------------------------------------------------
def tables_to_markdown(blocks: List[Dict[str, Any]]) -> List[Dict[str, Any]]:
    by_id = {b["Id"]: b for b in blocks}
    tables_out: List[Dict[str, Any]] = []

    for block in blocks:
        if block["BlockType"] != "TABLE":
            continue

        cell_ids = [
            cid
            for rel in block.get("Relationships", [])
            if rel["Type"] == "CHILD"
            for cid in rel["Ids"]
        ]
        cells = [by_id[cid] for cid in cell_ids if by_id.get(cid, {}).get("BlockType") == "CELL"]
        if not cells:
            continue

        max_row = max(c["RowIndex"] for c in cells)
        max_col = max(c["ColumnIndex"] for c in cells)
        grid = [["" for _ in range(max_col)] for _ in range(max_row)]

        for cell in cells:
            text = _cell_text(cell, by_id)
            r, c = cell["RowIndex"] - 1, cell["ColumnIndex"] - 1
            grid[r][c] = text

        tables_out.append({
            "table_id": block["Id"],
            "page": block.get("Page", 1),
            "rows": max_row,
            "cols": max_col,
            "markdown": _grid_to_markdown(grid),
        })

    return tables_out


def _cell_text(cell: Dict[str, Any], by_id: Dict[str, Dict[str, Any]]) -> str:
    word_ids = [
        wid
        for rel in cell.get("Relationships", [])
        if rel["Type"] == "CHILD"
        for wid in rel["Ids"]
    ]
    words = []
    for wid in word_ids:
        child = by_id.get(wid)
        if not child:
            continue
        if child["BlockType"] == "WORD":
            words.append(child.get("Text", ""))
        elif child["BlockType"] == "SELECTION_ELEMENT":
            words.append("[x]" if child.get("SelectionStatus") == "SELECTED" else "[ ]")
    return " ".join(words).strip()


def _grid_to_markdown(grid: List[List[str]]) -> str:
    if not grid:
        return ""
    lines = ["| " + " | ".join(grid[0]) + " |"]
    lines.append("|" + "|".join(["---"] * len(grid[0])) + "|")
    for row in grid[1:]:
        lines.append("| " + " | ".join(row) + " |")
    return "\n".join(lines)


# ---------------------------------------------------------------------------
# Forms — KEY_VALUE_SET blocks. AWS's documented schema: a KEY block has
# EntityTypes=["KEY"], a CHILD relationship to its own WORD blocks (the
# label text), and a VALUE relationship pointing at the paired VALUE
# block's Id. The VALUE block then has its own CHILD relationship to the
# WORD/SELECTION_ELEMENT blocks making up the actual value text.
# ---------------------------------------------------------------------------
def _clean_kv_key(key_text: str) -> str:
    """Strip a trailing colon Textract may have already included in the key
    text (e.g. source label was literally "Talking Points:") so rendering
    doesn't produce a double colon like "Talking Points::".
    """
    return key_text.rstrip(":").rstrip()


# A real form field's value is normally short — "$5.03", "M00351053". If
# FORMS groups a bold label with a long multi-sentence block of prose that
# follows it (e.g. "Talking Points:" + two bullet points merged into one
# run-on VALUE string), that's FORMS mis-triggering on prose, not a genuine
# label:value pair. We can't recover the ORIGINAL bullet boundaries — that
# structure is gone once Textract merges it into one VALUE block, before we
# ever see the response — but we can stop rendering it as an awkward
# "**Key:** <giant paragraph>" line and fall back to plain paragraph text,
# which at least reads naturally.
_KV_VALUE_MAX_LEN_FOR_FORM_FIELD = 150

# The mirror image of the above: a "key" this long, standing alone with NO
# matched value at all, is not really a label either — it's a full sentence
# Textract misread as a form field name (e.g. "Only the Chapter 13
# Trustee's Office may request a refund of PY T payments." as a "key" with
# an empty value). Left unhandled, this renders as a broken half-line:
# "**<the whole sentence>:**" with nothing after the colon.
_KV_KEY_MIN_LEN_FOR_STANDALONE_SENTENCE = 40


def _build_kv_sequence_item(page: int, key_text: str, value_text: str) -> Dict[str, Any]:
    clean_key = _clean_kv_key(key_text)
    if len(value_text) > _KV_VALUE_MAX_LEN_FOR_FORM_FIELD:
        return {"type": "paragraph", "page": page, "text": f"{clean_key}: {value_text}"}
    if not value_text.strip() and len(clean_key) > _KV_KEY_MIN_LEN_FOR_STANDALONE_SENTENCE:
        return {"type": "paragraph", "page": page, "text": clean_key}
    return {"type": "key_value", "page": page, "key": clean_key, "value": value_text}


def forms_to_keyvalues(blocks: List[Dict[str, Any]]) -> List[Dict[str, Any]]:
    by_id = {b["Id"]: b for b in blocks}
    pairs: List[Dict[str, Any]] = []

    for block in blocks:
        if block["BlockType"] != "KEY_VALUE_SET":
            continue
        if "KEY" not in (block.get("EntityTypes") or []):
            continue

        key_text = _cell_text(block, by_id)
        value_id = next(
            (rel["Ids"][0] for rel in block.get("Relationships", [])
             if rel["Type"] == "VALUE" and rel.get("Ids")),
            None,
        )
        value_block = by_id.get(value_id) if value_id else None
        value_text = _cell_text(value_block, by_id) if value_block else ""

        if key_text:
            pairs.append({
                "key_id": block["Id"],
                "page": block.get("Page", 1),
                "key": key_text,
                "value": value_text,
            })

    return pairs


# ---------------------------------------------------------------------------
# Plain text — titles, headers, paragraphs, lists, in reading order
# ---------------------------------------------------------------------------
_TEXT_LAYOUT_TYPES = {
    "LAYOUT_TITLE",
    "LAYOUT_HEADER",
    "LAYOUT_SECTION_HEADER",
    "LAYOUT_TEXT",
    "LAYOUT_LIST",
    "LAYOUT_FOOTER",
    "LAYOUT_PAGE_NUMBER",
}


def extract_text(blocks: List[Dict[str, Any]]) -> List[Dict[str, Any]]:
    """Reconstruct titles/headers/paragraphs/lists in reading order, per page.

    Textract's Layout feature already returns these blocks in the correct
    reading order (left-to-right, top-to-bottom, column-aware) — we just
    walk them in the order they appear and resolve each one's child LINE
    text. TABLE and LAYOUT_FIGURE content is intentionally excluded here;
    those are handled by tables_to_markdown() and caption_figures().
    """
    by_id = {b["Id"]: b for b in blocks}
    pages: Dict[int, List[Dict[str, str]]] = {}

    for block in blocks:
        if block["BlockType"] not in _TEXT_LAYOUT_TYPES:
            continue

        line_ids = [
            lid
            for rel in block.get("Relationships", [])
            if rel["Type"] == "CHILD"
            for lid in rel["Ids"]
        ]
        lines = [by_id[lid]["Text"] for lid in line_ids
                if by_id.get(lid, {}).get("BlockType") == "LINE"]
        if not lines:
            continue

        page_no = block.get("Page", 1)
        if block["BlockType"] == "LAYOUT_LIST":
            # Each LINE under a LAYOUT_LIST is one list item — keep them
            # separate instead of joining into a run-on sentence.
            for line_text in lines:
                if line_text.strip():
                    pages.setdefault(page_no, []).append({
                        "type": "LAYOUT_LIST", "text": line_text.strip(),
                    })
            continue

        text = " ".join(lines).strip()
        if not text:
            continue
        pages.setdefault(page_no, []).append({
            "type": block["BlockType"],
            "text": text,
        })

    return [
        {"page": page_no, "elements": elements}
        for page_no, elements in sorted(pages.items())
    ]


def text_to_markdown(pages: List[Dict[str, Any]]) -> str:
    """Flatten extract_text() output into one markdown string for the whole doc."""
    heading_types = {"LAYOUT_TITLE": "# ", "LAYOUT_HEADER": "## ",
                     "LAYOUT_SECTION_HEADER": "### "}
    lines = []
    for page in pages:
        for el in page["elements"]:
            prefix = heading_types.get(el["type"], "")
            if el["type"] == "LAYOUT_LIST":
                lines.append(f"- {el['text']}")
            else:
                lines.append(f"{prefix}{el['text']}")
        lines.append("")
    return "\n".join(lines).strip()


# ---------------------------------------------------------------------------
# Sequential ordering.
#
# AWS's own documentation states LAYOUT_* blocks are returned "in the implied
# reading order of the document... top to bottom, left to right" — meaning
# Textract ALREADY resolves multi-column ordering correctly at the LAYOUT
# block level. A previous version of this function re-sorted everything by
# (Page, Top) globally, which THREW THAT AWAY: Top-only sorting interleaves
# two side-by-side columns by vertical position instead of finishing column
# 1 before starting column 2 — a regression for genuine multi-column pages.
#
# Fix: trust Textract's native block order for LAYOUT_TITLE/SECTION_HEADER/
# TEXT/LIST/TABLE/FIGURE — walk `blocks` as given, don't re-sort.
#
# Remaining risk: some tables appear as a raw TABLE block with NO LAYOUT_TABLE
# wrapper at all (confirmed happens — see the test for this file). Native
# order alone would silently drop those, same bug as before. Fallback: after
# the native-order walk, find any TABLE block never matched to a LAYOUT_TABLE
# and insert it via (Page, Top) position instead of dropping it. This case is
# logged so we can tell how often it happens on real documents.
# ---------------------------------------------------------------------------
_SEQUENCE_TEXT_KIND = {
    "LAYOUT_TITLE": "title",
    "LAYOUT_SECTION_HEADER": "section_header",
    "LAYOUT_TEXT": "paragraph",
}


def _block_top(block: Dict[str, Any]) -> float:
    try:
        return block["Geometry"]["BoundingBox"]["Top"]
    except KeyError:
        return 0.0


def _bbox_overlap(box_a: Dict[str, float], box_b: Dict[str, float]) -> float:
    """Intersection-over-union of two Textract fractional bounding boxes."""
    ax0, ay0 = box_a["Left"], box_a["Top"]
    ax1, ay1 = ax0 + box_a["Width"], ay0 + box_a["Height"]
    bx0, by0 = box_b["Left"], box_b["Top"]
    bx1, by1 = bx0 + box_b["Width"], by0 + box_b["Height"]

    ix = max(0.0, min(ax1, bx1) - max(ax0, bx0))
    iy = max(0.0, min(ay1, by1) - max(ay0, by0))
    intersection = ix * iy
    union = box_a["Width"] * box_a["Height"] + box_b["Width"] * box_b["Height"] - intersection
    return intersection / union if union > 0 else 0.0


def _match_layout_block(
    layout_block: Dict[str, Any], candidate_blocks: List[Dict[str, Any]]
) -> Optional[str]:
    """Find the raw block (TABLE or KEY_VALUE_SET) that best overlaps a
    LAYOUT_* wrapper's bounding box on the same page — geometry-based,
    doesn't assume any specific relationship schema.
    """
    page = layout_block.get("Page", 1)
    box = layout_block["Geometry"]["BoundingBox"]
    candidates = [c for c in candidate_blocks if c.get("Page", 1) == page]
    if not candidates:
        return None
    best = max(candidates, key=lambda c: _bbox_overlap(box, c["Geometry"]["BoundingBox"]))
    return best["Id"] if _bbox_overlap(box, best["Geometry"]["BoundingBox"]) > 0 else None


def build_sequential_content(
    blocks: List[Dict[str, Any]],
    tables: List[Dict[str, Any]],
    form_fields: Optional[List[Dict[str, Any]]] = None,
) -> List[Dict[str, Any]]:
    """Order text, tables, form key/value fields, and figures using
    Textract's own native block order (correct per AWS docs, including
    multi-column) rather than re-deriving a global position sort. Tables
    and form fields lacking a LAYOUT_TABLE/LAYOUT_KEY_VALUE wrapper are
    still recovered via a position-based fallback so nothing silently
    disappears.
    """
    form_fields = form_fields or []
    by_id = {b["Id"]: b for b in blocks}
    tables_by_id = {t["table_id"]: t for t in tables}
    kv_by_id = {f["key_id"]: f for f in form_fields}
    table_blocks = [b for b in blocks if b["BlockType"] == "TABLE"]
    kv_blocks = [
        b for b in blocks
        if b["BlockType"] == "KEY_VALUE_SET" and "KEY" in (b.get("EntityTypes") or [])
    ]

    sequence: List[Dict[str, Any]] = []
    matched_table_ids: set = set()
    matched_kv_ids: set = set()

    for block in blocks:
        block_type = block["BlockType"]
        page = block.get("Page", 1)

        if block_type in _SEQUENCE_TEXT_KIND:
            text = _resolve_line_text(block, by_id)
            if text:
                sequence.append({"type": _SEQUENCE_TEXT_KIND[block_type],
                                 "page": page, "text": text})

        elif block_type == "LAYOUT_LIST":
            for line_id in _child_ids(block):
                child = by_id.get(line_id)
                if child and child["BlockType"] == "LINE" and child.get("Text", "").strip():
                    sequence.append({"type": "list_item", "page": page,
                                     "text": child["Text"].strip()})

        elif block_type == "LAYOUT_TABLE":
            table_id = _match_layout_block(block, table_blocks)
            table = tables_by_id.get(table_id)
            if table:
                matched_table_ids.add(table_id)
                sequence.append({"type": "table", "page": page, "table_id": table_id,
                                 "markdown": table["markdown"], "rows": table["rows"],
                                 "cols": table["cols"]})

        elif block_type == "LAYOUT_KEY_VALUE":
            key_id = _match_layout_block(block, kv_blocks)
            kv_field = kv_by_id.get(key_id)
            if kv_field:
                matched_kv_ids.add(key_id)
                sequence.append(_build_kv_sequence_item(page, kv_field["key"], kv_field["value"]))

        elif block_type == "LAYOUT_FIGURE":
            sequence.append({"type": "figure", "page": page, "figure_id": block["Id"]})

    # Fallback: any TABLE/KEY_VALUE_SET block never matched to a LAYOUT
    # wrapper would otherwise vanish entirely. Recover by (Page, Top)
    # position — best effort, not native-order-perfect, but never dropped.
    unmatched_tables = [t for t in table_blocks if t["Id"] not in matched_table_ids]
    unmatched_kv = [k for k in kv_blocks if k["Id"] not in matched_kv_ids]
    if unmatched_tables:
        log.warning("%d table(s) had no LAYOUT_TABLE wrapper; recovered by "
                   "position instead of native order.", len(unmatched_tables))
    if unmatched_kv:
        log.warning("%d form field(s) had no LAYOUT_KEY_VALUE wrapper; "
                   "recovered by position instead of native order.", len(unmatched_kv))

    for table_block in unmatched_tables:
        table = tables_by_id.get(table_block["Id"])
        if not table:
            continue
        fallback_item = {"type": "table", "page": table_block.get("Page", 1),
                         "table_id": table_block["Id"], "markdown": table["markdown"],
                         "rows": table["rows"], "cols": table["cols"]}
        insert_at = _fallback_insert_index(sequence, table_block.get("Page", 1))
        sequence.insert(insert_at, fallback_item)

    for kv_block in unmatched_kv:
        kv_field = kv_by_id.get(kv_block["Id"])
        if not kv_field:
            continue
        fallback_item = _build_kv_sequence_item(
            kv_block.get("Page", 1), kv_field["key"], kv_field["value"])
        insert_at = _fallback_insert_index(sequence, kv_block.get("Page", 1))
        sequence.insert(insert_at, fallback_item)

    return sequence


def _fallback_insert_index(sequence: List[Dict[str, Any]], page: int) -> int:
    """Best-effort position for an unmatched table: insert right after the
    last sequence item on the same page or earlier. Appends at the end if
    no such item exists (e.g. table is on the very first page/position).
    """
    last_qualifying = -1
    for i, item in enumerate(sequence):
        if item.get("page", 1) <= page:
            last_qualifying = i
    return last_qualifying + 1 if last_qualifying >= 0 else len(sequence)



def _child_ids(block: Dict[str, Any]) -> List[str]:
    return [
        cid
        for rel in block.get("Relationships", [])
        if rel["Type"] == "CHILD"
        for cid in rel["Ids"]
    ]


def _resolve_line_text(block: Dict[str, Any], by_id: Dict[str, Dict[str, Any]]) -> str:
    lines = [
        by_id[cid]["Text"]
        for cid in _child_ids(block)
        if by_id.get(cid, {}).get("BlockType") == "LINE"
    ]
    return " ".join(lines).strip()


def _render_figure_markdown(item: Dict[str, Any]) -> str:
    ocr_text = item.get("ocr_text")
    key_values = item.get("key_values") or []
    tables = item.get("tables") or []

    if not ocr_text and not key_values and not tables:
        return (f"**[Figure — page {item['page']}, no text detected by OCR; "
               f"may be a diagram, photo, or icon]**")

    parts = [f"**[Figure — page {item['page']}]**"]
    if item.get("needs_vision_review"):
        parts.append("*(short/fragmented OCR text — may be a diagram; "
                     "consider manual or vision review)*")
    for kv_field in key_values:
        parts.append(f"**{_clean_kv_key(kv_field['key'])}:** {kv_field['value']}")
    for table in tables:
        parts.append(table["markdown"])
    if not key_values and not tables and ocr_text:
        parts.append(ocr_text)
    return "\n".join(parts)


def _render_figure_plain(item: Dict[str, Any]) -> str:
    ocr_text = item.get("ocr_text")
    key_values = item.get("key_values") or []
    tables = item.get("tables") or []

    if not ocr_text and not key_values and not tables:
        return f"Figure (page {item['page']}) - no text detected by OCR."

    parts = [f"Figure (page {item['page']})"]
    if item.get("needs_vision_review"):
        parts.append("(short/fragmented OCR text - may be a diagram; consider review)")
    for kv_field in key_values:
        parts.append(f"{_clean_kv_key(kv_field['key'])}: {kv_field['value']}")
    for table in tables:
        parts.append(_markdown_table_to_plain(table["markdown"]))
    if not key_values and not tables and ocr_text:
        parts.append(ocr_text)
    return "\n".join(parts)


def _slugify(text: str, seen_slugs: Dict[str, int]) -> str:
    """GitHub-Flavored-Markdown style anchor slug: lowercase, spaces to
    hyphens, strip punctuation, de-duplicate with -1/-2/... suffixes so
    repeated heading text still gets unique, working anchor links.
    """
    slug = text.strip().lower()
    slug = re.sub(r"[^\w\s-]", "", slug)
    slug = re.sub(r"[\s]+", "-", slug).strip("-")
    if slug in seen_slugs:
        seen_slugs[slug] += 1
        return f"{slug}-{seen_slugs[slug]}"
    seen_slugs[slug] = 0
    return slug


def extract_toc_entries(sequence: List[Dict[str, Any]]) -> List[Dict[str, Any]]:
    """Pull just the title/section_header items out of a sequence, in
    order, as structured data (not rendered markdown). This is the shared
    heading data both build_table_of_contents() (per-document markdown ToC)
    and the corpus-level tree builder consume — extracted once here so
    both stay consistent.
    """
    return [
        {"type": item["type"], "text": item["text"], "page": item.get("page", 1)}
        for item in sequence
        if item["type"] in ("title", "section_header")
    ]


def build_table_of_contents(sequence: List[Dict[str, Any]]) -> str:
    """Build a markdown Table of Contents from a document's title/section
    header items, with GFM-compatible anchor links (# renders correctly on
    GitHub and most markdown viewers that auto-generate heading anchors).
    Titles are top-level bullets; section headers are indented one level.
    """
    entries = extract_toc_entries(sequence)
    if not entries:
        return ""

    seen_slugs: Dict[str, int] = {}
    lines: List[str] = ["## Table of Contents", ""]

    for entry in entries:
        slug = _slugify(entry["text"], seen_slugs)
        indent = "" if entry["type"] == "title" else "  "
        lines.append(f"{indent}- [{entry['text']}](#{slug})")

    return "\n".join(lines)


def build_table_of_contents_plain(sequence: List[Dict[str, Any]]) -> str:
    """Same heading outline as build_table_of_contents(), no markdown links
    — for the plain-text output, where anchors don't apply.
    """
    lines: List[str] = ["Table of Contents", ""]
    for item in sequence:
        if item["type"] not in ("title", "section_header"):
            continue
        indent = "" if item["type"] == "title" else "  "
        lines.append(f"{indent}- {item['text']}")

    if len(lines) == 2:
        return ""

    return "\n".join(lines)


def sequence_to_markdown(sequence: List[Dict[str, Any]]) -> str:
    """Flatten build_sequential_content() output into one markdown document,
    with tables and figure captions embedded at their correct reading-order
    position — not appended at the end.
    """
    heading_prefix = {"title": "# ", "section_header": "## "}
    lines: List[str] = []

    for item in sequence:
        kind = item["type"]
        if kind == "list_item":
            lines.append(f"- {item['text']}")
        elif kind in ("title", "section_header", "paragraph"):
            lines.append(f"{heading_prefix.get(kind, '')}{item['text']}")
        elif kind == "table":
            lines.append(item["markdown"])
        elif kind == "key_value":
            lines.append(f"**{_clean_kv_key(item['key'])}:** {item['value']}")
        elif kind == "figure":
            lines.append(_render_figure_markdown(item))
        lines.append("")

    return "\n".join(lines).strip()


def sequence_to_plain_text(sequence: List[Dict[str, Any]]) -> str:
    """Same ordering as sequence_to_markdown(), but with no markdown syntax
    (#, -, |). Tables render as simple space-padded columns; figures render
    as 'Figure: <caption>' with no bracket/bold decoration.
    """
    lines: List[str] = []

    for item in sequence:
        kind = item["type"]
        if kind == "list_item":
            lines.append(f"  - {item['text']}")
        elif kind in ("title", "section_header", "paragraph"):
            lines.append(item["text"])
        elif kind == "table":
            lines.append(_markdown_table_to_plain(item["markdown"]))
        elif kind == "key_value":
            lines.append(f"{_clean_kv_key(item['key'])}: {item['value']}")
        elif kind == "figure":
            lines.append(_render_figure_plain(item))
        lines.append("")

    return "\n".join(lines).strip()


def _markdown_table_to_plain(markdown_table: str) -> str:
    rows = [
        [cell.strip() for cell in line.strip("|").split("|")]
        for line in markdown_table.splitlines()
        if line.strip() and not set(line.replace("|", "").strip()) <= {"-"}
    ]
    if not rows:
        return ""
    col_widths = [max(len(row[i]) for row in rows) for i in range(len(rows[0]))]
    return "\n".join(
        "  ".join(cell.ljust(col_widths[i]) for i, cell in enumerate(row))
        for row in rows
    )


# ---------------------------------------------------------------------------
# Figures — Textract detects location only; it does not read image content.
# No captioning/vision model of any kind is used in this pipeline.
# ---------------------------------------------------------------------------
def find_figures(blocks: List[Dict[str, Any]]) -> List[Dict[str, Any]]:
    figures = []
    for block in blocks:
        if block["BlockType"] == "LAYOUT_FIGURE":
            box = block["Geometry"]["BoundingBox"]
            figures.append({
                "figure_id": block["Id"],
                "page": block.get("Page", 1),
                "bbox": box,
            })
    return figures


def crop_figure(pdf_bytes: bytes, page_no: int, bbox: Dict[str, float]) -> bytes:
    """Crop one figure region out of the source PDF page as a PNG image."""
    doc = fitz.open(stream=pdf_bytes, filetype="pdf")
    try:
        page = doc[page_no - 1]
        # bbox values are fractions of page size (Textract convention).
        # fitz clip rects are in PDF point space (page.rect), NOT the pixel
        # space of a rendered pixmap — scaling to pixels happens via dpi=.
        page_w, page_h = page.rect.width, page.rect.height
        x0 = bbox["Left"] * page_w
        y0 = bbox["Top"] * page_h
        x1 = x0 + bbox["Width"] * page_w
        y1 = y0 + bbox["Height"] * page_h
        clip = fitz.Rect(x0, y0, x1, y1)
        crop_pix = page.get_pixmap(dpi=FIGURE_RENDER_DPI, clip=clip)
        return crop_pix.tobytes("png")
    finally:
        doc.close()


@retry(
    retry=retry_if_exception_type(TextractTransientError),
    wait=wait_exponential(multiplier=2, min=2, max=30),
    stop=stop_after_attempt(4),
    reraise=True,
)
def ocr_figure_region(textract, image_bytes: bytes) -> Dict[str, Any]:
    """Run Textract AnalyzeDocument (synchronous, FORMS+TABLES) against one
    cropped figure image. Still pure Textract, no vision/LLM model — this
    covers figures that turn out to be embedded screenshots of forms or
    tables (e.g. a pasted account-detail screen), not just plain text.
    Falls back gracefully to plain line text when no forms/tables are found.
    """
    try:
        response = textract.analyze_document(
            Document={"Bytes": image_bytes}, FeatureTypes=["FORMS", "TABLES"],
        )
    except Exception as exc:
        if _is_transient(exc):
            raise TextractTransientError(str(exc)) from exc
        raise

    crop_blocks = response.get("Blocks", [])
    lines = [b["Text"] for b in crop_blocks if b["BlockType"] == "LINE"]
    word_count = sum(1 for b in crop_blocks if b["BlockType"] == "WORD")
    key_values = forms_to_keyvalues(crop_blocks)
    tables = tables_to_markdown(crop_blocks)

    return {
        "text": "\n".join(lines).strip(),
        "line_count": len(lines),
        "word_count": word_count,
        "key_values": key_values,
        "tables": tables,
    }


# Heuristic only — approximate signal for "this figure is probably a
# flowchart/diagram, not a text-heavy screenshot." OCR on a diagram tends to
# return few, short, scattered words (box labels, arrows) rather than
# coherent lines of prose. A figure with recovered key/value pairs or
# tables is treated as clearly NOT a diagram — forms/tables imply real
# structured content was found, not a plain graphic. Not precise; meant to
# flag candidates for manual or vision-model review, not to decide alone.
_DIAGRAM_MIN_WORD_COUNT = 8
_DIAGRAM_MAX_AVG_WORDS_PER_LINE = 2.0


def _looks_like_diagram(ocr_result: Dict[str, Any]) -> bool:
    if ocr_result.get("key_values") or ocr_result.get("tables"):
        return False
    word_count = ocr_result["word_count"]
    if word_count == 0:
        return True
    if word_count < _DIAGRAM_MIN_WORD_COUNT:
        return True
    avg_words_per_line = word_count / max(ocr_result["line_count"], 1)
    return avg_words_per_line < _DIAGRAM_MAX_AVG_WORDS_PER_LINE


def ocr_figures(
    textract, pdf_bytes: bytes, figures: List[Dict[str, Any]]
) -> List[Dict[str, Any]]:
    """Crop and analyze every detected figure (FORMS+TABLES+plain text).
    Adds ocr_text, recovered key_values/tables, and a needs_vision_review
    flag (heuristic — see _looks_like_diagram) to each. A figure whose
    crop/analysis fails is marked needs_vision_review=True rather than
    silently losing the flag for review.
    """
    enriched = []
    for fig in figures:
        try:
            crop_bytes = crop_figure(pdf_bytes, fig["page"], fig["bbox"])
            ocr_result = ocr_figure_region(textract, crop_bytes)
            enriched.append({
                **fig,
                "ocr_text": ocr_result["text"] or None,
                "ocr_word_count": ocr_result["word_count"],
                "key_values": ocr_result["key_values"],
                "tables": ocr_result["tables"],
                "needs_vision_review": _looks_like_diagram(ocr_result),
            })
        except Exception as exc:  # noqa: BLE001 — one bad figure shouldn't kill the batch
            log.warning("figure analysis failed (page %s, id %s): %s",
                       fig["page"], fig["figure_id"], exc)
            enriched.append({**fig, "ocr_text": None, "key_values": [], "tables": [],
                             "error": str(exc), "needs_vision_review": True})
    return enriched



def merge_figure_ocr(sequence: List[Dict[str, Any]], ocr_figures_list: List[Dict[str, Any]]) -> None:
    """Attach ocr_text/key_values/tables/needs_vision_review onto the figure
    items already in a built sequence, matched by figure_id. Mutates
    sequence in place. Kept as a separate step so build_sequential_content()
    stays independent of whether OCR enrichment happened at all.
    """
    ocr_by_id = {f["figure_id"]: f for f in ocr_figures_list}
    for item in sequence:
        if item["type"] != "figure":
            continue
        info = ocr_by_id.get(item["figure_id"])
        if info:
            item["ocr_text"] = info.get("ocr_text")
            item["key_values"] = info.get("key_values", [])
            item["tables"] = info.get("tables", [])
            item["needs_vision_review"] = info.get("needs_vision_review", True)


# ---------------------------------------------------------------------------
# Output naming + job tracking

def _utc_timestamp() -> str:
    return datetime.now(timezone.utc).strftime("%Y%m%d_%H%M%S")


def _safe_stem(source_key: str) -> str:
    stem = Path(source_key).stem
    return "".join(c if c.isalnum() or c in "-_" else "_" for c in stem)


def output_key_for(
    source_key: str, output_prefix: str, ext: str = ".json", timestamp: Optional[str] = None
) -> str:
    """Build an output S3 key: {prefix}/{source_stem}_{timestamp}{ext}.

    Pass the SAME `timestamp` value across json/md/txt calls for one
    document so all three output files share one identical timestamp,
    making them an obviously-matched set.
    """
    ts = timestamp or _utc_timestamp()
    return f"{output_prefix.rstrip('/')}/{_safe_stem(source_key)}_{ts}{ext}"


def load_job_map(s3, bucket: str, key: str = JOB_MAP_KEY) -> Dict[str, Any]:
    try:
        obj = s3.get_object(Bucket=bucket, Key=key)
        return json.loads(obj["Body"].read())
    except s3.exceptions.NoSuchKey:
        return {}
    except s3.exceptions.ClientError as exc:
        if exc.response["Error"]["Code"] in ("NoSuchKey", "404"):
            return {}
        raise


def save_job_map(s3, bucket: str, job_map: Dict[str, Any], key: str = JOB_MAP_KEY) -> None:
    s3.put_object(
        Bucket=bucket, Key=key,
        Body=json.dumps(job_map, indent=2, ensure_ascii=False).encode(),
        ContentType="application/json",
    )


# ---------------------------------------------------------------------------
# Orchestration
# ---------------------------------------------------------------------------
def process_document(
    source_bucket: str,
    source_key: str,
    output_bucket: str,
    output_prefix: str,
    region: str = "us-east-1",
) -> DocumentResult:
    textract = boto3.client("textract", region_name=region)
    s3 = boto3.client("s3", region_name=region)

    job_map = load_job_map(s3, output_bucket)
    request_token = _request_token(source_bucket, source_key, DEFAULT_FEATURE_TYPES)

    job_id = start_analysis(textract, source_bucket, source_key)
    log.info("started job %s for %s", job_id, source_key)

    result_raw = wait_for_job(textract, job_id)
    status = result_raw["JobStatus"]

    if status != "SUCCEEDED":
        error = result_raw.get("StatusMessage", f"job ended with status {status}")
        job_map[request_token] = {
            "source_key": source_key, "job_id": job_id, "status": status,
            "error": error, "processed_at": datetime.now(timezone.utc).isoformat(),
        }
        save_job_map(s3, output_bucket, job_map)
        return DocumentResult(source_bucket, source_key, job_id, status, error=error)

    blocks = result_raw["Blocks"]
    line_count = sum(1 for b in blocks if b["BlockType"] == "LINE")
    text_pages = extract_text(blocks)
    tables = tables_to_markdown(blocks)
    form_fields = forms_to_keyvalues(blocks)
    figures = find_figures(blocks)  # location only, before OCR enrichment

    sequence = build_sequential_content(blocks, tables, form_fields)

    if figures:
        pdf_obj = s3.get_object(Bucket=source_bucket, Key=source_key)
        pdf_bytes = pdf_obj["Body"].read()
        figures = ocr_figures(textract, pdf_bytes, figures)  # Textract OCR only
        merge_figure_ocr(sequence, figures)
        needing_review = sum(1 for f in figures if f.get("needs_vision_review"))
        if needing_review:
            log.info("%d/%d figures flagged as likely diagrams (short/"
                     "fragmented OCR) — may need manual or vision review.",
                     needing_review, len(figures))

    toc_markdown = build_table_of_contents(sequence)
    toc_plain = build_table_of_contents_plain(sequence)
    body_markdown = sequence_to_markdown(sequence)
    body_plain_text = sequence_to_plain_text(sequence)

    document_markdown = f"{toc_markdown}\n\n---\n\n{body_markdown}" if toc_markdown else body_markdown
    document_plain_text = f"{toc_plain}\n\n---\n\n{body_plain_text}" if toc_plain else body_plain_text

    # One shared timestamp -> json/md/txt for this document are an obvious
    # matched set, e.g. Phone_Guide_20260719_143012.{json,md,txt}
    timestamp = _utc_timestamp()
    json_key = output_key_for(source_key, output_prefix, ".json", timestamp)
    markdown_key = output_key_for(source_key, output_prefix, ".md", timestamp)
    text_key = output_key_for(source_key, output_prefix, ".txt", timestamp)

    output_body = {
        "source_bucket": source_bucket,
        "source_key": source_key,
        "job_id": job_id,
        "processed_at": datetime.now(timezone.utc).isoformat(),
        "line_count": line_count,
        "text": text_pages,
        "text_markdown": text_to_markdown(text_pages),
        "tables": tables,
        "form_fields": form_fields,
        "figures": figures,
        "content_sequence": sequence,
        "table_of_contents": toc_markdown,
        "document_markdown": document_markdown,
        "raw_block_count": len(blocks),
    }
    s3.put_object(
        Bucket=output_bucket, Key=json_key,
        Body=json.dumps(output_body, indent=2, ensure_ascii=False).encode(),
        ContentType="application/json",
    )
    s3.put_object(
        Bucket=output_bucket, Key=markdown_key,
        Body=document_markdown.encode("utf-8"),
        ContentType="text/markdown",
    )
    s3.put_object(
        Bucket=output_bucket, Key=text_key,
        Body=document_plain_text.encode("utf-8"),
        ContentType="text/plain",
    )

    job_map[request_token] = {
        "source_key": source_key, "job_id": job_id, "status": "SUCCEEDED",
        "output_key": json_key, "markdown_key": markdown_key, "text_key": text_key,
        "table_count": len(tables), "figure_count": len(figures),
        "processed_at": output_body["processed_at"],
    }
    save_job_map(s3, output_bucket, job_map)

    log.info("done: %s -> %s / %s / %s (%d tables, %d figures)",
             source_key, json_key, markdown_key, text_key, len(tables), len(figures))

    return DocumentResult(
        source_bucket, source_key, job_id, "SUCCEEDED",
        output_key=json_key, markdown_key=markdown_key, text_key=text_key,
        tables=tables, figures=figures, line_count=line_count,
    )


def parse_s3_uri(uri: str) -> Tuple[str, str]:
    """s3://bucket/key/path.pdf -> (bucket, key/path.pdf)."""
    if not uri.startswith("s3://"):
        raise ValueError(f"not an s3:// URI: {uri}")
    without_scheme = uri[len("s3://"):]
    bucket, _, key = without_scheme.partition("/")
    if not bucket or not key:
        raise ValueError(f"malformed s3:// URI: {uri}")
    return bucket, key


def _process_key_pairs(
    bucket_key_pairs: List[Tuple[str, str]],
    output_bucket: str, output_prefix: str,
    region: str, skip_existing: bool,
) -> List[DocumentResult]:
    """Shared per-file loop — process_batch() and process_key_list() both
    call this so the retry/skip logic exists in exactly one place.
    """
    s3 = boto3.client("s3", region_name=region)
    job_map = load_job_map(s3, output_bucket) if skip_existing else {}

    results = []
    for source_bucket, key in bucket_key_pairs:
        token = _request_token(source_bucket, key, DEFAULT_FEATURE_TYPES)
        if skip_existing and job_map.get(token, {}).get("status") == "SUCCEEDED":
            log.info("skip (already processed): %s", key)
            continue
        try:
            results.append(process_document(source_bucket, key, output_bucket, output_prefix, region))
        except Exception as exc:  # noqa: BLE001 — one bad file shouldn't kill the batch
            log.exception("failed: %s", key)
            results.append(DocumentResult(source_bucket, key, "", "ERROR", error=str(exc)))

    return results


def process_batch(
    source_bucket: str,
    source_prefix: str,
    output_bucket: str,
    output_prefix: str,
    region: str = "us-east-1",
    skip_existing: bool = True,
) -> List[DocumentResult]:
    """Discover every PDF under one S3 prefix, then process each."""
    s3 = boto3.client("s3", region_name=region)
    paginator = s3.get_paginator("list_objects_v2")
    keys = [
        obj["Key"]
        for page in paginator.paginate(Bucket=source_bucket, Prefix=source_prefix)
        for obj in page.get("Contents", [])
        if obj["Key"].lower().endswith(".pdf")
    ]
    pairs = [(source_bucket, key) for key in keys]
    return _process_key_pairs(pairs, output_bucket, output_prefix, region, skip_existing)


def process_key_list(
    s3_uris: List[str],
    output_bucket: str,
    output_prefix: str,
    region: str = "us-east-1",
    skip_existing: bool = True,
) -> List[DocumentResult]:
    """Process an EXPLICIT list of PDF paths — no S3 prefix scanning. Each
    entry is a full s3://bucket/key URI, so the list can span multiple
    buckets/prefixes freely (exactly the shape of a printed file listing
    like "Total files loaded for Textract Pipeline: 13").
    """
    pairs = [parse_s3_uri(uri) for uri in s3_uris if uri.lower().endswith(".pdf")]
    skipped_non_pdf = len(s3_uris) - len(pairs)
    if skipped_non_pdf:
        log.info("ignored %d non-.pdf path(s) in the list", skipped_non_pdf)
    return _process_key_pairs(pairs, output_bucket, output_prefix, region, skip_existing)

def load_json_from_s3(bucket: str, key: str) -> Dict[str, Any]:
    """Load JSON object from S3.

    Args:
        bucket: S3 bucket name.
        key: S3 object key.

    Returns:
        Parsed JSON dict.
    """
    s3 = get_s3_client()
    obj = s3.get_object(Bucket=bucket, Key=key)
    return json.loads(obj["Body"].read())


def save_json_to_s3(bucket: str, key: str, data: Dict[str, Any]) -> None:
    """Save JSON object to S3.

    Args:
        bucket: S3 bucket name.
        key: S3 object key.
        data: JSON-serializable dict.
    """
    s3 = get_s3_client()
    s3.put_object(
        Bucket=bucket, Key=key,
        Body=json.dumps(data, indent=2, ensure_ascii=False).encode(),
        ContentType="application/json",
    )


def list_s3_objects(bucket: str, prefix: str) -> List[str]:
    """List all JSON keys under an S3 prefix.

    Args:
        bucket: S3 bucket name.
        prefix: S3 prefix path.

    Returns:
        List of matching S3 keys.
    """
    s3 = get_s3_client()
    paginator = s3.get_paginator("list_objects_v2")
    keys = []
    for page in paginator.paginate(Bucket=bucket, Prefix=prefix):
        for obj in page.get("Contents", []):
            if obj["Key"].endswith(".json"):
                keys.append(obj["Key"])
    return keys

def bedrock_chat(model_id: str, prompt: str, max_tokens: int = 1024) -> str:
    """Call Bedrock Converse API.

    Args:
        model_id: Bedrock model ARN.
        prompt: User prompt text.
        max_tokens: Max tokens for response.

    Returns:
        Model response text.
    """
    client = get_bedrock_client()
    try:
        resp = client.converse(
            modelId=model_id,
            messages=[{"role": "user", "content": [{"text": prompt}]}],
            inferenceConfig={"maxTokens": max_tokens, "temperature": 0.0}
        )
        return resp["output"]["message"]["content"][0]["text"]
    except Exception as exc:
        logger.error("Bedrock error with %s: %s", model_id, exc)
        return ""

def detect_section_boundaries(sequence: List[Dict[str, Any]]) -> List[Dict[str, Any]]:
    """Split sequence into sections at heading boundaries.

    Args:
        sequence: Content sequence from Step 1 corpus.

    Returns:
        List of section dicts with title, text, page.
    """
    heading_types = {"title", "section_header"}
    sections = []
    current_title = "Untitled"
    current_lines = []
    current_page = 1

    for item in sequence:
        if item["type"] in heading_types:
            if current_lines:
                sections.append({
                    "title": current_title,
                    "text": "\n".join(current_lines).strip(),
                    "page": current_page,
                })
            current_title = item["text"]
            current_page = item.get("page", 1)
            current_lines = []
        elif item["type"] in ("paragraph", "list_item", "key_value"):
            if item["type"] == "key_value":
                current_lines.append(f"{item['key']}: {item['value']}")
            else:
                current_lines.append(item["text"])
        elif item["type"] == "table":
            current_lines.append(f"[TABLE: {item.get('rows', 0)} rows x {item.get('cols', 0)} cols]")
        elif item["type"] == "figure":
            current_lines.append(f"[FIGURE: page {item.get('page', '')}]")

    if current_lines:
        sections.append({
            "title": current_title,
            "text": "\n".join(current_lines).strip(),
            "page": current_page,
        })

    return sections


def summarize_with_claude(title: str, text: str) -> str:
    """Ask Claude to summarize a document section.

    Args:
        title: Section title.
        text: Section full text.

    Returns:
        1-2 sentence summary.
    """
    prompt = (
        f"Summarize the following ECMC bankruptcy guide section in 1-2 sentences. "
        f"Be specific about what information it contains. Do not use generic phrases.\n\n"
        f"Title: {title}\n"
        f"Content: {text[:3000]}\n\n"
        f"Summary:"
    )
    summary = bedrock_chat(CLAUDE_MODEL_ID, prompt, max_tokens=150)
    return summary or f"Section about {title}"


def build_tree_index(corpus_data: Dict[str, Any]) -> Dict[str, Any]:
    """Build hierarchical tree index from corpus.

    Args:
        corpus_data: Parsed corpus.json from Step 1.

    Returns:
        Tree index dict with summaries.
    """
    doc_id = Path(corpus_data["source_key"]).stem
    sequence = corpus_data.get("sequence", [])

    logger.info("Building tree index for %s", doc_id)

    sections = detect_section_boundaries(sequence)
    nodes = []

    for idx, section in enumerate(sections):
        if idx % 5 == 0:
            logger.info("  Summarizing section %d/%d", idx + 1, len(sections))

        summary = summarize_with_claude(section["title"], section["text"])
        nodes.append({
            "node_id": f"{doc_id}_S{idx:04d}",
            "title": section["title"],
            "page": section["page"],
            "summary": summary,
            "full_text": section["text"],
        })

    return {
        "root": doc_id,
        "source_key": corpus_data["source_key"],
        "processed_at": datetime.now(timezone.utc).isoformat(),
        "section_count": len(nodes),
        "nodes": nodes,
    }

def load_glossary(glossary_bucket: str, glossary_key: str) -> str:
    """Load glossary text from S3 if available.

    Args:
        glossary_bucket: S3 bucket.
        glossary_key: S3 key to glossary doc.

    Returns:
        Glossary text or empty string.
    """
    if not glossary_bucket or not glossary_key:
        return ""
    try:
        s3 = get_s3_client()
        obj = s3.get_object(Bucket=glossary_bucket, Key=glossary_key)
        return obj["Body"].read().decode("utf-8")
    except Exception as exc:
        logger.warning("Glossary not found: %s", exc)
        return ""


def extract_terms_with_nova(text: str, glossary_text: str = "") -> List[Dict[str, Any]]:
    """Ask Nova Lite to extract domain terms from document text.

    Args:
        text: Document text to analyze.
        glossary_text: Optional glossary for context.

    Returns:
        List of term dicts with aliases and category.
    """
    glossary_context = ""
    if glossary_text:
        glossary_context = (
            f"\nReference glossary (use these as known terms):\n"
            f"{glossary_text[:2000]}\n\n"
        )

    prompt = (
        "You are analyzing an ECMC bankruptcy student loan document. "
        "Extract all important domain-specific terms that a borrower or agent "
        "might ask about. Include legal terms, procedures, forms, and loan statuses. "
        f"{glossary_context}"
        "Return ONLY a JSON array with this exact shape:\n"
        '[{"term": "Exact Term", "aliases": ["alias1"], "category": "legal|procedure|form|status"}]\n\n'
        "Document text:\n"
        f"{text[:4000]}\n\n"
        "Terms (JSON only):"
    )

    resp = bedrock_chat(NOVA_MODEL_ID, prompt, max_tokens=1024)
    terms = []

    try:
        cleaned = re.sub(r"```json\s*|\s*```", "", resp).strip()
        parsed = json.loads(cleaned)
        if isinstance(parsed, list):
            terms = parsed
        elif isinstance(parsed, dict) and "terms" in parsed:
            terms = parsed["terms"]
    except Exception as exc:
        logger.error("Failed to parse Nova term response: %s — raw: %s", exc, resp[:200])

    return terms


def build_term_index(corpus_data: Dict[str, Any], glossary_text: str = "") -> Dict[str, Any]:
    """Build flat term index from corpus.

    Args:
        corpus_data: Parsed corpus.json from Step 1.
        glossary_text: Optional glossary text for context.

    Returns:
        Term index dict with inverted mapping.
    """
    doc_id = Path(corpus_data["source_key"]).stem
    full_text = corpus_data.get("document_markdown", "")

    if not full_text:
        # Fallback: reconstruct from sequence
        sequence = corpus_data.get("sequence", [])
        full_text = "\n".join(item.get("text", "") for item in sequence)

    logger.info("Building term index for %s", doc_id)

    terms = extract_terms_with_nova(full_text, glossary_text)
    term_map: Dict[str, Dict[str, Any]] = {}

    for term_item in terms:
        term_name = term_item.get("term", "").strip()
        if not term_name:
            continue

        if term_name not in term_map:
            term_map[term_name] = {
                "term": term_name,
                "aliases": term_item.get("aliases", []),
                "category": term_item.get("category", "unknown"),
                "sources": [],
                "occurrence_count": 0,
            }

        term_map[term_name]["sources"].append({
            "doc_id": doc_id,
            "source_key": corpus_data["source_key"],
            "context_snippet": full_text[:300].replace("\n", " "),
        })
        term_map[term_name]["occurrence_count"] += 1

    return {
        "source_key": corpus_data["source_key"],
        "processed_at": datetime.now(timezone.utc).isoformat(),
        "total_terms": len(term_map),
        "terms": list(term_map.values()),
    }

def process_single_document(corpus_key: str, glossary_text: str = "") -> Dict[str, Any]:
    """Process one corpus file into tree + term indices.

    Args:
        corpus_key: S3 key to corpus.json from Step 1.
        glossary_text: Optional glossary text.

    Returns:
        Result dict with output keys and status.
    """
    logger.info("Processing corpus: %s", corpus_key)

    # Load corpus
    corpus_data = load_json_from_s3(CORPUS_BUCKET, corpus_key)

    # Build tree index
    tree = build_tree_index(corpus_data)

    # Build term index
    terms = build_term_index(corpus_data, glossary_text)

    # Save outputs
    doc_stem = Path(corpus_data["source_key"]).stem
    timestamp = datetime.now(timezone.utc).strftime("%Y%m%d_%H%M%S")

    tree_key = f"{OUTPUT_PREFIX.rstrip('/')}/{doc_stem}_tree_{timestamp}.json"
    term_key = f"{OUTPUT_PREFIX.rstrip('/')}/{doc_stem}_terms_{timestamp}.json"

    save_json_to_s3(OUTPUT_BUCKET, tree_key, tree)
    save_json_to_s3(OUTPUT_BUCKET, term_key, terms)

    logger.info("Saved tree: %s, terms: %s", tree_key, term_key)

    return {
        "source_key": corpus_data["source_key"],
        "tree_key": tree_key,
        "term_key": term_key,
        "section_count": tree["section_count"],
        "term_count": terms["total_terms"],
        "status": "SUCCEEDED",
    }


def process_all_documents(glossary_bucket: str = "", glossary_key: str = "") -> List[Dict[str, Any]]:
    """Process all corpus files under prefix.

    Args:
        glossary_bucket: Optional glossary S3 bucket.
        glossary_key: Optional glossary S3 key.

    Returns:
        List of result dicts per document.
    """
    glossary_text = load_glossary(glossary_bucket, glossary_key)

    corpus_keys = list_s3_objects(CORPUS_BUCKET, CORPUS_PREFIX)
    logger.info("Found %d corpus files to process", len(corpus_keys))

    results = []
    for key in corpus_keys:
        try:
            result = process_single_document(key, glossary_text)
            results.append(result)
        except Exception as exc:
            logger.exception("Failed to process %s: %s", key, exc)
            results.append({
                "corpus_key": key,
                "status": "ERROR",
                "error": str(exc),
            })

    return results

def start_analysis(bucket: str, key: str) -> str:
    """Submit PDF to Textract async analysis.

    Args:
        bucket: S3 bucket containing the PDF.
        key: S3 key of the PDF.

    Returns:
        Textract JobId string.
    """
    textract = get_textract_client()
    features = ["LAYOUT", "TABLES", "FORMS"]
    if USE_QUERIES:
        features.append("QUERIES")

    resp = textract.start_document_analysis(
        DocumentLocation={"S3Object": {"Bucket": bucket, "Name": key}},
        FeatureTypes=features,
    )
    job_id = resp["JobId"]
    logger.info("Started Textract job %s for s3://%s/%s", job_id, bucket, key)
    return job_id


def wait_for_job(job_id: str) -> Dict[str, Any]:
    """Poll Textract job until completion or timeout.

    Args:
        job_id: Textract JobId to poll.

    Returns:
        Textract response dict with Blocks list.
    """
    textract = get_textract_client()
    started = time.time()

    while True:
        resp = textract.get_document_analysis(JobId=job_id)
        status = resp["JobStatus"]
        if status in ("SUCCEEDED", "FAILED"):
            break
        if time.time() - started > MAX_POLL_SECONDS:
            logger.error("Job %s timed out after %s seconds", job_id, MAX_POLL_SECONDS)
            return {"JobStatus": "TIMEOUT", "Blocks": []}
        time.sleep(POLL_SECONDS)

    if status != "SUCCEEDED":
        logger.error("Job %s failed: %s", job_id, resp.get("StatusMessage", "Unknown"))
        return resp

    # Paginate all blocks
    blocks = list(resp.get("Blocks", []))
    next_token = resp.get("NextToken")
    while next_token:
        page = textract.get_document_analysis(JobId=job_id, NextToken=next_token)
        blocks.extend(page.get("Blocks", []))
        next_token = page.get("NextToken")

    resp["Blocks"] = blocks
    logger.info("Job %s completed: %d blocks", job_id, len(blocks))
    return resp


# ---------------------------------------------------------------------------
# BLOCK INDEXING
# ---------------------------------------------------------------------------
def index_blocks(blocks: List[Dict[str, Any]]) -> Dict[str, Dict[str, Any]]:
    """Build id-to-block lookup map.

    Args:
        blocks: Raw Textract Blocks list.

    Returns:
        Dict mapping Block Id to Block dict.
    """
    return {b["Id"]: b for b in blocks}


# ---------------------------------------------------------------------------
# TEXT EXTRACTION
# ---------------------------------------------------------------------------
def extract_text_elements(blocks: List[Dict[str, Any]]) -> List[Dict[str, Any]]:
    """Extract titles, headers, paragraphs, lists in reading order.

    Args:
        blocks: Raw Textract Blocks list.

    Returns:
        List of text elements with type, page, and text.
    """
    by_id = index_blocks(blocks)
    layout_types = {
        "LAYOUT_TITLE", "LAYOUT_HEADER", "LAYOUT_SECTION_HEADER",
        "LAYOUT_TEXT", "LAYOUT_LIST", "LAYOUT_FOOTER", "LAYOUT_PAGE_NUMBER",
    }
    pages: Dict[int, List[Dict[str, str]]] = {}

    for block in blocks:
        if block["BlockType"] not in layout_types:
            continue

        line_ids = [
            cid
            for rel in block.get("Relationships", [])
            if rel["Type"] == "CHILD"
            for cid in rel["Ids"]
        ]
        lines = [
            by_id[lid]["Text"]
            for lid in line_ids
            if by_id.get(lid, {}).get("BlockType") == "LINE"
        ]
        if not lines:
            continue

        page_no = block.get("Page", 1)
        if block["BlockType"] == "LAYOUT_LIST":
            for line in lines:
                if line.strip():
                    pages.setdefault(page_no, []).append({
                        "type": "list_item", "text": line.strip(),
                    })
        else:
            text = " ".join(lines).strip()
            if text:
                pages.setdefault(page_no, []).append({
                    "type": block["BlockType"], "text": text,
                })

    return [
        {"page": page_no, "elements": elements}
        for page_no, elements in sorted(pages.items())
    ]


# ---------------------------------------------------------------------------
# TABLE EXTRACTION
# ---------------------------------------------------------------------------
def extract_tables(blocks: List[Dict[str, Any]]) -> List[Dict[str, Any]]:
    """Reconstruct tables from TABLE and CELL blocks.

    Args:
        blocks: Raw Textract Blocks list.

    Returns:
        List of table dicts with markdown, rows, cols, page.
    """
    by_id = index_blocks(blocks)
    tables = []

    for block in blocks:
        if block["BlockType"] != "TABLE":
            continue

        cell_ids = [
            cid
            for rel in block.get("Relationships", [])
            if rel["Type"] == "CHILD"
            for cid in rel["Ids"]
        ]
        cells = [by_id[cid] for cid in cell_ids if by_id.get(cid, {}).get("BlockType") == "CELL"]
        if not cells:
            continue

        max_row = max(c["RowIndex"] for c in cells)
        max_col = max(c["ColumnIndex"] for c in cells)
        grid = [["" for _ in range(max_col)] for _ in range(max_row)]

        for cell in cells:
            text = extract_cell_text(cell, by_id)
            r, c = cell["RowIndex"] - 1, cell["ColumnIndex"] - 1
            grid[r][c] = text

        tables.append({
            "table_id": block["Id"],
            "page": block.get("Page", 1),
            "rows": max_row,
            "cols": max_col,
            "markdown": grid_to_markdown(grid),
        })

    return tables


def extract_cell_text(cell: Dict[str, Any], by_id: Dict[str, Dict[str, Any]]) -> str:
    """Extract text from a single table cell.

    Args:
        cell: Textract CELL block.
        by_id: Block lookup map.

    Returns:
        Concatenated cell text.
    """
    word_ids = [
        wid
        for rel in cell.get("Relationships", [])
        if rel["Type"] == "CHILD"
        for wid in rel["Ids"]
    ]
    words = []
    for wid in word_ids:
        child = by_id.get(wid)
        if not child:
            continue
        if child["BlockType"] == "WORD":
            words.append(child.get("Text", ""))
        elif child["BlockType"] == "SELECTION_ELEMENT":
            words.append("[x]" if child.get("SelectionStatus") == "SELECTED" else "[ ]")
    return " ".join(words).strip()


def grid_to_markdown(grid: List[List[str]]) -> str:
    """Convert 2D grid to markdown table.

    Args:
        grid: List of rows, each row is list of cell strings.

    Returns:
        Markdown table string.
    """
    if not grid:
        return ""
    lines = ["| " + " | ".join(grid[0]) + " |"]
    lines.append("|" + "|".join([" --- "] * len(grid[0])) + "|")
    for row in grid[1:]:
        lines.append("| " + " | ".join(row) + " |")
    return "\n".join(lines)


# ---------------------------------------------------------------------------
# FORM EXTRACTION
# ---------------------------------------------------------------------------
def extract_forms(blocks: List[Dict[str, Any]]) -> List[Dict[str, Any]]:
    """Extract key-value pairs from KEY_VALUE_SET blocks.

    Args:
        blocks: Raw Textract Blocks list.

    Returns:
        List of form field dicts with key, value, page.
    """
    by_id = index_blocks(blocks)
    fields = []

    for block in blocks:
        if block["BlockType"] != "KEY_VALUE_SET":
            continue
        if "KEY" not in (block.get("EntityTypes") or []):
            continue

        key_text = extract_cell_text(block, by_id)
        value_id = None
        for rel in block.get("Relationships", []):
            if rel["Type"] == "VALUE" and rel.get("Ids"):
                value_id = rel["Ids"][0]
                break

        value_text = ""
        if value_id:
            value_block = by_id.get(value_id)
            if value_block:
                value_text = extract_cell_text(value_block, by_id)

        if key_text:
            fields.append({
                "key_id": block["Id"],
                "page": block.get("Page", 1),
                "key": clean_key(key_text),
                "value": value_text,
            })

    return fields


def clean_key(text: str) -> str:
    """Strip trailing colon from key text to avoid double colons.

    Args:
        text: Raw key text from Textract.

    Returns:
        Cleaned key text.
    """
    return text.rstrip(":").rstrip()


# ---------------------------------------------------------------------------
# FIGURE DETECTION (location only, no OCR or vision)
# ---------------------------------------------------------------------------
def detect_figures(blocks: List[Dict[str, Any]]) -> List[Dict[str, Any]]:
    """Detect figure locations from LAYOUT_FIGURE blocks.

    Args:
        blocks: Raw Textract Blocks list.

    Returns:
        List of figure dicts with page and bounding box.
    """
    figures = []
    for block in blocks:
        if block["BlockType"] == "LAYOUT_FIGURE":
            figures.append({
                "figure_id": block["Id"],
                "page": block.get("Page", 1),
                "bbox": block["Geometry"]["BoundingBox"],
            })
    return figures


# ---------------------------------------------------------------------------
# SEQUENTIAL CONTENT BUILDER
# ---------------------------------------------------------------------------
def build_sequence(
    blocks: List[Dict[str, Any]],
    tables: List[Dict[str, Any]],
    forms: List[Dict[str, Any]],
) -> List[Dict[str, Any]]:
    """Build unified content sequence in reading order.

    Walks Textract blocks in native order, inserting tables and forms
    at their detected positions.

    Args:
        blocks: Raw Textract Blocks list.
        tables: Extracted tables.
        forms: Extracted form fields.

    Returns:
        Ordered list of content elements.
    """
    tables_by_id = {t["table_id"]: t for t in tables}
    forms_by_id = {f["key_id"]: f for f in forms}
    text_kind_map = {
        "LAYOUT_TITLE": "title",
        "LAYOUT_SECTION_HEADER": "section_header",
        "LAYOUT_TEXT": "paragraph",
    }

    sequence = []
    matched_tables = set()
    matched_forms = set()

    for block in blocks:
        btype = block["BlockType"]
        page = block.get("Page", 1)

        if btype in text_kind_map:
            text = resolve_line_text(block, index_blocks(blocks))
            if text:
                sequence.append({
                    "type": text_kind_map[btype],
                    "page": page,
                    "text": text,
                })

        elif btype == "LAYOUT_LIST":
            for line_text in resolve_list_items(block, index_blocks(blocks)):
                sequence.append({
                    "type": "list_item",
                    "page": page,
                    "text": line_text,
                })

        elif btype == "LAYOUT_TABLE":
            table_id = match_layout_child(block, blocks, "TABLE")
            if table_id and table_id in tables_by_id:
                matched_tables.add(table_id)
                table = tables_by_id[table_id]
                sequence.append({
                    "type": "table",
                    "page": page,
                    "table_id": table_id,
                    "markdown": table["markdown"],
                    "rows": table["rows"],
                    "cols": table["cols"],
                })

        elif btype == "LAYOUT_KEY_VALUE":
            key_id = match_layout_child(block, blocks, "KEY_VALUE_SET")
            if key_id and key_id in forms_by_id:
                matched_forms.add(key_id)
                form = forms_by_id[key_id]
                sequence.append({
                    "type": "key_value",
                    "page": page,
                    "key": form["key"],
                    "value": form["value"],
                })

        elif btype == "LAYOUT_FIGURE":
            sequence.append({
                "type": "figure",
                "page": page,
                "figure_id": block["Id"],
            })

    # Append unmatched tables at end of their page
    for table in tables:
        if table["table_id"] not in matched_tables:
            insert_at = find_page_end(sequence, table["page"])
            sequence.insert(insert_at, {
                "type": "table",
                "page": table["page"],
                "table_id": table["table_id"],
                "markdown": table["markdown"],
                "rows": table["rows"],
                "cols": table["cols"],
            })

    # Append unmatched forms at end of their page
    for form in forms:
        if form["key_id"] not in matched_forms:
            insert_at = find_page_end(sequence, form["page"])
            sequence.insert(insert_at, {
                "type": "key_value",
                "page": form["page"],
                "key": form["key"],
                "value": form["value"],
            })

    return sequence


def resolve_line_text(block: Dict[str, Any], by_id: Dict[str, Dict[str, Any]]) -> str:
    """Concatenate child LINE text from a layout block.

    Args:
        block: Textract layout block.
        by_id: Block lookup map.

    Returns:
        Joined line text.
    """
    line_ids = [
        cid
        for rel in block.get("Relationships", [])
        if rel["Type"] == "CHILD"
        for cid in rel["Ids"]
    ]
    lines = [
        by_id[lid]["Text"]
        for lid in line_ids
        if by_id.get(lid, {}).get("BlockType") == "LINE"
    ]
    return " ".join(lines).strip()


def resolve_list_items(block: Dict[str, Any], by_id: Dict[str, Dict[str, Any]]) -> List[str]:
    """Extract individual list items from LAYOUT_LIST block.

    Args:
        block: Textract LAYOUT_LIST block.
        by_id: Block lookup map.

    Returns:
        List of item text strings.
    """
    line_ids = [
        cid
        for rel in block.get("Relationships", [])
        if rel["Type"] == "CHILD"
        for cid in rel["Ids"]
    ]
    return [
        by_id[lid]["Text"].strip()
        for lid in line_ids
        if by_id.get(lid, {}).get("BlockType") == "LINE"
        and by_id[lid].get("Text", "").strip()
    ]


def match_layout_child(
    layout_block: Dict[str, Any],
    blocks: List[Dict[str, Any]],
    target_type: str,
) -> Optional[str]:
    """Find raw block that overlaps with a LAYOUT wrapper.

    Args:
        layout_block: LAYOUT_TABLE or LAYOUT_KEY_VALUE block.
        blocks: All blocks.
        target_type: BlockType to match (TABLE or KEY_VALUE_SET).

    Returns:
        Matching block Id or None.
    """
    layout_box = layout_block["Geometry"]["BoundingBox"]
    layout_page = layout_block.get("Page", 1)
    best_id = None
    best_overlap = 0.0

    for candidate in blocks:
        if candidate["BlockType"] != target_type:
            continue
        if candidate.get("Page", 1) != layout_page:
            continue
        overlap = bbox_overlap(layout_box, candidate["Geometry"]["BoundingBox"])
        if overlap > best_overlap:
            best_overlap = overlap
            best_id = candidate["Id"]

    return best_id if best_overlap > 0.0 else None


def bbox_overlap(box_a: Dict[str, float], box_b: Dict[str, float]) -> float:
    """Calculate intersection-over-union of two bounding boxes.

    Args:
        box_a: First bounding box dict with Left, Top, Width, Height.
        box_b: Second bounding box dict.

    Returns:
        IoU ratio (0.0 to 1.0).
    """
    ax0, ay0 = box_a["Left"], box_a["Top"]
    ax1, ay1 = ax0 + box_a["Width"], ay0 + box_a["Height"]
    bx0, by0 = box_b["Left"], box_b["Top"]
    bx1, by1 = bx0 + box_b["Width"], by0 + box_b["Height"]

    ix = max(0.0, min(ax1, bx1) - max(ax0, bx0))
    iy = max(0.0, min(ay1, by1) - max(ay0, by0))
    intersection = ix * iy
    union = (
        box_a["Width"] * box_a["Height"]
        + box_b["Width"] * box_b["Height"]
        - intersection
    )
    return intersection / union if union > 0 else 0.0


def find_page_end(sequence: List[Dict[str, Any]], page: int) -> int:
    """Find insertion point at end of given page.

    Args:
        sequence: Current content sequence.
        page: Target page number.

    Returns:
        Index to insert at (end of page or end of sequence).
    """
    last_idx = -1
    for idx, item in enumerate(sequence):
        if item.get("page", 1) <= page:
            last_idx = idx
    return last_idx + 1 if last_idx >= 0 else len(sequence)


# ---------------------------------------------------------------------------
# OUTPUT FORMATTERS
# ---------------------------------------------------------------------------
def sequence_to_markdown(sequence: List[Dict[str, Any]]) -> str:
    """Convert sequence to markdown document.

    Args:
        sequence: Content sequence from build_sequence().

    Returns:
        Markdown string.
    """
    heading_prefix = {"title": "# ", "section_header": "## "}
    lines = []

    for item in sequence:
        kind = item["type"]
        if kind == "list_item":
            lines.append(f"- {item['text']}")
        elif kind in ("title", "section_header", "paragraph"):
            prefix = heading_prefix.get(kind, "")
            lines.append(f"{prefix}{item['text']}")
        elif kind == "table":
            lines.append(item["markdown"])
        elif kind == "key_value":
            lines.append(f"**{item['key']}:** {item['value']}")
        elif kind == "figure":
            lines.append(f"**[Figure — page {item['page']}]**")
        lines.append("")

    return "\n".join(lines).strip()


def sequence_to_text(sequence: List[Dict[str, Any]]) -> str:
    """Convert sequence to plain text document.

    Args:
        sequence: Content sequence from build_sequence().

    Returns:
        Plain text string.
    """
    lines = []
    current_page = 0

    for item in sequence:
        page = item.get("page", 1)
        if page != current_page:
            lines.append(f"[PAGE {page}]")
            current_page = page

        kind = item["type"]
        if kind == "list_item":
            lines.append(f"  - {item['text']}")
        elif kind in ("title", "section_header", "paragraph"):
            lines.append(item["text"])
        elif kind == "table":
            lines.append(table_to_plain(item["markdown"]))
        elif kind == "key_value":
            lines.append(f"{item['key']}: {item['value']}")
        elif kind == "figure":
            lines.append(f"Figure (page {page})")
        lines.append("")

    return "\n".join(lines).strip()


def table_to_plain(markdown_table: str) -> str:
    """Convert markdown table to plain text columns.

    Args:
        markdown_table: Markdown table string.

    Returns:
        Plain text table with space-padded columns.
    """
    rows = [
        [cell.strip() for cell in line.strip("|").split("|")]
        for line in markdown_table.splitlines()
        if line.strip() and not set(line.replace("|", "").strip()) <= {"-"}
    ]
    if not rows:
        return ""
    col_widths = [max(len(row[i]) for row in rows) for i in range(len(rows[0]))]
    return "\n".join(
        "  ".join(cell.ljust(col_widths[i]) for i, cell in enumerate(row))
        for row in rows
    )


# ---------------------------------------------------------------------------
# S3 OUTPUT
# ---------------------------------------------------------------------------
def write_outputs(
    bucket: str,
    prefix: str,
    source_key: str,
    timestamp: str,
    data: Dict[str, Any],
    markdown: str,
    text: str,
) -> Dict[str, str]:
    """Write JSON, Markdown, and Text outputs to S3.

    Args:
        bucket: Output S3 bucket.
        prefix: Output S3 prefix.
        source_key: Original source key (for naming).
        timestamp: Shared timestamp string.
        data: JSON-serializable dict.
        markdown: Markdown content.
        text: Plain text content.

    Returns:
        Dict with keys json_key, md_key, txt_key.
    """
    s3 = get_s3_client()
    stem = Path(source_key).stem
    safe_stem = re.sub(r"[^\w\-]", "_", stem)

    json_key = f"{prefix.rstrip('/')}/{safe_stem}_{timestamp}.json"
    md_key = f"{prefix.rstrip('/')}/{safe_stem}_{timestamp}.md"
    txt_key = f"{prefix.rstrip('/')}/{safe_stem}_{timestamp}.txt"

    s3.put_object(
        Bucket=bucket, Key=json_key,
        Body=json.dumps(data, indent=2, ensure_ascii=False).encode(),
        ContentType="application/json",
    )
    s3.put_object(
        Bucket=bucket, Key=md_key,
        Body=markdown.encode("utf-8"),
        ContentType="text/markdown",
    )
    s3.put_object(
        Bucket=bucket, Key=txt_key,
        Body=text.encode("utf-8"),
        ContentType="text/plain",
    )

    return {"json_key": json_key, "md_key": md_key, "txt_key": txt_key}


# ---------------------------------------------------------------------------
# JOB TRACKING
# ---------------------------------------------------------------------------
def load_job_map(bucket: str) -> Dict[str, Any]:
    """Load job tracking map from S3.

    Args:
        bucket: S3 bucket.

    Returns:
        Job map dict or empty dict.
    """
    s3 = get_s3_client()
    try:
        obj = s3.get_object(Bucket=bucket, Key=JOB_MAP_KEY)
        return json.loads(obj["Body"].read())
    except s3.exceptions.NoSuchKey:
        return {}
    except s3.exceptions.ClientError as exc:
        if exc.response["Error"]["Code"] in ("NoSuchKey", "404"):
            return {}
        raise


def save_job_map(bucket: str, job_map: Dict[str, Any]) -> None:
    """Save job tracking map to S3.

    Args:
        bucket: S3 bucket.
        job_map: Job map dict.
    """
    s3 = get_s3_client()
    s3.put_object(
        Bucket=bucket, Key=JOB_MAP_KEY,
        Body=json.dumps(job_map, indent=2, ensure_ascii=False).encode(),
        ContentType="application/json",
    )


# ---------------------------------------------------------------------------
# MAIN PROCESSOR
# ---------------------------------------------------------------------------
def process_document(source_bucket: str, source_key: str) -> Dict[str, Any]:
    """Process single PDF through full Textract pipeline.

    Args:
        source_bucket: S3 bucket with PDF.
        source_key: S3 key of PDF.

    Returns:
        Result dict with status, output keys, counts.
    """
    logger.info("Processing s3://%s/%s", source_bucket, source_key)
    timestamp = datetime.now(timezone.utc).strftime("%Y%m%d_%H%M%S")

    # Start Textract
    job_id = start_analysis(source_bucket, source_key)

    # Wait for completion
    result = wait_for_job(job_id)
    status = result["JobStatus"]

    if status != "SUCCEEDED":
        logger.error("Job failed for %s: %s", source_key, status)
        return {
            "source_key": source_key,
            "job_id": job_id,
            "status": status,
            "error": result.get("StatusMessage", "Unknown"),
        }

    blocks = result["Blocks"]

    # Extract components
    text_elements = extract_text_elements(blocks)
    tables = extract_tables(blocks)
    forms = extract_forms(blocks)
    figures = detect_figures(blocks)

    # Build unified sequence
    sequence = build_sequence(blocks, tables, forms)

    # Format outputs
    markdown = sequence_to_markdown(sequence)
    plain_text = sequence_to_text(sequence)

    # Build JSON data
    output_data = {
        "source_bucket": source_bucket,
        "source_key": source_key,
        "job_id": job_id,
        "processed_at": datetime.now(timezone.utc).isoformat(),
        "block_count": len(blocks),
        "line_count": sum(1 for b in blocks if b["BlockType"] == "LINE"),
        "table_count": len(tables),
        "form_count": len(forms),
        "figure_count": len(figures),
        "text_elements": text_elements,
        "tables": tables,
        "forms": forms,
        "figures": figures,
        "sequence": sequence,
    }

    # Write to S3
    keys = write_outputs(
        OUTPUT_BUCKET, OUTPUT_PREFIX, source_key, timestamp,
        output_data, markdown, plain_text,
    )

    # Update job map
    job_map = load_job_map(OUTPUT_BUCKET)
    job_map[source_key] = {
        "job_id": job_id,
        "status": "SUCCEEDED",
        "timestamp": timestamp,
        **keys,
    }
    save_job_map(OUTPUT_BUCKET, job_map)

    logger.info("Completed %s -> %s", source_key, keys["json_key"])
    return {
        "source_key": source_key,
        "job_id": job_id,
        "status": "SUCCEEDED",
        **keys,
    }


# ---------------------------------------------------------------------------
# BATCH PROCESSOR
# ---------------------------------------------------------------------------
def process_batch(source_bucket: str, source_prefix: str) -> List[Dict[str, Any]]:
    """Process all PDFs under an S3 prefix.

    Args:
        source_bucket: S3 bucket.
        source_prefix: S3 prefix (folder path).

    Returns:
        List of result dicts per document.
    """
    s3 = get_s3_client()
    paginator = s3.get_paginator("list_objects_v2")
    pdf_keys = [
        obj["Key"]
        for page in paginator.paginate(Bucket=source_bucket, Prefix=source_prefix)
        for obj in page.get("Contents", [])
        if obj["Key"].lower().endswith(".pdf")
    ]

    logger.info("Found %d PDFs under s3://%s/%s", len(pdf_keys), source_bucket, source_prefix)

    results = []
    for key in pdf_keys:
        try:
            result = process_document(source_bucket, key)
            results.append(result)
        except Exception as exc:
            logger.exception("Failed to process %s: %s", key, exc)
            results.append({
                "source_key": key,
                "status": "ERROR",
                "error": str(exc),
            })

    return results


# ---------------------------------------------------------------------------
# LAMBDA HANDLER
# ---------------------------------------------------------------------------
def lambda_handler(event: Dict[str, Any], context: Any) -> Dict[str, Any]:
    """AWS Lambda entry point.

    Expected event format:
        {
            "source_bucket": "my-bucket",
            "source_key": "documents/file.pdf",
            "source_prefix": "documents/"
        }

    Args:
        event: Lambda event dict.

def generate_folder_name(url: str) -> str:
    parsed = urlparse(url)
    path = parsed.path.strip("/")
    timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")

    if path:
        last_segment = path.split("/")[-1] if "/" in path else path
        last_segment = re.sub(
            r"\.(html|htm|php|asp|aspx|jsp)$", "", last_segment, flags=re.IGNORECASE
        )
        clean_name = re.sub(r"[^a-zA-Z0-9]+", "_", last_segment).strip("_")[:80]
        if not clean_name:
            clean_name = re.sub(r"[^a-zA-Z0-9]+", "_", parsed.netloc).strip("_")
    else:
        clean_name = re.sub(r"[^a-zA-Z0-9]+", "_", parsed.netloc).strip("_")

    return f"{clean_name}_{timestamp}"


def derive_section_prefix(url: str) -> str:
    parsed = urlparse(url)
    path = parsed.path.rstrip("/")
    if "." in path.split("/")[-1]:
        path = "/".join(path.split("/")[:-1])
    return path or "/"


def normalize_url(url: str) -> str:
    return url.split("#")[0].rstrip("/")


def normalize_prefix(prefix: str) -> str:
    p = prefix.strip()
    if not p.startswith("/"):
        p = "/" + p
    return p.rstrip("/") or "/"


def canon_language(s: str) -> str:
    """Reduce a Bedrock-returned language label to a canonical form."""
    if not s:
        return ""
    cleaned = re.sub(r"[^\w\-]", "", s.strip().lower())
    for canon, aliases in _LANG_ALIASES.items():
        if cleaned in aliases:
            return canon
    return cleaned


def is_excluded_language(detected: str, exclude_list: Iterable[str]) -> bool:
    if not exclude_list:
        return False
    detected_canon = canon_language(detected)
    if not detected_canon:
        return False
    excluded_canon = {canon_language(x) for x in exclude_list}
    return detected_canon in excluded_canon


def sanitize_doc_name(name: str) -> str:
    """Bedrock document name spec: alphanumeric, whitespace, hyphens, parens, brackets."""
    cleaned = re.sub(r"[^a-zA-Z0-9\s\-\(\)\[\]]", " ", name or "").strip()
    cleaned = re.sub(r"\s+", " ", cleaned)
    return cleaned[:50] or "document"


def s3_put_object(key: str, body, content_type: str = "text/plain") -> str:
    if isinstance(body, str):
        body = body.encode("utf-8")
    s3_client.put_object(
        Bucket=S3_BUCKET, Key=key, Body=body, ContentType=content_type
    )
    return f"s3://{S3_BUCKET}/{key}"

def bedrock_call(prompt: str, max_tokens: int = 500) -> str:
    try:
        resp = bedrock_client.converse(
            modelId=BEDROCK_MODEL_ID,
            messages=[{"role": "user", "content": [{"text": prompt}]}],
            inferenceConfig={"maxTokens": max_tokens, "temperature": 0.3},
        )
        return resp["output"]["message"]["content"][0]["text"]
    except Exception as e:
        return f"[ERROR: {e}]"


def bedrock_pdf_ocr(pdf_bytes: bytes, title_hint: str = "document",
                    max_tokens: int = 4096) -> str:
    """
    OCR / text extraction via Bedrock Claude multimodal.

    Uses the converse() API's document content block — accepts raw PDF bytes,
    no rasterisation needed. Handles scanned PDFs, form fields, and mixed
    text+image documents that PyPDF2 cannot.

    Returns the extracted text, or "[OCR ERROR: ...]" on failure.
    """
    if not pdf_bytes:
        return "[OCR ERROR: empty bytes]"
    if len(pdf_bytes) > BEDROCK_DOC_SIZE_LIMIT_BYTES:
        return (f"[OCR ERROR: PDF size {len(pdf_bytes)/1024/1024:.1f} MB exceeds "
                f"Bedrock document limit ({BEDROCK_DOC_SIZE_LIMIT_BYTES/1024/1024:.1f} MB)]")

    prompt = (
        "Extract all visible text from this PDF exactly as it appears. "
        "Preserve structure: paragraphs, headings, form labels, table cells, "
        "checkbox options, signature lines. If the document is not in English, "
        "transcribe it in its original language without translating. "
        "Return only the extracted text — no commentary, no markdown wrappers."
    )

    try:
        resp = bedrock_client.converse(
            modelId=BEDROCK_MODEL_ID,
            messages=[{
                "role": "user",
                "content": [
                    {"document": {
                        "format": "pdf",
                        "name": sanitize_doc_name(title_hint),
                        "source": {"bytes": pdf_bytes},
                    }},
                    {"text": prompt},
                ],
            }],
            inferenceConfig={"maxTokens": max_tokens, "temperature": 0.0},
        )
        text = resp["output"]["message"]["content"][0]["text"]
        return text.strip() if text and text.strip() else "[OCR ERROR: empty response]"
    except Exception as e:
        return f"[OCR ERROR: {e}]"


def detect_language(text: str) -> str:
    if not text or len(text.strip()) < 10:
        return "unknown"
    prompt = (
        "What language is the following text? Respond with ONLY the language name.\n\n"
        f"Text sample: {text[:500]}\n\nLanguage:"
    )
    return bedrock_call(prompt, max_tokens=20).strip().lower().replace(".", "")


def summarize_pdf(text: str, title: str) -> str:
    prompt = (
        f"Summarize the following document in 3-4 sentences.\n\n"
        f"Title: {title}\nContent: {text[:8000]}\n\n"
        f"Highlight: main topic, key advice, and target audience."
    )
    return bedrock_call(prompt, max_tokens=500)

class SectionCrawler:
    """Walks HTML pages within scope and harvests every PDF link along the way."""

    def __init__(
        self,
        s3_prefix: str,
        base_domain: str,
        section_prefixes: Optional[List[str]] = None,
        allow_full_domain: bool = False,
        delay: float = 0.5,
        max_pages: Optional[int] = None,
        max_depth: Optional[int] = None,
        request_timeout: int = 20,
        verbose: bool = False,
    ):
        self.s3_prefix = s3_prefix
        self.base_domain = base_domain.replace("www.", "")
        self.allow_full_domain = allow_full_domain
        self.section_prefixes = [normalize_prefix(p) for p in (section_prefixes or [])]
        self.delay = delay
        self.max_pages = max_pages
        self.max_depth = max_depth
        self.request_timeout = request_timeout
        self.verbose = verbose

        self.visited: Set[str] = set()
        self.queued: Set[str] = set()
        self.pdf_catalog: Dict[str, Dict] = {}
        self.reject_stats: Counter = Counter()
        self._to_visit: deque = deque()

        self.session = requests.Session()
        self.session.headers.update({
            "User-Agent": (
                "Mozilla/5.0 (Windows NT 10.0; Win64; x64) "
                "AppleWebKit/537.36 (KHTML, like Gecko) "
                "Chrome/124.0 Safari/537.36"
            )
        })

    def _same_site(self, url: str) -> bool:
        parsed = urlparse(url)
        if parsed.scheme not in ("http", "https"):
            return False
        host = parsed.netloc.replace("www.", "")
        return host == self.base_domain or host.endswith("." + self.base_domain)

    def _in_scope(self, url: str) -> bool:
        if self.allow_full_domain:
            return True
        if not self.section_prefixes:
            return True
        path = urlparse(url).path.rstrip("/") or "/"
        for prefix in self.section_prefixes:
            if path == prefix or path.startswith(prefix + "/"):
                return True
        return False

    def _classify_link(self, url: str) -> Tuple[str, str]:
        parsed = urlparse(url)
        if parsed.scheme not in ("http", "https"):
            return "reject", "bad_scheme"
        if not self._same_site(url):
            return "reject", "off_site"

        path_lower = parsed.path.lower()
        if path_lower.endswith(".pdf"):
            return "pdf", "pdf"
        if path_lower.endswith(SKIP_EXTENSIONS):
            return "reject", "binary_extension"
        if not self._in_scope(url):
            return "reject", "out_of_scope"

        return "crawl", "in_scope"

    def _html_filename(self, url: str) -> str:
        parsed = urlparse(url)
        path = parsed.path.strip("/")
        if path:
            safe_path = re.sub(r"[^a-zA-Z0-9._-]+", "_", path.replace("/", "_"))
            base = safe_path[:100]
        else:
            base = "index"
        domain = parsed.netloc.replace("www.", "")
        full = f"{domain}_{base}"
        if len(full) > 150:
            full = f"{domain}_{hashlib.md5(url.encode()).hexdigest()[:8]}"
        return f"{full}.txt"

    def _record_pdf(self, anchor_tag, full_url: str, source_page: str) -> None:
        if full_url in self.pdf_catalog:
            self.pdf_catalog[full_url]["referrers"].add(source_page)
            return

        title = anchor_tag.get_text(strip=True)
        if not title:
            parent = anchor_tag.find_parent(["h1", "h2", "h3", "h4", "li"])
            if parent:
                title = parent.get_text(strip=True)
        if not title:
            title = Path(urlparse(full_url).path).stem.replace("-", " ").title()

        self.pdf_catalog[full_url] = {
            "title": title[:150],
            "url": full_url,
            "filename": Path(urlparse(full_url).path).name,
            "first_seen_on": source_page,
            "referrers": {source_page},
        }

    def _scrape_page(self, url: str, depth: int) -> Tuple[bool, str]:
        self.visited.add(url)
        link_counts: Counter = Counter()

        try:
            print(f"  [HTML d={depth}] {url}")
            resp = self.session.get(url, timeout=self.request_timeout, allow_redirects=True)
            resp.raise_for_status()

            ctype = resp.headers.get("content-type", "").lower()
            if "text/html" not in ctype and "application/xhtml" not in ctype:
                return False, f"Non-HTML: {ctype}"

            final_url = normalize_url(resp.url)
            if not self._same_site(final_url):
                return False, f"Redirected off-site: {final_url}"

            soup = BeautifulSoup(resp.text, "html.parser")

            for a in soup.find_all("a", href=True):
                href = a["href"].strip()
                if not href or href.startswith(("javascript:", "mailto:", "tel:", "#")):
                    link_counts["non_http"] += 1
                    continue
                full = normalize_url(urljoin(url, href))
                category, reason = self._classify_link(full)
                link_counts[reason] += 1

                if category == "pdf":
                    self._record_pdf(a, full, source_page=url)
                elif category == "crawl":
                    if full not in self.visited and full not in self.queued:
                        if self.max_depth is None or depth + 1 <= self.max_depth:
                            self.queued.add(full)
                            self._to_visit.append((full, depth + 1))
                        else:
                            link_counts["depth_capped"] += 1
                            self.reject_stats["depth_capped"] += 1
                else:
                    self.reject_stats[reason] += 1
                    if self.verbose:
                        print(f"        rejected ({reason}): {full}")

            for el in soup(["script", "style", "noscript", "meta", "link"]):
                el.decompose()
            text = soup.get_text(separator="\n", strip=True)
            lines = [ln.strip() for ln in text.splitlines() if ln.strip()]
            clean_text = "\n".join(lines)

            key = f"{self.s3_prefix}/web_text/{self._html_filename(url)}"
            content = f"Source URL: {url}\n{'=' * 80}\n\n{clean_text}"
            s3_path = s3_put_object(key, content, "text/plain")

            print(
                f"        links: {sum(link_counts.values())} total | "
                f"in_scope={link_counts.get('in_scope', 0)} | "
                f"pdf={link_counts.get('pdf', 0)} | "
                f"out_of_scope={link_counts.get('out_of_scope', 0)} | "
                f"off_site={link_counts.get('off_site', 0)}"
            )
            return True, s3_path

        except Exception as e:
            return False, f"Error: {e}"

    def crawl(self, seed_urls: List[str]) -> Dict:
        scope_desc = (
            "FULL DOMAIN" if self.allow_full_domain
            else f"prefixes={self.section_prefixes}"
        )
        print(
            f"\n{'=' * 60}\n"
            f"Crawling {self.base_domain}\n"
            f"  scope:     {scope_desc}\n"
            f"  max_depth: {self.max_depth if self.max_depth is not None else 'unlimited'}\n"
            f"  max_pages: {self.max_pages if self.max_pages else 'unlimited'}\n"
            f"  seeds:     {seed_urls}\n"
            f"{'=' * 60}"
        )

        for seed in seed_urls:
            seed = normalize_url(seed)
            if seed not in self.queued:
                self.queued.add(seed)
                self._to_visit.append((seed, 0))

        result = {"scraped": 0, "failed": 0, "s3_files": [], "max_depth_seen": 0}

        while self._to_visit:
            if self.max_pages and result["scraped"] >= self.max_pages:
                print(f"  Reached max_pages={self.max_pages}; stopping.")
                break

            url, depth = self._to_visit.popleft()
            if url in self.visited:
                continue

            ok, msg = self._scrape_page(url, depth)
            result["max_depth_seen"] = max(result["max_depth_seen"], depth)

            if ok:
                result["scraped"] += 1
                result["s3_files"].append(msg)
            else:
                result["failed"] += 1
                print(f"      skipped: {msg}")

            if self._to_visit and self.delay > 0:
                time.sleep(self.delay)

        result["pdf_count"] = len(self.pdf_catalog)
        result["reject_stats"] = dict(self.reject_stats)

        print(
            f"\n  HTML pages scraped: {result['scraped']}"
            f" | failed: {result['failed']}"
            f" | unique PDFs: {result['pdf_count']}"
            f" | deepest reached: d={result['max_depth_seen']}"
        )
        if self.reject_stats:
            print(f"  Rejected-link tally: {dict(self.reject_stats)}")
        return result


# ----------------------------------------------------------------------------
# PDF processor — PyPDF2 first, optional Bedrock OCR fallback for image PDFs.
# ----------------------------------------------------------------------------
class PDFProcessor:
    """
    Download PDFs, extract text, language-detect, optionally filter, summarise.

    Extraction strategy:
        1. PyPDF2 (fast, free, handles text-based PDFs).
        2. If PyPDF2 fails AND bedrock_ocr_fallback=True, retry via
           Bedrock Claude multimodal (handles scanned / image-only PDFs).
        3. If both fail, record status="processed_no_text" and move on.

    Language filtering, when enabled, applies AFTER extraction and works
    identically regardless of which path produced the text — so a scanned
    Spanish PDF is now caught.
    """

    def __init__(
        self,
        s3_prefix: str,
        exclude_languages: Optional[List[str]] = None,
        bedrock_ocr_fallback: bool = False,
        request_timeout: int = 60,
    ):
        self.s3_prefix = s3_prefix
        self.exclude_languages = [canon_language(x) for x in (exclude_languages or [])]
        self.bedrock_ocr_fallback = bedrock_ocr_fallback
        self.request_timeout = request_timeout
        self.session = requests.Session()
        self.session.headers.update({
            "User-Agent": (
                "Mozilla/5.0 (Windows NT 10.0; Win64; x64) "
                "AppleWebKit/537.36 (KHTML, like Gecko) "
                "Chrome/124.0 Safari/537.36"
            )
        })

    @staticmethod
    def _clean_pdf_filename(name: str) -> str:
        clean = "".join(c for c in name if c.isalnum() or c in (".", "-", "_")).strip()
        if not clean.endswith(".pdf"):
            clean += ".pdf"
        return clean or "document.pdf"

    def _extract_text_pypdf2(self, pdf_bytes: bytes, max_chars: int = 5000) -> str:
        try:
            from PyPDF2 import PdfReader
            reader = PdfReader(BytesIO(pdf_bytes))
            text = ""
            for page in reader.pages:
                t = page.extract_text() or ""
                text += t + "\n"
                if len(text) > max_chars:
                    break
            return text[:max_chars] if text.strip() else "[No text extractable]"
        except Exception as e:
            return f"[Error: {e}]"

    def _extract_text(
        self, pdf_bytes: bytes, title_hint: str
    ) -> Tuple[str, str]:
        """
        Returns (text, extraction_method).

        extraction_method is one of: "pypdf2", "bedrock_ocr", "failed".
        """
        # First pass: PyPDF2
        text = self._extract_text_pypdf2(pdf_bytes)
        if text and not text.startswith("["):
            return text, "pypdf2"

        # Second pass: Bedrock OCR (if enabled)
        if self.bedrock_ocr_fallback:
            print(f"    PyPDF2 failed ({text!r:.60}...) — trying Bedrock OCR...")
            ocr_text = bedrock_pdf_ocr(pdf_bytes, title_hint=title_hint)
            if ocr_text and not ocr_text.startswith("["):
                print(f"    OCR succeeded — {len(ocr_text)} chars extracted.")
                # Match the PyPDF2 char cap for downstream consistency.
                return ocr_text[:5000], "bedrock_ocr"
            else:
                print(f"    OCR also failed: {ocr_text[:120]}")
                return ocr_text, "failed"

        return text, "failed"

    def process(self, pdf_catalog: Dict[str, Dict]) -> List[Dict]:
        if not pdf_catalog:
            print("  No PDFs to process.")
            return []

        if self.exclude_languages:
            print(f"  Language exclude filter active: {self.exclude_languages}")
        if self.bedrock_ocr_fallback:
            print("  Bedrock OCR fallback: ENABLED for image / scanned PDFs.")

        results: List[Dict] = []
        total = len(pdf_catalog)
        skipped_by_language = 0
        ocr_used = 0
        seen_filenames: Dict[str, str] = {}

        for i, (pdf_url, meta) in enumerate(pdf_catalog.items(), 1):
            print(f"\n  [PDF {i}/{total}] {meta['title']}")
            print(f"    url: {pdf_url}")
            print(f"    discovered on: {meta['first_seen_on']}")

            try:
                resp = self.session.get(pdf_url, timeout=self.request_timeout)
                resp.raise_for_status()
                pdf_bytes = resp.content
                size_mb = len(pdf_bytes) / (1024 * 1024)

                # --- Extract text (PyPDF2, then OCR if enabled) ---
                text, extraction_method = self._extract_text(
                    pdf_bytes, title_hint=meta["title"]
                )
                text_ok = bool(text) and not text.startswith("[")
                if extraction_method == "bedrock_ocr":
                    ocr_used += 1

                # --- Language detection ---
                if text_ok:
                    language = detect_language(text)
                else:
                    language = "unknown_or_image_pdf"
                print(f"    language: {language}  (via {extraction_method})")

                # --- Language exclusion filter ---
                if is_excluded_language(language, self.exclude_languages):
                    canon = canon_language(language)
                    print(
                        f"    SKIPPED — '{language}' (canonical: '{canon}') "
                        f"is in exclude list. Not uploaded to S3."
                    )
                    results.append({
                        "title": meta["title"],
                        "url": pdf_url,
                        "first_seen_on": meta["first_seen_on"],
                        "referrers": sorted(meta["referrers"]),
                        "language": language,
                        "language_canonical": canon,
                        "extraction_method": extraction_method,
                        "status": "skipped_excluded_language",
                        "size_mb": round(size_mb, 3),
                    })
                    skipped_by_language += 1
                    continue

                # --- Upload PDF bytes to S3 ---
                clean_name = self._clean_pdf_filename(meta["filename"])
                if clean_name in seen_filenames and seen_filenames[clean_name] != pdf_url:
                    suffix = hashlib.md5(pdf_url.encode()).hexdigest()[:6]
                    clean_name = clean_name.replace(".pdf", f"_{suffix}.pdf")
                seen_filenames[clean_name] = pdf_url

                pdf_key = f"{self.s3_prefix}/pdfs/{clean_name}"
                s3_client.put_object(
                    Bucket=S3_BUCKET, Key=pdf_key, Body=pdf_bytes,
                    ContentType="application/pdf",
                )
                pdf_s3 = f"s3://{S3_BUCKET}/{pdf_key}"
                print(f"    saved: {pdf_s3} ({size_mb:.2f} MB)")

                record = {
                    "title": meta["title"],
                    "pdf_s3_path": pdf_s3,
                    "url": pdf_url,
                    "first_seen_on": meta["first_seen_on"],
                    "referrers": sorted(meta["referrers"]),
                    "language": language,
                    "language_canonical": canon_language(language),
                    "extraction_method": extraction_method,
                    "size_mb": round(size_mb, 3),
                }

                if text_ok:
                    summary = summarize_pdf(text, meta["title"])
                    print(f"    summary: {summary[:120]}...")
                    text_key = (
                        f"{self.s3_prefix}/pdfs/"
                        f"{clean_name.replace('.pdf', '_extracted.txt')}"
                    )
                    s3_put_object(text_key, text, "text/plain")
                    record.update({
                        "text_s3_path": f"s3://{S3_BUCKET}/{text_key}",
                        "summary": summary,
                        "text_preview": text[:500],
                        "status": "processed",
                    })
                else:
                    print(f"    note: no text after PyPDF2 + OCR. Stored as bytes only.")
                    record["status"] = "processed_no_text"
                    record["error"] = text

                results.append(record)

            except Exception as e:
                print(f"    failed: {e}")
                results.append({
                    "title": meta["title"],
                    "url": pdf_url,
                    "first_seen_on": meta["first_seen_on"],
                    "status": "error",
                    "error": str(e),
                })

        print(f"\n  Extraction summary:")
        method_counter = Counter(r.get("extraction_method") for r in results
                                 if r.get("extraction_method"))
        for method, count in method_counter.most_common():
            print(f"    {method}: {count}")
        if self.exclude_languages:
            print(f"  PDFs filtered by language: {skipped_by_language}/{total}")
        if self.bedrock_ocr_fallback:
            print(f"  PDFs rescued via OCR:      {ocr_used}")

        return results

class UnifiedScraperPipeline:
    """HTML crawl + PDF processing with optional language filter + OCR fallback."""

    def run(
        self,
        url: str,
        section_prefixes: Optional[List[str]] = None,
        allow_full_domain: bool = False,
        max_pages: Optional[int] = None,
        max_depth: Optional[int] = None,
        web_delay: float = 0.5,
        scrape_pdfs: bool = True,
        verbose: bool = False,
        extra_seed_urls: Optional[List[str]] = None,
        exclude_languages: Optional[List[str]] = None,
        bedrock_ocr_fallback: bool = False,
    ) -> Dict:
        if section_prefixes is None and not allow_full_domain:
            section_prefixes = [derive_section_prefix(url)]

        base_domain = urlparse(url).netloc.replace("www.", "")
        folder_name = generate_folder_name(url)
        s3_prefix = f"{S3_BASE_PREFIX}/{folder_name}"

        excluded_canon = [canon_language(x) for x in (exclude_languages or [])]

        print("=" * 70)
        print("UNIFIED SCRAPER PIPELINE")
        print("=" * 70)
        print(f"Start URL:         {url}")
        print(f"Full domain?:      {allow_full_domain}")
        print(f"Section prefixes:  {section_prefixes}")
        print(f"Base domain:       {base_domain}")
        print(f"Max pages:         {max_pages if max_pages else 'unlimited'}")
        print(f"Max depth:         {max_depth if max_depth is not None else 'unlimited'}")
        print(f"Exclude languages: {excluded_canon if excluded_canon else 'none'}")
        print(f"OCR fallback:      {bedrock_ocr_fallback}")
        print(f"S3 output:         s3://{S3_BUCKET}/{s3_prefix}")
        print("=" * 70)

        results = {
            "url": url,
            "section_prefixes": section_prefixes,
            "allow_full_domain": allow_full_domain,
            "max_depth": max_depth,
            "exclude_languages": excluded_canon,
            "bedrock_ocr_fallback": bedrock_ocr_fallback,
            "s3_bucket": S3_BUCKET,
            "s3_prefix": s3_prefix,
            "s3_full_path": f"s3://{S3_BUCKET}/{s3_prefix}",
            "folder_name": folder_name,
            "web_results": None,
            "pdf_results": None,
        }

        print("\n[STEP 1] Crawling HTML and discovering PDFs...")
        crawler = SectionCrawler(
            s3_prefix=s3_prefix,
            base_domain=base_domain,
            section_prefixes=section_prefixes,
            allow_full_domain=allow_full_domain,
            delay=web_delay,
            max_pages=max_pages,
            max_depth=max_depth,
            verbose=verbose,
        )
        seeds = [url] + (extra_seed_urls or [])
        web_results = crawler.crawl(seeds)
        results["web_results"] = web_results

        if scrape_pdfs and crawler.pdf_catalog:
            print("\n[STEP 2] Downloading & summarising PDFs...")
            processor = PDFProcessor(
                s3_prefix=s3_prefix,
                exclude_languages=excluded_canon,
                bedrock_ocr_fallback=bedrock_ocr_fallback,
            )
            pdf_results = processor.process(crawler.pdf_catalog)
            results["pdf_results"] = pdf_results

            summaries_key = f"{s3_prefix}/pdf_summaries.json"
            s3_put_object(summaries_key, json.dumps(pdf_results, indent=2), "application/json")
            print(f"\n  PDF summaries saved: s3://{S3_BUCKET}/{summaries_key}")
        elif not crawler.pdf_catalog:
            print("\n[STEP 2] No PDFs discovered.")

        report = self._generate_report(results)
        report_key = f"{s3_prefix}/pipeline_report.txt"
        report_path = s3_put_object(report_key, report, "text/plain")

        print("\n" + "=" * 70)
        print("FINAL REPORT")
        print("=" * 70)
        print(f"S3 output:      s3://{S3_BUCKET}/{s3_prefix}")
        print(f"Report:         {report_path}")
        print(f"HTML scraped:   {web_results['scraped']}")
        print(f"PDFs found:     {web_results.get('pdf_count', 0)}")
        print(f"Deepest depth:  {web_results.get('max_depth_seen', 0)}")
        if web_results.get("reject_stats"):
            print(f"Rejected links: {web_results['reject_stats']}")

        if results["pdf_results"]:
            stats = Counter(r.get("status", "unknown") for r in results["pdf_results"])
            print(f"PDF outcomes:   {dict(stats)}")
            methods = Counter(r.get("extraction_method") for r in results["pdf_results"]
                              if r.get("extraction_method"))
            print(f"Extraction:     {dict(methods)}")
            if excluded_canon:
                print(f"Language filter dropped: "
                      f"{stats.get('skipped_excluded_language', 0)} PDF(s) "
                      f"({excluded_canon})")
        print("=" * 70)
        return results

    def _generate_report(self, results: Dict) -> str:
        lines = [
            "=" * 70,
            "UNIFIED SCRAPER PIPELINE REPORT",
            "=" * 70,
            f"URL:                  {results['url']}",
            f"Full domain?:         {results['allow_full_domain']}",
            f"Section prefixes:     {results['section_prefixes']}",
            f"Max depth:            {results['max_depth']}",
            f"Exclude languages:    {results.get('exclude_languages') or 'none'}",
            f"OCR fallback:         {results.get('bedrock_ocr_fallback', False)}",
            f"S3 path:              {results['s3_full_path']}",
            f"Timestamp:            {datetime.now().isoformat()}",
            "",
        ]

        wr = results.get("web_results") or {}
        lines += [
            "HTML CRAWL",
            "-" * 40,
            f"  Pages scraped:  {wr.get('scraped', 0)}",
            f"  Failed:         {wr.get('failed', 0)}",
            f"  PDFs found:     {wr.get('pdf_count', 0)}",
            f"  Deepest depth:  {wr.get('max_depth_seen', 0)}",
            f"  Rejected links: {wr.get('reject_stats', {})}",
            "  Files:",
        ]
        for f in wr.get("s3_files", []):
            lines.append(f"    - {f}")
        lines.append("")

        if results.get("pdf_results"):
            status_counter = Counter(r.get("status", "unknown")
                                     for r in results["pdf_results"])
            method_counter = Counter(r.get("extraction_method")
                                     for r in results["pdf_results"]
                                     if r.get("extraction_method"))
            lines += [
                "PDF PROCESSING",
                "-" * 40,
                f"  Status breakdown:     {dict(status_counter)}",
                f"  Extraction methods:   {dict(method_counter)}",
                "",
            ]
            for r in results["pdf_results"]:
                status = r.get("status", "unknown")
                lines.append(f"  [{status}] {r['title']}")
                lines.append(f"    URL:        {r['url']}")
                lines.append(f"    Found on:   {r.get('first_seen_on', 'N/A')}")
                if "language" in r:
                    lang_extra = ""
                    if r.get("language_canonical") and r["language_canonical"] != r["language"]:
                        lang_extra = f" (canonical: {r['language_canonical']})"
                    lines.append(f"    Language:   {r['language']}{lang_extra}")
                if "extraction_method" in r:
                    lines.append(f"    Extracted:  {r['extraction_method']}")
                if "pdf_s3_path" in r:
                    lines.append(f"    PDF:        {r['pdf_s3_path']}")
                if "text_s3_path" in r:
                    lines.append(f"    Text:       {r['text_s3_path']}")
                if "summary" in r:
                    lines.append(f"    Summary:    {r['summary'][:200]}...")
                if "error" in r:
                    lines.append(f"    ERROR:      {r['error']}")
                lines.append("")

        lines.append("=" * 70)
        return "\n".join(lines)

"""Unsupervised Labeling Utility — the Index (flat term -> sources map).

Implements the pipeline from Section 3 of the design doc:
    EXTRACT candidate terms (Nova, per document)
    EMBED candidate terms + short context (Titan v2)
    MATCH against existing index entries (cosine similarity)
        high similarity  -> auto-merge as alias
        low similarity   -> create new entry
        borderline       -> Nova judge decides merge vs new entry

This is the complement to tree_builder.py's Tree (Section 2's rationale):
the Tree is free/deterministic navigation; the Index is the LLM-assisted
layer that resolves a term mentioned in multiple, unrelated branches of
the Tree back to every source that discusses it.

Usage:
    from index_builder import build_index_from_job_map, save_index, validate_index

    index = build_index_from_job_map(bucket="bankrupcy-data-bkt", region="us-east-1")
    warnings = validate_index(index)
    save_index("bankrupcy-data-bkt", index)
"""
import json
import logging
import re
from dataclasses import dataclass
from typing import Any, Dict, List, Optional, Tuple

import boto3
from tenacity import retry, retry_if_exception_type, stop_after_attempt, wait_exponential

from textract_pipeline import TextractTransientError, load_job_map

log = logging.getLogger(__name__)

INDEX_KEY = "textract-jobs/index.json"
INDEX_INTERNAL_KEY = "textract-jobs/index_internal.json"
NOVA_MODEL_ID = "amazon.nova-2-lite-v1:0"
TITAN_EMBED_MODEL_ID = "amazon.titan-embed-text-v2:0"
EMBED_DIM = 1024

# Cosine similarity thresholds — starting points, not proven. Calibrate
# against real extracted terms once available; these are not tuned yet.
SIM_AUTO_MERGE = 0.92
SIM_NO_MATCH = 0.75

MAX_DOC_CHARS_FOR_EXTRACTION = 30_000  # size-based fallback threshold
CONTEXT_SNIPPET_CHARS = 200


# ---------------------------------------------------------------------------
# Nova extraction — per document, using the already-assembled document text
# ---------------------------------------------------------------------------
_EXTRACTION_SYSTEM_PROMPT = (
    "You extract domain-specific terminology from student loan servicing "
    "and bankruptcy documents. Respond with valid JSON only, no prose, "
    "no code fences."
)

_EXTRACTION_INSTRUCTIONS = """\
Extract the domain-specific terms, form names, and defined concepts from
the document below — the kind of vocabulary a call-center agent would
need a glossary entry for (e.g. "Automatic Stay", "Loan Rehabilitation",
"Chapter 13 Plan"). Do not extract generic words or common phrases.

Respond with a JSON array. Each item:
{
  "term": "canonical term name",
  "context": "the sentence or phrase where it appears, verbatim, max 200 chars"
}

Extract at most 30 terms. If the document has none, return [].

Document:
"""


def _is_transient(exc: BaseException) -> bool:
    msg = str(exc).lower()
    return any(m in msg for m in ("throttl", "429", "500", "502", "503", "504", "timeout"))


@retry(
    retry=retry_if_exception_type(TextractTransientError),
    wait=wait_exponential(multiplier=2, min=2, max=30),
    stop=stop_after_attempt(4),
    reraise=True,
)
def _call_nova(bedrock_runtime, system_prompt: str, user_prompt: str,
               max_tokens: int = 2048, temperature: float = 0.0) -> str:
    try:
        response = bedrock_runtime.converse(
            modelId=NOVA_MODEL_ID,
            messages=[{"role": "user", "content": [{"text": user_prompt}]}],
            system=[{"text": system_prompt}],
            inferenceConfig={"maxTokens": max_tokens, "temperature": temperature},
        )
    except Exception as exc:
        if _is_transient(exc):
            raise TextractTransientError(str(exc)) from exc
        raise
    blocks = response["output"]["message"]["content"]
    return "".join(b.get("text", "") for b in blocks).strip()


def _parse_json_array(raw: str) -> List[Dict[str, Any]]:
    """Extract a JSON array from a model response, tolerating code fences
    and stray prose around it.
    """
    text = raw.strip()
    if text.startswith("```"):
        text = re.sub(r"^```(?:json)?\s*", "", text)
        text = re.sub(r"```\s*$", "", text)
    start = text.find("[")
    end = text.rfind("]")
    if start == -1 or end == -1 or end <= start:
        return []
    try:
        parsed = json.loads(text[start:end + 1])
    except json.JSONDecodeError:
        return []
    return parsed if isinstance(parsed, list) else []


def extract_candidate_terms(
    bedrock_runtime, document_text: str,
) -> List[Dict[str, str]]:
    """Extract {term, context} pairs from one document's assembled text.
    Falls back to truncating oversized documents rather than failing —
    see MAX_DOC_CHARS_FOR_EXTRACTION.
    """
    text = document_text
    if len(text) > MAX_DOC_CHARS_FOR_EXTRACTION:
        log.warning("document is %d chars, truncating to %d for extraction",
                   len(text), MAX_DOC_CHARS_FOR_EXTRACTION)
        text = text[:MAX_DOC_CHARS_FOR_EXTRACTION]

    raw = _call_nova(bedrock_runtime, _EXTRACTION_SYSTEM_PROMPT,
                     _EXTRACTION_INSTRUCTIONS + text)
    items = _parse_json_array(raw)

    terms = []
    for item in items:
        term = (item.get("term") or "").strip()
        context = (item.get("context") or "").strip()[:CONTEXT_SNIPPET_CHARS]
        if term:
            terms.append({"term": term, "context": context})
    return terms


# ---------------------------------------------------------------------------
# Titan embedding
# ---------------------------------------------------------------------------
@retry(
    retry=retry_if_exception_type(TextractTransientError),
    wait=wait_exponential(multiplier=2, min=2, max=30),
    stop=stop_after_attempt(4),
    reraise=True,
)
def embed_term(bedrock_runtime, term: str, context: str) -> List[float]:
    """Embed a term together with its short context — matches the doc's
    Section 3 step ("EMBED candidate terms + short context").
    """
    text = f"{term}: {context}" if context else term
    payload = {"inputText": text or " ", "dimensions": EMBED_DIM, "normalize": True}
    try:
        response = bedrock_runtime.invoke_model(
            modelId=TITAN_EMBED_MODEL_ID,
            body=json.dumps(payload),
            contentType="application/json",
            accept="application/json",
        )
    except Exception as exc:
        if _is_transient(exc):
            raise TextractTransientError(str(exc)) from exc
        raise
    body = json.loads(response["body"].read())
    vector = body.get("embedding")
    if not vector:
        raise RuntimeError(f"Titan returned no embedding: {body}")
    return vector


def _cosine(vec_a: List[float], vec_b: List[float]) -> float:
    # Titan v2 vectors are L2-normalized, so cosine similarity == dot product.
    return sum(a * b for a, b in zip(vec_a, vec_b))


# ---------------------------------------------------------------------------
# Matching — auto-merge / no-match / judge
# ---------------------------------------------------------------------------
@dataclass
class MatchResult:
    decision: str          # "auto_merge" | "new_entry" | "judged_merge" | "judged_new"
    matched_term_id: Optional[str]
    similarity: Optional[float]


def find_best_match(
    term_vector: List[float], index_entries: Dict[str, Dict[str, Any]],
) -> Tuple[Optional[str], float]:
    """Return (best_matching_term_id, similarity) against existing index
    entries' stored vectors, or (None, 0.0) if the index is empty.
    """
    best_id, best_sim = None, 0.0
    for term_id, entry in index_entries.items():
        sim = _cosine(term_vector, entry["embedding"])
        if sim > best_sim:
            best_id, best_sim = term_id, sim
    return best_id, best_sim


_JUDGE_SYSTEM_PROMPT = (
    "You decide whether two terms from student loan/bankruptcy documents "
    "refer to the same underlying concept. Respond with JSON only."
)

_JUDGE_INSTRUCTIONS = """\
Term A: "{term_a}" (context: {context_a})
Term B: "{term_b}" (context: {context_b})

Do these refer to the SAME concept (e.g. "Auto Stay" and "Automatic Stay"
are the same; "Chapter 13 Plan" and "Chapter 7 Bankruptcy" are NOT the
same, despite being semantically close)?

Respond with: {{"same_concept": true}} or {{"same_concept": false}}
"""


def judge_same_concept(
    bedrock_runtime, term_a: str, context_a: str, term_b: str, context_b: str,
) -> bool:
    prompt = _JUDGE_INSTRUCTIONS.format(
        term_a=term_a, context_a=context_a or "(none)",
        term_b=term_b, context_b=context_b or "(none)",
    )
    raw = _call_nova(bedrock_runtime, _JUDGE_SYSTEM_PROMPT, prompt,
                     max_tokens=64, temperature=0.0)
    parsed = _parse_json_object(raw)
    return bool(parsed.get("same_concept", False))


def _parse_json_object(raw: str) -> Dict[str, Any]:
    text = raw.strip()
    if text.startswith("```"):
        text = re.sub(r"^```(?:json)?\s*", "", text)
        text = re.sub(r"```\s*$", "", text)
    start = text.find("{")
    end = text.rfind("}")
    if start == -1 or end == -1 or end <= start:
        return {}
    try:
        return json.loads(text[start:end + 1])
    except json.JSONDecodeError:
        return {}


def match_term_against_index(
    bedrock_runtime, term_vector: List[float], term: str, context: str,
    index_entries: Dict[str, Dict[str, Any]],
) -> MatchResult:
    """Full matching decision for one candidate term against the existing
    index — the auto_merge / no_match / judge branch from Section 3.
    """
    best_id, best_sim = find_best_match(term_vector, index_entries)

    if best_id is None or best_sim <= SIM_NO_MATCH:
        return MatchResult("new_entry", None, best_sim)

    if best_sim >= SIM_AUTO_MERGE:
        return MatchResult("auto_merge", best_id, best_sim)

    # Borderline — ask Nova to decide, rather than guess from similarity alone.
    existing = index_entries[best_id]
    same = judge_same_concept(
        bedrock_runtime, term, context,
        existing["term"], existing.get("sample_context", ""),
    )
    if same:
        return MatchResult("judged_merge", best_id, best_sim)
    return MatchResult("judged_new", None, best_sim)


# ---------------------------------------------------------------------------
# Page-location recovery — no LLM call, just string search against the
# document's already page-tagged content_sequence.
# ---------------------------------------------------------------------------
def find_term_pages(term: str, content_sequence: List[Dict[str, Any]]) -> List[int]:
    pages: List[int] = []
    needle = term.lower()
    for item in content_sequence:
        # Concatenate every text-bearing field rather than picking the
        # first that exists — a key_value item's match might be in its
        # VALUE even though it also has a (non-matching) KEY.
        haystack = " ".join(
            str(item.get(field, "")) for field in ("text", "key", "value")
        ).lower()
        if needle in haystack:
            page = item.get("page")
            if page is not None and page not in pages:
                pages.append(page)
    return pages


# ---------------------------------------------------------------------------
# Per-document processing + top-level orchestration
# ---------------------------------------------------------------------------
def process_document_for_index(
    bedrock_runtime, source_key: str, document_text: str,
    content_sequence: List[Dict[str, Any]], index_entries: Dict[str, Dict[str, Any]],
) -> Dict[str, Any]:
    """Extract terms from one document and merge them into index_entries
    (mutated in place). Returns a small per-document summary.
    """
    candidates = extract_candidate_terms(bedrock_runtime, document_text)
    new_count, merged_count = 0, 0

    for candidate in candidates:
        term, context = candidate["term"], candidate["context"]
        vector = embed_term(bedrock_runtime, term, context)
        result = match_term_against_index(
            bedrock_runtime, vector, term, context, index_entries,
        )
        pages = find_term_pages(term, content_sequence)
        source = {"document": source_key, "pages": pages}

        if result.decision in ("auto_merge", "judged_merge"):
            entry = index_entries[result.matched_term_id]
            if term not in entry["aliases"] and term != entry["term"]:
                entry["aliases"].append(term)
            entry["sources"].append(source)
            entry["occurrence_count"] += 1
            merged_count += 1
        else:
            term_id = re.sub(r"[^a-z0-9]+", "_", term.lower()).strip("_")
            index_entries[term_id] = {
                "term": term,
                "aliases": [],
                "sources": [source],
                "occurrence_count": 1,
                "embedding": vector,
                "sample_context": context,
            }
            new_count += 1

    log.info("%s: %d candidates -> %d new, %d merged",
             source_key, len(candidates), new_count, merged_count)
    return {"source_key": source_key, "candidates": len(candidates),
            "new_entries": new_count, "merged": merged_count}


def _to_public_index(index_entries: Dict[str, Dict[str, Any]]) -> Dict[str, Any]:
    """Strip embedding vectors for the human-readable public index."""
    return {
        term_id: {k: v for k, v in entry.items() if k not in ("embedding", "sample_context")}
        for term_id, entry in index_entries.items()
    }


def build_index_from_job_map(
    bucket: str, region: str = "us-east-1",
) -> Dict[str, Any]:
    """Walk every successfully-processed document (via job_map.json, same
    registry tree_builder.py uses) and build the term index across all of
    them, from scratch. For adding ONE new document to an already-existing
    index without reprocessing everything, use index_single_document()
    instead — this function always starts empty.
    """
    s3 = boto3.client("s3", region_name=region)
    bedrock_runtime = boto3.client("bedrock-runtime", region_name=region)
    job_map = load_job_map(s3, bucket)

    index_entries: Dict[str, Dict[str, Any]] = {}
    summaries: List[Dict[str, Any]] = []

    for entry in job_map.values():
        if entry.get("status") != "SUCCEEDED":
            continue
        output_key = entry.get("output_key")
        if not output_key:
            continue
        obj = s3.get_object(Bucket=bucket, Key=output_key)
        doc_output = json.loads(obj["Body"].read())

        source_key = doc_output.get("source_key", output_key)
        document_text = doc_output.get("document_markdown", "")
        content_sequence = doc_output.get("content_sequence", [])

        summary = process_document_for_index(
            bedrock_runtime, source_key, document_text, content_sequence, index_entries,
        )
        summaries.append(summary)

    return {"terms": _to_public_index(index_entries), "_document_summaries": summaries}


def index_single_document(
    bucket: str, output_key: str, region: str = "us-east-1",
) -> Dict[str, Any]:
    """Add ONE already-Textract-processed document to the index — the
    incremental path the design doc describes ("a new or edited document
    only triggers extraction + matching for itself"). Loads whatever index
    already exists (with embeddings intact, from the internal file) so new
    terms are correctly matched against everything indexed so far, not
    matched against nothing.

    output_key is the .json output path from process_document() /
    process_document_for_index() — i.e. result.output_key from
    textract_pipeline, or an entry's "output_key" field in job_map.json.
    """
    s3 = boto3.client("s3", region_name=region)
    bedrock_runtime = boto3.client("bedrock-runtime", region_name=region)

    try:
        index_entries = load_index_internal(bucket, region=region)
        log.info("loaded existing index: %d terms", len(index_entries))
    except s3.exceptions.NoSuchKey:
        index_entries = {}
        log.info("no existing index found — starting fresh")

    obj = s3.get_object(Bucket=bucket, Key=output_key)
    doc_output = json.loads(obj["Body"].read())
    source_key = doc_output.get("source_key", output_key)
    document_text = doc_output.get("document_markdown", "")
    content_sequence = doc_output.get("content_sequence", [])

    summary = process_document_for_index(
        bedrock_runtime, source_key, document_text, content_sequence, index_entries,
    )

    save_index_internal(bucket, index_entries, region=region)
    public_index = {"terms": _to_public_index(index_entries), "_last_document": summary}
    save_index(bucket, public_index, region=region)

    return public_index


def save_index(bucket: str, index: Dict[str, Any], region: str = "us-east-1",
              key: str = INDEX_KEY) -> str:
    """Save the PUBLIC index — clean, no embeddings, safe to read directly."""
    s3 = boto3.client("s3", region_name=region)
    s3.put_object(
        Bucket=bucket, Key=key,
        Body=json.dumps(index, indent=2, ensure_ascii=False).encode("utf-8"),
        ContentType="application/json",
    )
    return f"s3://{bucket}/{key}"

# ---------------------------------------------------------------------------
# Querying — two different tools for two different questions.
# ---------------------------------------------------------------------------
def query_index_by_name(index: Dict[str, Any], query: str) -> List[Dict[str, Any]]:
    """Free, instant lookup: case-insensitive substring match against each
    term's name and its aliases. No embedding/LLM call. Answers "does this
    exact term exist in the index" — NOT a natural-language question.
    Works against the PUBLIC index from load_index().
    """
    query_lower = query.lower().strip()
    matches = []
    for term_id, entry in index.get("terms", {}).items():
        names_to_check = [entry.get("term", "")] + entry.get("aliases", [])
        if any(query_lower in name.lower() for name in names_to_check if name):
            matches.append({"term_id": term_id, **entry})
    return matches


def query_index_semantic(
    bedrock_runtime, index_entries: Dict[str, Dict[str, Any]], question: str, top_k: int = 3,
) -> List[Dict[str, Any]]:
    """Natural-language lookup: embeds the question and returns the top_k
    closest terms by cosine similarity — answers a real question like "what
    happens when someone files bankruptcy", not just an exact term name.
    Needs the INTERNAL index (embeddings intact) from load_index_internal(),
    not the public one — this is the one real use for cosine at query time,
    separate from its role during index-building.
    """
    query_vector = embed_term(bedrock_runtime, question, "")
    scored = [
        (term_id, _cosine(query_vector, entry["embedding"]), entry)
        for term_id, entry in index_entries.items()
    ]
    scored.sort(key=lambda pair: pair[1], reverse=True)
    return [
        {"term_id": tid, "similarity": round(sim, 3), "term": entry["term"],
         "aliases": entry["aliases"], "sources": entry["sources"]}
        for tid, sim, entry in scored[:top_k]
    ]




# ---------------------------------------------------------------------------
# Validation — the deterministic checks. Precision/recall against ground
# truth needs a human-reviewed sample; this only catches structural issues.
# ---------------------------------------------------------------------------
def validate_index(index: Dict[str, Any]) -> List[str]:
    warnings: List[str] = []
    terms = index.get("terms", {})

    if not terms:
        warnings.append("Index has zero terms — extraction may have failed for every document.")
        return warnings

    seen_term_names: Dict[str, str] = {}
    for term_id, entry in terms.items():
        term_text = entry.get("term", "").strip()
        if not term_text:
            warnings.append(f"{term_id}: empty term text.")
            continue
        if len(term_text) <= 2:
            warnings.append(f"{term_id}: suspiciously short term '{term_text}' — "
                           f"possible extraction noise.")

        lower = term_text.lower()
        if lower in seen_term_names:
            warnings.append(f"Possible duplicate entries: '{term_text}' and "
                           f"'{seen_term_names[lower]}' — matching may have missed this pair.")
        seen_term_names[lower] = term_text

        if not entry.get("sources"):
            warnings.append(f"{term_id} ('{term_text}'): no sources recorded.")

        if entry.get("occurrence_count", 0) != len(entry.get("sources", [])):
            warnings.append(f"{term_id} ('{term_text}'): occurrence_count does not match "
                           f"the number of recorded sources.")

    return warnings

"""Textract-only ingestion pipeline: Layout + Tables extraction, ordered by
physical page position, written as three timestamped output files per
document (.json/.md/.txt). No Bedrock, Claude, or Nova model of any kind —
text, tables, and figure locations come exclusively from Amazon Textract.

Flow per document:
    start_analysis          -> submit to Textract (LAYOUT + TABLES), get job_id
    wait_for_job             -> poll until SUCCEEDED/FAILED, paginate all Blocks
    tables_to_markdown       -> TABLE/CELL blocks -> list of markdown tables
    find_figures             -> LAYOUT_FIGURE blocks -> page + location only,
                                 no captioning (Textract does not read images)
    build_sequential_content -> orders text/tables/figures by (Page, Top) —
                                 physical position on the page, not by an
                                 assumed block relationship
    process_document         -> writes {source}_{timestamp}.json/.md/.txt,
                                 all three sharing one timestamp, and updates
                                 the job tracking map

process_document() runs the full flow for one S3 PDF and returns a summary.
process_batch() runs it across every PDF under an S3 prefix.

class TextractTransientError(Exception):
    """Retryable Textract failure (throttling, transient service errors)."""


def _is_transient(exc: BaseException) -> bool:
    code = getattr(exc, "response", {}).get("Error", {}).get("Code", "")
    return code in (
        "ProvisionedThroughputExceededException",
        "ThrottlingException",
        "InternalServerError",
        "LimitExceededException",
    )


@dataclass
class DocumentResult:
    source_bucket: str
    source_key: str
    job_id: str
    status: str
    output_key: Optional[str] = None
    markdown_key: Optional[str] = None
    text_key: Optional[str] = None
    tables: List[Dict[str, Any]] = field(default_factory=list)
    figures: List[Dict[str, Any]] = field(default_factory=list)
    line_count: int = 0
    error: Optional[str] = None


# ---------------------------------------------------------------------------
# Textract job lifecycle
# ---------------------------------------------------------------------------
def _request_token(bucket: str, key: str, feature_types: List[str]) -> str:
    # Deterministic per (bucket, key, feature_types) so re-running a batch
    # with the SAME request doesn't start duplicate (billable) jobs — but a
    # genuinely different request (e.g. FeatureTypes changed) gets a new
    # token instead of colliding with an old one and raising
    # IdempotentParameterMismatchException.
    features_part = ",".join(sorted(feature_types))
    digest = hashlib.sha256(f"{bucket}/{key}/{features_part}".encode()).hexdigest()
    return digest[:64]


DEFAULT_FEATURE_TYPES = ["LAYOUT", "TABLES", "FORMS"]


@retry(
    retry=retry_if_exception_type(TextractTransientError),
    wait=wait_exponential(multiplier=2, min=2, max=60),
    stop=stop_after_attempt(5),
    reraise=True,
)
def start_analysis(textract, bucket: str, key: str) -> str:
    try:
        resp = textract.start_document_analysis(
            DocumentLocation={"S3Object": {"Bucket": bucket, "Name": key}},
            FeatureTypes=DEFAULT_FEATURE_TYPES,
            ClientRequestToken=_request_token(bucket, key, DEFAULT_FEATURE_TYPES),
        )
    except Exception as exc:
        if _is_transient(exc):
            raise TextractTransientError(str(exc)) from exc
        raise
    return resp["JobId"]


def wait_for_job(textract, job_id: str) -> Dict[str, Any]:
    started = time.time()
    while True:
        resp = textract.get_document_analysis(JobId=job_id)
        status = resp["JobStatus"]
        if status in ("SUCCEEDED", "FAILED"):
            break
        if time.time() - started > MAX_POLL_SECONDS:
            return {"JobStatus": "TIMEOUT", "Blocks": [], "StatusMessage": "Polling timed out"}
        time.sleep(POLL_SECONDS)

    if status != "SUCCEEDED":
        return resp

    blocks = list(resp.get("Blocks", []))
    next_token = resp.get("NextToken")
    while next_token:
        page = textract.get_document_analysis(JobId=job_id, NextToken=next_token)
        blocks.extend(page.get("Blocks", []))
        next_token = page.get("NextToken")

    resp["Blocks"] = blocks
    return resp


# ---------------------------------------------------------------------------
# Table reconstruction — TABLE/CELL blocks -> markdown grid
# ---------------------------------------------------------------------------
def tables_to_markdown(blocks: List[Dict[str, Any]]) -> List[Dict[str, Any]]:
    by_id = {b["Id"]: b for b in blocks}
    tables_out: List[Dict[str, Any]] = []

    for block in blocks:
        if block["BlockType"] != "TABLE":
            continue

        cell_ids = [
            cid
            for rel in block.get("Relationships", [])
            if rel["Type"] == "CHILD"
            for cid in rel["Ids"]
        ]
        cells = [by_id[cid] for cid in cell_ids if by_id.get(cid, {}).get("BlockType") == "CELL"]
        if not cells:
            continue

        max_row = max(c["RowIndex"] for c in cells)
        max_col = max(c["ColumnIndex"] for c in cells)
        grid = [["" for _ in range(max_col)] for _ in range(max_row)]

        for cell in cells:
            text = _cell_text(cell, by_id)
            r, c = cell["RowIndex"] - 1, cell["ColumnIndex"] - 1
            grid[r][c] = text

        tables_out.append({
            "table_id": block["Id"],
            "page": block.get("Page", 1),
            "rows": max_row,
            "cols": max_col,
            "markdown": _grid_to_markdown(grid),
        })

    return tables_out


def _cell_text(cell: Dict[str, Any], by_id: Dict[str, Dict[str, Any]]) -> str:
    word_ids = [
        wid
        for rel in cell.get("Relationships", [])
        if rel["Type"] == "CHILD"
        for wid in rel["Ids"]
    ]
    words = []
    for wid in word_ids:
        child = by_id.get(wid)
        if not child:
            continue
        if child["BlockType"] == "WORD":
            words.append(child.get("Text", ""))
        elif child["BlockType"] == "SELECTION_ELEMENT":
            words.append("[x]" if child.get("SelectionStatus") == "SELECTED" else "[ ]")
    return " ".join(words).strip()


def _grid_to_markdown(grid: List[List[str]]) -> str:
    if not grid:
        return ""
    lines = ["| " + " | ".join(grid[0]) + " |"]
    lines.append("|" + "|".join(["---"] * len(grid[0])) + "|")
    for row in grid[1:]:
        lines.append("| " + " | ".join(row) + " |")
    return "\n".join(lines)


# ---------------------------------------------------------------------------
# Forms — KEY_VALUE_SET blocks. AWS's documented schema: a KEY block has
# EntityTypes=["KEY"], a CHILD relationship to its own WORD blocks (the
# label text), and a VALUE relationship pointing at the paired VALUE
# block's Id. The VALUE block then has its own CHILD relationship to the
# WORD/SELECTION_ELEMENT blocks making up the actual value text.
# ---------------------------------------------------------------------------
def _clean_kv_key(key_text: str) -> str:
    """Strip a trailing colon Textract may have already included in the key
    text (e.g. source label was literally "Talking Points:") so rendering
    doesn't produce a double colon like "Talking Points::".
    """
    return key_text.rstrip(":").rstrip()


# A real form field's value is normally short — "$5.03", "M00351053". If
# FORMS groups a bold label with a long multi-sentence block of prose that
# follows it (e.g. "Talking Points:" + two bullet points merged into one
# run-on VALUE string), that's FORMS mis-triggering on prose, not a genuine
# label:value pair. We can't recover the ORIGINAL bullet boundaries — that
# structure is gone once Textract merges it into one VALUE block, before we
# ever see the response — but we can stop rendering it as an awkward
# "**Key:** <giant paragraph>" line and fall back to plain paragraph text,
# which at least reads naturally.
_KV_VALUE_MAX_LEN_FOR_FORM_FIELD = 150

# The mirror image of the above: a "key" this long, standing alone with NO
# matched value at all, is not really a label either — it's a full sentence
# Textract misread as a form field name (e.g. "Only the Chapter 13
# Trustee's Office may request a refund of PY T payments." as a "key" with
# an empty value). Left unhandled, this renders as a broken half-line:
# "**<the whole sentence>:**" with nothing after the colon.
_KV_KEY_MIN_LEN_FOR_STANDALONE_SENTENCE = 40


def _build_kv_sequence_item(page: int, key_text: str, value_text: str) -> Dict[str, Any]:
    clean_key = _clean_kv_key(key_text)
    if len(value_text) > _KV_VALUE_MAX_LEN_FOR_FORM_FIELD:
        return {"type": "paragraph", "page": page, "text": f"{clean_key}: {value_text}"}
    if not value_text.strip() and len(clean_key) > _KV_KEY_MIN_LEN_FOR_STANDALONE_SENTENCE:
        return {"type": "paragraph", "page": page, "text": clean_key}
    return {"type": "key_value", "page": page, "key": clean_key, "value": value_text}


def forms_to_keyvalues(blocks: List[Dict[str, Any]]) -> List[Dict[str, Any]]:
    by_id = {b["Id"]: b for b in blocks}
    pairs: List[Dict[str, Any]] = []

    for block in blocks:
        if block["BlockType"] != "KEY_VALUE_SET":
            continue
        if "KEY" not in (block.get("EntityTypes") or []):
            continue

        key_text = _cell_text(block, by_id)
        value_id = next(
            (rel["Ids"][0] for rel in block.get("Relationships", [])
             if rel["Type"] == "VALUE" and rel.get("Ids")),
            None,
        )
        value_block = by_id.get(value_id) if value_id else None
        value_text = _cell_text(value_block, by_id) if value_block else ""

        if key_text:
            pairs.append({
                "key_id": block["Id"],
                "page": block.get("Page", 1),
                "key": key_text,
                "value": value_text,
            })

    return pairs


# ---------------------------------------------------------------------------
# Plain text — titles, headers, paragraphs, lists, in reading order
# ---------------------------------------------------------------------------
_TEXT_LAYOUT_TYPES = {
    "LAYOUT_TITLE",
    "LAYOUT_HEADER",
    "LAYOUT_SECTION_HEADER",
    "LAYOUT_TEXT",
    "LAYOUT_LIST",
    "LAYOUT_FOOTER",
    "LAYOUT_PAGE_NUMBER",
}


def extract_text(blocks: List[Dict[str, Any]]) -> List[Dict[str, Any]]:
    """Reconstruct titles/headers/paragraphs/lists in reading order, per page.

    Textract's Layout feature already returns these blocks in the correct
    reading order (left-to-right, top-to-bottom, column-aware) — we just
    walk them in the order they appear and resolve each one's child LINE
    text. TABLE and LAYOUT_FIGURE content is intentionally excluded here;
    those are handled by tables_to_markdown() and caption_figures().
    """
    by_id = {b["Id"]: b for b in blocks}
    pages: Dict[int, List[Dict[str, str]]] = {}

    for block in blocks:
        if block["BlockType"] not in _TEXT_LAYOUT_TYPES:
            continue

        line_ids = [
            lid
            for rel in block.get("Relationships", [])
            if rel["Type"] == "CHILD"
            for lid in rel["Ids"]
        ]
        lines = [by_id[lid]["Text"] for lid in line_ids
                if by_id.get(lid, {}).get("BlockType") == "LINE"]
        if not lines:
            continue

        page_no = block.get("Page", 1)
        if block["BlockType"] == "LAYOUT_LIST":
            # Each LINE under a LAYOUT_LIST is one list item — keep them
            # separate instead of joining into a run-on sentence.
            for line_text in lines:
                if line_text.strip():
                    pages.setdefault(page_no, []).append({
                        "type": "LAYOUT_LIST", "text": line_text.strip(),
                    })
            continue

        text = " ".join(lines).strip()
        if not text:
            continue
        pages.setdefault(page_no, []).append({
            "type": block["BlockType"],
            "text": text,
        })

    return [
        {"page": page_no, "elements": elements}
        for page_no, elements in sorted(pages.items())
    ]


def text_to_markdown(pages: List[Dict[str, Any]]) -> str:
    """Flatten extract_text() output into one markdown string for the whole doc."""
    heading_types = {"LAYOUT_TITLE": "# ", "LAYOUT_HEADER": "## ",
                     "LAYOUT_SECTION_HEADER": "### "}
    lines = []
    for page in pages:
        for el in page["elements"]:
            prefix = heading_types.get(el["type"], "")
            if el["type"] == "LAYOUT_LIST":
                lines.append(f"- {el['text']}")
            else:
                lines.append(f"{prefix}{el['text']}")
        lines.append("")
    return "\n".join(lines).strip()


# ---------------------------------------------------------------------------
# Sequential ordering.
#
# AWS's own documentation states LAYOUT_* blocks are returned "in the implied
# reading order of the document... top to bottom, left to right" — meaning
# Textract ALREADY resolves multi-column ordering correctly at the LAYOUT
# block level. A previous version of this function re-sorted everything by
# (Page, Top) globally, which THREW THAT AWAY: Top-only sorting interleaves
# two side-by-side columns by vertical position instead of finishing column
# 1 before starting column 2 — a regression for genuine multi-column pages.
#
# Fix: trust Textract's native block order for LAYOUT_TITLE/SECTION_HEADER/
# TEXT/LIST/TABLE/FIGURE — walk `blocks` as given, don't re-sort.
#
# Remaining risk: some tables appear as a raw TABLE block with NO LAYOUT_TABLE
# wrapper at all (confirmed happens — see the test for this file). Native
# order alone would silently drop those, same bug as before. Fallback: after
# the native-order walk, find any TABLE block never matched to a LAYOUT_TABLE
# and insert it via (Page, Top) position instead of dropping it. This case is
# logged so we can tell how often it happens on real documents.
# ---------------------------------------------------------------------------
_SEQUENCE_TEXT_KIND = {
    "LAYOUT_TITLE": "title",
    "LAYOUT_SECTION_HEADER": "section_header",
    "LAYOUT_TEXT": "paragraph",
}


def _block_top(block: Dict[str, Any]) -> float:
    try:
        return block["Geometry"]["BoundingBox"]["Top"]
    except KeyError:
        return 0.0


def _bbox_overlap(box_a: Dict[str, float], box_b: Dict[str, float]) -> float:
    """Intersection-over-union of two Textract fractional bounding boxes."""
    ax0, ay0 = box_a["Left"], box_a["Top"]
    ax1, ay1 = ax0 + box_a["Width"], ay0 + box_a["Height"]
    bx0, by0 = box_b["Left"], box_b["Top"]
    bx1, by1 = bx0 + box_b["Width"], by0 + box_b["Height"]

    ix = max(0.0, min(ax1, bx1) - max(ax0, bx0))
    iy = max(0.0, min(ay1, by1) - max(ay0, by0))
    intersection = ix * iy
    union = box_a["Width"] * box_a["Height"] + box_b["Width"] * box_b["Height"] - intersection
    return intersection / union if union > 0 else 0.0


def _match_layout_block(
    layout_block: Dict[str, Any], candidate_blocks: List[Dict[str, Any]]
) -> Optional[str]:
    """Find the raw block (TABLE or KEY_VALUE_SET) that best overlaps a
    LAYOUT_* wrapper's bounding box on the same page — geometry-based,
    doesn't assume any specific relationship schema.
    """
    page = layout_block.get("Page", 1)
    box = layout_block["Geometry"]["BoundingBox"]
    candidates = [c for c in candidate_blocks if c.get("Page", 1) == page]
    if not candidates:
        return None
    best = max(candidates, key=lambda c: _bbox_overlap(box, c["Geometry"]["BoundingBox"]))
    return best["Id"] if _bbox_overlap(box, best["Geometry"]["BoundingBox"]) > 0 else None


def build_sequential_content(
    blocks: List[Dict[str, Any]],
    tables: List[Dict[str, Any]],
    form_fields: Optional[List[Dict[str, Any]]] = None,
) -> List[Dict[str, Any]]:
    """Order text, tables, form key/value fields, and figures using
    Textract's own native block order (correct per AWS docs, including
    multi-column) rather than re-deriving a global position sort. Tables
    and form fields lacking a LAYOUT_TABLE/LAYOUT_KEY_VALUE wrapper are
    still recovered via a position-based fallback so nothing silently
    disappears.
    """
    form_fields = form_fields or []
    by_id = {b["Id"]: b for b in blocks}
    tables_by_id = {t["table_id"]: t for t in tables}
    kv_by_id = {f["key_id"]: f for f in form_fields}
    table_blocks = [b for b in blocks if b["BlockType"] == "TABLE"]
    kv_blocks = [
        b for b in blocks
        if b["BlockType"] == "KEY_VALUE_SET" and "KEY" in (b.get("EntityTypes") or [])
    ]

    sequence: List[Dict[str, Any]] = []
    matched_table_ids: set = set()
    matched_kv_ids: set = set()

    for block in blocks:
        block_type = block["BlockType"]
        page = block.get("Page", 1)

        if block_type in _SEQUENCE_TEXT_KIND:
            text = _resolve_line_text(block, by_id)
            if text:
                sequence.append({"type": _SEQUENCE_TEXT_KIND[block_type],
                                 "page": page, "text": text})

        elif block_type == "LAYOUT_LIST":
            for line_id in _child_ids(block):
                child = by_id.get(line_id)
                if child and child["BlockType"] == "LINE" and child.get("Text", "").strip():
                    sequence.append({"type": "list_item", "page": page,
                                     "text": child["Text"].strip()})

        elif block_type == "LAYOUT_TABLE":
            table_id = _match_layout_block(block, table_blocks)
            table = tables_by_id.get(table_id)
            if table:
                matched_table_ids.add(table_id)
                sequence.append({"type": "table", "page": page, "table_id": table_id,
                                 "markdown": table["markdown"], "rows": table["rows"],
                                 "cols": table["cols"]})

        elif block_type == "LAYOUT_KEY_VALUE":
            key_id = _match_layout_block(block, kv_blocks)
            kv_field = kv_by_id.get(key_id)
            if kv_field:
                matched_kv_ids.add(key_id)
                sequence.append(_build_kv_sequence_item(page, kv_field["key"], kv_field["value"]))

        elif block_type == "LAYOUT_FIGURE":
            sequence.append({"type": "figure", "page": page, "figure_id": block["Id"]})

    # Fallback: any TABLE/KEY_VALUE_SET block never matched to a LAYOUT
    # wrapper would otherwise vanish entirely. Recover by (Page, Top)
    # position — best effort, not native-order-perfect, but never dropped.
    unmatched_tables = [t for t in table_blocks if t["Id"] not in matched_table_ids]
    unmatched_kv = [k for k in kv_blocks if k["Id"] not in matched_kv_ids]
    if unmatched_tables:
        log.warning("%d table(s) had no LAYOUT_TABLE wrapper; recovered by "
                   "position instead of native order.", len(unmatched_tables))
    if unmatched_kv:
        log.warning("%d form field(s) had no LAYOUT_KEY_VALUE wrapper; "
                   "recovered by position instead of native order.", len(unmatched_kv))

    for table_block in unmatched_tables:
        table = tables_by_id.get(table_block["Id"])
        if not table:
            continue
        fallback_item = {"type": "table", "page": table_block.get("Page", 1),
                         "table_id": table_block["Id"], "markdown": table["markdown"],
                         "rows": table["rows"], "cols": table["cols"]}
        insert_at = _fallback_insert_index(sequence, table_block.get("Page", 1))
        sequence.insert(insert_at, fallback_item)

    for kv_block in unmatched_kv:
        kv_field = kv_by_id.get(kv_block["Id"])
        if not kv_field:
            continue
        fallback_item = _build_kv_sequence_item(
            kv_block.get("Page", 1), kv_field["key"], kv_field["value"])
        insert_at = _fallback_insert_index(sequence, kv_block.get("Page", 1))
        sequence.insert(insert_at, fallback_item)

    return sequence


def _fallback_insert_index(sequence: List[Dict[str, Any]], page: int) -> int:
    """Best-effort position for an unmatched table: insert right after the
    last sequence item on the same page or earlier. Appends at the end if
    no such item exists (e.g. table is on the very first page/position).
    """
    last_qualifying = -1
    for i, item in enumerate(sequence):
        if item.get("page", 1) <= page:
            last_qualifying = i
    return last_qualifying + 1 if last_qualifying >= 0 else len(sequence)



def _child_ids(block: Dict[str, Any]) -> List[str]:
    return [
        cid
        for rel in block.get("Relationships", [])
        if rel["Type"] == "CHILD"
        for cid in rel["Ids"]
    ]


def _resolve_line_text(block: Dict[str, Any], by_id: Dict[str, Dict[str, Any]]) -> str:
    lines = [
        by_id[cid]["Text"]
        for cid in _child_ids(block)
        if by_id.get(cid, {}).get("BlockType") == "LINE"
    ]
    return " ".join(lines).strip()


def _render_figure_markdown(item: Dict[str, Any]) -> str:
    ocr_text = item.get("ocr_text")
    key_values = item.get("key_values") or []
    tables = item.get("tables") or []

    if not ocr_text and not key_values and not tables:
        return (f"**[Figure — page {item['page']}, no text detected by OCR; "
               f"may be a diagram, photo, or icon]**")

    parts = [f"**[Figure — page {item['page']}]**"]
    if item.get("needs_vision_review"):
        parts.append("*(short/fragmented OCR text — may be a diagram; "
                     "consider manual or vision review)*")
    for kv_field in key_values:
        parts.append(f"**{_clean_kv_key(kv_field['key'])}:** {kv_field['value']}")
    for table in tables:
        parts.append(table["markdown"])
    if not key_values and not tables and ocr_text:
        parts.append(ocr_text)
    return "\n".join(parts)


def _render_figure_plain(item: Dict[str, Any]) -> str:
    ocr_text = item.get("ocr_text")
    key_values = item.get("key_values") or []
    tables = item.get("tables") or []

    if not ocr_text and not key_values and not tables:
        return f"Figure (page {item['page']}) - no text detected by OCR."

    parts = [f"Figure (page {item['page']})"]
    if item.get("needs_vision_review"):
        parts.append("(short/fragmented OCR text - may be a diagram; consider review)")
    for kv_field in key_values:
        parts.append(f"{_clean_kv_key(kv_field['key'])}: {kv_field['value']}")
    for table in tables:
        parts.append(_markdown_table_to_plain(table["markdown"]))
    if not key_values and not tables and ocr_text:
        parts.append(ocr_text)
    return "\n".join(parts)


def _slugify(text: str, seen_slugs: Dict[str, int]) -> str:
    """GitHub-Flavored-Markdown style anchor slug: lowercase, spaces to
    hyphens, strip punctuation, de-duplicate with -1/-2/... suffixes so
    repeated heading text still gets unique, working anchor links.
    """
    slug = text.strip().lower()
    slug = re.sub(r"[^\w\s-]", "", slug)
    slug = re.sub(r"[\s]+", "-", slug).strip("-")
    if slug in seen_slugs:
        seen_slugs[slug] += 1
        return f"{slug}-{seen_slugs[slug]}"
    seen_slugs[slug] = 0
    return slug


def extract_toc_entries(sequence: List[Dict[str, Any]]) -> List[Dict[str, Any]]:
    """Pull just the title/section_header items out of a sequence, in
    order, as structured data (not rendered markdown). This is the shared
    heading data both build_table_of_contents() (per-document markdown ToC)
    and the corpus-level tree builder consume — extracted once here so
    both stay consistent.
    """
    return [
        {"type": item["type"], "text": item["text"], "page": item.get("page", 1)}
        for item in sequence
        if item["type"] in ("title", "section_header")
    ]


def build_table_of_contents(sequence: List[Dict[str, Any]]) -> str:
    """Build a markdown Table of Contents from a document's title/section
    header items, with GFM-compatible anchor links (# renders correctly on
    GitHub and most markdown viewers that auto-generate heading anchors).
    Titles are top-level bullets; section headers are indented one level.
    """
    entries = extract_toc_entries(sequence)
    if not entries:
        return ""

    seen_slugs: Dict[str, int] = {}
    lines: List[str] = ["## Table of Contents", ""]

    for entry in entries:
        slug = _slugify(entry["text"], seen_slugs)
        indent = "" if entry["type"] == "title" else "  "
        lines.append(f"{indent}- [{entry['text']}](#{slug})")

    return "\n".join(lines)


def build_table_of_contents_plain(sequence: List[Dict[str, Any]]) -> str:
    """Same heading outline as build_table_of_contents(), no markdown links
    — for the plain-text output, where anchors don't apply.
    """
    lines: List[str] = ["Table of Contents", ""]
    for item in sequence:
        if item["type"] not in ("title", "section_header"):
            continue
        indent = "" if item["type"] == "title" else "  "
        lines.append(f"{indent}- {item['text']}")

    if len(lines) == 2:
        return ""

    return "\n".join(lines)


def sequence_to_markdown(sequence: List[Dict[str, Any]]) -> str:
    """Flatten build_sequential_content() output into one markdown document,
    with tables and figure captions embedded at their correct reading-order
    position — not appended at the end.
    """
    heading_prefix = {"title": "# ", "section_header": "## "}
    lines: List[str] = []

    for item in sequence:
        kind = item["type"]
        if kind == "list_item":
            lines.append(f"- {item['text']}")
        elif kind in ("title", "section_header", "paragraph"):
            lines.append(f"{heading_prefix.get(kind, '')}{item['text']}")
        elif kind == "table":
            lines.append(item["markdown"])
        elif kind == "key_value":
            lines.append(f"**{_clean_kv_key(item['key'])}:** {item['value']}")
        elif kind == "figure":
            lines.append(_render_figure_markdown(item))
        lines.append("")

    return "\n".join(lines).strip()


def sequence_to_plain_text(sequence: List[Dict[str, Any]]) -> str:
    """Same ordering as sequence_to_markdown(), but with no markdown syntax
    (#, -, |). Tables render as simple space-padded columns; figures render
    as 'Figure: <caption>' with no bracket/bold decoration.
    """
    lines: List[str] = []

    for item in sequence:
        kind = item["type"]
        if kind == "list_item":
            lines.append(f"  - {item['text']}")
        elif kind in ("title", "section_header", "paragraph"):
            lines.append(item["text"])
        elif kind == "table":
            lines.append(_markdown_table_to_plain(item["markdown"]))
        elif kind == "key_value":
            lines.append(f"{_clean_kv_key(item['key'])}: {item['value']}")
        elif kind == "figure":
            lines.append(_render_figure_plain(item))
        lines.append("")

    return "\n".join(lines).strip()


def _markdown_table_to_plain(markdown_table: str) -> str:
    rows = [
        [cell.strip() for cell in line.strip("|").split("|")]
        for line in markdown_table.splitlines()
        if line.strip() and not set(line.replace("|", "").strip()) <= {"-"}
    ]
    if not rows:
        return ""
    col_widths = [max(len(row[i]) for row in rows) for i in range(len(rows[0]))]
    return "\n".join(
        "  ".join(cell.ljust(col_widths[i]) for i, cell in enumerate(row))
        for row in rows
    )


# ---------------------------------------------------------------------------
# Figures — Textract detects location only; it does not read image content.
# No captioning/vision model of any kind is used in this pipeline.
# ---------------------------------------------------------------------------
def find_figures(blocks: List[Dict[str, Any]]) -> List[Dict[str, Any]]:
    figures = []
    for block in blocks:
        if block["BlockType"] == "LAYOUT_FIGURE":
            box = block["Geometry"]["BoundingBox"]
            figures.append({
                "figure_id": block["Id"],
                "page": block.get("Page", 1),
                "bbox": box,
            })
    return figures


def crop_figure(pdf_bytes: bytes, page_no: int, bbox: Dict[str, float]) -> bytes:
    """Crop one figure region out of the source PDF page as a PNG image."""
    doc = fitz.open(stream=pdf_bytes, filetype="pdf")
    try:
        page = doc[page_no - 1]
        # bbox values are fractions of page size (Textract convention).
        # fitz clip rects are in PDF point space (page.rect), NOT the pixel
        # space of a rendered pixmap — scaling to pixels happens via dpi=.
        page_w, page_h = page.rect.width, page.rect.height
        x0 = bbox["Left"] * page_w
        y0 = bbox["Top"] * page_h
        x1 = x0 + bbox["Width"] * page_w
        y1 = y0 + bbox["Height"] * page_h
        clip = fitz.Rect(x0, y0, x1, y1)
        crop_pix = page.get_pixmap(dpi=FIGURE_RENDER_DPI, clip=clip)
        return crop_pix.tobytes("png")
    finally:
        doc.close()


@retry(
    retry=retry_if_exception_type(TextractTransientError),
    wait=wait_exponential(multiplier=2, min=2, max=30),
    stop=stop_after_attempt(4),
    reraise=True,
)
def ocr_figure_region(textract, image_bytes: bytes) -> Dict[str, Any]:
    """Run Textract AnalyzeDocument (synchronous, FORMS+TABLES) against one
    cropped figure image. Still pure Textract, no vision/LLM model — this
    covers figures that turn out to be embedded screenshots of forms or
    tables (e.g. a pasted account-detail screen), not just plain text.
    Falls back gracefully to plain line text when no forms/tables are found.
    """
    try:
        response = textract.analyze_document(
            Document={"Bytes": image_bytes}, FeatureTypes=["FORMS", "TABLES"],
        )
    except Exception as exc:
        if _is_transient(exc):
            raise TextractTransientError(str(exc)) from exc
        raise

    crop_blocks = response.get("Blocks", [])
    lines = [b["Text"] for b in crop_blocks if b["BlockType"] == "LINE"]
    word_count = sum(1 for b in crop_blocks if b["BlockType"] == "WORD")
    key_values = forms_to_keyvalues(crop_blocks)
    tables = tables_to_markdown(crop_blocks)

    return {
        "text": "\n".join(lines).strip(),
        "line_count": len(lines),
        "word_count": word_count,
        "key_values": key_values,
        "tables": tables,
    }


# Heuristic only — approximate signal for "this figure is probably a
# flowchart/diagram, not a text-heavy screenshot." OCR on a diagram tends to
# return few, short, scattered words (box labels, arrows) rather than
# coherent lines of prose. A figure with recovered key/value pairs or
# tables is treated as clearly NOT a diagram — forms/tables imply real
# structured content was found, not a plain graphic. Not precise; meant to
# flag candidates for manual or vision-model review, not to decide alone.
_DIAGRAM_MIN_WORD_COUNT = 8
_DIAGRAM_MAX_AVG_WORDS_PER_LINE = 2.0


def _looks_like_diagram(ocr_result: Dict[str, Any]) -> bool:
    if ocr_result.get("key_values") or ocr_result.get("tables"):
        return False
    word_count = ocr_result["word_count"]
    if word_count == 0:
        return True
    if word_count < _DIAGRAM_MIN_WORD_COUNT:
        return True
    avg_words_per_line = word_count / max(ocr_result["line_count"], 1)
    return avg_words_per_line < _DIAGRAM_MAX_AVG_WORDS_PER_LINE


def ocr_figures(
    textract, pdf_bytes: bytes, figures: List[Dict[str, Any]]
) -> List[Dict[str, Any]]:
    """Crop and analyze every detected figure (FORMS+TABLES+plain text).
    Adds ocr_text, recovered key_values/tables, and a needs_vision_review
    flag (heuristic — see _looks_like_diagram) to each. A figure whose
    crop/analysis fails is marked needs_vision_review=True rather than
    silently losing the flag for review.
    """
    enriched = []
    for fig in figures:
        try:
            crop_bytes = crop_figure(pdf_bytes, fig["page"], fig["bbox"])
            ocr_result = ocr_figure_region(textract, crop_bytes)
            enriched.append({
                **fig,
                "ocr_text": ocr_result["text"] or None,
                "ocr_word_count": ocr_result["word_count"],
                "key_values": ocr_result["key_values"],
                "tables": ocr_result["tables"],
                "needs_vision_review": _looks_like_diagram(ocr_result),
            })
        except Exception as exc:  # noqa: BLE001 — one bad figure shouldn't kill the batch
            log.warning("figure analysis failed (page %s, id %s): %s",
                       fig["page"], fig["figure_id"], exc)
            enriched.append({**fig, "ocr_text": None, "key_values": [], "tables": [],
                             "error": str(exc), "needs_vision_review": True})
    return enriched



def merge_figure_ocr(sequence: List[Dict[str, Any]], ocr_figures_list: List[Dict[str, Any]]) -> None:
    """Attach ocr_text/key_values/tables/needs_vision_review onto the figure
    items already in a built sequence, matched by figure_id. Mutates
    sequence in place. Kept as a separate step so build_sequential_content()
    stays independent of whether OCR enrichment happened at all.
    """
    ocr_by_id = {f["figure_id"]: f for f in ocr_figures_list}
    for item in sequence:
        if item["type"] != "figure":
            continue
        info = ocr_by_id.get(item["figure_id"])
        if info:
            item["ocr_text"] = info.get("ocr_text")
            item["key_values"] = info.get("key_values", [])
            item["tables"] = info.get("tables", [])
            item["needs_vision_review"] = info.get("needs_vision_review", True)


# ---------------------------------------------------------------------------
# Output naming + job tracking
# ---------------------------------------------------------------------------
def _utc_timestamp() -> str:
    return datetime.now(timezone.utc).strftime("%Y%m%d_%H%M%S")


def _safe_stem(source_key: str) -> str:
    stem = Path(source_key).stem
    return "".join(c if c.isalnum() or c in "-_" else "_" for c in stem)


def output_key_for(
    source_key: str, output_prefix: str, ext: str = ".json", timestamp: Optional[str] = None
) -> str:
    """Build an output S3 key: {prefix}/{source_stem}_{timestamp}{ext}.

    Pass the SAME `timestamp` value across json/md/txt calls for one
    document so all three output files share one identical timestamp,
    making them an obviously-matched set.
    """
    ts = timestamp or _utc_timestamp()
    return f"{output_prefix.rstrip('/')}/{_safe_stem(source_key)}_{ts}{ext}"


def load_job_map(s3, bucket: str, key: str = JOB_MAP_KEY) -> Dict[str, Any]:
    try:
        obj = s3.get_object(Bucket=bucket, Key=key)
        return json.loads(obj["Body"].read())
    except s3.exceptions.NoSuchKey:
        return {}
    except s3.exceptions.ClientError as exc:
        if exc.response["Error"]["Code"] in ("NoSuchKey", "404"):
            return {}
        raise


def save_job_map(s3, bucket: str, job_map: Dict[str, Any], key: str = JOB_MAP_KEY) -> None:
    s3.put_object(
        Bucket=bucket, Key=key,
        Body=json.dumps(job_map, indent=2, ensure_ascii=False).encode(),
        ContentType="application/json",
    )


# ---------------------------------------------------------------------------
# Orchestration
# ---------------------------------------------------------------------------
def process_document(
    source_bucket: str,
    source_key: str,
    output_bucket: str,
    output_prefix: str,
    region: str = "us-east-1",
) -> DocumentResult:
    textract = boto3.client("textract", region_name=region)
    s3 = boto3.client("s3", region_name=region)

    job_map = load_job_map(s3, output_bucket)
    request_token = _request_token(source_bucket, source_key, DEFAULT_FEATURE_TYPES)

    job_id = start_analysis(textract, source_bucket, source_key)
    log.info("started job %s for %s", job_id, source_key)

    result_raw = wait_for_job(textract, job_id)
    status = result_raw["JobStatus"]

    if status != "SUCCEEDED":
        error = result_raw.get("StatusMessage", f"job ended with status {status}")
        job_map[request_token] = {
            "source_key": source_key, "job_id": job_id, "status": status,
            "error": error, "processed_at": datetime.now(timezone.utc).isoformat(),
        }
        save_job_map(s3, output_bucket, job_map)
        return DocumentResult(source_bucket, source_key, job_id, status, error=error)

    blocks = result_raw["Blocks"]
    line_count = sum(1 for b in blocks if b["BlockType"] == "LINE")
    text_pages = extract_text(blocks)
    tables = tables_to_markdown(blocks)
    form_fields = forms_to_keyvalues(blocks)
    figures = find_figures(blocks)  # location only, before OCR enrichment

    sequence = build_sequential_content(blocks, tables, form_fields)

    if figures:
        pdf_obj = s3.get_object(Bucket=source_bucket, Key=source_key)
        pdf_bytes = pdf_obj["Body"].read()
        figures = ocr_figures(textract, pdf_bytes, figures)  # Textract OCR only
        merge_figure_ocr(sequence, figures)
        needing_review = sum(1 for f in figures if f.get("needs_vision_review"))
        if needing_review:
            log.info("%d/%d figures flagged as likely diagrams (short/"
                     "fragmented OCR) — may need manual or vision review.",
                     needing_review, len(figures))

    toc_markdown = build_table_of_contents(sequence)
    toc_plain = build_table_of_contents_plain(sequence)
    body_markdown = sequence_to_markdown(sequence)
    body_plain_text = sequence_to_plain_text(sequence)

    document_markdown = f"{toc_markdown}\n\n---\n\n{body_markdown}" if toc_markdown else body_markdown
    document_plain_text = f"{toc_plain}\n\n---\n\n{body_plain_text}" if toc_plain else body_plain_text

    # One shared timestamp -> json/md/txt for this document are an obvious
    # matched set, e.g. Phone_Guide_20260719_143012.{json,md,txt}
    timestamp = _utc_timestamp()
    json_key = output_key_for(source_key, output_prefix, ".json", timestamp)
    markdown_key = output_key_for(source_key, output_prefix, ".md", timestamp)
    text_key = output_key_for(source_key, output_prefix, ".txt", timestamp)

    output_body = {
        "source_bucket": source_bucket,
        "source_key": source_key,
        "job_id": job_id,
        "processed_at": datetime.now(timezone.utc).isoformat(),
        "line_count": line_count,
        "text": text_pages,
        "text_markdown": text_to_markdown(text_pages),
        "tables": tables,
        "form_fields": form_fields,
        "figures": figures,
        "content_sequence": sequence,
        "table_of_contents": toc_markdown,
        "document_markdown": document_markdown,
        "raw_block_count": len(blocks),
    }
    s3.put_object(
        Bucket=output_bucket, Key=json_key,
        Body=json.dumps(output_body, indent=2, ensure_ascii=False).encode(),
        ContentType="application/json",
    )
    s3.put_object(
        Bucket=output_bucket, Key=markdown_key,
        Body=document_markdown.encode("utf-8"),
        ContentType="text/markdown",
    )
    s3.put_object(
        Bucket=output_bucket, Key=text_key,
        Body=document_plain_text.encode("utf-8"),
        ContentType="text/plain",
    )

    job_map[request_token] = {
        "source_key": source_key, "job_id": job_id, "status": "SUCCEEDED",
        "output_key": json_key, "markdown_key": markdown_key, "text_key": text_key,
        "table_count": len(tables), "figure_count": len(figures),
        "processed_at": output_body["processed_at"],
    }
    save_job_map(s3, output_bucket, job_map)

    log.info("done: %s -> %s / %s / %s (%d tables, %d figures)",
             source_key, json_key, markdown_key, text_key, len(tables), len(figures))

    return DocumentResult(
        source_bucket, source_key, job_id, "SUCCEEDED",
        output_key=json_key, markdown_key=markdown_key, text_key=text_key,
        tables=tables, figures=figures, line_count=line_count,
    )


def process_batch(
    source_bucket: str,
    source_prefix: str,
    output_bucket: str,
    output_prefix: str,
    region: str = "us-east-1",
    skip_existing: bool = True,
) -> List[DocumentResult]:
    s3 = boto3.client("s3", region_name=region)
    job_map = load_job_map(s3, output_bucket) if skip_existing else {}

    paginator = s3.get_paginator("list_objects_v2")
    keys = [
        obj["Key"]
        for page in paginator.paginate(Bucket=source_bucket, Prefix=source_prefix)
        for obj in page.get("Contents", [])
        if obj["Key"].lower().endswith(".pdf")
    ]

    results = []
    for key in keys:
        token = _request_token(source_bucket, key, DEFAULT_FEATURE_TYPES)
        if skip_existing and job_map.get(token, {}).get("status") == "SUCCEEDED":
            log.info("skip (already processed): %s", key)
            continue
        try:
            results.append(process_document(source_bucket, key, output_bucket, output_prefix, region))
        except Exception as exc:  # noqa: BLE001 — one bad file shouldn't kill the batch
            log.exception("failed: %s", key)
            results.append(DocumentResult(source_bucket, key, "", "ERROR", error=str(exc)))

    return results


#########
"""Corpus-level Tree: aggregates per-document structure (already extracted
by textract_pipeline.py) into one navigable Root -> Document -> Section
tree, per the Unsupervised Labeling Utility doc's Section 4d.

No LLM calls anywhere in this module — the tree is fully derived from
Textract's own detected LAYOUT_TITLE/LAYOUT_SECTION_HEADER blocks, already
extracted and tested in textract_pipeline.py. This is the "primary
navigation layer" the doc's Section 2 describes; the Index (separate,
LLM-based) is its complement, closing the cross-branch blind spot.

Usage:
    from tree_builder import build_corpus_tree_from_job_map, validate_tree

    tree = build_corpus_tree_from_job_map(
        bucket="bankrupcy-data-bkt", region="us-east-1",
    )
    warnings = validate_tree(tree)
    for w in warnings:
        print("WARNING:", w)
"""
import json
import logging
from pathlib import Path
from typing import Any, Dict, List

import boto3

from textract_pipeline import extract_toc_entries, load_job_map

log = logging.getLogger(__name__)

TREE_KEY = "textract-jobs/corpus_tree.json"


def _document_display_title(source_key: str, toc_entries: List[Dict[str, Any]]) -> str:
    """Prefer the document's own detected title; fall back to the filename
    if no LAYOUT_TITLE was found (some documents may only have section
    headers, no top-level title).
    """
    for entry in toc_entries:
        if entry["type"] == "title":
            return entry["text"]
    return Path(source_key).stem


def build_document_tree_node(source_key: str, sequence: List[Dict[str, Any]]) -> Dict[str, Any]:
    """One document's node: its display title plus an ordered list of its
    section headers (page-tagged).
    """
    toc_entries = extract_toc_entries(sequence)
    sections = [
        {"heading": e["text"], "page": e["page"]}
        for e in toc_entries if e["type"] == "section_header"
    ]
    return {
        "document": source_key,
        "title": _document_display_title(source_key, toc_entries),
        "sections": sections,
    }


def build_corpus_tree(document_nodes: List[Dict[str, Any]]) -> Dict[str, Any]:
    """Wrap per-document nodes into the corpus-level root structure."""
    return {"root": document_nodes}


def build_corpus_tree_from_job_map(
    bucket: str, region: str = "us-east-1", job_map_key: str = None,
) -> Dict[str, Any]:
    """Read every successfully-processed document's JSON output (via the
    existing job_map.json) and build one combined corpus tree. No new
    Textract or LLM calls — this only reads outputs already produced by
    process_document()/process_batch().
    """
    s3 = boto3.client("s3", region_name=region)
    job_map = load_job_map(s3, bucket, job_map_key) if job_map_key else load_job_map(s3, bucket)

    document_nodes: List[Dict[str, Any]] = []
    skipped = 0
    for token, entry in job_map.items():
        if entry.get("status") != "SUCCEEDED":
            skipped += 1
            continue
        output_key = entry.get("output_key")
        if not output_key:
            skipped += 1
            continue
        try:
            obj = s3.get_object(Bucket=bucket, Key=output_key)
            doc_output = json.loads(obj["Body"].read())
        except Exception as exc:  # noqa: BLE001 — one bad doc shouldn't kill the batch
            log.warning("could not load output for token %s (%s): %s", token, output_key, exc)
            skipped += 1
            continue

        sequence = doc_output.get("content_sequence", [])
        document_nodes.append(
            build_document_tree_node(doc_output.get("source_key", output_key), sequence)
        )

    if skipped:
        log.info("skipped %d job_map entries (not succeeded or output missing)", skipped)
    log.info("built tree from %d documents", len(document_nodes))

    return build_corpus_tree(document_nodes)


def save_tree(bucket: str, tree: Dict[str, Any], region: str = "us-east-1",
             key: str = TREE_KEY) -> str:
    s3 = boto3.client("s3", region_name=region)
    s3.put_object(
        Bucket=bucket, Key=key,
        Body=json.dumps(tree, indent=2, ensure_ascii=False).encode("utf-8"),
        ContentType="application/json",
    )
    return f"s3://{bucket}/{key}"


def load_tree(bucket: str, region: str = "us-east-1", key: str = TREE_KEY) -> Dict[str, Any]:
    s3 = boto3.client("s3", region_name=region)
    obj = s3.get_object(Bucket=bucket, Key=key)
    return json.loads(obj["Body"].read())


# ---------------------------------------------------------------------------
# Validation — fully automated, no LLM needed since the tree is deterministic
# ---------------------------------------------------------------------------
def validate_tree(tree: Dict[str, Any]) -> List[str]:
    """Sanity-check the tree structure. Returns a list of human-readable
    warning strings; an empty list means no issues found. Does not raise —
    callers decide whether warnings are acceptable for their use case.
    """
    warnings: List[str] = []
    documents = tree.get("root", [])

    if not documents:
        warnings.append("Tree has zero documents — job_map may be empty or all failed.")
        return warnings

    seen_document_keys: set = set()
    for doc in documents:
        doc_key = doc.get("document", "<unknown>")

        if doc_key in seen_document_keys:
            warnings.append(f"Duplicate document entry in tree: {doc_key}")
        seen_document_keys.add(doc_key)

        sections = doc.get("sections", [])
        if not sections:
            warnings.append(f"{doc_key}: no sections detected at all (title-only or "
                           f"heading detection may have failed for this document).")
            continue

        seen_headings_this_doc: set = set()
        for section in sections:
            heading = section.get("heading", "").strip()
            if not heading:
                warnings.append(f"{doc_key}: section with empty/missing heading text "
                               f"(page {section.get('page', '?')}).")
                continue
            if heading in seen_headings_this_doc:
                warnings.append(f"{doc_key}: duplicate section heading '{heading}' "
                               f"appears more than once — possible mis-parse.")
            seen_headings_this_doc.add(heading)

    return warnings
