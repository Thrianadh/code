def extract_paragraphs(doc):
    """Extract all paragraphs from the document."""
    paragraphs = []
    for para in doc.paragraphs:
        paragraphs.append({
            "text": para.text,
            "style": para.style.name,
            "alignment": para.alignment,
            "bold": any(run.bold for run in para.runs),
            "italic": any(run.italic for run in para.runs),
        })
    return paragraphs

def extract_tables(doc):
    """Extract all tables from the document."""
    tables = []
    for table in doc.tables:
        table_data = []
        for row in table.rows:
            row_data = [cell.text for cell in row.cells]
            table_data.append(row_data)
        tables.append(table_data)
    return tables

def extract_headings(doc):
    """Extract headings (paragraphs with Heading styles)."""
    headings = []
    for para in doc.paragraphs:
        if para.style.name.startswith("Heading"):
            headings.append({
                "text": para.text,
                "level": int(para.style.name.split(" ")[-1]),  # e.g., "Heading 1" -> 1
            })
    return headings

def save_to_json(data, output_file="extracted_data.json"):
    """Save extracted data to a JSON file."""
    with open(output_file, "w", encoding="utf-8") as f:
        json.dump(data, f, indent=4, ensure_ascii=False)
    print(f"Data saved to {output_file}")

def extract_docx(file_path):
    """Main function to extract data from a .docx file."""
    doc = Document(file_path)
    data = {
        "metadata": {
            "file_name": os.path.basename(file_path),
            "num_paragraphs": len(doc.paragraphs),
            "num_tables": len(doc.tables),
        },
        "headings": extract_headings(doc),
        "paragraphs": extract_paragraphs(doc),
        "tables": extract_tables(doc),
    }
    return data

def load_docx_as_markdown(path: Path) -> str:
    """Read .docx and return markdown (mammoth). Skips broken images."""
    def _ignore_image(_image):
        return []
    with path.open("rb") as fh:
        result = mammoth.convert_to_markdown(
            fh, convert_image=mammoth.images.img_element(_ignore_image)
        )
    return result.value


def looks_like_heading(line: str) -> Optional[str]:
    """Detect __bold__ lines as pseudo-headings (OneNote copy-paste style)."""
    match = re.match(r'^\s*__([^_].{2,120}?)__\s*$', line)
    if not match:
        return None
    text = match.group(1).strip()
    if len(text) > 120 or text.endswith('.'):
        return None
    return text


def normalize_whitespace(text: str) -> str:
    """Collapse blank lines and strip trailing whitespace."""
    lines = [line.rstrip() for line in text.splitlines()]
    return re.sub(r'\n{3,}', '\n\n', '\n'.join(lines)).strip()


def strip_markdown_bold(text: str) -> str:
    """Remove __bold__ markers."""
    return re.sub(r'__([^_]+)__', r'\1', text)

class DocxDiagnosticAnalyzer:
    """Analyzes markdown from mammoth to find structure and issues."""

    def __init__(self, markdown_text: str, filename: str):
        self.text = normalize_whitespace(markdown_text)
        self.filename = filename
        self.lines = self.text.splitlines()
        self.sections: List[Dict] = []
        self.tables: List[str] = []
        self.issues: List[str] = []

    def analyze(self):
        """Run all analysis steps."""
        self._find_tables()
        self._find_sections()
        self._find_issues()

    def _find_tables(self):
        """Extract markdown tables (lines starting with |)."""
        in_table = False
        current_table = []

        for line in self.lines:
            stripped = line.strip()
            if stripped.startswith('|'):
                in_table = True
                current_table.append(stripped)
            elif in_table:
                # Table ended
                if len(current_table) >= 2:  # Need at least header + separator
                    self.tables.append('\n'.join(current_table))
                in_table = False
                current_table = []

        # Catch last table
        if in_table and len(current_table) >= 2:
            self.tables.append('\n'.join(current_table))

    def _find_sections(self):
        """Split markdown into sections by heading lines."""
        current = {
            'heading': 'Introduction',
            'heading_level': 0,
            'lines': [],
            'tables': [],
            'word_count': 0
        }
        table_idx = 0
        tables_in_section = []

        for line in self.lines:
            heading_text = looks_like_heading(line)

            if heading_text:
                # Save previous section
                current['word_count'] = len(' '.join(current['lines']).split())
                current['tables'] = tables_in_section
                self.sections.append(current)

                # Start new section
                current = {
                    'heading': heading_text,
                    'heading_level': 1,  # All __bold__ treated as H1
                    'lines': [],
                    'tables': [],
                    'word_count': 0
                }
                tables_in_section = []

            elif line.strip().startswith('|'):
                # Table line - associate with current section
                if table_idx < len(self.tables):
                    tables_in_section.append(table_idx)
                    table_idx += 1
            else:
                current['lines'].append(line)

        # Save last section
        current['word_count'] = len(' '.join(current['lines']).split())
        current['tables'] = tables_in_section
        self.sections.append(current)

    def _find_issues(self):
        """Detect common problems in the markdown."""
        # 1. Check for images (mammoth usually drops them, but check for placeholders)
        image_markers = ['[image]', '![]', 'Image:', 'Figure:']
        image_count = sum(1 for line in self.lines if any(m in line for m in image_markers))
        if image_count > 0:
            self.issues.append(
                f"🔴 Found {image_count} image reference(s). "
                f"Images were skipped by mammoth — content may be missing. "
                f"Consider manual transcription or OCR."
            )

        # 2. Check for tables
        if self.tables:
            self.issues.append(
                f"🟡 Found {len(self.tables)} table(s). "
                f"Verify row alignment in markdown output."
            )

            # Check for very wide tables (many columns)
            for i, table in enumerate(self.tables):
                first_line = table.split('\n')[0]
                col_count = first_line.count('|') - 1
                if col_count > 5:
                    self.issues.append(
                        f"🟡 Table {i+1} has {col_count} columns — "
                        f"may be hard for LLM to read. Consider simplifying."
                    )

        # 3. Check for very long sections
        long_sections = [s for s in self.sections if s['word_count'] > 1000]
        if long_sections:
            self.issues.append(
                f"🟡 {len(long_sections)} section(s) have >1000 words. "
                f"Consider splitting for better RAG retrieval."
            )

        # 4. Check for very short sections
        short_sections = [s for s in self.sections if 0 < s['word_count'] < 20]
        if short_sections:
            self.issues.append(
                f"🟡 {len(short_sections)} section(s) have <20 words. "
                f"These may be formatting artifacts or empty headings."
            )

        # 5. Check for no headings at all
        heading_sections = [s for s in self.sections if s['heading_level'] > 0]
        if len(heading_sections) <= 1:
            self.issues.append(
                f"🔴 Only {len(heading_sections)} heading(s) detected. "
                f"Document may lack structure. Check if bold text was preserved in OneNote transfer."
            )

        # 6. Check for potential multi-column (alternating short lines)
        short_lines = [l for l in self.lines if 10 < len(l.strip()) < 60]
        if len(short_lines) > 20:
            # Check for pairs of lines with similar length (side-by-side)
            pairs = 0
            for i in range(0, len(short_lines)-1, 2):
                if abs(len(short_lines[i]) - len(short_lines[i+1])) < 10:
                    pairs += 1
            if pairs > 5:
                self.issues.append(
                    "🟡 Potential multi-column layout detected (pairs of similar-length short lines). "
                    "Verify addresses and paired data are not interleaved."
                )

        # 7. Check for embedded document references
        doc_refs = ['.docx', '.pdf', '.xlsx', 'Template', 'Attachment']
        ref_count = sum(1 for line in self.lines if any(r in line for r in doc_refs))
        if ref_count > 0:
            self.issues.append(
                f"🟡 Found {ref_count} reference(s) to embedded documents/templates. "
                f"Content from these files is not included in the markdown."
            )

        # 8. Check for URLs
        urls = re.findall(r'https?://[^\s]+', self.text)
        if urls:
            self.issues.append(
                f"🟡 Found {len(urls)} URL(s). External content is not included: "
                f"{', '.join(urls[:3])}{'...' if len(urls) > 3 else ''}"
            )

        # 9. Check for numbered lists that might be procedures
        numbered = [l for l in self.lines if re.match(r'^\s*\d+[\.\)]\s+', l.strip())]
        if len(numbered) > 5:
            self.issues.append(
                f"🟡 Found {len(numbered)} numbered list items. "
                f"Ensure these are grouped under correct section headings."
            )

    def generate_report(self) -> str:
        """Generate human-readable diagnostic report."""
        lines = [
            "=" * 70,
            "DOCX DIAGNOSTIC REPORT (Markdown-based)",
            "=" * 70,
            "",
            f"File: {self.filename}",
            f"",
            f"STATISTICS",
            f"  Total Lines:        {len(self.lines)}",
            f"  Total Words:        {len(self.text.split())}",
            f"  Sections Found:     {len(self.sections)}",
            f"  Tables Found:       {len(self.tables)}",
            f"  Issues Found:       {len(self.issues)}",
            f"",
            f"SECTIONS BREAKDOWN",
        ]

        for i, sec in enumerate(self.sections):
            lines.append(
                f"  [{i+1}] {sec['heading'][:50]} | "
                f"{len(sec['lines'])} lines | {sec['word_count']} words | "
                f"{len(sec['tables'])} tables"
            )

        lines.extend([
            f"",
            f"TABLES BREAKDOWN",
        ])

        for i, table in enumerate(self.tables):
            first_line = table.split('\n')[0]
            col_count = first_line.count('|') - 1
            row_count = len([l for l in table.split('\n') if l.strip().startswith('|')])
            lines.append(f"  [T{i+1}] {row_count} rows x {col_count} cols")

        lines.extend([
            f"",
            f"ISSUES & RECOMMENDATIONS",
        ])

        if self.issues:
            for issue in self.issues:
                lines.append(f"  {issue}")
        else:
            lines.append("  ✅ No major issues detected.")

        lines.extend([
            f"",
            f"RECOMMENDATIONS FOR RAG INGESTION",
            f"  1. If images are missing: Manually transcribe critical screenshots or use OCR.",
            f"  2. If tables are complex: Verify markdown row alignment before ingestion.",
            f"  3. If sections are too long (>1000 words): Split into sub-sections.",
            f"  4. If no headings detected: Add __bold__ lines as section markers in the docx.",
            f"  5. If URLs are present: Decide whether to crawl them or keep as references.",
            f"",
            f"=" * 70,
            f"END OF REPORT",
            f"=" * 70,
        ])

        return '\n'.join(lines)

def main():
    parser = argparse.ArgumentParser(description="Simplified DOCX Diagnostic Extractor")
    parser.add_argument("--input", "-i", required=True, help="Path to .docx file")
    parser.add_argument("--output", "-o", default="./docx_diagnostic", help="Output directory")
    args = parser.parse_args()

    input_path = Path(args.input)
    output_dir = Path(args.output)
    output_dir.mkdir(parents=True, exist_ok=True)

    print(f"Loading: {input_path}")
    markdown = load_docx_as_markdown(input_path)
    print(f"  Markdown length: {len(markdown)} chars, {len(markdown.splitlines())} lines")

    print(f"Analyzing...")
    analyzer = DocxDiagnosticAnalyzer(markdown, input_path.name)
    analyzer.analyze()

    # Save outputs
    base = input_path.stem

    # 1. Report
    report_path = output_dir / f"{base}_report.txt"
    report_path.write_text(analyzer.generate_report(), encoding="utf-8")
    print(f"  ✓ Report: {report_path}")

    # 2. Sections JSON
    sections_path = output_dir / f"{base}_sections.json"
    sections_data = [
        {
            "id": i+1,
            "heading": s["heading"],
            "word_count": s["word_count"],
            "line_count": len(s["lines"]),
            "table_count": len(s["tables"]),
            "content": "\\n".join(s["lines"])[:500] + "..." if len("\\n".join(s["lines"])) > 500 else "\\n".join(s["lines"])
        }
        for i, s in enumerate(analyzer.sections)
    ]
    sections_path.write_text(json.dumps(sections_data, indent=2, ensure_ascii=False), encoding="utf-8")
    print(f"  ✓ Sections: {sections_path}")

    # 3. Tables markdown
    if analyzer.tables:
        tables_path = output_dir / f"{base}_tables.md"
        tables_md = "\n\n---\n\n".join(
            f"## Table {i+1}\n\n{t}" for i, t in enumerate(analyzer.tables)
        )
        tables_path.write_text(tables_md, encoding="utf-8")
        print(f"  ✓ Tables: {tables_path}")

    # 4. Raw markdown
    raw_path = output_dir / f"{base}_raw.md"
    raw_path.write_text(markdown, encoding="utf-8")
    print(f"  ✓ Raw markdown: {raw_path}")

    # Print summary
    print(f"\n{'='*70}")
    print(f"SUMMARY")
    print(f"{'='*70}")
    print(f"Sections: {len(analyzer.sections)}")
    print(f"Tables:   {len(analyzer.tables)}")
    print(f"Issues:   {len(analyzer.issues)}")
    print(f"\nTop issues:")
    for issue in analyzer.issues[:5]:
        print(f"  {issue}")
    print(f"\nSee {output_dir} for full details.")


if __name__ == "__main__":
    main()

class OneNoteDocxExtractor:
    """
    Production-grade extractor for OneNote-sourced DOCX files.
    Preserves reading order by iterating the document XML in sequence.
    """

    def __init__(
        self,
        docx_path: Union[str, Path],
        output_dir: Union[str, Path] = "./output",
        use_ocr: bool = True,
        ocr_engine: str = "pytesseract",
        ocr_lang: str = "eng",
        extract_images: bool = True,
        image_format: str = "png"
    ):
        self.docx_path = Path(docx_path).resolve()
        self.output_dir = Path(output_dir).resolve()
        self.output_dir.mkdir(parents=True, exist_ok=True)
        (self.output_dir / "images").mkdir(exist_ok=True)

        self.use_ocr = use_ocr
        self.ocr_engine = ocr_engine
        self.ocr_lang = ocr_lang
        self.extract_images = extract_images
        self.image_format = image_format

        self.doc = Document(str(self.docx_path))
        self.elements: List[ExtractedElement] = []
        self.reading_order = 0
        self.image_counter = 0
        self._image_hashes = set()  # Deduplication

        # OCR pipeline (lazy init)
        self._baidu_pipeline = None

        # Namespace map for XPath
        self._nsmap = {
            "w": "http://schemas.openxmlformats.org/wordprocessingml/2006/main",
            "wp": "http://schemas.openxmlformats.org/drawingml/2006/wordprocessingDrawing",
            "a": "http://schemas.openxmlformats.org/drawingml/2006/main",

    # ------------------------------------------------------------------

    # ------------------------------------------------------------------
    def extract_all(self) -> Dict[str, Any]:
        """Run full extraction pipeline."""
        metadata = self._extract_metadata()

    def save_json(self, data: Dict[str, Any], filename: str = "extracted_data.json") -> Path:
        """Persist structured output to JSON."""
        out_path = self.output_dir / filename
        with open(out_path, "w", encoding="utf-8") as f:
            json.dump(data, f, indent=2, ensure_ascii=False, default=str)
        return out_path

    # ------------------------------------------------------------------
    # Metadata
    # ------------------------------------------------------------------
    def _extract_metadata(self) -> Dict[str, Any]:
        cp = self.doc.core_properties
        return {
            "title": cp.title or "",
            "author": cp.author or "",
            "subject": cp.subject or "",
            "created": str(cp.created) if cp.created else None,
            "modified": str(cp.modified) if cp.modified else None,
            "paragraphs": len(self.doc.paragraphs),
            "tables": len(self.doc.tables),
            "sections": len(self.doc.sections),
            "column_layout_detected": self._detect_multi_column()
        }

    def _detect_multi_column(self) -> bool:
        """Check if any section uses multi-column layout."""
        for section in self.doc.sections:
            sectPr = section._sectPr
            cols = sectPr.find(qn("w:cols"))
            if cols is not None:
                num = cols.get(qn("w:num"))
                if num and int(num) > 1:
                    return True
        return False

    # ------------------------------------------------------------------
    # Body Processing (Reading Order Preservation)
    # ------------------------------------------------------------------
    def _process_body(self):
        """
        Iterate body children in XML document order.
        This guarantees reading order is preserved.
        """
        body = self.doc.element.body
        for child in body:
            tag = etree.QName(child).localname
            if tag == "p":
                self._process_paragraph(Paragraph(child, self.doc))
            elif tag == "tbl":
                self._process_table(Table(child, self.doc))
            # sectPr and other tags are ignored

    def _process_paragraph(self, para: Paragraph):
        """Handle text, inline images, and runs."""
        text = para.text.strip()
        style = para.style.name if para.style else "Normal"
        images = self._extract_images_from_paragraph(para)

        # If paragraph has both text and images, emit separate elements
        # to preserve reading order (images usually appear inline)
        if text and images:
            # Heuristic: if images are at start, emit them first
            # We emit text block then images (or vice versa) based on XML order
            self._emit_text_element(para, text, style)
            for img in images:
                self._add_element(ElementType.IMAGE, img)
        elif text:
            self._emit_text_element(para, text, style)
        elif images:
            for img in images:
                self._add_element(ElementType.IMAGE, img)

    def _emit_text_element(self, para: Paragraph, text: str, style: str):
        """Classify and emit text element."""
        elem_type = self._classify_text_element(para, style)
        content = {
            "text": text,
            "style": style,
            "alignment": str(para.alignment) if para.alignment else None,
            "runs": self._extract_runs(para),
            "is_bold": any(r.bold for r in para.runs),
            "is_italic": any(r.italic for r in para.runs),
            "is_underline": any(r.underline for r in para.runs),
        }
        self._add_element(elem_type, content)

    def _classify_text_element(self, para: Paragraph, style: str) -> ElementType:
        """Determine if paragraph is heading, list, or normal text."""
        style_lower = style.lower()
        if "heading" in style_lower or style_lower.startswith("title"):
            return ElementType.HEADING
        if self._is_list_paragraph(para):
            return ElementType.LIST_ITEM
        return ElementType.TEXT

    def _is_list_paragraph(self, para: Paragraph) -> bool:
        """Check for numbering / bullet properties."""
        pPr = para._p.find(qn("w:pPr"))
        if pPr is None:
            return False
        numPr = pPr.find(qn("w:numPr"))
        return numPr is not None

    def _extract_runs(self, para: Paragraph) -> List[Dict[str, Any]]:
        """Extract per-run formatting for fine-grained reconstruction."""
        runs = []
        for run in para.runs:
            font = run.font
            runs.append({
                "text": run.text,
                "bold": bool(run.bold),
                "italic": bool(run.italic),
                "underline": bool(run.underline),
                "font_name": font.name,
                "font_size_pt": round(font.size.pt, 1) if font.size else None,
                "color_rgb": str(font.color.rgb) if font.color and font.color.rgb else None,
            })
        return runs

    # ------------------------------------------------------------------
    # Image Extraction
    # ------------------------------------------------------------------
    def _extract_images_from_paragraph(self, para: Paragraph) -> List[Dict[str, Any]]:
        """Pull all inline images from paragraph runs."""
        images = []
        for run in para.runs:
            img = self._extract_image_from_run(run)
            if img:
                images.append(img)
        return images

    def _extract_image_from_run(self, run) -> Optional[Dict[str, Any]]:
        """Extract image binary, save to disk, optionally OCR."""
        try:
            # Find blip (binary large image payload)
            blip = run._element.find(".//a:blip", self._nsmap)
            if blip is None:
                # Fallback: try vml (older Word format)
                return self._extract_vml_image(run)

            embed = blip.get("{http://schemas.openxmlformats.org/officeDocument/2006/relationships}embed")
            if not embed:
                return None

            image_part = self.doc.part.related_parts.get(embed)
            if not image_part:
                return None

            image_bytes = image_part.blob
            content_type = image_part.content_type  # e.g., image/png
            ext = content_type.split("/")[-1]
            if ext in ("jpeg", "jpg"):
                ext = "jpg"
            elif ext not in ("png", "gif", "bmp", "tiff"):
                ext = self.image_format

            # Deduplication via hash
            img_hash = hashlib.md5(image_bytes).hexdigest()
            if img_hash in self._image_hashes:
                return None
            self._image_hashes.add(img_hash)

            self.image_counter += 1
            filename = f"img_{self.image_counter:04d}.{ext}"
            img_path = self.output_dir / "images" / filename
            img_path.write_bytes(image_bytes)

            # Dimensions (optional)
            width, height = self._get_image_dimensions(image_bytes)

            # OCR
            ocr_text = ""
            if self.use_ocr and PILLOW_AVAILABLE:
                ocr_text = self._perform_ocr(image_bytes)

            return {
                "filename": filename,
                "relative_path": str(Path("images") / filename),
                "absolute_path": str(img_path),
                "width_px": width,
                "height_px": height,
                "content_type": content_type,
                "size_bytes": len(image_bytes),
                "md5": img_hash,
                "ocr_text": ocr_text,
                "base64": base64.b64encode(image_bytes).decode("utf-8")
            }

        except Exception as e:
            return {
                "filename": f"img_error_{self.image_counter:04d}",
                "error": str(e),
                "ocr_text": ""
            }

    def _extract_vml_image(self, run) -> Optional[Dict[str, Any]]:
        """Handle legacy VML shapes (common in copied OneNote content)."""
        # VML extraction is complex; return placeholder for now
        shape = run._element.find(".//v:shape", self._nsmap)
        if shape is not None:
            return {
                "filename": "vml_shape",
                "note": "VML image detected but not extracted (legacy format)",
                "ocr_text": ""
            }
        return None

    def _get_image_dimensions(self, image_bytes: bytes) -> Tuple[Optional[int], Optional[int]]:
        """Get image dimensions using Pillow."""
        if not PILLOW_AVAILABLE:
            return None, None
        try:
            img = PILImage.open(io.BytesIO(image_bytes))
            return img.width, img.height
        except Exception:
            return None, None

    def _perform_ocr(self, image_bytes: bytes) -> str:
        """Run OCR on image bytes."""
        if not PILLOW_AVAILABLE:
            return "[Pillow not installed]"

        try:
            img = PILImage.open(io.BytesIO(image_bytes))

            if self.ocr_engine == "pytesseract":
                if not PYTESSERACT_AVAILABLE:
                    return "[pytesseract not installed]"
                return pytesseract.image_to_string(img, lang=self.ocr_lang).strip()

            elif self.ocr_engine == "baidu":
                return self._baidu_ocr(img)

            else:
                return f"[Unknown OCR engine: {self.ocr_engine}]"

        except Exception as e:
            return f"[OCR Error: {str(e)}]"

    # ------------------------------------------------------------------
    # Table Processing
    # ------------------------------------------------------------------
    def _process_table(self, table: Table):
        """
        Extract table content.
        Detects borderless tables that are likely used for multi-column layouts
        (common when OneNote content is pasted into Word).
        """
        is_borderless = self._is_borderless_table(table)
        is_nested = self._is_nested_table(table)

        rows_data = []
        for r_idx, row in enumerate(table.rows):
            row_cells = []
            for c_idx, cell in enumerate(row.cells):
                cell_content = self._extract_cell_content(cell)
                row_cells.append({
                    "row_index": r_idx,
                    "col_index": c_idx,
                    "text": cell.text.strip(),
                    "content_blocks": cell_content,
                    "is_merged": self._is_merged_cell(cell)
                })
            rows_data.append(row_cells)

        table_data = {
            "row_count": len(table.rows),
            "col_count": len(table.columns),
            "is_borderless": is_borderless,
            "is_nested": is_nested,
            "is_likely_column_layout": is_borderless and len(table.columns) >= 2,
            "rows": rows_data
        }

        elem_type = ElementType.COLUMN_LAYOUT if (is_borderless and len(table.columns) >= 2) else ElementType.TABLE
        self._add_element(elem_type, table_data)

    def _extract_cell_content(self, cell) -> List[Dict[str, Any]]:
        """Recursively extract paragraphs, images, and nested tables from a cell."""
        blocks = []
        for para in cell.paragraphs:
            text = para.text.strip()
            images = self._extract_images_from_paragraph(para)
            if text:
                blocks.append({
                    "type": "text",
                    "text": text,
                    "style": para.style.name if para.style else "Normal"
                })
            for img in images:
                blocks.append({"type": "image", "data": img})

        for nested_table in cell.tables:
            # Recursively process nested table
            nested_data = {
                "type": "nested_table",
                "row_count": len(nested_table.rows),
                "col_count": len(nested_table.columns),
                "rows": []
            }
            for row in nested_table.rows:
                nested_data["rows"].append([c.text.strip() for c in row.cells])
            blocks.append(nested_data)

        return blocks

    def _is_borderless_table(self, table: Table) -> bool:
        """Heuristic: check if table has no visible borders."""
        try:
            tblPr = table._element.find(qn("w:tblPr"))
            if tblPr is None:
                return True
            borders = tblPr.find(qn("w:tblBorders"))
            if borders is None:
                return True
            # If all border values are "none" or missing, consider borderless
            for border in borders:
                val = border.get(qn("w:val"))
                if val and val != "none":
                    return False
            return True
        except Exception:
            return False

    def _is_nested_table(self, table: Table) -> bool:
        """Check if table is inside another table."""
        parent = table._element.getparent()
        while parent is not None:
            if etree.QName(parent).localname == "tbl":
                return True
            parent = parent.getparent()
        return False

    def _is_merged_cell(self, cell) -> bool:
        """Detect merged cells via gridSpan or vMerge."""
        tcPr = cell._tc.find(qn("w:tcPr"))
        if tcPr is None:
            return False
        return (tcPr.find(qn("w:gridSpan")) is not None or
                tcPr.find(qn("w:vMerge")) is not None)

    # ------------------------------------------------------------------
    # Headers & Footers
    # ------------------------------------------------------------------
    def _process_headers_and_footers(self):
        for section in self.doc.sections:
            for header in [section.header, section.first_page_header, section.even_page_header]:
                if header and header.paragraphs:
                    self._process_header_footer(header, ElementType.HEADER)
            for footer in [section.footer, section.first_page_footer, section.even_page_footer]:
                if footer and footer.paragraphs:
                    self._process_header_footer(footer, ElementType.FOOTER)

    def _process_header_footer(self, part, elem_type: ElementType):
        for para in part.paragraphs:
            self._process_paragraph(para)
        for table in part.tables:
            self._process_table(table)

    # ------------------------------------------------------------------
    # Helpers
    # ------------------------------------------------------------------
    def _add_element(self, elem_type: ElementType, content: Any):
        self.reading_order += 1
        self.elements.append(ExtractedElement(
            element_type=elem_type.value,
            content=content,
            reading_order=self.reading_order
        ))

def build_argument_parser() -> argparse.ArgumentParser:
    parser = argparse.ArgumentParser(
        prog="onenote_docx_extractor.py",
        description="Extract structured data from OneNote-exported DOCX files"
    )
    parser.add_argument("docx", help="Path to input .docx file")
    parser.add_argument("-o", "--output", default="./output", help="Output directory")
    parser.add_argument("--ocr", choices=["pytesseract", "baidu", "none"], default="pytesseract",
                        help="OCR engine for images")
    parser.add_argument("--ocr-lang", default="eng", help="OCR language code (e.g., eng, chi_sim)")
    parser.add_argument("--no-images", action="store_true", help="Skip image extraction")
    parser.add_argument("--json-name", default="extracted_data.json", help="Output JSON filename")
    return parser


def main():
    parser = build_argument_parser()
    args = parser.parse_args()

    if not os.path.exists(args.docx):
        print(f"Error: File not found: {args.docx}", file=sys.stderr)
        sys.exit(1)

    extractor = OneNoteDocxExtractor(
        docx_path=args.docx,
        output_dir=args.output,
        use_ocr=(args.ocr != "none"),
        ocr_engine=args.ocr if args.ocr != "none" else "pytesseract",
        ocr_lang=args.ocr_lang,
        extract_images=not args.no_images
    )

    print(f"Extracting: {args.docx}")
    data = extractor.extract_all()
    json_path = extractor.save_json(data, filename=args.json_name)

    print(f"\nExtraction complete!")
    print(f"  JSON output : {json_path}")
    print(f"  Images dir  : {extractor.output_dir / 'images'}")
    print(f"  Elements    : {data['total_elements']}")
    print(f"  Tables      : {data['metadata']['tables']}")
    print(f"  Multi-col   : {data['metadata']['column_layout_detected']}")

    # Print quick summary of element types
    type_counts = defaultdict(int)
    for el in data["elements"]:
        type_counts[el["element_type"]] += 1
    print("\nElement breakdown:")
    for t, c in sorted(type_counts.items()):
        print(f"  {t:20s}: {c}")


if __name__ == "__main__":
    main()
