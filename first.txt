SYSTEM_PROMPT = """You are an ECMC customer service agent specializing in student loans, bankruptcy, and default resolution.

def build_prompt(system: str, chunks: List[str], history: str, question: str) -> str:
    """
    Build LLM prompt with system instruction, retrieved context, conversation history, and current question.

    Args:
        system: System instruction
        chunks: Retrieved document chunks
        history: Formatted conversation history string (or empty)
        question: Current user question

    Returns:
        str: Complete prompt for LLM
    """
    parts = [system]

    # Retrieved context
    parts.append("\n[RETRIEVED CONTEXT]")
    for i, chunk in enumerate(chunks, 1):
        parts.append(f"Source {i}: {chunk}")

    # Conversation history (if any)
    if history:
        parts.append(history)

    # Current question
    parts.append(f"\n[CURRENT QUESTION]\nUser: {question}\nAssistant:")

    return "\n\n".join(parts)

"""
session_memory.py — Sliding window conversation memory for ECMC RAG chatbot.

Simple in-memory dict storage. Lost on process restart.
For persistence across restarts, replace dict with Redis later.

Usage:
    from session_memory import create_session, get_history, add_exchange, format_history_for_prompt

    session_id = create_session()
    history = format_history_for_prompt(session_id)
    add_exchange(session_id, question, answer)
"""

import uuid
from collections import deque
from typing import List, Dict, Optional


# ═══════════════════════════════════════════════════════════════════════════════
# CONFIGURATION
# ═══════════════════════════════════════════════════════════════════════════════

MAX_MESSAGES = 10   # 5 Q&A pairs (user + assistant = 2 messages per turn)
MAX_PAIRS = 5       # Convenience: 5 turns of conversation


# ═══════════════════════════════════════════════════════════════════════════════
# GLOBAL SESSION STORE
# ═══════════════════════════════════════════════════════════════════════════════

# session_id -> deque of message dicts
# Each message: {"role": "user"|"assistant", "content": str}
_session_store: Dict[str, deque] = {}


# ═══════════════════════════════════════════════════════════════════════════════
# PUBLIC API
# ═══════════════════════════════════════════════════════════════════════════════

def create_session() -> str:
    """
    Generate a new session ID and initialize empty history.

    Returns:
        str: 8-character session ID (e.g., "a3f7b2d1")
    """
    session_id = str(uuid.uuid4())[:8]
    _session_store[session_id] = deque(maxlen=MAX_MESSAGES)
    return session_id


def get_history(session_id: str) -> List[Dict[str, str]]:
    """
    Retrieve conversation history for a session.

    Args:
        session_id: The session ID returned by create_session()

    Returns:
        List of message dicts: [{"role": "user", "content": "..."}, ...]
        Empty list if session not found.
    """
    if session_id not in _session_store:
        _session_store[session_id] = deque(maxlen=MAX_MESSAGES)
    return list(_session_store[session_id])


def add_exchange(session_id: str, question: str, answer: str) -> None:
    """
    Append a Q&A pair to session history.
    Automatically trims to MAX_MESSAGES via deque.

    Args:
        session_id: Session ID
        question: User question
        answer: Assistant answer
    """
    if session_id not in _session_store:
        _session_store[session_id] = deque(maxlen=MAX_MESSAGES)

    _session_store[session_id].append({"role": "user", "content": question})
    _session_store[session_id].append({"role": "assistant", "content": answer})


def get_history_length(session_id: str) -> int:
    """
    Return number of messages in session history.

    Args:
        session_id: Session ID

    Returns:
        int: Number of messages (0 if session not found)
    """
    if session_id not in _session_store:
        return 0
    return len(_session_store[session_id])


def get_history_pairs(session_id: str) -> int:
    """
    Return number of Q&A pairs in session history.

    Args:
        session_id: Session ID

    Returns:
        int: Number of complete pairs (messages // 2)
    """
    return get_history_length(session_id) // 2


def clear_session(session_id: str) -> None:
    """
    Remove a session from memory entirely.

    Args:
        session_id: Session ID to remove
    """
    _session_store.pop(session_id, None)


def format_history_for_prompt(session_id: str) -> str:
    """
    Format history as string for LLM prompt.
    Returns empty string if no history.

    Args:
        session_id: Session ID

    Returns:
        str: Formatted history block or empty string
    """
    history = get_history(session_id)
    if not history:
        return ""

    lines = ["[CONVERSATION HISTORY]"]
    for msg in history:
        role = "User" if msg["role"] == "user" else "Assistant"
        # Truncate long messages for prompt brevity
        content = msg["content"]
        if len(content) > 300:
            content = content[:300] + "..."
        lines.append(f"{role}: {content}")

    return "\n".join(lines)


def list_active_sessions() -> List[str]:
    """
    Return list of all active session IDs.
    Useful for debugging and monitoring.

    Returns:
        List[str]: All session IDs in memory
    """
    return list(_session_store.keys())


def get_session_stats() -> Dict[str, any]:
    """
    Return statistics about all sessions.
    Useful for debugging memory usage.

    Returns:
        Dict: {"total_sessions": int, "total_messages": int, "avg_messages_per_session": float}
    """
    total_sessions = len(_session_store)
    total_messages = sum(len(d) for d in _session_store.values())
    avg = total_messages / total_sessions if total_sessions > 0 else 0

    return {
        "total_sessions": total_sessions,
        "total_messages": total_messages,
        "avg_messages_per_session": round(avg, 2),
    }

class MarkdownFileScanner:
    """Scan local folders for markdown files."""

# SECTION 2: Content Extractor — Three-chunk extraction for large files
# ═══════════════════════════════════════════════════════════════════════════════

class ContentExtractor:
    """Extract content from markdown file using 3-chunk strategy for large files."""

    def extract_single(self, file_path, max_chars=1500):
        """Read first N chars from file (fallback for small files)."""
        try:
            with open(file_path, 'r', encoding='utf-8') as f:
                content = f.read(max_chars)
            return content.strip()
        except Exception as e:
            return f"[ERROR reading file: {str(e)}]"

    def extract_chunks(self, file_path, max_total=3500):
        """
        Extract 3 chunks from file: beginning, middle, end.

        For large files, hierarchy clues are usually at top,
        but sub-sections might be in middle or end.

        Returns:
            str: Formatted string with [BEGINNING], [MIDDLE], [END] sections
        """
        try:
            with open(file_path, 'r', encoding='utf-8') as f:
                content = f.read()

            total_len = len(content)

            # If file is small, just return it all
            if total_len <= max_total:
                return content.strip()

            # Calculate chunk sizes
            chunk_size = max_total // 3

            # Chunk 1: Beginning (usually has title, main headers)
            beginning = content[:chunk_size].strip()

            # Chunk 2: Middle (catches sub-sections)
            middle_start = total_len // 2 - chunk_size // 2
            middle = content[middle_start:middle_start + chunk_size].strip()

            # Chunk 3: End (usually has conclusion, footer, references)
            end = content[-chunk_size:].strip()

            # Combine with markers
            combined = f"""[BEGINNING - first {len(beginning)} chars]
{beginning}

[MIDDLE - sample from ~50% position, {len(middle)} chars]
{middle}

[END - last {len(end)} chars]
{end}"""

            return combined

        except Exception as e:
            return f"[ERROR reading file: {str(e)}]"

# SECTION 4: Nova Analyzer — Bedrock Converse API caller
# ═══════════════════════════════════════════════════════════════════════════════

class NovaAnalyzer:
    """Send analysis requests to Amazon Nova via Bedrock Converse API."""

    def __init__(self, model_id, region="us-east-1"):
        self.bedrock = boto3.client("bedrock-runtime", region_name=region)
        self.model_id = model_id

    def analyze(self, prompt, max_tokens=500, temperature=0.1):
        """Send prompt to Nova and parse JSON response."""
        try:
            response = self.bedrock.converse(
                modelId=self.model_id,
                messages=[
                    {
                        "role": "user",
                        "content": [{"text": prompt}]
                    }
                ],
                inferenceConfig={
                    "maxTokens": max_tokens,
                    "temperature": temperature,
                    "topP": 0.9,
                }
            )

            # Extract text response
            content = response["output"]["message"]["content"]
            text = content[0]["text"] if content else ""

            # Parse JSON
            result = self._extract_json(text)
            return result

        except Exception as e:
            return {
                "source": "Unknown",
                "title": "[ERROR]",
                "parent_path": [],
                "type": "page",
                "confidence": 0.0,
                "reasoning": f"Nova API error: {str(e)}",
                "error": True,
            }

    def _extract_json(self, text):
        """Extract and parse JSON from Nova response."""
        text = text.strip()

        # Remove markdown code blocks if present
        if text.startswith("```json"):
            text = text[7:]
        if text.startswith("```"):
            text = text[3:]
        if text.endswith("```"):
            text = text[:-3]

        text = text.strip()

        try:
            return json.loads(text)
        except json.JSONDecodeError:
            # Try regex fallback
            match = re.search(r'\{.*\}', text, re.DOTALL)
            if match:
                try:
                    return json.loads(match.group())
                except json.JSONDecodeError:
                    pass

            return {
                "source": "Unknown",
                "title": "[PARSE ERROR]",
                "parent_path": [],
                "type": "page",
                "confidence": 0.0,
                "reasoning": f"Failed to parse JSON. Raw: {text[:200]}",
                "error": True,
            }

# SECTION 5: Result Processor — Confidence filter + Deduplication
# ═══════════════════════════════════════════════════════════════════════════════

class ResultProcessor:
    """Filter by confidence, deduplicate, and sort results."""

    def __init__(self, auto_threshold=0.70, review_threshold=0.40):
        self.auto_threshold = auto_threshold
        self.review_threshold = review_threshold

    def classify(self, nova_result, file_path, source):
        """Classify Nova result into approved, review, or rejected."""
        confidence = nova_result.get("confidence", 0)

        entry = {
            "file_path": str(file_path),
            "filename": file_path.name,
            "source": nova_result.get("source", source),
            "title": nova_result.get("title", "Untitled"),
            "parent_path": nova_result.get("parent_path", []),
            "type": nova_result.get("type", "page"),
            "confidence": confidence,
            "reasoning": nova_result.get("reasoning", ""),
            "nova_raw": nova_result,
        }

        if confidence >= self.auto_threshold:
            return "approved", entry
        elif confidence >= self.review_threshold:
            return "review", entry
        else:
            return "rejected", entry

    def deduplicate(self, approved_list):
        """Remove duplicates based on (source + parent_path + title)."""
        seen = OrderedDict()
        unique = []

        for entry in approved_list:
            path_tuple = tuple(entry.get("parent_path", []))
            key = (entry["source"], path_tuple, entry["title"])

            if key not in seen:
                seen[key] = entry
                unique.append(entry)
            else:
                existing = seen[key]
                if "duplicate_files" not in existing:
                    existing["duplicate_files"] = []
                existing["duplicate_files"].append(entry["file_path"])

        return unique

    def sort(self, entries):
        """Sort by source, then path, then title."""
        return sorted(entries, key=lambda e: (
            e["source"],
            tuple(e.get("parent_path", [])),
            e["title"],
        ))

# SECTION 6: Tree Renderer — ASCII + JSON + Markdown outputs
# ═══════════════════════════════════════════════════════════════════════════════

class TreeRenderer:
    """Render TOC entries as ASCII tree, JSON, and Markdown."""

    def build_tree_dict(self, entries):
        """Build nested dictionary from flat entries."""
        tree = {"ROOT": {}}

        for entry in entries:
            source = entry["source"]
            parent_path = entry.get("parent_path", [])
            title = entry["title"]
            doc_type = entry.get("type", "page")

            branch = "OneNote" if source == "OneNote" else "Website (ecmc.org)"

            if branch not in tree["ROOT"]:
                tree["ROOT"][branch] = {}

            current = tree["ROOT"][branch]
            for level in parent_path:
                if level not in current:
                    current[level] = {}
                current = current[level]

            display_title = title
            if doc_type == "pdf_attachment":
                display_title = f"[PDF] {title}"

            current[display_title] = {
                "_file_path": entry.get("file_path", ""),
                "_type": doc_type,
                "_confidence": entry.get("confidence", 0),
            }

        return tree

    def render_ascii(self, tree):
        """Render nested dict as ASCII tree."""
        lines = ["ROOT"]
        root_children = tree.get("ROOT", {})

        items = list(root_children.items())
        for i, (key, value) in enumerate(items):
            is_last = (i == len(items) - 1)
            connector = "└─ " if is_last else "├─ "
            lines.append(connector + key)
            child_prefix = "   " if is_last else "│  "
            self._render_ascii_node(lines, value, child_prefix)

        return "\n".join(lines)

    def _render_ascii_node(self, lines, node, prefix):
        """Recursively render ASCII tree nodes."""
        children = []
        for k, v in node.items():
            if k.startswith("_"):
                continue
            is_branch = isinstance(v, dict) and any(not sk.startswith("_") for sk in v.keys())
            children.append((k, v, is_branch))

        for i, (key, value, is_branch) in enumerate(children):
            is_last = (i == len(children) - 1)
            connector = "└─ " if is_last else "├─ "
            lines.append(prefix + connector + key)

            if is_branch:
                new_prefix = prefix + ("   " if is_last else "│  ")
                self._render_ascii_node(lines, value, new_prefix)

    def render_json(self, tree):
        """Render as pretty-printed JSON."""
        return json.dumps(tree, indent=2, ensure_ascii=False)

    def render_markdown(self, entries):
        """Render as Markdown document."""
        lines = [
            "# Table of Contents — ECMC Knowledge Base",
            "",
            "Generated from OneNote and Website (ecmc.org) markdown files via Nova analysis.",
            "",
        ]

        by_source = {}
        for entry in entries:
            source = entry["source"]
            if source not in by_source:
                by_source[source] = []
            by_source[source].append(entry)

        for source, source_entries in by_source.items():
            branch_name = "OneNote" if source == "OneNote" else "Website (ecmc.org)"
            lines.append(f"## {branch_name}")
            lines.append("")

            source_entries.sort(key=lambda e: (tuple(e.get("parent_path", [])), e["title"]))

            for entry in source_entries:
                depth = len(entry.get("parent_path", [])) + 2
                hashes = "#" * min(depth, 6)
                title = entry["title"]
                if entry.get("type") == "pdf_attachment":
                    title = f"[PDF] {title}"

                lines.append(f"{hashes} {title}")
                lines.append(f"- **File:** `{entry.get('file_path', '')}`")
                lines.append(f"- **Confidence:** {entry.get('confidence', 0)}")
                lines.append(f"- **Reasoning:** {entry.get('reasoning', '')}")
                lines.append("")

        return "\n".join(lines)

# SECTION 7: Main Pipeline — Orchestrates everything
# ═══════════════════════════════════════════════════════════════════════════════

class TOCPipeline:
    """Main pipeline — Nova-based TOC builder with chunking."""

    def __init__(self, config):
        self.config = config
        self.scanner = MarkdownFileScanner(config["onenote_dir"], config["website_dir"])
        self.extractor = ContentExtractor()
        self.prompt_builder = NovaPromptBuilder()
        self.nova = NovaAnalyzer(config["nova_model_id"], config["bedrock_region"])
        self.processor = ResultProcessor(config["auto_accept_threshold"], config["review_threshold"])
        self.renderer = TreeRenderer()

    def run(self):
        """Execute the full Nova-based pipeline."""
        print("=" * 70)
        print("ECMC TOC BUILDER — Nova-Powered Markdown Analysis")
        print("=" * 70)
        print(f"Model: {self.config['nova_model_id']}")
        print(f"Region: {self.config['bedrock_region']}")
        print(f"Chunking: 3-chunk (beginning + middle + end)")
        print(f"Auto-accept threshold: {self.config['auto_accept_threshold']}")
        print(f"Review threshold: {self.config['review_threshold']}")

        all_files = []
        approved = []
        review = []
        rejected = []

        # Collect all files
        print("\n[Scan] Collecting markdown files...")
        for file_path, source in self.scanner.scan_onenote():
            all_files.append((file_path, source))
        for file_path, source in self.scanner.scan_website():
            all_files.append((file_path, source))

        if not all_files:
            print("\n❌ No markdown files found. Check your paths.")
            return

        print(f"\nTotal files to process: {len(all_files)}")

        # Process each file with Nova
        print("\n[Nova] Analyzing files...")
        for i, (file_path, source) in enumerate(all_files, 1):
            print(f"\n  [{i}/{len(all_files)}] {file_path.name}")
            print(f"      Source: {source}")

            # Extract content with chunking
            content = self.extractor.extract_chunks(file_path, self.config["content_max_chars"])
            print(f"      Content: {len(content)} chars (3-chunk)")

            # Build prompt
            prompt = self.prompt_builder.build_prompt(file_path, source, content)

            # Call Nova
            print(f"      🤖 Calling Nova...")
            result = self.nova.analyze(
                prompt,
                max_tokens=self.config["nova_max_tokens"],
                temperature=self.config["nova_temperature"],
            )

            if result.get("error"):
                print(f"      ❌ Nova error: {result.get('reasoning', 'Unknown')}")
                rejected.append({"pdf": file_path, "result": result})
                continue

            print(f"      ✅ Title: '{result.get('title')}', Confidence: {result.get('confidence')}")
            print(f"      📍 Path: {' > '.join(result.get('parent_path', []))}")

            # Classify
            status, entry = self.processor.classify(result, file_path, source)

            if status == "approved":
                print(f"      ✓ Auto-approved")
                approved.append(entry)
            elif status == "review":
                print(f"      ⚠️  Flagged for review")
                review.append(entry)
            else:
                print(f"      ❌ Rejected (confidence too low)")
                rejected.append(entry)

        # Deduplicate and sort approved
        print("\n[Process] Deduplicating approved entries...")
        approved = self.processor.deduplicate(approved)
        approved = self.processor.sort(approved)
        print(f"  {len(approved)} unique approved entries")
        print(f"  {len(review)} flagged for review")
        print(f"  {len(rejected)} rejected")

        # Build tree
        print("\n[Render] Building tree structures...")
        tree = self.renderer.build_tree_dict(approved)

        # Create output directory
        output_dir = Path(self.config["output_dir"])
        output_dir.mkdir(parents=True, exist_ok=True)

        # Save ASCII tree
        ascii_tree = self.renderer.render_ascii(tree)
        ascii_path = output_dir / "toc_tree.txt"
        with open(ascii_path, 'w', encoding='utf-8') as f:
            f.write(ascii_tree)
        print(f"  ✓ {ascii_path}")

        # Save JSON tree
        json_tree = self.renderer.render_json(tree)
        json_path = output_dir / "toc_tree.json"
        with open(json_path, 'w', encoding='utf-8') as f:
            f.write(json_tree)
        print(f"  ✓ {json_path}")

        # Save Markdown
        md_tree = self.renderer.render_markdown(approved)
        md_path = output_dir / "toc_tree.md"
        with open(md_path, 'w', encoding='utf-8') as f:
            f.write(md_tree)
        print(f"  ✓ {md_path}")

        # Save index
        index_path = Path(self.config["toc_index_file"])
        with open(index_path, 'w', encoding='utf-8') as f:
            json.dump({"approved": approved, "review": review, "rejected": rejected}, f, indent=2, ensure_ascii=False)
        print(f"  ✓ {index_path}")

        # Save review queue
        review_path = Path(self.config["review_queue_file"])
        with open(review_path, 'w', encoding='utf-8') as f:
            json.dump(review, f, indent=2, ensure_ascii=False)
        print(f"  ✓ {review_path}")

        # Summary
        print("\n" + "=" * 70)
        print("PIPELINE COMPLETE")
        print("=" * 70)
        print(f"  Total files:      {len(all_files)}")
        print(f"  Approved:         {len(approved)}")
        print(f"  Review queue:     {len(review)}")
        print(f"  Rejected:         {len(rejected)}")
        print(f"  Output directory: {self.config['output_dir']}")
        print("\n  Preview of ASCII tree:")
        print("-" * 70)
        print(ascii_tree[:2000])
        if len(ascii_tree) > 2000:
            print("...")

# SECTION 1: Markdown Header Extractor
# ═══════════════════════════════════════════════════════════════════════════════

class MarkdownHeaderExtractor:
    """Extract headers (#, ##, ###, ####) from markdown files."""

    HEADER_PATTERN = re.compile(r'^(#{1,6})\s+(.+)$', re.MULTILINE)

    def extract(self, file_path):
        """
        Extract all headers from a markdown file.

        Returns:
            list of dicts: [{"level": 1, "text": "Title", "line": 5}, ...]
        """
        headers = []
        try:
            with open(file_path, 'r', encoding='utf-8') as f:
                for line_num, line in enumerate(f, 1):
                    line = line.rstrip('\n')
                    match = self.HEADER_PATTERN.match(line)
                    if match:
                        hashes = match.group(1)
                        text = match.group(2).strip()
                        level = len(hashes)
                        headers.append({
                            "level": level,
                            "text": text,
                            "line": line_num,
                        })
        except Exception as e:
            print(f"    ⚠️  Error reading {file_path}: {e}")

        return headers

# SECTION 2: Path Builder from Headers
# ═══════════════════════════════════════════════════════════════════════════════

class PathBuilder:
    """Build tree path from extracted markdown headers."""

    def build_path(self, headers):
        """
        Build path from markdown headers.

        Expected structure:
            # Level 1 (Notebook / Category)
            ## Level 2 (Section Group / Subcategory)
            ### Level 3 (Section / Page)
            #### Level 4 (Page / Subtopic)

        Returns:
            {"parent_path": [...], "title": "leaf title", "depth": N, "all_headers": [...]}
        """
        if not headers:
            return {"parent_path": [], "title": "[No Headers]", "depth": 0, "all_headers": []}

        # Sort by line number to maintain order
        headers = sorted(headers, key=lambda h: h["line"])

        # Build path from all headers
        path_parts = []
        for h in headers:
            level = h["level"]
            text = h["text"]

            # Map level to path position (1-based index)
            idx = level - 1
            if idx >= len(path_parts):
                path_parts.append(text)
            else:
                path_parts[idx] = text

            # Trim any deeper levels that might exist from earlier headers
            path_parts = path_parts[:level]

        # Parent path = all except last
        # Leaf title = last header (deepest level)
        if len(path_parts) >= 2:
            parent_path = path_parts[:-1]
            title = path_parts[-1]
        elif len(path_parts) == 1:
            parent_path = []
            title = path_parts[0]
        else:
            parent_path = []
            title = "[Empty]"

        return {
            "parent_path": parent_path,
            "title": title,
            "depth": len(path_parts),
            "all_headers": [h["text"] for h in headers],
        }

# SECTION 3: PDF Attachment Detector
# ═══════════════════════════════════════════════════════════════════════════════

class PDFAttachmentDetector:
    """Detect PDF links referenced in markdown files."""

    PDF_LINK_PATTERN = re.compile(
        r'\[([^\]]+)\]\(([^)]+\.pdf)\)',
        re.IGNORECASE
    )
    PDF_URL_PATTERN = re.compile(
        r'(https?://[^\s]+\.pdf)',
        re.IGNORECASE
    )

    def detect(self, file_path):
        """
        Find all PDF references in a markdown file.

        Returns:
            list of dicts: [{"title": "PDF Title", "url": "..."}, ...]
        """
        attachments = []
        try:
            with open(file_path, 'r', encoding='utf-8') as f:
                content = f.read()

            # Match [Title](path/to/file.pdf)
            for match in self.PDF_LINK_PATTERN.finditer(content):
                title = match.group(1).strip()
                url = match.group(2).strip()
                attachments.append({
                    "title": title,
                    "url": url,
                    "type": "pdf_attachment",
                })

            # Match bare PDF URLs
            for match in self.PDF_URL_PATTERN.finditer(content):
                url = match.group(1).strip()
                # Extract filename as title
                title = os.path.basename(url).replace('.pdf', '').replace('-', ' ').replace('_', ' ').title()
                attachments.append({
                    "title": title,
                    "url": url,
                    "type": "pdf_attachment",
                })

        except Exception as e:
            print(f"    ⚠️  Error scanning PDF links in {file_path}: {e}")

        return attachments

# SECTION 5: Result Processor (Deduplication)

class ResultProcessor:
    """Process, deduplicate, and merge TOC entries."""

    def deduplicate(self, entries):
        """
        Remove duplicate entries based on (source + parent_path + title).
        If same title exists under different paths, both are kept.
        """
        seen = OrderedDict()
        unique = []

        for entry in entries:
            # Deduplication key: source + full path + title
            path_tuple = tuple(entry.get("parent_path", []))
            key = (entry["source"], path_tuple, entry["title"])

            if key not in seen:
                seen[key] = entry
                unique.append(entry)
            else:
                # Same entry from different file — merge file paths
                existing = seen[key]
                if "duplicate_files" not in existing:
                    existing["duplicate_files"] = []
                existing["duplicate_files"].append(entry.get("file_path", ""))

        return unique

    def sort_entries(self, entries):
        """Sort entries for consistent tree rendering."""
        return sorted(entries, key=lambda e: (
            e["source"],
            tuple(e.get("parent_path", [])),
            e["title"],
        ))

class TreeRenderer:
    """Render TOC entries as ASCII tree, JSON, and Markdown."""

    def build_tree_dict(self, entries):
        """
        Build nested dictionary from flat entries list.

        Structure:
        {
            "ROOT": {
                "OneNote": {
                    "Borrower Call Playbook": {
                        "Payments During Bankruptcy": {
                            "Setting Up Payments": {"_meta": {...}}
                        }
                    }
                },
                "Website (ecmc.org)": {...}
            }
        }
        """
        tree = {"ROOT": {}}

        for entry in entries:
            source = entry["source"]
            parent_path = entry.get("parent_path", [])
            title = entry["title"]

            # Determine branch name
            if source == "OneNote":
                branch = "OneNote"
            else:
                branch = "Website (ecmc.org)"

            # Initialize branch
            if branch not in tree["ROOT"]:
                tree["ROOT"][branch] = {}

            # Navigate/create path
            current = tree["ROOT"][branch]
            for level in parent_path:
                if level not in current:
                    current[level] = {}
                current = current[level]

            # Add leaf with metadata
            display_title = title
            if entry.get("type") == "pdf_attachment":
                display_title = f"[PDF] {title}"

            current[display_title] = {
                "_file_path": entry.get("file_path", ""),
                "_type": entry.get("type", "page"),
                "_headers": entry.get("all_headers", []),
            }

            # Add PDF attachments as children
            for attachment in entry.get("pdf_attachments", []):
                attach_title = f"[PDF] {attachment['title']}"
                current[attach_title] = {
                    "_url": attachment["url"],
                    "_type": "pdf_attachment",
                }

        return tree

    def render_ascii(self, tree):
        """Render nested dict as ASCII tree."""
        lines = ["ROOT"]
        root_children = tree.get("ROOT", {})

        items = list(root_children.items())
        for i, (key, value) in enumerate(items):
            is_last = (i == len(items) - 1)
            connector = "└─ " if is_last else "├─ "
            lines.append(connector + key)
            child_prefix = "   " if is_last else "│  "
            self._render_ascii_node(lines, value, child_prefix)

        return "\n".join(lines)

    def _render_ascii_node(self, lines, node, prefix):
        """Recursively render ASCII tree nodes."""
        # Separate branch children from leaf metadata
        children = []
        for k, v in node.items():
            if k.startswith("_"):
                continue
            # Check if v has non-metadata children (is a branch)
            is_branch = isinstance(v, dict) and any(not sk.startswith("_") for sk in v.keys())
            children.append((k, v, is_branch))

        for i, (key, value, is_branch) in enumerate(children):
            is_last = (i == len(children) - 1)
            connector = "└─ " if is_last else "├─ "
            lines.append(prefix + connector + key)

            if is_branch:
                new_prefix = prefix + ("   " if is_last else "│  ")
                self._render_ascii_node(lines, value, new_prefix)

    def render_json(self, tree):
        """Render as pretty-printed JSON."""
        return json.dumps(tree, indent=2, ensure_ascii=False)

    def render_markdown(self, entries):
        """Render as Markdown document with headers."""
        lines = [
            "# Table of Contents — ECMC Knowledge Base",
            "",
            "Generated from OneNote and Website (ecmc.org) markdown files.",
            "",
        ]

        # Group by source
        by_source = {}
        for entry in entries:
            source = entry["source"]
            if source not in by_source:
                by_source[source] = []
            by_source[source].append(entry)

        for source, source_entries in by_source.items():
            branch_name = "OneNote" if source == "OneNote" else "Website (ecmc.org)"
            lines.append(f"## {branch_name}")
            lines.append("")

            # Sort by path depth
            source_entries.sort(key=lambda e: (tuple(e.get("parent_path", [])), e["title"]))

            for entry in source_entries:
                depth = len(entry.get("parent_path", [])) + 2  # ## = level 0
                hashes = "#" * min(depth, 6)
                title = entry["title"]
                if entry.get("type") == "pdf_attachment":
                    title = f"[PDF] {title}"

                lines.append(f"{hashes} {title}")
                lines.append(f"- **File:** `{entry.get('file_path', '')}`")
                lines.append(f"- **Headers:** {', '.join(entry.get('all_headers', [])[:3])}")
                lines.append("")

                # Add PDF attachments
                for attachment in entry.get("pdf_attachments", []):
                    lines.append(f"- **[PDF]** {attachment['title']} — `{attachment['url']}`")

                if entry.get("pdf_attachments"):
                    lines.append("")

        return "\n".join(lines)

class TOCPipeline:
    """Main pipeline — orchestrates the entire TOC building process."""

    def __init__(self, config):
        self.config = config
        self.extractor = MarkdownHeaderExtractor()
        self.path_builder = PathBuilder()
        self.pdf_detector = PDFAttachmentDetector()
        self.file_iterator = MarkdownFileIterator(
            config["onenote_dir"],
            config["website_dir"],
        )
        self.processor = ResultProcessor()
        self.renderer = TreeRenderer()

    def run(self):
        """Execute the full pipeline."""
        print("=" * 70)
        print("ECMC TOC BUILDER — Local Markdown Parser")
        print("=" * 70)

        all_entries = []

        # Process OneNote files
        print("\n[OneNote] Scanning markdown files...")
        onenote_count = 0
        for file_path, source in self.file_iterator.iterate_onenote():
            onenote_count += 1
            print(f"  [{onenote_count}] {file_path.name}")

            headers = self.extractor.extract(file_path)
            path_info = self.path_builder.build_path(headers)
            pdf_attachments = self.pdf_detector.detect(file_path)

            entry = {
                "source": source,
                "file_path": str(file_path),
                "title": path_info["title"],
                "parent_path": path_info["parent_path"],
                "depth": path_info["depth"],
                "all_headers": path_info["all_headers"],
                "type": "page",
                "pdf_attachments": pdf_attachments,
            }
            all_entries.append(entry)
            path_str = " > ".join(path_info["parent_path"]) if path_info["parent_path"] else "[root]"
            print(f"      → {path_str} > {path_info['title']}")

        # Process Website files
        print("\n[Website] Scanning markdown files...")
        website_count = 0
        for file_path, source in self.file_iterator.iterate_website():
            website_count += 1
            print(f"  [{website_count}] {file_path.name}")

            headers = self.extractor.extract(file_path)
            path_info = self.path_builder.build_path(headers)
            pdf_attachments = self.pdf_detector.detect(file_path)

            entry = {
                "source": source,
                "file_path": str(file_path),
                "title": path_info["title"],
                "parent_path": path_info["parent_path"],
                "depth": path_info["depth"],
                "all_headers": path_info["all_headers"],
                "type": "page",
                "pdf_attachments": pdf_attachments,
            }
            all_entries.append(entry)
            path_str = " > ".join(path_info["parent_path"]) if path_info["parent_path"] else "[root]"
            print(f"      → {path_str} > {path_info['title']}")

        if not all_entries:
            print("\n❌ No markdown files found. Check your paths.")
            return

        # Deduplicate and sort
        print("\n[Process] Deduplicating entries...")
        all_entries = self.processor.deduplicate(all_entries)
        all_entries = self.processor.sort_entries(all_entries)
        print(f"  {len(all_entries)} unique entries")

        # Build tree
        print("\n[Render] Building tree structures...")
        tree = self.renderer.build_tree_dict(all_entries)

        # Create output directory
        output_dir = Path(self.config["output_dir"])
        output_dir.mkdir(parents=True, exist_ok=True)

        # ASCII tree
        ascii_tree = self.renderer.render_ascii(tree)
        ascii_path = output_dir / "toc_tree.txt"
        with open(ascii_path, 'w', encoding='utf-8') as f:
            f.write(ascii_tree)
        print(f"  ✓ {ascii_path}")

        # JSON tree
        json_tree = self.renderer.render_json(tree)
        json_path = output_dir / "toc_tree.json"
        with open(json_path, 'w', encoding='utf-8') as f:
            f.write(json_tree)
        print(f"  ✓ {json_path}")

        # Markdown
        md_tree = self.renderer.render_markdown(all_entries)
        md_path = output_dir / "toc_tree.md"
        with open(md_path, 'w', encoding='utf-8') as f:
            f.write(md_tree)
        print(f"  ✓ {md_path}")

        # Index JSON
        index_path = Path(self.config["toc_index_file"])
        with open(index_path, 'w', encoding='utf-8') as f:
            json.dump(all_entries, f, indent=2, ensure_ascii=False)
        print(f"  ✓ {index_path}")

        # Summary
        print("\n" + "=" * 70)
        print("PIPELINE COMPLETE")
        print("=" * 70)
        print(f"  OneNote files:  {onenote_count}")
        print(f"  Website files:  {website_count}")
        print(f"  Total entries:  {len(all_entries)}")
        print(f"  Output dir:     {self.config['output_dir']}")
        print("\n  Preview of ASCII tree:")
        print("-" * 70)
        print(ascii_tree[:2000])
        if len(ascii_tree) > 2000:
            print("...")

# FAISS Index Loader

def load_faiss_and_metadata(index_dir: str) -> Tuple[Any, List[Dict], Dict]:
    """
    Load FAISS index and metadata from local directory.
    Returns (index, metadata_list, index_info).
    """
    import faiss

    dir_path = Path(index_dir)
    if not dir_path.exists():
        raise FileNotFoundError(f"Index directory not found: {index_dir}")

    # Find latest files
    index_files = sorted(dir_path.glob("faiss_index_*.bin"), reverse=True)
    meta_files = sorted(dir_path.glob("chunk_metadata_*.json"), reverse=True)

    if not index_files:
        raise FileNotFoundError(f"No FAISS index files in {index_dir}")
    if not meta_files:
        raise FileNotFoundError(f"No metadata files in {index_dir}")

    index_path = index_files[0]
    meta_path = meta_files[0]

    log.info("Loading FAISS index: %s", index_path)
    index = faiss.read_index(str(index_path))

    log.info("Loading metadata: %s", meta_path)
    with open(meta_path, "r", encoding="utf-8") as f:
        meta_data = json.load(f)

    metadata = meta_data.get("metadata", [])
    index_info = {
        "total_vectors": meta_data.get("total_vectors", 0),
        "dimensions": meta_data.get("dimensions", 0),
        "index_type": meta_data.get("index_type", "unknown"),
        "metric_type": meta_data.get("metric_type", "unknown"),
        "index_file": str(index_path),
        "metadata_file": str(meta_path)
    }

    log.info("Loaded: %d vectors, %d dimensions", index_info["total_vectors"], index_info["dimensions"])
    return index, metadata, index_info

# Embedding (Single Text)
# ============================================================================
def embed_query(text: str, model_id: str = EMBED_MODEL_ID) -> np.ndarray:
    """Embed a single query text using Titan v2."""
    client = get_bedrock_client()

    body = {
        "inputText": text,
        "dimensions": 1024,
        "normalize": True
    }

    try:
        resp = client.invoke_model(
            modelId=model_id,
            body=json.dumps(body),
            contentType="application/json",
            accept="application/json"
        )
        result = json.loads(resp["body"].read())
        embedding = result.get("embedding", [])

        if not embedding:
            log.warning("Empty embedding for query")
            return np.array([], dtype=np.float32)

        return np.array(embedding, dtype=np.float32).reshape(1, -1)

    except Exception as exc:
        log.error("Embedding failed: %s", exc)
        return np.array([], dtype=np.float32)


# ============================================================================
# Vector Search
# ============================================================================
def search_faiss(index: Any, query_vector: np.ndarray, metadata: List[Dict], top_k: int = 5) -> List[Dict]:
    """
    Search FAISS index and return top-K chunks with metadata.
    """
    if query_vector.size == 0:
        log.error("Empty query vector")
        return []

    # Search
    distances, indices = index.search(query_vector, top_k)

    results = []
    for i, idx in enumerate(indices[0]):
        if idx < 0 or idx >= len(metadata):
            continue

        meta = metadata[idx]
        results.append({
            **meta,
            "_distance": float(distances[0][i]),
            "_rank": i + 1
        })

    return results

# Context Builder

def build_context(retrieved_chunks: List[Dict]) -> str:
    """Build structured context string from retrieved chunks."""
    parts = []

    for i, chunk in enumerate(retrieved_chunks, 1):
        chunk_type = chunk.get("chunk_type", "text")
        content = chunk.get("content", "")
        meta = chunk.get("metadata", {})

        part_lines = [f"[SOURCE {i}]"]
        part_lines.append(f"Document: {meta.get('source_document', 'unknown')}")
        part_lines.append(f"Section: {meta.get('parent_heading', chunk.get('chunk_id', 'unknown'))}")
        part_lines.append(f"Type: {chunk_type}")
        part_lines.append("")

        if chunk_type == "table":
            part_lines.append("[TABLE]")
            part_lines.append(content)
        elif chunk_type == "figure":
            part_lines.append("[FIGURE]")
            part_lines.append(content)
        elif chunk_type == "form":
            part_lines.append("[FORM]")
            part_lines.append(content)
        else:
            part_lines.append(content)

        parts.append("\n".join(part_lines))

    return "\n\n".join(parts)

# Answer Generation
# ============================================================================
def generate_answer(query: str, context: str, citations: List[Dict]) -> Tuple[str, str]:
    """Generate answer using Claude with retrieved context."""
    client = get_bedrock_client()

    prompt = (
        "You are an ECMC bankruptcy student loan document assistant. "
        "Answer the user\'s question using ONLY the provided context below. "
        "Be specific and accurate. Cite the source number [SOURCE N] when referencing information. "
        "If the context does not contain the answer, say so clearly.\n\n"
        f"{context}\n\n"
        f"[USER QUESTION]\n{query}\n\n"
        "[ANSWER]"
    )

    try:
        resp = client.converse(
            modelId=CLAUDE_MODEL_ID,
            messages=[{"role": "user", "content": [{"text": prompt}]}],
            inferenceConfig={"maxTokens": 2048, "temperature": 0.0}
        )
        answer = resp["output"]["message"]["content"][0]["text"]

        # Assess confidence
        confidence = assess_confidence(answer, citations)

        return answer, confidence

    except Exception as exc:
        log.error("Answer generation failed: %s", exc)
        return "Error generating answer.", "none"


def assess_confidence(answer: str, citations: List[Dict]) -> str:
    """Assess confidence based on answer quality and citations."""
    if not answer or len(answer) < 20:
        return "none"
    if "not found" in answer.lower() or "not contain" in answer.lower():
        return "low"
    if len(citations) >= 3:
        return "high"
    if len(citations) >= 1:
        return "medium"
    return "low"

    def to_dict(self) -> Dict[str, Any]:

            "query": self.query,
            "answer": self.answer,
            "confidence": self.confidence,
            "sources_used": self.sources_used,
            "citations": self.citations,
            "latency_ms": round(self.latency_ms, 2),
            "query_type": self.query_type
        }


def run_query(query: str, index: Any = None, metadata: List[Dict] = None, index_info: Dict = None) -> Dict[str, Any]:
    """
    Main entry point: process query and return answer.
    Loads index on first call if not provided.
    """
    start_time = time.time()

    # Load index if not provided
    if index is None or metadata is None:
        index, metadata, index_info = load_faiss_and_metadata(INDEX_DIR)

    log.info("Query: %s", query)

    # Step 1: Embed query
    embed_start = time.time()
    query_vector = embed_query(query)
    embed_time = (time.time() - embed_start) * 1000

    # Step 2: Search FAISS
    search_start = time.time()
    retrieved = search_faiss(index, query_vector, metadata, TOP_K)
    search_time = (time.time() - search_start) * 1000

    if not retrieved:
        total_time = (time.time() - start_time) * 1000
        return QueryResult(
            query=query,
            answer="No relevant information found in the documents.",
            confidence="none",
            sources_used=0,
            citations=[],
            latency_ms=total_time
        ).to_dict()

    # Step 3: Build context
    context = build_context(retrieved)

    # Step 4: Generate answer
    gen_start = time.time()
    answer, confidence = generate_answer(query, context, retrieved)
    gen_time = (time.time() - gen_start) * 1000

    total_time = (time.time() - start_time) * 1000

    # Build citations
    citations = []
    for chunk in retrieved:
        meta = chunk.get("metadata", {})
        citations.append({
            "chunk_id": chunk.get("chunk_id", ""),
            "chunk_type": chunk.get("chunk_type", ""),
            "document": meta.get("source_document", "unknown"),
            "section": meta.get("parent_heading", "unknown"),
            "distance": chunk.get("_distance", 0)
        })

    log.info("Answer generated in %.1f ms (embed: %.1f, search: %.1f, gen: %.1f)",
             total_time, embed_time, search_time, gen_time)

    return QueryResult(
        query=query,
        answer=answer,
        confidence=confidence,
        sources_used=len(retrieved),
        citations=citations,
        latency_ms=total_time
    ).to_dict()

    """Build FAISS index from embedding vectors."""
    try:
        import faiss
    except ImportError:

        raise

    n, d = vectors.shape
    log.info("Building FAISS index: %d vectors, %d dimensions", n, d)

    # Select metric

    else:

    # Build index
    if index_type == "Flat":
        if metric_type == "IP":

    # Add vectors
    log.info("Adding vectors to index...")
    if index_type == "IVF":
        index.train(vectors)
    index.add(vectors)

    log.info("Index built: %d vectors stored", index.ntotal)
    return index

        resp = client.invoke_model(
            modelId=model_id,
            body=json.dumps(body),
            contentType="application/json",
            accept="application/json"
        )
        result = json.loads(resp["body"].read())

        # Titan v2 response format
        embedding = result.get("embedding", [])

        if not embedding:
            log.warning("Empty embedding returned for text: %s...", text[:50])
            return []

        return embedding

    except Exception as exc:
        log.error("Embedding failed: %s", exc)
        log.error("Text preview: %s...", text[:100])
        return []


def embed_chunk(chunk: Dict[str, Any], model_id: str = EMBED_MODEL_ID) -> Dict[str, Any]:
    """
    Embed a single chunk and return enriched chunk with embedding.
    """
    content = chunk.get("content", "")
    chunk_type = chunk.get("chunk_type", "text")
    metadata = chunk.get("metadata", {})

    # Build rich embedding text
    parts = []
    if chunk_type == "table":
        parts.append(f"Table: {metadata.get('parent_heading', '')}")
        parts.append(content)
    elif chunk_type == "figure":
        parts.append(f"Figure: {content}")
    elif chunk_type == "form":
        parts.append(f"Form: {metadata.get('parent_heading', '')}")
        parts.append(content)
    else:
        parts.append(content)

    embedding_text = " ".join(parts)

    # Call embedding
    embedding = embed_single_text(embedding_text, model_id)

    return {
        "chunk_id": chunk["chunk_id"],
        "chunk_type": chunk_type,
        "content": content,
        "metadata": metadata,
        "token_count": chunk.get("token_count", 0),
        "embedding": embedding,
        "embedding_model": model_id,
        "embedding_dimensions": len(embedding) if embedding else 0
    }

# Main Orchestrator
# ============================================================================
def process_embeddings() -> Dict[str, Any]:
    log.info("=" * 60)
    log.info("Embeddings Pipeline (Titan v2 — Single Text)")
    log.info("Model: %s", EMBED_MODEL_ID)
    log.info("Max workers: %d", MAX_WORKERS)
    log.info("=" * 60)

    # Auto-discover chunk files
    chunk_keys = find_chunk_files(INPUT_BUCKET, CHUNKS_PREFIX)

    if not chunk_keys:
        log.error("No chunk files found anywhere in s3://%s/", INPUT_BUCKET)
        return {"status": "ERROR", "error": "No chunk files found"}

    log.info("Found %d potential chunk files", len(chunk_keys))

    # Try loading each until we find one with valid chunks
    chunk_data = None
    loaded_key = None

    for key in chunk_keys:
        try:
            log.info("Attempting to load: s3://%s/%s", INPUT_BUCKET, key)
            data = load_s3_json(INPUT_BUCKET, key)

            if "chunks" in data and isinstance(data["chunks"], list) and len(data["chunks"]) > 0:
                chunk_data = data
                loaded_key = key
                log.info("✓ Successfully loaded: %s (%d chunks)", key, len(data["chunks"]))
                break
            else:
                log.warning("  File has no valid chunks: %s", key)
        except Exception as exc:
            log.warning("  Failed to load %s: %s", key, exc)
            continue

    if not chunk_data:
        log.error("Could not load any valid chunk file!")
        return {"status": "ERROR", "error": "No valid chunk files could be loaded"}

    chunks = chunk_data["chunks"]
    total = len(chunks)
    log.info("Total chunks to embed: %d", total)

    # Process chunks in parallel using ThreadPoolExecutor
    embedded_chunks = []
    completed = 0
    failed = 0

    log.info("Starting parallel embedding with %d workers...", MAX_WORKERS)
    start_time = time.time()

    with ThreadPoolExecutor(max_workers=MAX_WORKERS) as executor:
        # Submit all tasks
        future_to_chunk = {
            executor.submit(embed_chunk, chunk): chunk 
            for chunk in chunks
        }

        # Collect results as they complete
        for future in as_completed(future_to_chunk):
            try:
                result = future.result()
                embedded_chunks.append(result)
                completed += 1

                if completed % 100 == 0:
                    elapsed = time.time() - start_time
                    rate = completed / elapsed if elapsed > 0 else 0
                    log.info("  Progress: %d/%d chunks (%.1f chunks/sec)", 
                             completed, total, rate)

            except Exception as exc:
                failed += 1
                log.error("Chunk embedding failed: %s", exc)

    total_time = time.time() - start_time
    log.info("Embedding complete: %d succeeded, %d failed in %.1f seconds", 
             completed, failed, total_time)

    # Build output
    output_data = {
        "pipeline": "embeddings",
        "processed_at": datetime.now(timezone.utc).isoformat(),
        "source_chunk_file": loaded_key,
        "embedding_model": EMBED_MODEL_ID,
        "total_chunks": len(embedded_chunks),
        "dimensions": 1024,
        "chunks": embedded_chunks
    }

    # Save to S3
    timestamp = datetime.now(timezone.utc).strftime("%Y%m%d_%H%M%S")
    output_key = f"{OUTPUT_PREFIX.rstrip('/')}/chunk_embeddings_{timestamp}.json"
    output_uri = save_s3_json(OUTPUT_BUCKET, output_key, output_data)

    log.info("=" * 60)
    log.info("COMPLETE: %d chunks embedded", len(embedded_chunks))
    log.info("Source: s3://%s/%s", INPUT_BUCKET, loaded_key)
    log.info("Output: %s", output_uri)
    log.info("Avg rate: %.1f chunks/sec", total / total_time if total_time > 0 else 0)
    log.info("=" * 60)

    return {
        "status": "SUCCEEDED",
        "output_uri": output_uri,
        "source_file": loaded_key,
        "total_chunks": len(embedded_chunks),
        "embedding_model": EMBED_MODEL_ID,
        "dimensions": 1024
    }

def find_chunk_files(bucket: str, prefix: str = "") -> List[str]:
    """
    Auto-discover chunk object files in S3.
    Searches for files containing 'chunk' and 'object' in name, ending with .json.
    """
    s3 = get_s3_client()
    paginator = s3.get_paginator("list_objects_v2")

    matching_keys = []

    # If prefix provided, search under it; otherwise search entire bucket
    search_prefix = prefix if prefix else ""

    log.info("Searching for chunk files under s3://%s/%s", bucket, search_prefix)

    for page in paginator.paginate(Bucket=bucket, Prefix=search_prefix):
        for obj in page.get("Contents", []):
            k = obj["Key"]
            # Match files that look like chunk objects
            filename = Path(k).name.lower()
            if k.endswith(".json") and ("chunk" in filename or "object" in filename):
                # Exclude embedding outputs and other pipeline files
                if "embedding" not in filename and "vector" not in filename:
                    matching_keys.append(k)
                    log.info("  Found: %s", k)

    # Sort by last modified (newest first) if possible
    # For now, sort by key name which includes timestamp
    matching_keys.sort(reverse=True)

    return matching_keys


def load_s3_json(bucket: str, key: str) -> Dict[str, Any]:
    s3 = get_s3_client()
    return json.loads(s3.get_object(Bucket=bucket, Key=key)["Body"].read())


def save_s3_json(bucket: str, key: str, data: Dict[str, Any]) -> str:
    s3 = get_s3_client()
    body = json.dumps(data, indent=2, ensure_ascii=False).encode("utf-8")
    s3.put_object(Bucket=bucket, Key=key, Body=body, ContentType="application/json")
    return f"s3://{bucket}/{key}"

def embed_texts(texts: List[str], model_id: str = EMBED_MODEL_ID) -> List[List[float]]:

    Returns list of embedding vectors.
    """
    client = get_bedrock_client()

    body = {
        "inputText": texts,
        "dimensions": 1024,
        "normalize": True
    }

    try:
        resp = client.invoke_model(

def bedrock_converse(model_id: str, prompt: str, image_bytes: Optional[bytes] = None, max_tokens: int = 1024) -> str:
    client = get_bedrock_client()
    
    # Build content list
    content = []
    
    # Add text
    if prompt:
        content.append({"text": prompt})
    
    # Add image - Nova expects image as bytes with proper format
    if image_bytes:
        import base64
        
        # Encode image to base64
        b64_image = base64.b64encode(image_bytes).decode("utf-8")
        
        # For Nova models, image must be in specific format
        # The 'bytes' field should contain the raw base64 string
        content.append({
            "image": {
                "format": "png",  # or "jpeg" depending on your image
                "source": {
                    "bytes": b64_image
                }
            }
        })
    
    try:
        resp = client.converse(
            modelId=model_id,
            messages=[{
                "role": "user",
                "content": content
            }],
            inferenceConfig={
                "maxTokens": max_tokens,
                "temperature": 0.0
            }
        )
        return resp["output"]["message"]["content"][0]["text"]
    except Exception as exc:
        log.error("Bedrock error: %s", exc)
        return ""

Chunking Pipeline — Vector RAG Step 2

Converts Textract JSON output into structured chunks for embedding.

Input:
    - Textract .json files (from Pipeline 1)
    - Scrapy Markdown files (4 files)

Output:
    - chunk_objects.json: List of chunks with metadata

Chunk types:
    - text: 512 tokens, 10% overlap
    - table: full table <= 1024 tokens (Strategy A)
    - figure: Nova description (only if NOVA_ENABLED=true)
    - form: key-value pairs

Design decisions (locked):
    - Text chunk size: 512 tokens
    - Text overlap: 10% (50 tokens)
    - Table chunk max: 1024 tokens
    - Figures: only if NOVA_ENABLED=true
    - Sequence preserved from Textract block order

def list_json_keys(bucket: str, prefix: str) -> List[str]:
    """List all .json keys under S3 prefix."""
    s3 = get_s3_client()
    paginator = s3.get_paginator("list_objects_v2")
    keys = []
    for page in paginator.paginate(Bucket=bucket, Prefix=prefix):
        for obj in page.get("Contents", []):
            k = obj["Key"]
            if k.endswith(".json") and not k.endswith("_tables.json"):
                keys.append(k)
    return keys


def list_md_keys(bucket: str, prefix: str) -> List[str]:
    s3 = get_s3_client()
    paginator = s3.get_paginator("list_objects_v2")
    keys = []
    for page in paginator.paginate(Bucket=bucket, Prefix=prefix):
        for obj in page.get("Contents", []):
            k = obj["Key"]
            if k.lower().endswith((".md", ".markdown")):
                keys.append(k)
    return keys


def load_s3_json(bucket: str, key: str) -> Dict[str, Any]:
    s3 = get_s3_client()

    body = json.dumps(data, indent=2, ensure_ascii=False).encode("utf-8")
    s3.put_object(Bucket=bucket, Key=key, Body=body, ContentType="application/json")
    return f"s3://{bucket}/{key}"

def estimate_tokens(text: str) -> int:
    """Rough token estimate: ~4 chars per token for English."""
    return len(text) // 4

def index_blocks(blocks: List[Dict[str, Any]]) -> Dict[str, Dict[str, Any]]:
    return {b["Id"]: b for b in blocks}


def extract_cell_text(cell: Dict[str, Any], by_id: Dict[str, Dict[str, Any]]) -> str:
    words = []
    for rel in cell.get("Relationships", []):
        if rel["Type"] == "CHILD":
            for wid in rel["Ids"]:
                child = by_id.get(wid)
                if not child:
                    continue
                if child["BlockType"] == "WORD":
                    words.append(child.get("Text", ""))
                elif child["BlockType"] == "SELECTION_ELEMENT":
                    words.append("[x]" if child.get("SelectionStatus") == "SELECTED" else "[ ]")
    return " ".join(words).strip()


def grid_to_markdown(grid: List[List[str]]) -> str:
    if not grid:
        return ""
    lines = ["| " + " | ".join(grid[0]) + " |"]
    lines.append("|" + "|".join([" --- "] * len(grid[0])) + "|")
    for row in grid[1:]:
        lines.append("| " + " | ".join(row) + " |")
    return "\n".join(lines)

def analyze_figure_with_nova(pdf_bytes: bytes, figure: Dict[str, Any], doc_context: str = "") -> str:
    page_num = figure["page"] - 1
    bbox = figure["bbox"]

    doc = fitz.open(stream=pdf_bytes, filetype="pdf")
    try:
        page = doc[page_num]
        page_rect = page.rect

        x0 = bbox["Left"] * page_rect.width
        y0 = bbox["Top"] * page_rect.height
        x1 = x0 + bbox["Width"] * page_rect.width
        y1 = y0 + bbox["Height"] * page_rect.height

        clip_rect = fitz.Rect(x0, y0, x1, y1)
        pix = page.get_pixmap(clip=clip_rect, dpi=200)
        img_bytes = pix.tobytes("png")

        prompt = (
            f"You are analyzing a figure from an ECMC bankruptcy document. "
            f"Document context: {doc_context[:500]}\n\n"
            f"Analyze this image and describe: "
            f"1. What type of content is this? "
            f"2. If UI elements, list them. "
            f"3. If form/table, extract visible data. "
            f"4. If flowchart, describe process step-by-step. "
            f"5. If screenshot, describe user actions."
        )

        return bedrock_converse(NOVA_MODEL_ID, prompt, image_bytes=img_bytes, max_tokens=2048)
    finally:
        doc.close()

# Chunk Builders
# ============================================================================
@dataclass
class Chunk:
    chunk_id: str
    chunk_type: str
    content: str
    metadata: Dict[str, Any]
    token_count: int = 0


def build_text_chunks(text: str, doc_id: str, start_idx: int, metadata_base: Dict[str, Any]) -> List[Chunk]:
    """Split text into overlapping chunks of ~512 tokens."""
    # Simple sentence-based splitting
    sentences = re.split(r"(?<=[.!?])\s+", text)
    chunks = []
    current_text = ""
    chunk_idx = 0

    for sent in sentences:
        if estimate_tokens(current_text + " " + sent) > TEXT_CHUNK_SIZE:
            # Save current chunk
            chunks.append(Chunk(
                chunk_id=f"{doc_id}_text_{start_idx + chunk_idx:04d}",
                chunk_type="text",
                content=current_text.strip(),
                metadata={**metadata_base, "chunk_index": chunk_idx},
                token_count=estimate_tokens(current_text)
            ))
            # Start new chunk with overlap (last ~50 tokens)
            words = current_text.split()
            overlap_text = " ".join(words[-TEXT_OVERLAP:]) if len(words) > TEXT_OVERLAP else current_text
            current_text = overlap_text + " " + sent
            chunk_idx += 1
        else:
            current_text += " " + sent if current_text else sent

    # Final chunk
    if current_text.strip():
        chunks.append(Chunk(
            chunk_id=f"{doc_id}_text_{start_idx + chunk_idx:04d}",
            chunk_type="text",
            content=current_text.strip(),
            metadata={**metadata_base, "chunk_index": chunk_idx},
            token_count=estimate_tokens(current_text)
        ))

    return chunks


def build_table_chunk(table_data: Dict[str, Any], doc_id: str, chunk_idx: int, metadata_base: Dict[str, Any]) -> Chunk:
    """Build single table chunk. If exceeds 1024 tokens, will need Strategy B later."""
    markdown = table_data.get("markdown", "")
    token_count = estimate_tokens(markdown)

    return Chunk(
        chunk_id=f"{doc_id}_table_{chunk_idx:04d}",
        chunk_type="table",
        content=markdown,
        metadata={
            **metadata_base,
            "row_count": table_data.get("rows", 0),
            "col_count": table_data.get("cols", 0),
            "table_id": table_data.get("table_id", ""),
            "token_count": token_count
        },
        token_count=token_count
    )


def build_form_chunk(form_fields: List[Dict[str, Any]], doc_id: str, chunk_idx: int, metadata_base: Dict[str, Any]) -> Chunk:
    """Build form chunk from key-value pairs."""
    lines = [f"{f['key']}: {f['value']}" for f in form_fields if f.get("key")]
    content = "\n".join(lines)

    return Chunk(
        chunk_id=f"{doc_id}_form_{chunk_idx:04d}",
        chunk_type="form",
        content=content,
        metadata={
            **metadata_base,
            "field_count": len(form_fields),
            "token_count": estimate_tokens(content)
        },
        token_count=estimate_tokens(content)
    )


def build_figure_chunk(description: str, figure_data: Dict[str, Any], doc_id: str, chunk_idx: int, metadata_base: Dict[str, Any]) -> Chunk:
    """Build figure chunk from Nova description."""
    content = f"[FIGURE: {description}]"

    return Chunk(
        chunk_id=f"{doc_id}_figure_{chunk_idx:04d}",
        chunk_type="figure",
        content=content,
        metadata={
            **metadata_base,
            "figure_id": figure_data.get("figure_id", ""),
            "page": figure_data.get("page", 1),
            "nova_analyzed": True,
            "token_count": estimate_tokens(content)
        },
        token_count=estimate_tokens(content)
    )

# Textract JSON Processor
# ============================================================================
def process_textract_json(json_key: str, json_data: Dict[str, Any]) -> List[Chunk]:
    """Process Textract JSON output into chunks."""
    doc_id = Path(json_key).stem
    source_key = json_data.get("source_key", json_key)
    blocks = json_data.get("Blocks", [])
    sequence = json_data.get("sequence", [])

    log.info("Processing Textract JSON: %s (%d blocks, %d sequence items)",
             doc_id, len(blocks), len(sequence))

    chunks = []
    chunk_counter = {"text": 0, "table": 0, "form": 0, "figure": 0}

    # Load PDF bytes for figure analysis if Nova enabled
    pdf_bytes = None
    if NOVA_ENABLED:
        try:
            pdf_key = json_data.get("source_key", "").replace("s3://" + INPUT_BUCKET + "/", "")
            if pdf_key:
                pdf_bytes = load_s3_bytes(INPUT_BUCKET, pdf_key)
        except Exception as e:
            log.warning("Could not load PDF for figure analysis: %s", e)

    # Build context from text for Nova
    doc_context = " ".join([
        item.get("text", "") for item in sequence
        if item.get("type") in ("title", "section_header", "paragraph")
    ])[:1000]

    # Process sequence items in order
    for item in sequence:
        item_type = item.get("type", "")
        page = item.get("page", 1)

        metadata_base = {
            "source_document": doc_id,
            "source_key": source_key,
            "page_number": page,
            "sequence_position": sequence.index(item)
        }

        if item_type == "paragraph":
            text = item.get("text", "")
            if text:
                text_chunks = build_text_chunks(
                    text, doc_id, chunk_counter["text"], metadata_base
                )
                chunks.extend(text_chunks)
                chunk_counter["text"] += len(text_chunks)

        elif item_type == "table":
            table_data = {
                "markdown": item.get("markdown", ""),
                "rows": item.get("rows", 0),
                "cols": item.get("cols", 0),
                "table_id": item.get("table_id", "")
            }
            chunk = build_table_chunk(table_data, doc_id, chunk_counter["table"], metadata_base)
            chunks.append(chunk)
            chunk_counter["table"] += 1

        elif item_type == "key_value":
            # Group consecutive key_values into form chunks
            form_fields = [{"key": item.get("key", ""), "value": item.get("value", "")}]
            chunk = build_form_chunk(form_fields, doc_id, chunk_counter["form"], metadata_base)
            chunks.append(chunk)
            chunk_counter["form"] += 1

        elif item_type == "figure":
            if NOVA_ENABLED and pdf_bytes:
                try:
                    description = analyze_figure_with_nova(
                        pdf_bytes,
                        {"page": page, "bbox": item.get("bbox", {})},
                        doc_context
                    )
                    chunk = build_figure_chunk(
                        description, item, doc_id, chunk_counter["figure"], metadata_base
                    )
                    chunks.append(chunk)
                    chunk_counter["figure"] += 1
                except Exception as e:
                    log.warning("Figure analysis failed: %s", e)
            # If NOVA disabled, skip figure entirely (no placeholder)

    log.info("Chunks created: text=%d, table=%d, form=%d, figure=%d",
             chunk_counter["text"], chunk_counter["table"],
             chunk_counter["form"], chunk_counter["figure"])

    return chunks


# ============================================================================
# Markdown Processor (Scrapy files)
# ============================================================================
def process_markdown(md_key: str, md_text: str) -> List[Chunk]:
    """Process Scrapy Markdown files into chunks."""
    doc_id = Path(md_key).stem
    log.info("Processing Markdown: %s", doc_id)

    chunks = []
    chunk_counter = 0

    # Extract metadata from HTML comments
    metadata = {}
    for match in re.finditer(r"<!--\s*(\w+):\s*(.*?)\s*-->", md_text):
        metadata[match.group(1).strip().lower()] = match.group(2).strip()

    # Split by headings
    sections = re.split(r"(\n##\s+.+\n|\n###\s+.+\n)", md_text)
    current_heading = "Untitled"

    for section in sections:
        if section.strip().startswith("## ") or section.strip().startswith("### "):
            current_heading = section.strip().strip("#").strip()
            continue

        if not section.strip():
            continue

        # Detect tables in markdown
        table_pattern = r"(\|.+\|\n)+"
        tables = re.finditer(table_pattern, section)
        table_positions = [(m.start(), m.end()) for m in tables]

        # Split section into text and table parts
        last_end = 0
        for start, end in table_positions:
            # Text before table
            text_before = section[last_end:start]
            if text_before.strip():
                text_chunks = build_text_chunks(
                    text_before, doc_id, chunk_counter,
                    {
                        "source_document": doc_id,
                        "source_key": md_key,
                        "parent_heading": current_heading,
                        "page_number": None
                    }
                )
                chunks.extend(text_chunks)
                chunk_counter += len(text_chunks)

            # Table
            table_md = section[start:end]
            table_lines = [l.strip() for l in table_md.strip().split("\n") if l.strip().startswith("|")]
            if table_lines:
                # Parse simple table
                grid = []
                for line in table_lines:
                    if "---" in line:
                        continue
                    cells = [c.strip() for c in line.strip("|").split("|")]
                    grid.append(cells)

                if grid:
                    markdown = grid_to_markdown(grid)
                    chunk = Chunk(
                        chunk_id=f"{doc_id}_table_{chunk_counter:04d}",
                        chunk_type="table",
                        content=markdown,
                        metadata={
                            "source_document": doc_id,
                            "source_key": md_key,
                            "parent_heading": current_heading,
                            "page_number": None,
                            "row_count": len(grid) - 1,
                            "col_count": len(grid[0]) if grid else 0
                        },
                        token_count=estimate_tokens(markdown)
                    )
                    chunks.append(chunk)
                    chunk_counter += 1

            last_end = end

        # Text after last table
        text_after = section[last_end:]
        if text_after.strip():
            text_chunks = build_text_chunks(
                text_after, doc_id, chunk_counter,
                {
                    "source_document": doc_id,
                    "source_key": md_key,
                    "parent_heading": current_heading,
                    "page_number": None
                }
            )
            chunks.extend(text_chunks)
            chunk_counter += len(text_chunks)

    log.info("Markdown chunks: %d", len(chunks))
    return chunks

# Main Orchestrator
# ============================================================================
def process_all_documents() -> Dict[str, Any]:
    log.info("=" * 60)
    log.info("Chunking Pipeline")
    log.info("NOVA_ENABLED=%s", NOVA_ENABLED)
    log.info("Text chunk size=%d, overlap=%d", TEXT_CHUNK_SIZE, TEXT_OVERLAP)
    log.info("Table chunk max=%d", TABLE_CHUNK_MAX)
    log.info("=" * 60)

    all_chunks = []

    # Process Textract JSON files (9 PDFs)
    json_keys = list_json_keys(INPUT_BUCKET, TEXTRACT_JSON_PREFIX)
    log.info("Found %d Textract JSON files", len(json_keys))

    for key in json_keys:
        try:
            json_data = load_s3_json(INPUT_BUCKET, key)
            chunks = process_textract_json(key, json_data)
            all_chunks.extend(chunks)
        except Exception as exc:
            log.exception("Failed to process %s: %s", key, exc)

    # Process Markdown files (4 Scrapy)
    md_keys = list_md_keys(INPUT_BUCKET, MD_PREFIX)
    log.info("Found %d Markdown files", len(md_keys))

    for key in md_keys:
        try:
            md_text = load_s3_text(INPUT_BUCKET, key)
            chunks = process_markdown(key, md_text)
            all_chunks.extend(chunks)
        except Exception as exc:
            log.exception("Failed to process %s: %s", key, exc)

    if not all_chunks:
        log.error("No chunks generated!")
        return {"status": "ERROR", "error": "No chunks"}

    # Build output
    output_data = {
        "pipeline": "chunking",
        "processed_at": datetime.now(timezone.utc).isoformat(),
        "nova_enabled": NOVA_ENABLED,
        "text_chunk_size": TEXT_CHUNK_SIZE,
        "text_overlap": TEXT_OVERLAP,
        "table_chunk_max": TABLE_CHUNK_MAX,
        "total_chunks": len(all_chunks),
        "chunk_types": {},
        "chunks": []
    }

    # Count types
    type_counts = {}
    for chunk in all_chunks:
        type_counts[chunk.chunk_type] = type_counts.get(chunk.chunk_type, 0) + 1
        output_data["chunks"].append({
            "chunk_id": chunk.chunk_id,
            "chunk_type": chunk.chunk_type,
            "content": chunk.content,
            "metadata": chunk.metadata,
            "token_count": chunk.token_count
        })

    output_data["chunk_types"] = type_counts

    log.info("=" * 60)
    log.info("COMPLETE: %d chunks", len(all_chunks))
    for ctype, count in type_counts.items():
        log.info("  %s: %d", ctype, count)
    log.info("=" * 60)

    # Save to S3
    timestamp = datetime.now(timezone.utc).strftime("%Y%m%d_%H%M%S")
    output_key = f"{OUTPUT_PREFIX.rstrip('/')}/chunk_objects_{timestamp}.json"
    output_uri = save_s3_json(OUTPUT_BUCKET, output_key, output_data)

    return {
        "status": "SUCCEEDED",
        "output_uri": output_uri,
        "total_chunks": len(all_chunks),
        "chunk_types": type_counts
    }

def robust_json_parse(text: str) -> List[Dict[str, Any]]:
    """
    Robustly parse JSON from Claude response.
    Handles markdown wrappers, preamble, trailing text, and common JSON errors.
    """
    if not text or not text.strip():
        log.warning("Empty response from Claude")
        return []
    
    # Step 1: Extract JSON array from text
    json_text = extract_json_array(text)
    
    if not json_text:
        log.error("No JSON array found in response. Raw: %s", text[:500])
        return []
    
    # Step 2: Clean common JSON issues
    json_text = clean_json_text(json_text)
    
    # Step 3: Parse
    try:
        parsed = json.loads(json_text)
        if isinstance(parsed, list):
            return parsed
        elif isinstance(parsed, dict) and "terms" in parsed:
            return parsed["terms"]
        else:
            log.error("Parsed JSON is not a list. Type: %s", type(parsed))
            return []
    except json.JSONDecodeError as e:
        log.error("JSON parse failed: %s", e)
        log.error("Cleaned JSON text: %s", json_text[:500])
        return []


def extract_json_array(text: str) -> str:
    """
    Extract JSON array from text that may contain markdown, preamble, etc.
    """
    text = text.strip()
    
    # Remove markdown code blocks: ```json ... ``` or ``` ... ```
    patterns = [
        r"```json\s*(.*?)\s*```",
        r"```\s*(.*?)\s*```",
    ]
    for pattern in patterns:
        match = re.search(pattern, text, re.DOTALL)
        if match:
            return match.group(1).strip()
    
    # Find raw array between [ and ]
    start = text.find("[")
    end = text.rfind("]")
    
    if start != -1 and end != -1 and end > start:
        return text[start:end+1]
    
    return ""


def clean_json_text(text: str) -> str:
    """
    Fix common JSON formatting issues from LLM output.
    """
    text = text.strip("\ufeff").strip()
    
    # Fix trailing commas before ] or }
    text = re.sub(r",\s*]", "]", text)
    text = re.sub(r",\s*}", "}", text)
    
    # Fix single quotes to double quotes (only where it looks like JSON delimiters)
    text = re.sub(r"(?<=[:,\[\{])\s*'", '"', text)
    text = re.sub(r"'(?=\s*[:,}\]])", '"', text)
    
    # Fix unescaped newlines inside strings
    text = fix_newlines_in_json(text)
    
    return text.strip()


def fix_newlines_in_json(text: str) -> str:
    """
    Replace literal newlines inside JSON string values with \n.
    """
    result = []
    in_string = False
    escape = False
    
    for char in text:
        if escape:
            result.append(char)
            escape = False
            continue
            
        if char == "\\":
            result.append(char)
            escape = True
            continue
            
        if char == '"':
            in_string = not in_string
            result.append(char)
            continue
            
        if char == "\n" and in_string:
            result.append("\\n")
            continue
            
        if char == "\r" and in_string:
            continue
            
        result.append(char)
    
    return "".join(result)

=============================================
Answers user questions using Tree Index + Term Index.

Input: User question (text) + tree_index.json + term_index.json
Output: Answer with citations and confidence score

Supports:
    - Term queries: "What is a T4 letter?" → Term Index lookup
    - Section queries: "How do I make payments?" → Tree Index keyword search
    - Hybrid: combines both paths

Design decisions (confirmed):
    - Input: CLI + Lambda compatible
    - Output: JSON with answer, citations, confidence
    - Citation: detailed (source_key + section_title)
    - Top-K: 3 for term, 5 for section
    - Fallback: "not found" + suggest related terms

def list_index_keys(bucket: str, prefix: str) -> List[str]:
    """List latest index file under prefix."""
    s3 = get_s3_client()
    paginator = s3.get_paginator("list_objects_v2")
    keys = []
    for page in paginator.paginate(Bucket=bucket, Prefix=prefix):
        for obj in page.get("Contents", []):
            if obj["Key"].endswith(".json"):
                keys.append(obj["Key"])
    # Return latest (sorted by key name which includes timestamp)
    return sorted(keys, reverse=True)


def load_s3_json(bucket: str, key: str) -> Dict[str, Any]:
    s3 = get_s3_client()
    return json.loads(s3.get_object(Bucket=bucket, Key=key)["Body"].read())


def load_index(bucket: str, prefix: str) -> Optional[Dict[str, Any]]:
    """Load latest index file from S3."""
    keys = list_index_keys(bucket, prefix)
    if not keys:
        log.error("No index found for prefix: %s", prefix)
        return None
    latest = keys[0]
    log.info("Loading index: s3://%s/%s", bucket, latest)
    return load_s3_json(bucket, latest)

def bedrock_chat(model_id: str, prompt: str, max_tokens: int = 1024) -> str:
    client = get_bedrock_client()
    try:
        resp = client.converse(
            modelId=model_id,
            messages=[{"role": "user", "content": [{"text": prompt}]}],
            inferenceConfig={"maxTokens": max_tokens, "temperature": 0.0}
        )
        return resp["output"]["message"]["content"][0]["text"]
    except Exception as exc:
        log.error("Bedrock error: %s", exc)
        return ""

# Term Lookup

class TermLookup:
    """Search Term Index for canonical terms and aliases."""

    def __init__(self, term_index: Dict[str, Any]):
        self.terms = term_index.get("terms", [])
        self._build_index()

    def _build_index(self):
        """Build normalized lookup maps."""
        self.by_term: Dict[str, Dict] = {}      # normalized canonical -> term
        self.by_alias: Dict[str, str] = {}     # normalized alias -> canonical

        for term in self.terms:
            canonical = self._normalize(term["term"])
            self.by_term[canonical] = term

            for alias in term.get("aliases", []):
                norm_alias = self._normalize(alias)
                self.by_alias[norm_alias] = canonical

    @staticmethod
    def _normalize(text: str) -> str:
        return re.sub(r"[^\w\s]", "", text.lower()).strip()

    def search(self, query: str) -> Optional[Dict[str, Any]]:
        """
        Search for term by query.
        Returns term entry with all sources if found, None otherwise.
        """
        norm_query = self._normalize(query)

        # Direct match on canonical term
        if norm_query in self.by_term:
            return self.by_term[norm_query]

        # Match on alias
        if norm_query in self.by_alias:
            canonical = self.by_alias[norm_query]
            return self.by_term.get(canonical)

        # Partial match: check if query is substring of any term/alias
        for canonical, term in self.by_term.items():
            if norm_query in canonical:
                return term
            for alias in term.get("aliases", []):
                if norm_query in self._normalize(alias):
                    return term

        return None

    def get_related_terms(self, query: str, top_k: int = 3) -> List[str]:
        """Suggest related terms if exact match not found."""
        norm_query = self._normalize(query)
        scored = []

        for term in self.terms:
            canonical = self._normalize(term["term"])
            score = 0
            if norm_query in canonical or canonical in norm_query:
                score += 2
            for alias in term.get("aliases", []):
                if norm_query in self._normalize(alias):
                    score += 1
            if score > 0:
                scored.append((score, term["term"]))

        scored.sort(reverse=True)
        return [t[1] for t in scored[:top_k]]

# Section Retriever

class SectionRetriever:
    """Search Tree Index for relevant sections by keywords."""

    def __init__(self, tree_index: Dict[str, Any]):
        self.documents = tree_index.get("documents", [])
        self._flatten_nodes()

    def _flatten_nodes(self):
        """Flatten all nodes from all documents into searchable list."""
        self.all_nodes = []
        for doc in self.documents:
            for node in doc.get("nodes", []):
                self.all_nodes.append({
                    **node,
                    "doc_id": doc.get("doc_id", "unknown"),
                    "doc_title": doc.get("doc_title", "unknown")
                })

    def search(self, query: str, top_k: int = 5) -> List[Dict[str, Any]]:
        """
        Score nodes by keyword overlap and return top-K.
        """
        query_words = set(self._tokenize(query))
        scored = []

        for node in self.all_nodes:
            score = self._score_node(node, query_words)
            if score > 0:
                scored.append({**node, "_score": score})

        scored.sort(key=lambda x: x["_score"], reverse=True)
        return scored[:top_k]

    @staticmethod
    def _tokenize(text: str) -> List[str]:
        """Simple tokenization."""
        return re.findall(r"\b\w+\b", text.lower())

    def _score_node(self, node: Dict, query_words: set) -> float:
        """Score how relevant a node is to query."""
        title_words = set(self._tokenize(node.get("title", "")))
        summary_words = set(self._tokenize(node.get("summary", "")))
        text_words = set(self._tokenize(node.get("full_text", "")))

        # Weighted scoring
        title_match = len(query_words & title_words) * 3.0
        summary_match = len(query_words & summary_words) * 2.0
        text_match = len(query_words & text_words) * 1.0

        return title_match + summary_match + text_match

# Context Builder

class ContextBuilder:
    """Format retrieved sections into Claude prompt context."""

    @staticmethod
    def build_context(sections: List[Dict[str, Any]]) -> str:
        """Build structured context string from sections."""
        parts = []
        for i, sec in enumerate(sections, 1):
            part = ContextBuilder._format_section(sec, i)
            parts.append(part)
        return "\n\n".join(parts)

    @staticmethod
    def _format_section(sec: Dict, index: int) -> str:
        """Format single section for context."""
        lines = [f"[SOURCE {index}]"]
        lines.append(f"Document: {sec.get('doc_title', 'unknown')} ({sec.get('doc_id', 'unknown')})")
        lines.append(f"Section: {sec.get('title', 'Untitled')}")
        lines.append(f"Summary: {sec.get('summary', 'No summary')}")
        lines.append("")

        # Add sequence blocks
        for block in sec.get("sequence", []):
            btype = block.get("type", "unknown")
            content = block.get("content", {})

            if btype == "paragraph":
                lines.append(content)
            elif btype == "table":
                cols = ", ".join(content.get("columns", []))
                lines.append(f"[TABLE: columns = {cols}]")
            elif btype == "list":
                items = content.get("items", [])
                for item in items:
                    lines.append(f"  - {item}")
            elif btype == "figure":
                alt = content.get("alt_text", "no description")
                lines.append(f"[FIGURE: {alt}]")
            elif btype == "key_value":
                lines.append(f"{content['key']}: {content['value']}")

        return "\n".join(lines)

class AnswerGenerator:
    """Generate answer using Claude with retrieved context."""

    def __init__(self, model_id: str = CLAUDE_MODEL_ID):
        self.model_id = model_id

    def generate(self, query: str, context: str, citations: List[Dict]) -> Tuple[str, str]:
        """
        Generate answer from context.
        Returns (answer, confidence).
        """
        prompt = self._build_prompt(query, context)
        response = bedrock_chat(self.model_id, prompt, max_tokens=2048)

        # Simple confidence based on response quality
        confidence = self._assess_confidence(response, citations)

        return response, confidence

    def _build_prompt(self, query: str, context: str) -> str:
        return (
            "You are an ECMC bankruptcy student loan document assistant. "
            "Answer the user's question using ONLY the provided context below. "
            "Be specific and accurate. If the context does not contain the answer, say so clearly. "
            "Cite the source number [SOURCE N] when referencing information.\n\n"
            f"{context}\n\n"
            f"[USER QUESTION]\n{query}\n\n"
            "[ANSWER]"
        )

    @staticmethod
    def _assess_confidence(answer: str, citations: List[Dict]) -> str:
        """Assess confidence based on answer quality and citations."""
        if not answer or len(answer) < 20:
            return "none"
        if "not found" in answer.lower() or "not contain" in answer.lower():
            return "low"
        if len(citations) >= 3:
            return "high"
        if len(citations) >= 1:
            return "medium"
        return "low"

# Query Engine (Main)
# ============================================================================
class QueryEngine:
    """
    Main query engine combining Term Index + Tree Index.
    """

    def __init__(self, tree_index: Dict[str, Any], term_index: Dict[str, Any]):
        self.term_lookup = TermLookup(term_index)
        self.section_retriever = SectionRetriever(tree_index)
        self.context_builder = ContextBuilder()
        self.answer_generator = AnswerGenerator()

    def query(self, user_query: str) -> QueryResult:
        """
        Process user query and return answer.
        """
        log.info("Query: %s", user_query)

        # Try term lookup first
        term_result = self.term_lookup.search(user_query)

        if term_result:
            return self._handle_term_query(user_query, term_result)
        else:
            return self._handle_section_query(user_query)

    def _handle_term_query(self, query: str, term: Dict) -> QueryResult:
        """Handle term-based query."""
        log.info("Term match: %s", term["term"])

        # Load sections from tree index using term sources
        sources = term.get("sources", [])
        sections = self._load_sections_from_sources(sources)

        if not sections:
            # Fallback to keyword search
            sections = self.section_retriever.search(query, top_k=3)

        return self._generate_answer(query, sections, "term")

    def _handle_section_query(self, query: str) -> QueryResult:
        """Handle general section query."""
        log.info("No term match, using section search")

        sections = self.section_retriever.search(query, top_k=5)

        if not sections:
            # Suggest related terms
            suggestions = self.term_lookup.get_related_terms(query)
            return QueryResult(
                query=query,
                answer=f"No relevant information found. Did you mean: {', '.join(suggestions)}?" if suggestions else "No relevant information found.",
                confidence="none",
                query_type="section"
            )

        return self._generate_answer(query, sections, "section")

    def _load_sections_from_sources(self, sources: List[Dict]) -> List[Dict]:
        """Load section nodes from Tree Index matching term sources."""
        sections = []
        for src in sources:
            section_title = src.get("section_title", "")
            # Find matching node in tree
            for node in self.section_retriever.all_nodes:
                if node.get("title") == section_title:
                    sections.append(node)
                    break
        return sections

    def _generate_answer(self, query: str, sections: List[Dict], query_type: str) -> QueryResult:
        """Build context, generate answer, format result."""
        # Build citations
        citations = []
        for sec in sections:
            citations.append({
                "document": sec.get("doc_title", "unknown"),
                "doc_id": sec.get("doc_id", "unknown"),
                "section": sec.get("title", "Untitled"),
                "node_id": sec.get("node_id", "unknown")
            })

        # Build context and generate answer
        context = self.context_builder.build_context(sections)
        answer, confidence = self.answer_generator.generate(query, context, citations)

        return QueryResult(
            query=query,
            answer=answer,
            citations=citations,
            confidence=confidence,
            sources_used=len(sections),
            query_type=query_type
        )

def run_query(user_query: str) -> Dict[str, Any]:
    """
    Load indices and run query.
    """
    log.info("=" * 60)
    log.info("Query Engine")
    log.info("Query: %s", user_query)
    log.info("=" * 60)

    # Load indices from S3
    tree_index = load_index(INPUT_BUCKET, TREE_INDEX_PREFIX)
    term_index = load_index(INPUT_BUCKET, TERM_INDEX_PREFIX)

    if not tree_index or not term_index:
        return {
            "status": "ERROR",
            "error": "Failed to load indices",
            "tree_loaded": tree_index is not None,
            "term_loaded": term_index is not None
        }

    # Run query

# Section Parser (same logic as Tree Index Builder)

def extract_sections(md_text: str) -> List[Dict[str, Any]]:
    """
    Split Markdown into sections at ## and ### headings.
    Returns list of {title, level, text} dicts.
    """
    lines = md_text.splitlines()
    sections = []
    current_title = "Untitled"
    current_level = 0
    current_lines = []
    section_counter = 0

    for line in lines:
        stripped = line.strip()

        # ## Section
        m2 = re.match(r"^##\s+(.+)", stripped)
        if m2:
            if current_lines:
                sections.append({
                    "node_id": f"S{section_counter:04d}",
                    "title": current_title,
                    "level": current_level,
                    "text": "\n".join(current_lines).strip()
                })
                section_counter += 1
            current_title = m2.group(1).strip()
            current_level = 2
            current_lines = []
            continue

        # ### Subsection
        m3 = re.match(r"^###\s+(.+)", stripped)
        if m3:
            if current_lines:
                sections.append({
                    "node_id": f"S{section_counter:04d}",
                    "title": current_title,
                    "level": current_level,
                    "text": "\n".join(current_lines).strip()
                })
                section_counter += 1
            current_title = m3.group(1).strip()
            current_level = 3
            current_lines = []
            continue

        # Skip title and comments
        if re.match(r"^#\s+", stripped) or stripped.startswith("<!--") or not stripped:
            continue

        current_lines.append(stripped)

    # Final section
    if current_lines:
        sections.append({
            "node_id": f"S{section_counter:04d}",
            "title": current_title,
            "level": current_level,
            "text": "\n".join(current_lines).strip()
        })

    return sections

def bedrock_chat(model_id: str, prompt: str, max_tokens: int = 1024) -> str:
    client = get_bedrock_client()
    try:
        resp = client.converse(
            modelId=model_id,
            messages=[{"role": "user", "content": [{"text": prompt}]}],
            inferenceConfig={"maxTokens": max_tokens, "temperature": 0.0}
        )
        return resp["output"]["message"]["content"][0]["text"]
    except Exception as exc:
        log.error("Bedrock error: %s", exc)
        return ""


def extract_terms_from_section(section_title: str, section_text: str) -> List[Dict[str, Any]]:
    """
    Ask Claude to extract domain terms from a section.
    Returns list of {term, aliases, category} dicts.
    """
    prompt = (
        "You are analyzing an ECMC bankruptcy student loan document section. "
        "Extract all important domain-specific terms that a borrower or agent "
        "might ask about. Include legal terms, procedures, forms, loan statuses, "
        "and specific document names or codes.\n\n"
        "For each term, provide:\n"
        "- The canonical term name\n"
        "- Common aliases or abbreviations\n"
        - A category: legal, procedure, form, status, or document\n\n"
        "Return ONLY a JSON array with this exact shape:\n"
        '[{"term": "Exact Term", "aliases": ["alias1", "alias2"], "category": "legal|procedure|form|status|document"}]\n\n'
        f"Section title: {section_title}\n"
        f"Section text:\n{section_text[:3000]}\n\n"
        "Terms (JSON only):"
    )

    resp = bedrock_chat(CLAUDE_MODEL_ID, prompt, max_tokens=1024)

    terms = []
    try:
        cleaned = re.sub(r"```json\s*|\s*```", "", resp).strip()
        parsed = json.loads(cleaned)
        if isinstance(parsed, list):
            terms = parsed
        elif isinstance(parsed, dict) and "terms" in parsed:
            terms = parsed["terms"]
    except Exception as exc:
        log.error("Failed to parse Claude response: %s — raw: %s", exc, resp[:200])

    return terms

# Term Deduplication (Simple String Match)
# ============================================================================
def normalize_term(term: str) -> str:
    """Normalize term for deduplication."""
    return re.sub(r"[^\w\s]", "", term.lower()).strip()


def merge_term_into_index(
    term_item: Dict[str, Any],
    source_key: str,
    section_title: str,
    context_snippet: str,
    term_index: Dict[str, Dict[str, Any]]
) -> None:
    """
    Merge term into index. Simple string match on normalized term name.
    If term exists, add source. If new, create entry.
    """
    term_name = term_item.get("term", "").strip()
    if not term_name:
        return

    normalized = normalize_term(term_name)

    if normalized not in term_index:
        term_index[normalized] = {
            "term": term_name,
            "aliases": term_item.get("aliases", []),
            "category": term_item.get("category", "unknown"),
            "sources": [],
            "occurrence_count": 0
        }

    term_index[normalized]["sources"].append({
        "source_key": source_key,
        "section_title": section_title,
        "context_snippet": context_snippet[:300].replace("\n", " ")
    })
    term_index[normalized]["occurrence_count"] += 1

    if not all_keys:
        log.error("NO Markdown files found!")
        return {"status": "ERROR", "error": "No files found"}

    # Build term index
    term_index: Dict[str, Dict[str, Any]] = {}

    for key in all_keys:
        try:
            log.info("Processing: %s", key)
            md_text = load_s3_text(INPUT_BUCKET, key)
            doc_id = Path(key).stem

                    continue

                terms = extract_terms_from_section(sec["title"], sec["text"])

                for term_item in terms:
                    merge_term_into_index(
                        term_item=term_item,
                        source_key=key,
                        section_title=sec["title"],
                        context_snippet=sec["text"],
                        term_index=term_index
                    )

        except Exception as exc:
            log.exception("Failed to process %s: %s", key, exc)

        log.error("No terms extracted!")
        return {"status": "ERROR", "error": "No terms extracted"}

    # Build output
    unified_terms = {
        "index_type": "term",
        "processed_at": datetime.now(timezone.utc).isoformat(),
        "total_terms": len(term_index),
        "terms": list(term_index.values())
    }

    log.info("COMPLETE: %d unique terms", len(term_index))
    log.info("Output: %s", term_uri)
    log.info("=" * 60)

    return {
        "status": "SUCCEEDED",
        "term_index_uri": term_uri,
        "total_terms": len(term_index),
        "sample_terms": [t["term"] for t in list(term_index.values())[:5]]
    }

def extract_metadata(text: str) -> Dict[str, Any]:
    """Parse HTML comments into metadata dict."""
    metadata = {}
    pattern = r"<!--\s*(\w+):\s*(.*?)\s*-->"
    for match in re.finditer(pattern, text):

# ============================================================================
# Block Parser
# ============================================================================
def parse_blocks(md_text: str) -> List[Dict[str, Any]]:
    """
    Parse Markdown into typed blocks.
    Returns list of {type, content} dicts.
    """
    lines = md_text.splitlines()
    blocks = []
    i = 0

    while i < len(lines):
        line = lines[i]
        stripped = line.strip()

        # Skip empty lines and comments
        if not stripped or stripped.startswith("<!--"):
            i += 1
            continue

        # Skip headings (handled by section builder)
        if stripped.startswith("#"):
            i += 1
            continue

        # Table
        if stripped.startswith("|"):
            table_lines = []
            while i < len(lines) and lines[i].strip().startswith("|"):
                table_lines.append(lines[i].strip())
                i += 1

            table_data = parse_pipe_table(table_lines)
            if table_data:
                blocks.append({
                    "type": "table",
                    "content": table_data
                })
            continue

        # Unordered list
        if stripped.startswith(("- ", "* ", "+ ")):
            items = []
            while i < len(lines):
                current = lines[i].strip()
                if not current:
                    i += 1
                    continue
                if current.startswith(("- ", "* ", "+ ")):
                    items.append(current[2:].strip())
                    i += 1
                elif current.startswith("  ") and items:
                    items[-1] += " " + current.strip()
                    i += 1
                else:
                    break

            if items:
                blocks.append({
                    "type": "list",
                    "content": {"items": items, "ordered": False}
                })
            continue

        # Ordered list
        ordered_match = re.match(r"^(\d+)\.\s+(.+)", stripped)
        if ordered_match:
            items = []
            while i < len(lines):
                current = lines[i].strip()
                if not current:
                    i += 1
                    continue
                m = re.match(r"^(\d+)\.\s+(.+)", current)
                if m:
                    items.append(m.group(2).strip())
                    i += 1
                elif current.startswith("  ") and items:
                    items[-1] += " " + current.strip()
                    i += 1
                else:
                    break

            if items:
                blocks.append({
                    "type": "list",
                    "content": {"items": items, "ordered": True}
                })
            continue

        # Figure / Image
        img_match = re.match(r"^!\[(.*?)\]\((.*?)\)", stripped)
        if img_match:
            blocks.append({
                "type": "figure",
                "content": {
                    "alt_text": img_match.group(1),
                    "source_url": img_match.group(2),
                    "description": None
                }
            })
            i += 1
            continue

        # Key-Value
        kv_match = re.match(r"^\*\*(.+?)\*\*:\s*(.+)", stripped)
        if kv_match:
            blocks.append({
                "type": "key_value",
                "content": {
                    "key": kv_match.group(1).strip(),
                    "value": kv_match.group(2).strip()
                }
            })
            i += 1
            continue

        # Paragraph (default)
        para_lines = [stripped]
        i += 1
        while i < len(lines):
            current = lines[i].strip()
            if not current:
                i += 1
                break
            # Stop at block-level elements
            if current.startswith(("#", "|", "- ", "* ", "+ ", "!")):
                break
            if re.match(r"^\d+\.\s+", current):
                break
            if re.match(r"^\*\*(.+?)\*\*:\s*", current):
                break
            para_lines.append(current)
            i += 1

        content_text = strip_inline_markdown(" ".join(para_lines))
        if content_text:
            blocks.append({
                "type": "paragraph",
                "content": content_text
            })

    return blocks


def parse_pipe_table(lines: List[str]) -> Optional[Dict[str, Any]]:
    """Parse pipe-delimited table lines into structured data."""
    data_lines = []
    for line in lines:
        cells = [c.strip() for c in line.strip("|").split("|")]
        cells = [c for c in cells]
        # Skip separator
        is_sep = all(re.match(r"^[-:]+$", c.strip()) for c in cells if c.strip())
        if not is_sep:
            data_lines.append(cells)

    if len(data_lines) < 1:
        return None

    return {
        "columns": data_lines[0],
        "rows": data_lines[1:] if len(data_lines) > 1 else [],
        "row_count": len(data_lines) - 1,
        "col_count": len(data_lines[0]) if data_lines else 0
    }


def strip_inline_markdown(text: str) -> str:
    """Strip inline formatting for clean LLM text."""
    text = re.sub(r"\*\*(.+?)\*\*", r"\1", text)
    text = re.sub(r"__(.+?)__", r"\1", text)
    text = re.sub(r"\*(.+?)\*", r"\1", text)
    text = re.sub(r"_(.+?)_", r"\1", text)
    text = re.sub(r"`(.+?)`", r"\1", text)
    text = re.sub(r"\[(.+?)\]\(.*?\)", r"\1", text)
    text = re.sub(r"~~(.+?)~~", r"\1", text)
    return text.strip()

def build_sections(md_text: str, source_key: str, metadata: Dict[str, Any]) -> Tuple[List[Dict], str]:
    """
    Split Markdown into sections at heading boundaries.
    Returns (sections, doc_title).
    """
    lines = md_text.splitlines()
    sections = []
    current_title = "Untitled"
    current_level = 0
    current_blocks: List[Dict] = []
    section_counter = 0

    # Extract doc title from first # heading
    doc_title = "Untitled"
    for line in lines:
        m = re.match(r"^#\s+(.+)", line.strip())
        if m:
            doc_title = m.group(1).strip()
            break

    # Parse all blocks first
    all_blocks = parse_blocks(md_text)
    block_idx = 0

    i = 0
    while i < len(lines):
        line = lines[i]
        stripped = line.strip()

        # ## Section heading
        m2 = re.match(r"^##\s+(.+)", stripped)
        if m2:
            # Save previous section
            if current_blocks:
                sections.append({
                    "node_id": f"S{section_counter:04d}",
                    "title": current_title,
                    "level": current_level,
                    "blocks": current_blocks,
                    "metadata": {"source_key": source_key, **metadata}
                })
                section_counter += 1

            current_title = m2.group(1).strip()
            current_level = 2
            current_blocks = []
            i += 1
            continue

        # ### Subsection heading
        m3 = re.match(r"^###\s+(.+)", stripped)
        if m3:
            if current_blocks:
                sections.append({
                    "node_id": f"S{section_counter:04d}",
                    "title": current_title,
                    "level": current_level,
                    "blocks": current_blocks,
                    "metadata": {"source_key": source_key, **metadata}
                })
                section_counter += 1

            current_title = m3.group(1).strip()
            current_level = 3
            current_blocks = []
            i += 1
            continue

        # Skip title and comments
        if re.match(r"^#\s+", stripped) or stripped.startswith("<!--") or not stripped:
            i += 1
            continue

        # Assign block to current section
        # Find matching block by line content
        if block_idx < len(all_blocks):
            current_blocks.append(all_blocks[block_idx])
            block_idx += 1

        i += 1

    # Save final section
    if current_blocks:
        sections.append({
            "node_id": f"S{section_counter:04d}",
            "title": current_title,
            "level": current_level,
            "blocks": current_blocks,
            "metadata": {"source_key": source_key, **metadata}
        })

    return sections, doc_title

def build_document_tree(source_key: str, doc_id: str, md_text: str) -> Dict[str, Any]:
    """Build tree index for one Markdown document."""
    log.info("Building tree for %s", doc_id)

    metadata = extract_metadata(md_text)
    sections, doc_title = build_sections(md_text, source_key, metadata)

    nodes = []
    for sec in sections:
        # Build full text for summary
        text_parts = []
        for block in sec["blocks"]:
            btype = block["type"]
            content = block["content"]
            if btype == "paragraph":
                text_parts.append(content)
            elif btype == "table":
                text_parts.append(f"[TABLE: {content.get('row_count', 0)} rows x {content.get('col_count', 0)} cols]")
            elif btype == "list":
                text_parts.append(f"[LIST: {len(content.get('items', []))} items]")
            elif btype == "figure":
                text_parts.append(f"[FIGURE: {content.get('alt_text', 'no description')}]")
            elif btype == "key_value":
                text_parts.append(f"{content['key']}: {content['value']}")

        full_text = "\n".join(text_parts)
        summary = summarize_with_claude(sec["title"], full_text)

        nodes.append({
            "node_id": f"{doc_id}_{sec['node_id']}",
            "title": sec["title"],
            "level": sec["level"],
            "summary": summary,
            "full_text": full_text,
            "sequence": sec["blocks"],
            "metadata": sec["metadata"]
        })

    return {
        "doc_id": doc_id,
        "doc_title": doc_title,
        "source_key": source_key,
        "source_type": "markdown",
        "processed_at": datetime.now(timezone.utc).isoformat(),
        "section_count": len(nodes),
        "nodes": nodes
    }

def process_all_documents() -> Dict[str, Any]:

    log.info("Tree Index Builder")
    log.info("Input: s3://%s/%s, s3://%s/%s", INPUT_BUCKET, TEXTRACT_MD_PREFIX, INPUT_BUCKET, MD_PREFIX)
    log.info("Output: s3://%s/%s", OUTPUT_BUCKET, OUTPUT_PREFIX)
    log.info("=" * 60)

    # List ALL md files from BOTH prefixes
    textract_keys = list_md_keys(INPUT_BUCKET, TEXTRACT_MD_PREFIX)
    scrapy_keys = list_md_keys(INPUT_BUCKET, MD_PREFIX)
    all_keys = textract_keys + scrapy_keys

    log.info("TOTAL files found: %d (Textract: %d, Scrapy: %d)", 
             len(all_keys), len(textract_keys), len(scrapy_keys))

    if not all_keys:
        log.error("NO Markdown files found! Check prefixes.")
        return {"status": "ERROR", "error": "No files found", "textract_count": len(textract_keys), "scrapy_count": len(scrapy_keys)}

    documents = []
    for key in all_keys:
        try:
            log.info("Processing: %s", key)
            md_text = load_s3_text(INPUT_BUCKET, key)
            doc_id = Path(key).stem

            tree = build_document_tree(key, doc_id, md_text)
            documents.append(tree)

            log.info("  -> %d sections", tree["section_count"])

        except Exception as exc:
            log.exception("Failed to process %s: %s", key, exc)

    if not documents:
        log.error("No documents processed successfully!")
        return {"status": "ERROR", "error": "All documents failed"}

    # Build unified tree
    unified_tree = {
        "index_type": "tree",
        "processed_at": datetime.now(timezone.utc).isoformat(),
        "document_count": len(documents),
        "documents": documents
    }

    # Save
    timestamp = datetime.now(timezone.utc).strftime("%Y%m%d_%H%M%S")
    tree_key = f"{OUTPUT_PREFIX.rstrip('/')}/tree_index_{timestamp}.json"
    tree_uri = save_s3_json(OUTPUT_BUCKET, tree_key, unified_tree)

    total_sections = sum(d["section_count"] for d in documents)

    log.info("=" * 60)
    log.info("COMPLETE: %d documents, %d sections", len(documents), total_sections)
    log.info("Output: %s", tree_uri)
    log.info("=" * 60)

    return {
        "status": "SUCCEEDED",
        "tree_index_uri": tree_uri,
        "document_count": len(documents),
        "total_sections": total_sections,
        "documents": [d["doc_id"] for d in documents]
    }

"""
Tree Index Builder — Markdown to Hierarchical Index
====================================================
Builds tree index from Markdown files with zero information loss.

Input: Markdown files (from Textract .md or Scrapy HTML→Markdown)
Output: tree_index.json with typed sequence blocks, metadata, raw preservation

Design:
    - sequence[]: typed blocks (paragraph, table, list, figure, key_value)
    - metadata: source tracking per node
    - sequence[].raw: original Markdown per block
    - raw_markdown: full section backup
    - No Nova during indexing. Figure alt_text used as-is.
    - Nova on-demand only if query fails against alt_text.

Heading levels:
    #   → document title (root level, not a node)
    ##  → section node boundary (new node_id, sequence starts)
    ### → nested section node (child of ##, own node_id)
"""

def list_md_keys(bucket: str, prefix: str) -> List[str]:
    """List all .md and .markdown keys under S3 prefix."""
    s3 = get_s3_client()
    paginator = s3.get_paginator("list_objects_v2")
    keys = []
    for page in paginator.paginate(Bucket=bucket, Prefix=prefix):
        for obj in page.get("Contents", []):
            k = obj["Key"]
            if k.lower().endswith((".md", ".markdown")):
                keys.append(k)
    return keys


def load_s3_text(bucket: str, key: str) -> str:
    """Load and decode text from S3."""
    s3 = get_s3_client()
    return s3.get_object(Bucket=bucket, Key=key)["Body"].read().decode("utf-8")


def save_s3_json(bucket: str, key: str, data: Dict[str, Any]) -> str:
    """Save JSON to S3, return s3:// URI."""
    s3 = get_s3_client()
    body = json.dumps(data, indent=2, ensure_ascii=False).encode("utf-8")
    s3.put_object(Bucket=bucket, Key=key, Body=body, ContentType="application/json")
    return f"s3://{bucket}/{key}"

# Metadata Extraction from HTML Comments
# ============================================================================
def extract_metadata_from_comments(text: str) -> Dict[str, Any]:
    """
    Parse HTML comments like <!-- source_url: https://... --> into metadata dict.
    Returns empty dict if no comments found.
    """
    metadata = {}
    pattern = r"<!--\s*(\w+):\s*(.*?)\s*-->"
    for match in re.finditer(pattern, text):
        key = match.group(1).strip().lower()
        value = match.group(2).strip()
        metadata[key] = value
    return metadata


# ============================================================================
# Markdown Block Parser — Zero Information Loss
# ============================================================================
@dataclass
class ParsedBlock:
    type: str
    content: Any
    raw: str


def parse_markdown_to_blocks(md_text: str) -> Tuple[List[ParsedBlock], Dict[str, Any]]:
    """
    Parse Markdown into typed blocks with raw preservation.

    Returns:
        blocks: List of ParsedBlock (type, content, raw)
        metadata: Dict from HTML comments
    """
    lines = md_text.splitlines()
    blocks: List[ParsedBlock] = []
    metadata = extract_metadata_from_comments(md_text)

    i = 0
    while i < len(lines):
        line = lines[i]
        stripped = line.strip()

        # Skip empty lines and HTML comments (already extracted)
        if not stripped or stripped.startswith("<!--"):
            i += 1
            continue

        # Heading — detected but not stored as block (triggers section boundary)
        if stripped.startswith("#"):
            # Headings are section boundaries, not content blocks
            # They will be handled by section builder
            i += 1
            continue

        # Table (pipe format)
        if stripped.startswith("|"):
            table_lines = []
            while i < len(lines) and lines[i].strip().startswith("|"):
                table_lines.append(lines[i])
                i += 1

            raw_table = "\n".join(table_lines)
            table_data = parse_pipe_table(raw_table)
            if table_data:
                blocks.append(ParsedBlock(
                    type="table",
                    content=table_data,
                    raw=raw_table
                ))
            continue

        # Unordered list
        if stripped.startswith(("- ", "* ", "+ ")):
            list_lines = []
            while i < len(lines):
                current = lines[i].strip()
                if not current:
                    i += 1
                    continue
                if current.startswith(("- ", "* ", "+ ")):
                    list_lines.append(current[2:].strip())
                    i += 1
                elif current.startswith("  ") and list_lines:
                    # Continuation of list item
                    list_lines[-1] += " " + current.strip()
                    i += 1
                else:
                    break

            if list_lines:
                raw_list = "\n".join([f"- {item}" for item in list_lines])
                blocks.append(ParsedBlock(
                    type="list",
                    content={"items": list_lines, "ordered": False},
                    raw=raw_list
                ))
            continue

        # Ordered list
        ordered_match = re.match(r"^(\d+)\.\s+(.+)", stripped)
        if ordered_match:
            list_lines = []
            while i < len(lines):
                current = lines[i].strip()
                if not current:
                    i += 1
                    continue
                m = re.match(r"^(\d+)\.\s+(.+)", current)
                if m:
                    list_lines.append(m.group(2).strip())
                    i += 1
                elif current.startswith("  ") and list_lines:
                    list_lines[-1] += " " + current.strip()
                    i += 1
                else:
                    break

            if list_lines:
                raw_list = "\n".join([f"{idx+1}. {item}" for idx, item in enumerate(list_lines)])
                blocks.append(ParsedBlock(
                    type="list",
                    content={"items": list_lines, "ordered": True},
                    raw=raw_list
                ))
            continue

        # Figure / Image
        img_match = re.match(r"^!\[(.*?)\]\((.*?)\)", stripped)
        if img_match:
            alt_text = img_match.group(1)
            url = img_match.group(2)
            raw = stripped
            blocks.append(ParsedBlock(
                type="figure",
                content={
                    "alt_text": alt_text,
                    "source_url": url,
                    "description": None  # Nova on-demand only
                },
                raw=raw
            ))
            i += 1
            continue

        # Key-Value pattern (bold key + colon + value)
        kv_match = re.match(r"^\*\*(.+?)\*\*:\s*(.+)", stripped)
        if kv_match:
            key = kv_match.group(1).strip()
            value = kv_match.group(2).strip()
            raw = stripped
            blocks.append(ParsedBlock(
                type="key_value",
                content={"key": key, "value": value},
                raw=raw
            ))
            i += 1
            continue

        # Paragraph (default)
        para_lines = [stripped]
        i += 1
        while i < len(lines):
            current = lines[i].strip()
            if not current:
                i += 1
                break
            # Stop at block-level elements
            if current.startswith(("#", "|", "- ", "* ", "+ ", "!")):
                break
            if re.match(r"^\d+\.\s+", current):
                break
            if re.match(r"^\*\*(.+?)\*\*:\s*", current):
                break
            para_lines.append(current)
            i += 1

        raw_para = " ".join(para_lines)
        # Strip inline markdown for content, keep raw for exact reproduction
        content_text = strip_inline_markdown(raw_para)
        blocks.append(ParsedBlock(
            type="paragraph",
            content=content_text,
            raw=raw_para
        ))

    return blocks, metadata


def parse_pipe_table(raw_table: str) -> Optional[Dict[str, Any]]:
    """
    Parse pipe-delimited Markdown table into structured data.
    Handles basic tables. Merged cells will create empty strings.
    """
    lines = [line.strip() for line in raw_table.splitlines() if line.strip().startswith("|")]
    if len(lines) < 2:
        return None

    # Filter out separator line (| --- | --- |)
    data_lines = []
    for line in lines:
        cells = [c.strip() for c in line.strip("|").split("|")]
        cells = [c for c in cells if c or c == ""]  # Keep empty cells
        # Check if this is a separator line
        is_separator = all(re.match(r"^[-:]+$", c.strip()) for c in cells if c.strip())
        if not is_separator:
            data_lines.append(cells)

    if not data_lines:
        return None

    return {
        "columns": data_lines[0],
        "rows": data_lines[1:],
        "row_count": len(data_lines) - 1,
        "col_count": len(data_lines[0]) if data_lines else 0
    }


def strip_inline_markdown(text: str) -> str:
    """
    Strip inline Markdown formatting for clean LLM consumption.
    Preserves raw Markdown in block.raw for exact reconstruction.
    """
    # Bold **text** or __text__
    text = re.sub(r"\*\*(.+?)\*\*", r"\1", text)
    text = re.sub(r"__(.+?)__", r"\1", text)
    # Italic *text* or _text_
    text = re.sub(r"\*(.+?)\*", r"\1", text)
    text = re.sub(r"_(.+?)_", r"\1", text)
    # Inline code `text`
    text = re.sub(r"`(.+?)`", r"\1", text)
    # Links [text](url) → keep text only
    text = re.sub(r"\[(.+?)\]\(.*?\)", r"\1", text)
    # Strikethrough ~~text~~
    text = re.sub(r"~~(.+?)~~", r"\1", text)
    return text.strip()


# ============================================================================
# Section Builder — Heading Boundaries
# ============================================================================
def build_sections_from_blocks(
    blocks: List[ParsedBlock],
    md_text: str,
    metadata: Dict[str, Any],
    source_key: str
) -> List[Dict[str, Any]]:
    """
    Split blocks into sections at heading boundaries.

    #   → document title (not a section)
    ##  → section node
    ### → nested section node (child)
    """
    lines = md_text.splitlines()
    sections = []
    current_title = "Untitled"
    current_level = 0  # 0 = root, 2 = ##, 3 = ###
    current_blocks: List[ParsedBlock] = []
    current_raw_lines: List[str] = []
    section_counter = 0

    # Extract document title from first # heading
    doc_title = "Untitled"
    for line in lines:
        if line.strip().startswith("# "):
            doc_title = line.strip()[2:].strip()
            break

    i = 0
    while i < len(lines):
        line = lines[i]
        stripped = line.strip()

        # Detect heading
        if stripped.startswith("## "):
            # Save previous section
            if current_blocks:
                sections.append({
                    "node_id": f"S{section_counter:04d}",
                    "title": current_title,
                    "level": current_level,
                    "blocks": current_blocks,
                    "raw_markdown": "\n".join(current_raw_lines).strip(),
                    "metadata": {
                        "source_key": source_key,
                        **metadata
                    }
                })
                section_counter += 1

            current_title = stripped[3:].strip()
            current_level = 2
            current_blocks = []
            current_raw_lines = []
            i += 1
            continue

        if stripped.startswith("### "):
            # Save previous section
            if current_blocks:
                sections.append({
                    "node_id": f"S{section_counter:04d}",
                    "title": current_title,
                    "level": current_level,
                    "blocks": current_blocks,
                    "raw_markdown": "\n".join(current_raw_lines).strip(),
                    "metadata": {
                        "source_key": source_key,
                        **metadata
                    }
                })
                section_counter += 1

            current_title = stripped[4:].strip()
            current_level = 3
            current_blocks = []
            current_raw_lines = []
            i += 1
            continue

        # Skip title heading (already captured as doc_title)
        if stripped.startswith("# "):
            i += 1
            continue

        # Skip comments and empty lines
        if not stripped or stripped.startswith("<!--"):
            i += 1
            continue

        # Find which block this line belongs to
        # We rebuild blocks from lines to match with existing blocks
        current_raw_lines.append(line)
        i += 1

    # Save final section
    if current_blocks or current_raw_lines:
        sections.append({
            "node_id": f"S{section_counter:04d}",
            "title": current_title,
            "level": current_level,
            "blocks": current_blocks,
            "raw_markdown": "\n".join(current_raw_lines).strip(),
            "metadata": {
                "source_key": source_key,
                **metadata
            }
        })

    return sections, doc_title

def blocks_to_sequence(blocks: List[ParsedBlock]) -> List[Dict[str, Any]]:
    """Convert ParsedBlock list to JSON-serializable sequence."""
    return [
        {
            "type": b.type,
            "content": b.content,
            "raw": b.raw
        }
        for b in blocks
    ]

# Tree Index Builder
# ============================================================================
def build_tree_index_for_document(
    source_key: str,
    doc_id: str,
    md_text: str
) -> Dict[str, Any]:
    """
    Build tree index for a single Markdown document.

    Returns document node with sections as children.
    """
    log.info("Building tree index for %s", doc_id)

    # Parse blocks and metadata
    blocks, metadata = parse_markdown_to_blocks(md_text)

    # Build sections from headings
    sections_data, doc_title = build_sections_from_blocks(blocks, md_text, metadata, source_key)

    # Build nodes from sections
    nodes = []
    for sec in sections_data:
        # Build sequence from blocks
        sequence = blocks_to_sequence(sec["blocks"])

        # Build full text for summarization (concatenate content)
        text_parts = []
        for item in sequence:
            if item["type"] == "paragraph":
                text_parts.append(item["content"])
            elif item["type"] == "table":
                text_parts.append(f"[TABLE: {item['content'].get('row_count', 0)} rows x {item['content'].get('col_count', 0)} cols]")
            elif item["type"] == "list":
                text_parts.append(f"[LIST: {len(item['content'].get('items', []))} items]")
            elif item["type"] == "figure":
                text_parts.append(f"[FIGURE: {item['content'].get('alt_text', 'no description')}]")
            elif item["type"] == "key_value":
                text_parts.append(f"{item['content']['key']}: {item['content']['value']}")

        full_text = "\n".join(text_parts)

        # Generate summary
        summary = summarize_with_claude(sec["title"], full_text)

        nodes.append({
            "node_id": f"{doc_id}_{sec['node_id']}",
            "title": sec["title"],
            "level": sec["level"],
            "summary": summary,
            "full_text": full_text,
            "sequence": sequence,
            "raw_markdown": sec["raw_markdown"],
            "metadata": sec["metadata"]
        })

    return {
        "doc_id": doc_id,
        "doc_title": doc_title,
        "source_key": source_key,
        "source_type": "markdown",
        "processed_at": datetime.now(timezone.utc).isoformat(),
        "section_count": len(nodes),
        "nodes": nodes
    }

def process_all_documents() -> Dict[str, Any]:
    """
    Process all Markdown files from S3 and build unified tree index.

    Reads from:
        - INPUT_BUCKET/TEXTRACT_MD_PREFIX (Textract .md files)
        - INPUT_BUCKET/MD_PREFIX (Scrapy HTML→Markdown files)

    Outputs:
        - tree_index_{timestamp}.json to OUTPUT_BUCKET/OUTPUT_PREFIX
    """
    log.info("=" * 60)
    log.info("Tree Index Builder — Markdown Input")
    log.info("Input buckets: s3://%s/%s, s3://%s/%s", 
             INPUT_BUCKET, TEXTRACT_MD_PREFIX, INPUT_BUCKET, MD_PREFIX)
    log.info("Output: s3://%s/%s", OUTPUT_BUCKET, OUTPUT_PREFIX)
    log.info("=" * 60)

    # Collect all Markdown keys
    textract_keys = list_md_keys(INPUT_BUCKET, TEXTRACT_MD_PREFIX)
    scrapy_keys = list_md_keys(INPUT_BUCKET, MD_PREFIX)
    all_keys = textract_keys + scrapy_keys

    log.info("Found %d Textract .md files, %d Scrapy .md files", 
             len(textract_keys), len(scrapy_keys))

    documents = []
    for key in all_keys:
        try:
            md_text = load_s3_text(INPUT_BUCKET, key)
            doc_id = Path(key).stem

            tree = build_tree_index_for_document(key, doc_id, md_text)
            documents.append(tree)

            log.info("Document %s: %d sections", doc_id, tree["section_count"])

        except Exception as exc:
            log.exception("Failed to process %s: %s", key, exc)

    # Build unified tree index
    unified_tree = {
        "index_type": "tree",
        "processed_at": datetime.now(timezone.utc).isoformat(),
        "document_count": len(documents),
        "documents": documents
    }

    # Save to S3
    timestamp = datetime.now(timezone.utc).strftime("%Y%m%d_%H%M%S")
    tree_key = f"{OUTPUT_PREFIX.rstrip('/')}/tree_index_{timestamp}.json"
    tree_uri = save_s3_json(OUTPUT_BUCKET, tree_key, unified_tree)

    total_sections = sum(d["section_count"] for d in documents)

    log.info("=" * 60)
    log.info("Tree index complete: %d documents, %d sections", 
             len(documents), total_sections)
    log.info("Output: %s", tree_uri)
    log.info("=" * 60)

    return {
        "status": "SUCCEEDED",
        "tree_index_uri": tree_uri,
        "document_count": len(documents),
        "total_sections": total_sections,
        "documents": [d["doc_id"] for d in documents]
    }

def lambda_handler(event: Dict[str, Any], context: Any) -> Dict[str, Any]:
    """
    AWS Lambda entry point.

    Environment variables (or event overrides):
        INPUT_BUCKET, TEXTRACT_MD_PREFIX, MD_PREFIX
        OUTPUT_BUCKET, OUTPUT_PREFIX
        CLAUDE_MODEL_ID, AWS_REGION
    """
    global REGION, INPUT_BUCKET, TEXTRACT_MD_PREFIX, MD_PREFIX
    global OUTPUT_BUCKET, OUTPUT_PREFIX, CLAUDE_MODEL_ID
    global _s3_client, _bedrock_client

    REGION = event.get("aws_region", REGION)
    INPUT_BUCKET = event.get("input_bucket", INPUT_BUCKET)
    TEXTRACT_MD_PREFIX = event.get("textract_md_prefix", TEXTRACT_MD_PREFIX)
    MD_PREFIX = event.get("md_prefix", MD_PREFIX)
    OUTPUT_BUCKET = event.get("output_bucket", OUTPUT_BUCKET)
    OUTPUT_PREFIX = event.get("output_prefix", OUTPUT_PREFIX)
    CLAUDE_MODEL_ID = event.get("claude_model_id", CLAUDE_MODEL_ID)

    _s3_client = None
    _bedrock_client = None

    return process_all_documents()

======================================================================
Processes PDFs through AWS Textract (LAYOUT + TABLES + FORMS).
If NOVA_ENABLED=true, detected figures are analyzed via Nova Lite and
descriptions are embedded into the markdown output.

Outputs per PDF:
    - .json : Full Textract block data + sequence
    - .md   : Markdown with embedded figure descriptions (if Nova enabled)
    - .txt  : Plain text version

Environment:
    NOVA_ENABLED=true|false  (default: false)
"""

def list_pdf_keys(bucket: str, prefix: str) -> List[str]:
    s3 = get_s3_client()
    paginator = s3.get_paginator("list_objects_v2")
    keys = []
    for page in paginator.paginate(Bucket=bucket, Prefix=prefix):
        for obj in page.get("Contents", []):
            k = obj["Key"]
            if k.lower().endswith(".pdf"):
                keys.append(k)
    return keys


def load_s3_bytes(bucket: str, key: str) -> bytes:
    s3 = get_s3_client()
    return s3.get_object(Bucket=bucket, Key=key)["Body"].read()


def save_s3_object(bucket: str, key: str, body: bytes, content_type: str):
    s3 = get_s3_client()
    s3.put_object(Bucket=bucket, Key=key, Body=body, ContentType=content_type)
    return f"s3://{bucket}/{key}"

def start_textract_job(bucket: str, key: str) -> str:
    textract = get_textract_client()
    resp = textract.start_document_analysis(
        DocumentLocation={"S3Object": {"Bucket": bucket, "Name": key}},
        FeatureTypes=["LAYOUT", "TABLES", "FORMS"],
    )
    job_id = resp["JobId"]
    log.info("Started Textract job %s for s3://%s/%s", job_id, bucket, key)
    return job_id


def wait_for_textract_job(job_id: str) -> Dict[str, Any]:
    textract = get_textract_client()
    started = time.time()

    while True:
        resp = textract.get_document_analysis(JobId=job_id)
        status = resp["JobStatus"]
        if status in ("SUCCEEDED", "FAILED"):
            break
        if time.time() - started > TEXTRACT_MAX_POLL_SECONDS:
            log.error("Job %s timed out", job_id)
            return {"JobStatus": "TIMEOUT", "Blocks": []}
        time.sleep(TEXTRACT_POLL_SECONDS)

    if status != "SUCCEEDED":
        log.error("Job %s failed: %s", job_id, resp.get("StatusMessage", "Unknown"))
        return resp

    blocks = list(resp.get("Blocks", []))
    next_token = resp.get("NextToken")
    while next_token:
        page = textract.get_document_analysis(JobId=job_id, NextToken=next_token)
        blocks.extend(page.get("Blocks", []))
        next_token = page.get("NextToken")

    resp["Blocks"] = blocks
    log.info("Job %s completed: %d blocks", job_id, len(blocks))
    return resp


def index_blocks(blocks: List[Dict[str, Any]]) -> Dict[str, Dict[str, Any]]:
    return {b["Id"]: b for b in blocks}


def extract_cell_text(cell: Dict[str, Any], by_id: Dict[str, Dict[str, Any]]) -> str:
    words = []
    for rel in cell.get("Relationships", []):
        if rel["Type"] == "CHILD":
            for wid in rel["Ids"]:
                child = by_id.get(wid)
                if not child:
                    continue
                if child["BlockType"] == "WORD":
                    words.append(child.get("Text", ""))
                elif child["BlockType"] == "SELECTION_ELEMENT":
                    words.append("[x]" if child.get("SelectionStatus") == "SELECTED" else "[ ]")
    return " ".join(words).strip()


def grid_to_markdown(grid: List[List[str]]) -> str:
    if not grid:
        return ""
    lines = ["| " + " | ".join(grid[0]) + " |"]
    lines.append("|" + "|".join([" --- "] * len(grid[0])) + "|")
    for row in grid[1:]:
        lines.append("| " + " | ".join(row) + " |")
    return "\n".join(lines)


def extract_tables(blocks: List[Dict[str, Any]]) -> List[Dict[str, Any]]:
    by_id = index_blocks(blocks)
    tables = []
    for block in blocks:
        if block["BlockType"] != "TABLE":
            continue
        cells = []
        for rel in block.get("Relationships", []):
            if rel["Type"] == "CHILD":
                for cid in rel["Ids"]:
                    child = by_id.get(cid)
                    if child and child["BlockType"] == "CELL":
                        cells.append(child)
        if not cells:
            continue
        max_row = max(c["RowIndex"] for c in cells)
        max_col = max(c["ColumnIndex"] for c in cells)
        grid = [["" for _ in range(max_col)] for _ in range(max_row)]
        for cell in cells:
            text = extract_cell_text(cell, by_id)
            r, c = cell["RowIndex"] - 1, cell["ColumnIndex"] - 1
            grid[r][c] = text
        tables.append({
            "table_id": block["Id"],
            "page": block.get("Page", 1),
            "rows": max_row,
            "cols": max_col,
            "markdown": grid_to_markdown(grid),
            "grid": grid,
        })
    return tables


def extract_forms(blocks: List[Dict[str, Any]]) -> List[Dict[str, Any]]:
    by_id = index_blocks(blocks)
    fields = []
    for block in blocks:
        if block["BlockType"] != "KEY_VALUE_SET":
            continue
        if "KEY" not in (block.get("EntityTypes") or []):
            continue
        key_text = extract_cell_text(block, by_id)
        value_id = None
        for rel in block.get("Relationships", []):
            if rel["Type"] == "VALUE" and rel.get("Ids"):
                value_id = rel["Ids"][0]
                break
        value_text = ""
        if value_id:
            value_block = by_id.get(value_id)
            if value_block:
                value_text = extract_cell_text(value_block, by_id)
        if key_text:
            fields.append({
                "key_id": block["Id"],
                "page": block.get("Page", 1),
                "key": key_text.rstrip(":").rstrip(),
                "value": value_text,
            })
    return fields


def detect_figures(blocks: List[Dict[str, Any]]) -> List[Dict[str, Any]]:
    figures = []
    for block in blocks:
        if block["BlockType"] == "LAYOUT_FIGURE":
            figures.append({
                "figure_id": block["Id"],
                "page": block.get("Page", 1),
                "bbox": block["Geometry"]["BoundingBox"],
            })
    return figures


def bbox_overlap(box_a: Dict[str, float], box_b: Dict[str, float]) -> float:
    ax0, ay0 = box_a["Left"], box_a["Top"]
    ax1, ay1 = ax0 + box_a["Width"], ay0 + box_a["Height"]
    bx0, by0 = box_b["Left"], box_b["Top"]
    bx1, by1 = bx0 + box_b["Width"], by0 + box_b["Height"]
    ix = max(0.0, min(ax1, bx1) - max(ax0, bx0))
    iy = max(0.0, min(ay1, by1) - max(ay0, by0))
    intersection = ix * iy
    union = box_a["Width"] * box_a["Height"] + box_b["Width"] * box_b["Height"] - intersection
    return intersection / union if union > 0 else 0.0


def build_sequence(blocks: List[Dict[str, Any]], tables: List[Dict], forms: List[Dict]) -> List[Dict[str, Any]]:
    by_id = index_blocks(blocks)
    tables_by_id = {t["table_id"]: t for t in tables}
    forms_by_id = {f["key_id"]: f for f in forms}

    text_kind_map = {
        "LAYOUT_TITLE": "title",
        "LAYOUT_SECTION_HEADER": "section_header",
        "LAYOUT_TEXT": "paragraph",
    }

    sequence = []
    matched_tables = set()
    matched_forms = set()

    for block in blocks:
        btype = block["BlockType"]
        page = block.get("Page", 1)

        if btype in text_kind_map:
            line_ids = [cid for rel in block.get("Relationships", [])
                       if rel["Type"] == "CHILD" for cid in rel["Ids"]]
            lines = [by_id[lid]["Text"] for lid in line_ids
                    if by_id.get(lid, {}).get("BlockType") == "LINE"]
            text = " ".join(lines).strip()
            if text:
                sequence.append({"type": text_kind_map[btype], "page": page, "text": text})

        elif btype == "LAYOUT_LIST":
            for rel in block.get("Relationships", []):
                if rel["Type"] == "CHILD":
                    for lid in rel["Ids"]:
                        line = by_id.get(lid)
                        if line and line.get("BlockType") == "LINE" and line.get("Text", "").strip():
                            sequence.append({"type": "list_item", "page": page, "text": line["Text"].strip()})

        elif btype == "LAYOUT_TABLE":
            layout_box = block["Geometry"]["BoundingBox"]
            best_id = None
            best_overlap = 0.0
            for candidate in blocks:
                if candidate["BlockType"] != "TABLE":
                    continue
                if candidate.get("Page", 1) != page:
                    continue
                overlap = bbox_overlap(layout_box, candidate["Geometry"]["BoundingBox"])
                if overlap > best_overlap:
                    best_overlap = overlap
                    best_id = candidate["Id"]
            if best_id and best_id in tables_by_id:
                matched_tables.add(best_id)
                t = tables_by_id[best_id]
                sequence.append({
                    "type": "table", "page": page, "table_id": best_id,
                    "markdown": t["markdown"], "grid": t["grid"],
                    "rows": t["rows"], "cols": t["cols"],
                })

        elif btype == "LAYOUT_KEY_VALUE":
            layout_box = block["Geometry"]["BoundingBox"]
            best_id = None
            best_overlap = 0.0
            for candidate in blocks:
                if candidate["BlockType"] != "KEY_VALUE_SET":
                    continue
                if "KEY" not in (candidate.get("EntityTypes") or []):
                    continue
                if candidate.get("Page", 1) != page:
                    continue
                overlap = bbox_overlap(layout_box, candidate["Geometry"]["BoundingBox"])
                if overlap > best_overlap:
                    best_overlap = overlap
                    best_id = candidate["Id"]
            if best_id and best_id in forms_by_id:
                matched_forms.add(best_id)
                f = forms_by_id[best_id]
                sequence.append({
                    "type": "key_value", "page": page,
                    "key": f["key"], "value": f["value"],
                })

        elif btype == "LAYOUT_FIGURE":
            sequence.append({
                "type": "figure", "page": page,
                "figure_id": block["Id"],
                "bbox": block["Geometry"]["BoundingBox"],
                "description": None,  # Filled by Nova if enabled
            })

    # Append unmatched
    for t in tables:
        if t["table_id"] not in matched_tables:
            sequence.append({
                "type": "table", "page": t["page"],
                "table_id": t["table_id"], "markdown": t["markdown"],
                "grid": t["grid"], "rows": t["rows"], "cols": t["cols"],
            })
    for f in forms:
        if f["key_id"] not in matched_forms:
            sequence.append({
                "type": "key_value", "page": f["page"],
                "key": f["key"], "value": f["value"],
            })

    return sequence

def bedrock_converse(model_id: str, prompt: str, image_bytes: Optional[bytes] = None, max_tokens: int = 1024) -> str:
    client = get_bedrock_client()
    content = [{"text": prompt}]
    if image_bytes:
        import base64
        b64_image = base64.b64encode(image_bytes).decode("utf-8")
        content.append({
            "image": {
                "format": "png",
                "source": {"bytes": b64_image}
            }
        })
    try:
        resp = client.converse(
            modelId=model_id,
            messages=[{"role": "user", "content": content}],
            inferenceConfig={"maxTokens": max_tokens, "temperature": 0.0}
        )
        return resp["output"]["message"]["content"][0]["text"]
    except Exception as exc:
        log.error("Bedrock error: %s", exc)
        return ""


def analyze_figure_with_nova(pdf_bytes: bytes, figure: Dict[str, Any], doc_context: str = "") -> str:
    """
    Crop figure from PDF using bounding box, send to Nova for analysis.
    """
    page_num = figure["page"] - 1
    bbox = figure["bbox"]

    doc = fitz.open(stream=pdf_bytes, filetype="pdf")
    try:
        page = doc[page_num]
        page_rect = page.rect

        x0 = bbox["Left"] * page_rect.width
        y0 = bbox["Top"] * page_rect.height
        x1 = x0 + bbox["Width"] * page_rect.width
        y1 = y0 + bbox["Height"] * page_rect.height

        clip_rect = fitz.Rect(x0, y0, x1, y1)
        pix = page.get_pixmap(clip=clip_rect, dpi=200)
        img_bytes = pix.tobytes("png")

        prompt = (
            f"You are analyzing a figure from an ECMC bankruptcy document. "
            f"Document context: {doc_context[:500]}\n\n"
            f"Analyze this image and describe:\n"
            f"1. What type of content is this? (screenshot, diagram, form, table-image, flowchart, etc.)\n"
            f"2. If it contains UI elements, list them with labels and purposes.\n"
            f"3. If it contains a form or table, extract all visible fields/data.\n"
            f"4. If it is a flowchart or diagram, describe the process step-by-step.\n"
            f"5. If it is a software screenshot, describe user actions.\n\n"
            f"Return structured text. Be specific."
        )

        return bedrock_converse(NOVA_MODEL_ID, prompt, image_bytes=img_bytes, max_tokens=2048)

    finally:
        doc.close()


def enhance_figures_with_nova(pdf_bytes: bytes, sequence: List[Dict[str, Any]]) -> List[Dict[str, Any]]:
    """
    If NOVA_ENABLED=true, analyze all figure blocks and embed descriptions.
    If false, leave descriptions as None (converted to placeholder in markdown).
    """
    if not NOVA_ENABLED:
        log.info("NOVA_ENABLED=false, skipping figure analysis")
        for item in sequence:
            if item["type"] == "figure":
                item["description"] = None
        return sequence

    figures = [item for item in sequence if item["type"] == "figure"]
    if not figures:
        log.info("No figures detected, Nova not needed")
        return sequence

    log.info("NOVA_ENABLED=true, analyzing %d figures with Nova", len(figures))

    # Build context from surrounding text
    context_parts = []
    for item in sequence:
        if item["type"] in ("title", "section_header", "paragraph"):
            context_parts.append(item.get("text", ""))
    doc_context = " ".join(context_parts)[:1000]

    for item in figures:
        try:
            description = analyze_figure_with_nova(pdf_bytes, item, doc_context)
            item["description"] = description
            item["nova_analyzed"] = True
            log.info("Figure page %d analyzed successfully", item["page"])
        except Exception as e:
            log.warning("Failed to analyze figure page %d: %s", item["page"], e)
            item["description"] = f"[Figure analysis failed: {str(e)}]"
            item["nova_analyzed"] = False

    return sequence

def sequence_to_markdown(sequence: List[Dict[str, Any]]) -> str:
    lines = []
    for item in sequence:
        kind = item["type"]
        if kind == "title":
            lines.append(f"# {item['text']}")
        elif kind == "section_header":
            lines.append(f"## {item['text']}")
        elif kind == "paragraph":
            lines.append(item["text"])
        elif kind == "list_item":
            lines.append(f"- {item['text']}")
        elif kind == "table":
            lines.append(item["markdown"])
        elif kind == "key_value":
            lines.append(f"**{item['key']}:** {item['value']}")
        elif kind == "figure":
            desc = item.get("description")
            if desc:
                lines.append(f"**[Figure — page {item['page']}]**\n{desc}")
            else:
                lines.append(f"**[Figure — page {item['page']}]**")
        lines.append("")
    return "\n".join(lines).strip()


def sequence_to_text(sequence: List[Dict[str, Any]]) -> str:
    lines = []
    current_page = 0
    for item in sequence:
        page = item.get("page", 1)
        if page != current_page:
            lines.append(f"[PAGE {page}]")
            current_page = page

        kind = item["type"]
        if kind == "list_item":
            lines.append(f"  - {item['text']}")
        elif kind in ("title", "section_header", "paragraph"):
            lines.append(item["text"])
        elif kind == "table":
            lines.append("[TABLE]")
        elif kind == "key_value":
            lines.append(f"{item['key']}: {item['value']}")
        elif kind == "figure":
            desc = item.get("description")
            if desc:
                lines.append(f"[FIGURE: page {page}] {desc}")
            else:
                lines.append(f"[FIGURE: page {page}]")
        lines.append("")
    return "\n".join(lines).strip()

def process_single_pdf(bucket: str, key: str) -> Dict[str, Any]:
    log.info("Processing PDF: s3://%s/%s", bucket, key)
    timestamp = datetime.now(timezone.utc).strftime("%Y%m%d_%H%M%S")
    doc_id = Path(key).stem

    # Load PDF bytes for Nova cropping
    pdf_bytes = load_s3_bytes(bucket, key)

    # Start Textract
    job_id = start_textract_job(bucket, key)

    # Wait for completion
    result = wait_for_textract_job(job_id)
    status = result["JobStatus"]

    if status != "SUCCEEDED":
        return {
            "source_key": key,
            "job_id": job_id,
            "status": status,
            "error": result.get("StatusMessage", "Unknown"),
        }

    blocks = result["Blocks"]

    # Extract components
    tables = extract_tables(blocks)
    forms = extract_forms(blocks)
    figures = detect_figures(blocks)
    sequence = build_sequence(blocks, tables, forms)

    # Enhance figures with Nova if enabled
    sequence = enhance_figures_with_nova(pdf_bytes, sequence)

    # Format outputs
    markdown = sequence_to_markdown(sequence)
    plain_text = sequence_to_text(sequence)

    # Build JSON data
    output_data = {
        "source_bucket": bucket,
        "source_key": key,
        "job_id": job_id,
        "processed_at": datetime.now(timezone.utc).isoformat(),
        "nova_enabled": NOVA_ENABLED,
        "block_count": len(blocks),
        "table_count": len(tables),
        "form_count": len(forms),
        "figure_count": len(figures),
        "figures_analyzed": sum(1 for item in sequence if item.get("nova_analyzed", False)),
        "sequence": sequence,
        "tables": tables,
        "forms": forms,
        "figures": figures,
    }

    # Save to S3
    safe_stem = re.sub(r"[^\w\-]", "_", doc_id)
    base_prefix = f"{OUTPUT_PREFIX.rstrip('/')}/{safe_stem}_{timestamp}"

    json_uri = save_s3_object(

    )

        "figures_analyzed": output_data["figures_analyzed"],

    }

def canon_language(s: str) -> str:
    if not s:
        return ""
    cleaned = re.sub(r"[^\w\-]", "", s.strip().lower())
    for canon, aliases in _LANG_ALIASES.items():
        if cleaned in aliases:
            return canon
    if cleaned in _LANG_CODE_TO_NAME:
        return _LANG_CODE_TO_NAME[cleaned]
    return cleaned


def is_excluded_language(detected: str, exclude_list: List[str]) -> bool:
    if not exclude_list:
        return False
    detected_canon = canon_language(detected)
    if not detected_canon:
        return False
    excluded_canon = {canon_language(x) for x in exclude_list}
    return detected_canon in excluded_canon


def generate_folder_name(url: str) -> str:
    parsed = urlparse(url)
    path = parsed.path.strip("/")
    timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
    if path:
        last = path.split("/")[-1] if "/" in path else path
        last = re.sub(r"\.(html|htm|php|asp|aspx|jsp)$", "", last, flags=re.IGNORECASE)
        clean_name = re.sub(r"[^a-zA-Z0-9]+", "_", last).strip("_")[:80]
        if not clean_name:
            clean_name = re.sub(r"[^a-zA-Z0-9]+", "_", parsed.netloc).strip("_")
    else:
        clean_name = re.sub(r"[^a-zA-Z0-9]+", "_", parsed.netloc).strip("_")
    return f"{clean_name}_{timestamp}"

def detect_language(text: str) -> Tuple[str, float]:
    if not text or len(text.strip()) < 20:
        return "unknown", 0.0
    try:
        langs = detect_langs(text[:2000])
        if not langs:
            return "unknown", 0.0
        top = langs[0]
        return canon_language(top.lang), float(top.prob)
    except LangDetectException:
        return "unknown", 0.0

def html_to_markdown(html: str, source_url: str) -> str:
    soup = BeautifulSoup(html, "html.parser")
    for tag in soup(["script", "style", "noscript", "meta", "link",
                     "header", "footer", "nav", "aside", "form", "iframe"]):
        tag.decompose()
    content = soup.find("main") or soup.find("article") or soup.body or soup
    if content is None:
        return ""
    md_text = md(
        str(content),
        heading_style="ATX",
        strip=["a"],
        bullets="-",
    )
    md_text = re.sub(r"\n{3,}", "\n\n", md_text).strip()
    return f"<!-- source_url: {source_url} -->\n\n{md_text}"

def html_to_pdf(html_content: str, css_styles: str = None, output_path: str = None) -> bytes:
    """
    Convert HTML content to PDF using PyMuPDF (fitz).
    Zero system dependencies — fitz is already imported in this project.
    
    Args:
        html_content: HTML string to convert
        css_styles: Optional CSS for basic styling
        output_path: Optional file path to save PDF
    
    Returns:
        PDF bytes
    """
    if css_styles is None:
        css_styles = """
        body { font-family: Arial, sans-serif; font-size: 12px; line-height: 1.5; color: #333; }
        h1 { font-size: 18px; color: #222; margin-top: 20px; }
        h2 { font-size: 16px; color: #444; margin-top: 15px; }
        h3 { font-size: 14px; color: #555; margin-top: 10px; }
        p { margin: 8px 0; }
        table { border-collapse: collapse; width: 100%; margin: 10px 0; }
        td, th { border: 1px solid #ddd; padding: 6px; font-size: 11px; }
        th { background-color: #f5f5f5; }
        ul, ol { margin: 8px 0; padding-left: 20px; }
        li { margin: 4px 0; }
        a { color: #0066cc; text-decoration: underline; }
        """
    
    full_html = f"""<!DOCTYPE html>
<html>
<head>
    <meta charset="UTF-8">
    <style>{css_styles}</style>
</head>
<body>{html_content}</body>
</html>"""
    
    doc = fitz.open()
    page = doc.new_page(width=595, height=842)  # A4 in points
    
    # PyMuPDF's insert_htmlbox renders HTML with basic CSS support
    rect = fitz.Rect(50, 50, 545, 792)  # Margins: ~17mm all sides
    page.insert_htmlbox(rect, full_html, css=css_styles)
    
    pdf_bytes = doc.tobytes()
    doc.close()
    
    if output_path:
        with open(output_path, "wb") as f:
            f.write(pdf_bytes)
    
    return pdf_bytes

def classify_pdf_type(pdf_bytes: bytes) -> str:
    doc = fitz.open(stream=pdf_bytes, filetype="pdf")
    pages_with_text = 0
    pages_without_text = 0
    try:
        for page in doc:
            text = (page.get_text("text") or "").strip()
            has_real_text = False
            if len(text) >= 30:
                alnum = sum(c.isalnum() for c in text)
                if alnum / len(text) >= 0.20:
                    has_real_text = True
            if has_real_text:
                pages_with_text += 1
            else:
                pages_without_text += 1
    finally:
        doc.close()
    total = pages_with_text + pages_without_text
    if total == 0:
        return "empty"
    if pages_with_text == total:
        return "searchable"
    if pages_without_text == total:
        return "image"
    return "mixed"

class BoundingBox:
    """Helper to manipulate Textract bounding boxes for PyMuPDF overlay."""
    
    def __init__(self, left: float, bottom: float, right: float, top: float) -> None:
        self.bounds = [left, bottom, right, top]
    
    @classmethod
    def from_textract_bbox(cls, textract_bbox: Dict[str, float]) -> "BoundingBox":
        return cls(
            left=textract_bbox["Left"],
            bottom=textract_bbox["Top"] + textract_bbox["Height"],
            right=textract_bbox["Left"] + textract_bbox["Width"],
            top=textract_bbox["Top"],
        )
    
    def scale(self, x_scale: float, y_scale: Optional[float] = None) -> None:
        if y_scale is None:
            y_scale = x_scale
        self.bounds[0] *= x_scale
        self.bounds[1] *= y_scale
        self.bounds[2] *= x_scale
        self.bounds[3] *= y_scale
    
    @property
    def left(self) -> float:
        return self.bounds[0]
    
    @property
    def bottom(self) -> float:
        return self.bounds[1]
    
    @property
    def right(self) -> float:
        return self.bounds[2]
    
    @property
    def top(self) -> float:
        return self.bounds[3]
    
    @property
    def width(self) -> float:
        return abs(self.bounds[0] - self.bounds[2])
    
    @property
    def height(self) -> float:
        return abs(self.bounds[3] - self.bounds[1])


def _get_textract_client(region: str):
    return boto3.client("textract", region_name=region)


def _wait_for_textract_job(textract, job_id: str) -> Dict[str, any]:
    """Poll Textract async job until completion or timeout."""
    started = time.time()
    while True:
        resp = textract.get_document_text_detection(JobId=job_id)
        status = resp["JobStatus"]
        if status in ("SUCCEEDED", "FAILED"):
            break
        if time.time() - started > TEXTRACT_MAX_POLL_SECONDS:
            log.error("Textract job %s timed out after %d seconds", job_id, TEXTRACT_MAX_POLL_SECONDS)
            return {"JobStatus": "TIMEOUT", "Blocks": []}
        time.sleep(TEXTRACT_POLL_SECONDS)
    
    if status != "SUCCEEDED":
        log.error("Textract job %s failed: %s", job_id, resp.get("StatusMessage", "Unknown"))
        return resp
    
    blocks = list(resp.get("Blocks", []))
    next_token = resp.get("NextToken")
    while next_token:
        page = textract.get_document_text_detection(JobId=job_id, NextToken=next_token)
        blocks.extend(page.get("Blocks", []))
        next_token = page.get("NextToken")
    
    resp["Blocks"] = blocks
    log.info("Textract job %s completed: %d blocks", job_id, len(blocks))
    return resp


def textract_ocr_pdf(pdf_bytes: bytes, region: str = "us-east-1", dpi: int = 200) -> bytes:
    """
    Convert non-searchable PDF to searchable PDF using AWS Textract + PyMuPDF.
    
    Process:
        1. Upload PDF to temporary S3 (Textract requires S3 for PDFs)
        2. Start async Textract text detection
        3. Poll for completion
        4. Render each page as image
        5. Overlay invisible text from Textract WORD blocks at bounding box positions
        6. Return searchable PDF bytes
    """
    textract = _get_textract_client(region)
    s3 = _s3_client(region)
    
    temp_bucket = os.getenv("TEMP_BUCKET", os.getenv("BUCKET", "ecmc-dev-bucket"))
    temp_key = f"temp/textract_ocr_{hashlib.md5(pdf_bytes).hexdigest()[:16]}.pdf"
    
    log.info("Uploading to temporary S3 for Textract: s3://%s/%s", temp_bucket, temp_key)
    s3.put_object(Bucket=temp_bucket, Key=temp_key, Body=pdf_bytes, ContentType="application/pdf")
    
    try:
        resp = textract.start_document_text_detection(
            DocumentLocation={"S3Object": {"Bucket": temp_bucket, "Name": temp_key}}
        )
        job_id = resp["JobId"]
        log.info("Started Textract OCR job: %s", job_id)
        
        result = _wait_for_textract_job(textract, job_id)
        if result["JobStatus"] != "SUCCEEDED":
            raise RuntimeError(f"Textract job failed: {result.get('StatusMessage', 'Unknown')}")
        
        blocks = result["Blocks"]
        
        source_doc = fitz.open(stream=pdf_bytes, filetype="pdf")
        output_doc = fitz.open()
        
        try:
            for page_idx in range(len(source_doc)):
                source_page = source_doc[page_idx]
                pix = source_page.get_pixmap(dpi=dpi, alpha=False)
                new_page = output_doc.new_page(width=source_page.rect.width, height=source_page.rect.height)
                new_page.insert_image(new_page.rect, stream=pix.tobytes("png"))
            
            page_blocks: Dict[int, List[Dict]] = {}
            for block in blocks:
                if block["BlockType"] == "WORD":
                    page_num = block.get("Page", 1)
                    page_blocks.setdefault(page_num, []).append(block)
            
            for page_num, word_blocks in page_blocks.items():
                if page_num > len(output_doc):
                    continue
                    
                pdf_page = output_doc[page_num - 1]
                page_width = pdf_page.rect.width
                page_height = pdf_page.rect.height
                
                for block in word_blocks:
                    text = block.get("Text", "")
                    if not text:
                        continue
                    
                    bbox = BoundingBox.from_textract_bbox(block["Geometry"]["BoundingBox"])
                    bbox.scale(page_width, page_height)
                    
                    text_length = fitz.get_text_length(text, fontname="helv", fontsize=12)
                    if text_length > 0:
                        optimal_size = min(12, math.floor((bbox.width / text_length) * 12))
                        optimal_size = max(4, optimal_size)
                    else:
                        optimal_size = 8
                    
                    pdf_page.insert_text(
                        point=fitz.Point(bbox.left, bbox.bottom),
                        text=text,
                        fontname="helv",
                        fontsize=optimal_size,
                        color=(0, 0, 0),
                        fill_opacity=0,
                    )
            
            searchable_bytes = output_doc.tobytes()
            word_count = len([b for b in blocks if b["BlockType"] == "WORD"])
            log.info("Created searchable PDF: %d pages, %d words overlayed", len(output_doc), word_count)
            return searchable_bytes
            
        finally:
            source_doc.close()
            output_doc.close()
            
    finally:
        try:
            s3.delete_object(Bucket=temp_bucket, Key=temp_key)
            log.info("Cleaned up temporary S3 object: %s", temp_key)
        except Exception as e:
            log.warning("Failed to cleanup temp S3 object: %s", e)

# PDF -> Markdown via pymupdf4llm

def extract_markdown_from_pdf(pdf_bytes: bytes) -> Tuple[str, List[Dict]]:
    doc = fitz.open(stream=pdf_bytes, filetype="pdf")
    try:
        page_chunks = pymupdf4llm.to_markdown(
            doc,
            page_chunks=True,
            write_images=False,
            show_progress=False,
        )
    finally:
        doc.close()
    
    markdown_parts: List[str] = []
    tables: List[Dict] = []
    for i, page in enumerate(page_chunks, 1):
        text = (page.get("text") or "").strip()
        markdown_parts.append(f"\n\n<!-- page {i} -->\n\n{text}")
        for t in page.get("tables") or []:
            tables.append({
                "page": i,
                "bbox": t.get("bbox"),
                "rows": t.get("rows"),
            })
    
    return "\n".join(markdown_parts).strip(), tables

@dataclass
class CrawlConfig:
    bucket: str
    s3_prefix: str
    base_domain: str
    section_prefixes: List[str] = field(default_factory=list)
    allow_full_domain: bool = False
    delay: float = 0.5
    max_pages: Optional[int] = None
    max_depth: Optional[int] = None
    request_timeout: int = 20
    aws_region: str = "us-east-1"
    save_html_as_pdf: bool = True  # NEW: Convert HTML to PDF


def crawl_html_and_discover_pdfs(
    seed_urls: List[str], cfg: CrawlConfig
) -> Tuple[List[Dict], Dict[str, Dict], Dict]:
    s3 = _s3_client(cfg.aws_region)
    session = requests.Session()
    session.headers.update({
        "User-Agent": (
            "Mozilla/5.0 (Windows NT 10.0; Win64; x64) "
            "AppleWebKit/537.36 (KHTML, like Gecko) "
            "Chrome/124.0 Safari/537.36"
        )
    })

    visited: Set[str] = set()
    queued: Set[str] = set()
    pdf_catalog: Dict[str, Dict] = {}
    reject_stats: Counter = Counter()
    to_visit: deque = deque()
    html_outputs: List[Dict] = []
    scraped = 0
    failed = 0
    deepest = 0

    def same_site(url: str) -> bool:
        p = urlparse(url)
        if p.scheme not in ("http", "https"):
            return False
        host = p.netloc.replace("www.", "")
        return host == cfg.base_domain or host.endswith("." + cfg.base_domain)

    def in_scope(url: str) -> bool:
        if cfg.allow_full_domain:
            return True
        if not cfg.section_prefixes:
            return True
        path = urlparse(url).path.rstrip("/") or "/"
        for prefix in cfg.section_prefixes:
            if path == prefix or path.startswith(prefix + "/"):
                return True
        return False

    def classify_link(url: str) -> Tuple[str, str]:
        p = urlparse(url)
        if p.scheme not in ("http", "https"):
            return "reject", "bad_scheme"
        if not same_site(url):
            return "reject", "off_site"
        path_lower = p.path.lower()
        if path_lower.endswith(".pdf"):
            return "pdf", "pdf"
        if path_lower.endswith(SKIP_EXTENSIONS):
            return "reject", "binary_extension"
        if not in_scope(url):
            return "reject", "out_of_scope"
        return "crawl", "in_scope"

    def html_filename(url: str) -> str:
        p = urlparse(url)
        path = p.path.strip("/")
        if path:
            safe = re.sub(r"[^a-zA-Z0-9._-]+", "_", path.replace("/", "_"))[:100]
        else:
            safe = "index"
        domain = p.netloc.replace("www.", "")
        full = f"{domain}_{safe}"
        if len(full) > 150:
            full = f"{domain}_{hashlib.md5(url.encode()).hexdigest()[:8]}"
        return full

    def record_pdf(anchor, full_url: str, source_page: str):
        if full_url in pdf_catalog:
            pdf_catalog[full_url]["referrers"].add(source_page)
            return
        title = anchor.get_text(strip=True)
        if not title:
            parent = anchor.find_parent(["h1", "h2", "h3", "h4", "li"])
            if parent:
                title = parent.get_text(strip=True)
        if not title:
            title = Path(urlparse(full_url).path).stem.replace("-", " ").title()
        pdf_catalog[full_url] = {
            "title": title[:150],
            "url": full_url,
            "filename": Path(urlparse(full_url).path).name,
            "first_seen_on": source_page,
            "referrers": {source_page},
        }

    for seed in seed_urls:
        seed = normalize_url(seed)
        if seed not in queued:
            queued.add(seed)
            to_visit.append((seed, 0))

    log.info("Starting crawl (base=%s, scope=%s, max_depth=%s, max_pages=%s)",
             cfg.base_domain,
             "FULL" if cfg.allow_full_domain else cfg.section_prefixes,
             cfg.max_depth, cfg.max_pages)

    while to_visit:
        if cfg.max_pages and scraped >= cfg.max_pages:
            log.info("Reached max_pages=%d; stopping.", cfg.max_pages)
            break

        url, depth = to_visit.popleft()
        if url in visited:
            continue
        visited.add(url)
        deepest = max(deepest, depth)

        try:
            log.info("[HTML d=%d] %s", depth, url)
            resp = session.get(url, timeout=cfg.request_timeout, allow_redirects=True)
            resp.raise_for_status()

            ctype = resp.headers.get("content-type", "").lower()
            if "text/html" not in ctype and "application/xhtml" not in ctype:
                failed += 1
                log.info("  skip: non-HTML content-type %s", ctype)
                continue

            final_url = normalize_url(resp.url)
            if not same_site(final_url):
                failed += 1
                log.info("  skip: redirected off-site to %s", final_url)
                continue

            soup = BeautifulSoup(resp.text, "html.parser")

            for a in soup.find_all("a", href=True):
                href = a["href"].strip()
                if not href or href.startswith(("javascript:", "mailto:", "tel:", "#")):
                    continue
                full = normalize_url(urljoin(url, href))
                category, reason = classify_link(full)
                if category == "pdf":
                    record_pdf(a, full, source_page=url)
                elif category == "crawl":
                    if full not in visited and full not in queued:
                        if cfg.max_depth is None or depth + 1 <= cfg.max_depth:
                            queued.add(full)
                            to_visit.append((full, depth + 1))
                        else:
                            reject_stats["depth_capped"] += 1
                else:
                    reject_stats[reason] += 1

            page_markdown = html_to_markdown(resp.text, source_url=url)
            base = html_filename(url)
            
            # Save HTML as Markdown
            md_key = f"{cfg.s3_prefix}/web_md/{base}.md"
            md_s3 = s3_put(s3, cfg.bucket, md_key, page_markdown, "text/markdown")

            # NEW: Convert HTML to PDF and save
            html_pdf_s3 = None
            if cfg.save_html_as_pdf:
                try:
                    pdf_bytes = html_to_pdf(resp.text)
                    pdf_key = f"{cfg.s3_prefix}/html_as_pdf/{base}.pdf"
                    s3.put_object(
                        Bucket=cfg.bucket, Key=pdf_key,
                        Body=pdf_bytes, ContentType="application/pdf"
                    )
                    html_pdf_s3 = f"s3://{cfg.bucket}/{pdf_key}"
                    log.info("  Saved HTML as PDF: %s", pdf_key)
                except Exception as e:
                    log.warning("  Failed to convert HTML to PDF: %s", e)

            page_language, page_conf = detect_language(page_markdown)

            html_outputs.append({
                "url": url,
                "s3_markdown": md_s3,
                "s3_html_pdf": html_pdf_s3,
                "language": page_language,
                "language_confidence": round(page_conf, 4),
                "chars": len(page_markdown),
                "depth": depth,
            })
            scraped += 1

            if to_visit and cfg.delay > 0:
                time.sleep(cfg.delay)

        except Exception as e:
            failed += 1
            log.warning("  failed: %s", e)

    stats = {
        "scraped": scraped,
        "failed": failed,
        "deepest_depth_seen": deepest,
        "reject_stats": dict(reject_stats),
        "pdf_count_discovered": len(pdf_catalog),
    }
    log.info("HTML crawl done: scraped=%d failed=%d PDFs_found=%d",
             scraped, failed, len(pdf_catalog))
    return html_outputs, pdf_catalog, stats

@dataclass
class PdfConfig:
    bucket: str
    s3_prefix: str
    exclude_languages: List[str] = field(default_factory=list)
    enable_ocr_for_image_pdfs: bool = True
    ocr_dpi: int = 200
    request_timeout: int = 60
    aws_region: str = "us-east-1"


def process_pdfs(pdf_catalog: Dict[str, Dict], cfg: PdfConfig) -> List[Dict]:
    s3 = _s3_client(cfg.aws_region)
    session = requests.Session()
    session.headers.update({
        "User-Agent": (
            "Mozilla/5.0 (Windows NT 10.0; Win64; x64) "
            "AppleWebKit/537.36 (KHTML, like Gecko) "
            "Chrome/124.0 Safari/537.36"
        )
    })

    results: List[Dict] = []
    seen_filenames: Dict[str, str] = {}
    total = len(pdf_catalog)

    for i, (pdf_url, meta) in enumerate(pdf_catalog.items(), 1):
        log.info("[PDF %d/%d] %s", i, total, meta["title"])
        result: Dict = {
            "title": meta["title"],
            "url": pdf_url,
            "first_seen_on": meta["first_seen_on"],
            "referrers": sorted(meta["referrers"]),
        }

        try:
            resp = session.get(pdf_url, timeout=cfg.request_timeout)
            resp.raise_for_status()
            pdf_bytes = resp.content
            size_mb = len(pdf_bytes) / (1024 * 1024)
            result["size_mb"] = round(size_mb, 3)

            pdf_type = classify_pdf_type(pdf_bytes)
            result["pdf_type"] = pdf_type
            log.info("  type: %s | size: %.2f MB", pdf_type, size_mb)

            filename = clean_pdf_filename(meta["filename"])
            if filename in seen_filenames and seen_filenames[filename] != pdf_url:
                suffix = hashlib.md5(pdf_url.encode()).hexdigest()[:6]
                filename = filename.replace(".pdf", f"_{suffix}.pdf")
            seen_filenames[filename] = pdf_url
            base_no_ext = filename[:-4]

            original_key = f"{cfg.s3_prefix}/pdfs/{filename}"
            s3.put_object(
                Bucket=cfg.bucket, Key=original_key,
                Body=pdf_bytes, ContentType="application/pdf",
            )
            result["s3_pdf"] = f"s3://{cfg.bucket}/{original_key}"

            extraction_input_bytes = pdf_bytes
            extraction_method = "pymupdf4llm"

            if pdf_type in ("image", "mixed") and cfg.enable_ocr_for_image_pdfs:
                log.info("  OCR: running Textract at %d DPI...", cfg.ocr_dpi)
                try:
                    searchable_bytes = textract_ocr_pdf(
                        pdf_bytes, 
                        region=cfg.aws_region, 
                        dpi=cfg.ocr_dpi
                    )
                    searchable_key = f"{cfg.s3_prefix}/pdfs/{base_no_ext}_searchable.pdf"
                    s3.put_object(
                        Bucket=cfg.bucket, Key=searchable_key,
                        Body=searchable_bytes, ContentType="application/pdf",
                    )
                    result["s3_pdf_searchable"] = f"s3://{cfg.bucket}/{searchable_key}"
                    extraction_input_bytes = searchable_bytes
                    extraction_method = "textract_then_pymupdf4llm"
                    log.info("  Textract OCR done: %.1f KB", len(searchable_bytes) / 1024)
                except Exception as e:
                    log.warning("  Textract OCR failed, falling back to original: %s", e)
                    result["ocr_error"] = str(e)
            elif pdf_type == "empty":
                log.info("  empty PDF: no pages")
                result["status"] = "empty_pdf"
                results.append(result)
                continue

            try:
                markdown_text, tables = extract_markdown_from_pdf(extraction_input_bytes)
            except Exception as e:
                log.error("  pymupdf4llm extraction failed: %s", e)
                result["status"] = "extraction_failed"
                result["error"] = str(e)
                result["extraction_method"] = extraction_method
                results.append(result)
                continue

            result["extraction_method"] = extraction_method
            result["char_count"] = len(markdown_text)
            result["table_count"] = len(tables)

            language, confidence = detect_language(markdown_text)
            result["language"] = language
            result["language_confidence"] = round(confidence, 4)
            log.info("  language: %s (conf=%.3f)", language, confidence)

            if is_excluded_language(language, cfg.exclude_languages):
                log.info("  skipped: language '%s' is in exclude list", language)
                result["status"] = "skipped_excluded_language"
                results.append(result)
                continue

            md_key = f"{cfg.s3_prefix}/pdf_md/{base_no_ext}.md"
            s3_put(s3, cfg.bucket, md_key, markdown_text, "text/markdown")
            result["s3_markdown"] = f"s3://{cfg.bucket}/{md_key}"

            if tables:
                tables_key = f"{cfg.s3_prefix}/pdf_md/{base_no_ext}_tables.json"
                s3_put(s3, cfg.bucket, tables_key,
                       json.dumps(tables, indent=2, ensure_ascii=False),
                       "application/json")
                result["s3_tables_json"] = f"s3://{cfg.bucket}/{tables_key}"

            result["status"] = "success"

        except Exception as e:
            log.exception("  failed: %s", e)
            result["status"] = "error"
            result["error"] = str(e)

        results.append(result)

    return results

def _setup_logging():
    if not logging.getLogger().handlers:
        logging.basicConfig(
            level=logging.INFO,
            format="%(asctime)s [%(levelname)s] %(message)s",
            datefmt="%H:%M:%S",
            handlers=[logging.StreamHandler(sys.stdout)],
        )


def canon_language(s: str) -> str:
    if not s:
        return ""
    cleaned = re.sub(r"[^\w\-]", "", s.strip().lower())
    for canon, aliases in _LANG_ALIASES.items():
        if cleaned in aliases:
            return canon
    if cleaned in _LANG_CODE_TO_NAME:
        return _LANG_CODE_TO_NAME[cleaned]
    return cleaned


def is_excluded_language(detected: str, exclude_list: List[str]) -> bool:
    if not exclude_list:
        return False
    detected_canon = canon_language(detected)
    if not detected_canon:
        return False
    excluded_canon = {canon_language(x) for x in exclude_list}
    return detected_canon in excluded_canon


def generate_folder_name(url: str) -> str:
    parsed = urlparse(url)
    path = parsed.path.strip("/")
    timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
    if path:
        last = path.split("/")[-1] if "/" in path else path
        last = re.sub(r"\.(html|htm|php|asp|aspx|jsp)$", "", last, flags=re.IGNORECASE)
        clean_name = re.sub(r"[^a-zA-Z0-9]+", "_", last).strip("_")[:80]
        if not clean_name:
            clean_name = re.sub(r"[^a-zA-Z0-9]+", "_", parsed.netloc).strip("_")
    else:
        clean_name = re.sub(r"[^a-zA-Z0-9]+", "_", parsed.netloc).strip("_")
    return f"{clean_name}_{timestamp}"


def derive_section_prefix(url: str) -> str:
    parsed = urlparse(url)
    path = parsed.path.rstrip("/")
    if "." in path.split("/")[-1]:
        path = "/".join(path.split("/")[:-1])
    return path or "/"


def normalize_url(url: str) -> str:
    return url.split("#")[0].rstrip("/")


def normalize_prefix(prefix: str) -> str:
    p = prefix.strip()
    if not p.startswith("/"):
        p = "/" + p
    return p.rstrip("/") or "/"


def clean_pdf_filename(name: str) -> str:
    clean = "".join(c for c in name if c.isalnum() or c in (".", "-", "_")).strip()
    if not clean.endswith(".pdf"):
        clean += ".pdf"
    return clean or "document.pdf"

def s3_put(s3, bucket: str, key: str, body, content_type: str = "text/plain") -> str:
    if isinstance(body, str):
        body = body.encode("utf-8")
    s3.put_object(Bucket=bucket, Key=key, Body=body, ContentType=content_type)
    return f"s3://{bucket}/{key}"

def detect_language(text: str) -> Tuple[str, float]:
    if not text or len(text.strip()) < 20:
        return "unknown", 0.0
    try:
        langs = detect_langs(text[:2000])
        if not langs:
            return "unknown", 0.0
        top = langs[0]
        return canon_language(top.lang), float(top.prob)
    except LangDetectException:
        return "unknown", 0.0

def html_to_markdown(html: str, source_url: str) -> str:
    soup = BeautifulSoup(html, "html.parser")
    for tag in soup(["script", "style", "noscript", "meta", "link",
                     "header", "footer", "nav", "aside", "form", "iframe"]):
        tag.decompose()
    content = soup.find("main") or soup.find("article") or soup.body or soup
    if content is None:
        return ""
    md_text = md(
        str(content),
        heading_style="ATX",
        strip=["a"],
        bullets="-",
    )
    md_text = re.sub(r"\n{3,}", "\n\n", md_text).strip()
    return f"<!-- source_url: {source_url} -->\n\n{md_text}"

# PDF classification (unchanged)
# ============================================================================
def classify_pdf_type(pdf_bytes: bytes) -> str:
    doc = fitz.open(stream=pdf_bytes, filetype="pdf")
    pages_with_text = 0
    pages_without_text = 0
    try:
        for page in doc:
            text = (page.get_text("text") or "").strip()
            has_real_text = False
            if len(text) >= 30:
                alnum = sum(c.isalnum() for c in text)
                if alnum / len(text) >= 0.20:
                    has_real_text = True
            if has_real_text:
                pages_with_text += 1
            else:
                pages_without_text += 1
    finally:
        doc.close()
    total = pages_with_text + pages_without_text
    if total == 0:
        return "empty"
    if pages_with_text == total:
        return "searchable"
    if pages_without_text == total:
        return "image"
    return "mixed"

class BoundingBox:
    """Helper to manipulate Textract bounding boxes for PyMuPDF overlay."""
    
    def __init__(self, left: float, bottom: float, right: float, top: float) -> None:
        self.bounds = [left, bottom, right, top]
    
    @classmethod
    def from_textract_bbox(cls, textract_bbox: Dict[str, float]) -> "BoundingBox":
        """Textract uses upper-left origin; PyMuPDF uses lower-left."""
        return cls(
            left=textract_bbox["Left"],
            bottom=textract_bbox["Top"] + textract_bbox["Height"],
            right=textract_bbox["Left"] + textract_bbox["Width"],
            top=textract_bbox["Top"],
        )
    
    def scale(self, x_scale: float, y_scale: Optional[float] = None) -> None:
        if y_scale is None:
            y_scale = x_scale
        self.bounds[0] *= x_scale
        self.bounds[1] *= y_scale
        self.bounds[2] *= x_scale
        self.bounds[3] *= y_scale
    
    @property
    def left(self) -> float:
        return self.bounds[0]
    
    @property
    def bottom(self) -> float:
        return self.bounds[1]
    
    @property
    def right(self) -> float:
        return self.bounds[2]
    
    @property
    def top(self) -> float:
        return self.bounds[3]
    
    @property
    def width(self) -> float:
        return abs(self.bounds[0] - self.bounds[2])
    
    @property
    def height(self) -> float:
        return abs(self.bounds[3] - self.bounds[1])


def _get_textract_client(region: str):
    """Return cached Textract client."""
    return boto3.client("textract", region_name=region)


def _wait_for_textract_job(textract, job_id: str) -> Dict[str, any]:
    """Poll Textract async job until completion or timeout."""
    started = time.time()
    while True:
        resp = textract.get_document_text_detection(JobId=job_id)
        status = resp["JobStatus"]
        if status in ("SUCCEEDED", "FAILED"):
            break
        if time.time() - started > TEXTRACT_MAX_POLL_SECONDS:
            log.error("Textract job %s timed out after %d seconds", job_id, TEXTRACT_MAX_POLL_SECONDS)
            return {"JobStatus": "TIMEOUT", "Blocks": []}
        time.sleep(TEXTRACT_POLL_SECONDS)
    
    if status != "SUCCEEDED":
        log.error("Textract job %s failed: %s", job_id, resp.get("StatusMessage", "Unknown"))
        return resp
    
    # Paginate all blocks
    blocks = list(resp.get("Blocks", []))
    next_token = resp.get("NextToken")
    while next_token:
        page = textract.get_document_text_detection(JobId=job_id, NextToken=next_token)
        blocks.extend(page.get("Blocks", []))
        next_token = page.get("NextToken")
    
    resp["Blocks"] = blocks
    log.info("Textract job %s completed: %d blocks", job_id, len(blocks))
    return resp


def textract_ocr_pdf(pdf_bytes: bytes, region: str = "us-east-1", dpi: int = 200) -> bytes:
    """
    Convert non-searchable PDF to searchable PDF using AWS Textract + PyMuPDF.
    
    Process:
        1. Upload PDF to temporary S3 location (Textract requires S3 for PDFs)
        2. Start async Textract text detection
        3. Poll for completion
        4. Render each page as image
        5. Overlay invisible text from Textract WORD blocks at bounding box positions
        6. Return searchable PDF bytes
    
    Args:
        pdf_bytes: Raw PDF bytes (non-searchable or mixed)
        region: AWS region
        dpi: Resolution for rendering page images
    
    Returns:
        Searchable PDF bytes
    """
    textract = _get_textract_client(region)
    s3 = _s3_client(region)
    
    # Step 1: Upload to temporary S3 (Textract requires S3 for PDFs)
    temp_bucket = os.getenv("TEMP_BUCKET", os.getenv("BUCKET", "ecmc-dev-bucket"))
    temp_key = f"temp/textract_ocr_{hashlib.md5(pdf_bytes).hexdigest()[:16]}.pdf"
    
    log.info("Uploading to temporary S3 for Textract: s3://%s/%s", temp_bucket, temp_key)
    s3.put_object(Bucket=temp_bucket, Key=temp_key, Body=pdf_bytes, ContentType="application/pdf")
    
    try:
        # Step 2: Start async Textract job
        resp = textract.start_document_text_detection(
            DocumentLocation={"S3Object": {"Bucket": temp_bucket, "Name": temp_key}}
        )
        job_id = resp["JobId"]
        log.info("Started Textract OCR job: %s", job_id)
        
        # Step 3: Wait for completion
        result = _wait_for_textract_job(textract, job_id)
        if result["JobStatus"] != "SUCCEEDED":
            raise RuntimeError(f"Textract job failed: {result.get('StatusMessage', 'Unknown')}")
        
        blocks = result["Blocks"]
        
        # Step 4: Build searchable PDF
        source_doc = fitz.open(stream=pdf_bytes, filetype="pdf")
        output_doc = fitz.open()
        
        try:
            # Render each page as image
            for page_idx in range(len(source_doc)):
                source_page = source_doc[page_idx]
                pix = source_page.get_pixmap(dpi=dpi, alpha=False)
                
                # Create new page with same dimensions
                new_page = output_doc.new_page(width=source_page.rect.width, height=source_page.rect.height)
                # Insert the rendered image
                new_page.insert_image(new_page.rect, stream=pix.tobytes("png"))
            
            # Step 5: Overlay invisible text from Textract blocks
            # Group blocks by page
            page_blocks: Dict[int, List[Dict]] = {}
            for block in blocks:
                if block["BlockType"] == "WORD":
                    page_num = block.get("Page", 1)
                    page_blocks.setdefault(page_num, []).append(block)
            
            for page_num, word_blocks in page_blocks.items():
                if page_num > len(output_doc):
                    continue
                    
                pdf_page = output_doc[page_num - 1]  # 0-indexed
                page_width = pdf_page.rect.width
                page_height = pdf_page.rect.height
                
                for block in word_blocks:
                    text = block.get("Text", "")
                    if not text:
                        continue
                    
                    bbox = BoundingBox.from_textract_bbox(block["Geometry"]["BoundingBox"])
                    bbox.scale(page_width, page_height)
                    
                    # Calculate optimal font size to fit text in bounding box
                    text_length = fitz.get_text_length(text, fontname="helv", fontsize=12)
                    if text_length > 0:
                        optimal_size = min(12, math.floor((bbox.width / text_length) * 12))
                        optimal_size = max(4, optimal_size)  # Minimum readable size
                    else:
                        optimal_size = 8
                    
                    # Insert text with zero opacity (invisible but selectable)
                    pdf_page.insert_text(
                        point=fitz.Point(bbox.left, bbox.bottom),
                        text=text,
                        fontname="helv",
                        fontsize=optimal_size,
                        color=(0, 0, 0),  # Black
                        fill_opacity=0,   # Invisible overlay
                    )
            
            # Return searchable PDF bytes
            searchable_bytes = output_doc.tobytes()
            log.info("Created searchable PDF: %d pages, %d words overlayed", 
                     len(output_doc), len([b for b in blocks if b["BlockType"] == "WORD"]))
            return searchable_bytes
            
        finally:
            source_doc.close()
            output_doc.close()
            
    finally:
        # Cleanup temp S3 object
        try:
            s3.delete_object(Bucket=temp_bucket, Key=temp_key)
            log.info("Cleaned up temporary S3 object: %s", temp_key)
        except Exception as e:
            log.warning("Failed to cleanup temp S3 object: %s", e)

# PDF -> Markdown via pymupdf4llm (unchanged)
# ============================================================================
def extract_markdown_from_pdf(pdf_bytes: bytes) -> Tuple[str, List[Dict]]:
    doc = fitz.open(stream=pdf_bytes, filetype="pdf")
    try:
        page_chunks = pymupdf4llm.to_markdown(
            doc,
            page_chunks=True,
            write_images=False,
            show_progress=False,
        )
    finally:
        doc.close()
    
    markdown_parts: List[str] = []
    tables: List[Dict] = []
    for i, page in enumerate(page_chunks, 1):
        text = (page.get("text") or "").strip()
        markdown_parts.append(f"\n\n<!-- page {i} -->\n\n{text}")
        for t in page.get("tables") or []:
            tables.append({
                "page": i,
                "bbox": t.get("bbox"),
                "rows": t.get("rows"),
            })
    
    return "\n".join(markdown_parts).strip(), tables

@dataclass
class CrawlConfig:
    bucket: str
    s3_prefix: str
    base_domain: str
    section_prefixes: List[str] = field(default_factory=list)
    allow_full_domain: bool = False
    delay: float = 0.5
    max_pages: Optional[int] = None
    max_depth: Optional[int] = None
    request_timeout: int = 20
    aws_region: str = "us-east-1"


def crawl_html_and_discover_pdfs(
    seed_urls: List[str], cfg: CrawlConfig
) -> Tuple[List[Dict], Dict[str, Dict], Dict]:
    s3 = _s3_client(cfg.aws_region)
    session = requests.Session()
    session.headers.update({
        "User-Agent": (
            "Mozilla/5.0 (Windows NT 10.0; Win64; x64) "
            "AppleWebKit/537.36 (KHTML, like Gecko) "
            "Chrome/124.0 Safari/537.36"
        )
    })

    visited: Set[str] = set()
    queued: Set[str] = set()
    pdf_catalog: Dict[str, Dict] = {}
    reject_stats: Counter = Counter()
    to_visit: deque = deque()
    html_outputs: List[Dict] = []
    scraped = 0
    failed = 0
    deepest = 0

    def same_site(url: str) -> bool:
        p = urlparse(url)
        if p.scheme not in ("http", "https"):
            return False
        host = p.netloc.replace("www.", "")
        return host == cfg.base_domain or host.endswith("." + cfg.base_domain)

    def in_scope(url: str) -> bool:
        if cfg.allow_full_domain:
            return True
        if not cfg.section_prefixes:
            return True
        path = urlparse(url).path.rstrip("/") or "/"
        for prefix in cfg.section_prefixes:
            if path == prefix or path.startswith(prefix + "/"):
                return True
        return False

    def classify_link(url: str) -> Tuple[str, str]:
        p = urlparse(url)
        if p.scheme not in ("http", "https"):
            return "reject", "bad_scheme"
        if not same_site(url):
            return "reject", "off_site"
        path_lower = p.path.lower()
        if path_lower.endswith(".pdf"):
            return "pdf", "pdf"
        if path_lower.endswith(SKIP_EXTENSIONS):
            return "reject", "binary_extension"
        if not in_scope(url):
            return "reject", "out_of_scope"
        return "crawl", "in_scope"

    def html_filename(url: str) -> str:
        p = urlparse(url)
        path = p.path.strip("/")
        if path:
            safe = re.sub(r"[^a-zA-Z0-9._-]+", "_", path.replace("/", "_"))[:100]
        else:
            safe = "index"
        domain = p.netloc.replace("www.", "")
        full = f"{domain}_{safe}"
        if len(full) > 150:
            full = f"{domain}_{hashlib.md5(url.encode()).hexdigest()[:8]}"
        return full

    def record_pdf(anchor, full_url: str, source_page: str):
        if full_url in pdf_catalog:
            pdf_catalog[full_url]["referrers"].add(source_page)
            return
        title = anchor.get_text(strip=True)
        if not title:
            parent = anchor.find_parent(["h1", "h2", "h3", "h4", "li"])
            if parent:
                title = parent.get_text(strip=True)
        if not title:
            title = Path(urlparse(full_url).path).stem.replace("-", " ").title()
        pdf_catalog[full_url] = {
            "title": title[:150],
            "url": full_url,
            "filename": Path(urlparse(full_url).path).name,
            "first_seen_on": source_page,
            "referrers": {source_page},
        }

    for seed in seed_urls:
        seed = normalize_url(seed)
        if seed not in queued:
            queued.add(seed)
            to_visit.append((seed, 0))

    log.info("Starting crawl (base=%s, scope=%s, max_depth=%s, max_pages=%s)",
             cfg.base_domain,
             "FULL" if cfg.allow_full_domain else cfg.section_prefixes,
             cfg.max_depth, cfg.max_pages)

    while to_visit:
        if cfg.max_pages and scraped >= cfg.max_pages:
            log.info("Reached max_pages=%d; stopping.", cfg.max_pages)
            break

        url, depth = to_visit.popleft()
        if url in visited:
            continue
        visited.add(url)
        deepest = max(deepest, depth)

        try:
            log.info("[HTML d=%d] %s", depth, url)
            resp = session.get(url, timeout=cfg.request_timeout, allow_redirects=True)
            resp.raise_for_status()

            ctype = resp.headers.get("content-type", "").lower()
            if "text/html" not in ctype and "application/xhtml" not in ctype:
                failed += 1
                log.info("  skip: non-HTML content-type %s", ctype)
                continue

            final_url = normalize_url(resp.url)
            if not same_site(final_url):
                failed += 1
                log.info("  skip: redirected off-site to %s", final_url)
                continue

            soup = BeautifulSoup(resp.text, "html.parser")

            for a in soup.find_all("a", href=True):
                href = a["href"].strip()
                if not href or href.startswith(("javascript:", "mailto:", "tel:", "#")):
                    continue
                full = normalize_url(urljoin(url, href))
                category, reason = classify_link(full)
                if category == "pdf":
                    record_pdf(a, full, source_page=url)
                elif category == "crawl":
                    if full not in visited and full not in queued:
                        if cfg.max_depth is None or depth + 1 <= cfg.max_depth:
                            queued.add(full)
                            to_visit.append((full, depth + 1))
                        else:
                            reject_stats["depth_capped"] += 1
                else:
                    reject_stats[reason] += 1

            page_markdown = html_to_markdown(resp.text, source_url=url)
            base = html_filename(url)
            md_key = f"{cfg.s3_prefix}/web_md/{base}.md"
            md_s3 = s3_put(s3, cfg.bucket, md_key, page_markdown, "text/markdown")

            page_language, page_conf = detect_language(page_markdown)

            html_outputs.append({
                "url": url,
                "s3_markdown": md_s3,
                "language": page_language,
                "language_confidence": round(page_conf, 4),
                "chars": len(page_markdown),
                "depth": depth,
            })
            scraped += 1

            if to_visit and cfg.delay > 0:
                time.sleep(cfg.delay)

        except Exception as e:
            failed += 1
            log.warning("  failed: %s", e)

    stats = {
        "scraped": scraped,
        "failed": failed,
        "deepest_depth_seen": deepest,
        "reject_stats": dict(reject_stats),
        "pdf_count_discovered": len(pdf_catalog),
    }
    log.info("HTML crawl done: scraped=%d failed=%d PDFs_found=%d",
             scraped, failed, len(pdf_catalog))
    return html_outputs, pdf_catalog, stats

@dataclass
class PdfConfig:
    bucket: str
    s3_prefix: str
    exclude_languages: List[str] = field(default_factory=list)
    enable_ocr_for_image_pdfs: bool = True  # Now uses Textract, not Tesseract
    ocr_dpi: int = 200  # Textract rendering DPI
    request_timeout: int = 60
    aws_region: str = "us-east-1"


def process_pdfs(pdf_catalog: Dict[str, Dict], cfg: PdfConfig) -> List[Dict]:
    """
    Process PDFs with Textract OCR instead of Tesseract.
    
    For each PDF:
      1. Download
      2. Classify (searchable / image / mixed / empty)
      3. If image/mixed and OCR enabled -> Textract OCR -> searchable PDF
      4. Extract Markdown via pymupdf4llm
      5. Detect language; drop if excluded
      6. Save Markdown + tables JSON to S3
    """
    s3 = _s3_client(cfg.aws_region)
    session = requests.Session()
    session.headers.update({
        "User-Agent": (
            "Mozilla/5.0 (Windows NT 10.0; Win64; x64) "
            "AppleWebKit/537.36 (KHTML, like Gecko) "
            "Chrome/124.0 Safari/537.36"
        )
    })

    results: List[Dict] = []
    seen_filenames: Dict[str, str] = {}
    total = len(pdf_catalog)

    for i, (pdf_url, meta) in enumerate(pdf_catalog.items(), 1):
        log.info("[PDF %d/%d] %s", i, total, meta["title"])
        result: Dict = {
            "title": meta["title"],
            "url": pdf_url,
            "first_seen_on": meta["first_seen_on"],
            "referrers": sorted(meta["referrers"]),
        }

        try:
            # 1. Download
            resp = session.get(pdf_url, timeout=cfg.request_timeout)
            resp.raise_for_status()
            pdf_bytes = resp.content
            size_mb = len(pdf_bytes) / (1024 * 1024)
            result["size_mb"] = round(size_mb, 3)

            # 2. Classify
            pdf_type = classify_pdf_type(pdf_bytes)
            result["pdf_type"] = pdf_type
            log.info("  type: %s | size: %.2f MB", pdf_type, size_mb)

            # De-duplicate filenames
            filename = clean_pdf_filename(meta["filename"])
            if filename in seen_filenames and seen_filenames[filename] != pdf_url:
                suffix = hashlib.md5(pdf_url.encode()).hexdigest()[:6]
                filename = filename.replace(".pdf", f"_{suffix}.pdf")
            seen_filenames[filename] = pdf_url
            base_no_ext = filename[:-4]

            # Save original PDF
            original_key = f"{cfg.s3_prefix}/pdfs/{filename}"
            s3.put_object(
                Bucket=cfg.bucket, Key=original_key,
                Body=pdf_bytes, ContentType="application/pdf",
            )
            result["s3_pdf"] = f"s3://{cfg.bucket}/{original_key}"

            # 3. OCR with Textract if needed
            extraction_input_bytes = pdf_bytes
            extraction_method = "pymupdf4llm"

            if pdf_type in ("image", "mixed") and cfg.enable_ocr_for_image_pdfs:
                log.info("  OCR: running Textract at %d DPI...", cfg.ocr_dpi)
                try:
                    searchable_bytes = textract_ocr_pdf(
                        pdf_bytes, 
                        region=cfg.aws_region, 
                        dpi=cfg.ocr_dpi
                    )
                    searchable_key = f"{cfg.s3_prefix}/pdfs/{base_no_ext}_searchable.pdf"
                    s3.put_object(
                        Bucket=cfg.bucket, Key=searchable_key,
                        Body=searchable_bytes, ContentType="application/pdf",
                    )
                    result["s3_pdf_searchable"] = f"s3://{cfg.bucket}/{searchable_key}"
                    extraction_input_bytes = searchable_bytes
                    extraction_method = "textract_then_pymupdf4llm"
                    log.info("  Textract OCR done: %.1f KB", len(searchable_bytes) / 1024)
                except Exception as e:
                    log.warning("  Textract OCR failed, falling back to original: %s", e)
                    result["ocr_error"] = str(e)
            elif pdf_type == "empty":
                log.info("  empty PDF: no pages")
                result["status"] = "empty_pdf"
                results.append(result)
                continue

            # 4. Extract Markdown
            try:
                markdown_text, tables = extract_markdown_from_pdf(extraction_input_bytes)
            except Exception as e:
                log.error("  pymupdf4llm extraction failed: %s", e)
                result["status"] = "extraction_failed"
                result["error"] = str(e)
                result["extraction_method"] = extraction_method
                results.append(result)
                continue

            result["extraction_method"] = extraction_method
            result["char_count"] = len(markdown_text)
            result["table_count"] = len(tables)

            # 5. Language detection
            language, confidence = detect_language(markdown_text)
            result["language"] = language
            result["language_confidence"] = round(confidence, 4)
            log.info("  language: %s (conf=%.3f)", language, confidence)

            if is_excluded_language(language, cfg.exclude_languages):
                log.info("  skipped: language '%s' is in exclude list", language)
                result["status"] = "skipped_excluded_language"
                results.append(result)
                continue

            # 6. Save outputs
            md_key = f"{cfg.s3_prefix}/pdf_md/{base_no_ext}.md"
            s3_put(s3, cfg.bucket, md_key, markdown_text, "text/markdown")
            result["s3_markdown"] = f"s3://{cfg.bucket}/{md_key}"

            if tables:
                tables_key = f"{cfg.s3_prefix}/pdf_md/{base_no_ext}_tables.json"
                s3_put(s3, cfg.bucket, tables_key,
                       json.dumps(tables, indent=2, ensure_ascii=False),
                       "application/json")
                result["s3_tables_json"] = f"s3://{cfg.bucket}/{tables_key}"

            result["status"] = "success"

        except Exception as e:
            log.exception("  failed: %s", e)
            result["status"] = "error"
            result["error"] = str(e)

        results.append(result)

    return results

                                          for r in pdf_results

            "by_language": dict(Counter(r.get("language") for r in pdf_results
                                        if r.get("language"))),
            "outputs": pdf_results,
        },
    }
    s3 = _s3_client(aws_region)
    key = f"{s3_prefix}/pipeline_report.json"
    return s3_put(s3, bucket, key, json.dumps(report, indent=2, ensure_ascii=False),
                  "application/json")


# ============================================================================
# Top-level handler (unchanged interface)
# ============================================================================
def handler(event: Dict, context=None) -> Dict:
    """
    Lambda-style entry point. Same interface as before.
    
    Removed fields (no longer needed):
        - ocr_lang (Textract auto-detects language)
        - ocr_dpi still exists but now controls Textract rendering DPI
    
    New env var:
        - TEMP_BUCKET: S3 bucket for temporary Textract input (defaults to event.bucket)
    """
    _setup_logging()

    try:
        bucket = event.get("bucket", "ecmc-dev-bucket")
        s3_base_prefix = event.get("s3_base_prefix", "Ecmc_scrape_understand_bankruptcy")
        aws_region = event.get("aws_region", "us-east-1")

        start_url = event.get("start_url")
        if not start_url:
            return {"status": "error", "error": "event.start_url is required"}

        extra_seed_urls = event.get("extra_seed_urls") or []
        allow_full_domain = bool(event.get("allow_full_domain", False))
        section_prefixes = event.get("section_prefixes")
        if section_prefixes is None and not allow_full_domain:
            section_prefixes = [derive_section_prefix(start_url)]
        section_prefixes = [normalize_prefix(p) for p in (section_prefixes or [])]

        max_pages = event.get("max_pages")
        max_depth = event.get("max_depth", 4)
        web_delay = float(event.get("web_delay_seconds", 0.5))
        request_timeout = int(event.get("request_timeout_seconds", 60))

        exclude_languages = event.get("exclude_languages", ["spanish"])
        enable_ocr = bool(event.get("enable_ocr_for_image_pdfs", True))
        ocr_dpi = int(event.get("ocr_dpi", 200))  # Now Textract rendering DPI
        scrape_pdfs = bool(event.get("scrape_pdfs", True))

        base_domain = urlparse(start_url).netloc.replace("www.", "")
        run_folder = generate_folder_name(start_url)
        s3_prefix = f"{s3_base_prefix}/{run_folder}"

        log.info("=" * 70)
        log.info("ECMC SCRAPER v3 (Textract OCR) — starting run")
        log.info("  start_url:        %s", start_url)
        log.info("  s3 output prefix: s3://%s/%s/", bucket, s3_prefix)
        log.info("  scope:            %s",
                 "FULL DOMAIN" if allow_full_domain else section_prefixes)
        log.info("  max_pages:        %s", max_pages)
        log.info("  max_depth:        %s", max_depth)
        log.info("  exclude_langs:    %s", exclude_languages)
        log.info("  ocr enabled:      %s (Textract, dpi=%d)", enable_ocr, ocr_dpi)
        log.info("=" * 70)

        # Step 1: Crawl HTML
        crawl_cfg = CrawlConfig(
            bucket=bucket,
            s3_prefix=s3_prefix,
            base_domain=base_domain,
            section_prefixes=section_prefixes,
            allow_full_domain=allow_full_domain,
            delay=web_delay,
            max_pages=max_pages,
            max_depth=max_depth,
            request_timeout=request_timeout,
            aws_region=aws_region,
        )
        seeds = [start_url] + list(extra_seed_urls)
        html_outputs, pdf_catalog, html_stats = crawl_html_and_discover_pdfs(
            seeds, crawl_cfg,
        )

        # Step 2: Process PDFs with Textract OCR
        pdf_results: List[Dict] = []
        if scrape_pdfs and pdf_catalog:
            pdf_cfg = PdfConfig(
                bucket=bucket,
                s3_prefix=s3_prefix,
                exclude_languages=exclude_languages,
                enable_ocr_for_image_pdfs=enable_ocr,
                ocr_dpi=ocr_dpi,
                request_timeout=request_timeout,
                aws_region=aws_region,
            )
            pdf_results = process_pdfs(pdf_catalog, pdf_cfg)

        # Step 3: Write report
        report_uri = write_report(
            event=event,
            s3_prefix=s3_prefix,
            html_outputs=html_outputs,
            pdf_results=pdf_results,
            html_stats=html_stats,
            bucket=bucket,
            aws_region=aws_region,
        )

        return {
            "status": "success",
            "s3_output_prefix": f"s3://{bucket}/{s3_prefix}/",
            "run_folder": run_folder,
            "report_uri": report_uri,
            "html": {
                "pages_scraped": html_stats["scraped"],
                "pages_failed": html_stats["failed"],
                "deepest_depth_seen": html_stats["deepest_depth_seen"],
                "reject_stats": html_stats["reject_stats"],
                "outputs": html_outputs,
            },
            "pdfs": {
                "total_discovered": len(pdf_catalog),
                "processed": len(pdf_results),
                "by_status": dict(Counter(r.get("status", "unknown")
                                              for r in pdf_results)),
                "by_type": dict(Counter(r.get("pdf_type") for r in pdf_results
                                        if r.get("pdf_type"))),
                "by_extraction": dict(Counter(r.get("extraction_method")
                                              for r in pdf_results
                                              if r.get("extraction_method"))),
                "by_language": dict(Counter(r.get("language") for r in pdf_results
                                            if r.get("language"))),
                "outputs": pdf_results,
            },
        }

    except Exception as e:
        log.exception("Handler failed: %s", e)
        return {"status": "error", "error": str(e)}
